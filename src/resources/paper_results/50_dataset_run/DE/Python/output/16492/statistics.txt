import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid", context="notebook")

input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '16492', 'dataset.csv')
output_dir = os.path.join('.', 'src', 'resources', 'output', '16492')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '1_histogramm_odl_value.png')

df = pd.read_csv(input_path, sep=',', encoding='utf-8', dtype=str)
df = df.replace({"": np.nan, "NA": np.nan, "NaN": np.nan, "None": np.nan, "null": np.nan, "NULL": np.nan})

def _strip(x):
    return x.strip() if isinstance(x, str) else x

df = df.applymap(_strip)

for c in ['start_measure', 'end_measure']:
    if c in df.columns:
        df[c] = pd.to_datetime(df[c], errors='coerce')

if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

missing_share = df.isna().mean()

num_df = df.apply(pd.to_numeric, errors='coerce')
_ = num_df.corr(numeric_only=True)

value_series = df['value'] if 'value' in df.columns else pd.Series(dtype=float)
value_clean = value_series.dropna()

fig, ax = plt.subplots(figsize=(9, 6), dpi=150)

if value_clean.shape[0] == 0:
    ax.axis('off')
    ax.text(0.5, 0.5, 'Keine gültigen Messwerte in Spalte "value" verfügbar', ha='center', va='center', fontsize=13)
    plt.tight_layout()
    fig.savefig(output_path, bbox_inches='tight')
    plt.close(fig)
else:
    n = int(value_clean.shape[0])
    n_missing = int(value_series.shape[0] - n)
    missing_pct_value = 0.0 if value_series.shape[0] == 0 else n_missing / max(value_series.shape[0], 1)
    mean_v = float(np.nanmean(value_clean))
    median_v = float(np.nanmedian(value_clean))
    std_v = float(np.nanstd(value_clean, ddof=1)) if n > 1 else 0.0
    vmin = float(np.nanmin(value_clean))
    vmax = float(np.nanmax(value_clean))
    q1, q3 = np.nanpercentile(value_clean, [25, 75])
    iqr = float(q3 - q1)
    if n < 2:
        bins = 5
    else:
        if iqr <= 0:
            bins = int(np.clip(np.ceil(np.sqrt(n)), 5, 50))
        else:
            bw = 2 * iqr * (n ** (-1 / 3))
            if bw <= 0:
                bins = int(np.clip(np.ceil(np.sqrt(n)), 5, 50))
            else:
                bins = int(np.clip(np.ceil((vmax - vmin) / bw), 5, 60))
    hist_color = '#1f77b4'
    edge_color = '#ffffff'
    sns.histplot(value_clean, bins=bins, kde=False, color=hist_color, edgecolor=edge_color, alpha=0.9, ax=ax)
    ax.axvline(mean_v, color='#d62728', linestyle='-', linewidth=1.8, label='Mittelwert')
    ax.axvline(median_v, color='#2ca02c', linestyle='--', linewidth=1.8, label='Median')
    if std_v > 0 and vmax > vmin:
        xs = np.linspace(vmin, vmax, 300)
        bin_width = (vmax - vmin) / bins if bins > 0 else (vmax - vmin)
        pdf = (1.0 / (std_v * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((xs - mean_v) / std_v) ** 2)
        ax.plot(xs, pdf * n * max(bin_width, 1e-9), color='#9467bd', linewidth=1.6, label='Normalverteilung (geschätzt)')
    ax.set_title('Verteilung der Gamma-Ortsdosisleistung (ODL) – Lauf an der Pegnitz, Jan 2026', fontsize=13, pad=12)
    ax.set_xlabel('ODL-Wert [µSv/h]', fontsize=11)
    ax.set_ylabel('Häufigkeit', fontsize=11)
    span = vmax - vmin
    if span > 0:
        ax.set_xlim(vmin - 0.05 * span, vmax + 0.05 * span)
    ax.legend(frameon=True, fontsize=9)
    stats_text = (
        f"n: {n}\n"
        f"Fehlend: {n_missing} ({missing_pct_value:.1%})\n"
        f"Mittelwert: {mean_v:.3f} µSv/h\n"
        f"Median: {median_v:.3f} µSv/h\n"
        f"Std.-abw.: {std_v:.3f} µSv/h\n"
        f"Min/Max: {vmin:.3f} / {vmax:.3f} µSv/h"
    )
    ax.text(0.98, 0.98, stats_text, transform=ax.transAxes, ha='right', va='top', fontsize=9.5, bbox=dict(boxstyle='round,pad=0.35', fc='white', ec='#cccccc', alpha=0.95))
    plt.tight_layout()
    fig.savefig(output_path, bbox_inches='tight')
    plt.close(fig)

import os
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import geopandas as gpd
from shapely.geometry import Point

# Set style
sns.set_theme(style="whitegrid")

# Paths
input_path = r"./src/resources/data/study/16492/dataset.csv"
output_dir = r"./src/resources/output/16492"
os.makedirs(output_dir, exist_ok=True)
plot_filename = os.path.join(output_dir, "2_kartenansicht_messstation.png")

# Load data
df = pd.read_csv(input_path, sep=",", encoding="utf-8", dtype=str)

# Convert numeric and date columns
for col in ["value", "validated", "kenn"]:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors="coerce")
for col in ["start_measure", "end_measure"]:
    if col in df.columns:
        df[col] = pd.to_datetime(df[col], errors="coerce")

# Extract coordinates from 'geom' column (Format: POINT(lon lat))
geom_series = df.get("geom", pd.Series([np.nan] * len(df)))
coords = geom_series.fillna("").astype(str).str.extract(r"POINT\s*\(([-+]?\d*\.?\d+)\s+([-+]?\d*\.?\d+)\)")
coords.columns = ["lon", "lat"]
coords = coords.apply(pd.to_numeric, errors="coerce")

# Add coordinates to dataframe
df["lon"] = coords["lon"]
df["lat"] = coords["lat"]

# Filter for valid geometries
df_geo = df.dropna(subset=["lat", "lon"]).copy()

# Get the latest record for status display
if not df_geo.empty:
    if "end_measure" in df_geo.columns and not df_geo["end_measure"].isna().all():
        latest_row = df_geo.sort_values("end_measure").tail(1)
    else:
        latest_row = df_geo.tail(1)
else:
    latest_row = pd.DataFrame()

# Plotting
if latest_row.empty:
    fig, ax = plt.subplots(figsize=(8, 6))
    ax.text(0.5, 0.5, "Keine Geodaten verfügbar", ha="center", va="center")
    ax.axis("off")
    plt.savefig(plot_filename, dpi=300, bbox_inches="tight")
    plt.close(fig)
else:
    lat_pt = latest_row.iloc[0]["lat"]
    lon_pt = latest_row.iloc[0]["lon"]
    name_val = str(latest_row.iloc[0].get("name", "Lauf an der Pegnitz"))
    id_val = str(latest_row.iloc[0].get("id", "Unbekannt"))
    val_status = latest_row.iloc[0].get("validated", np.nan)
    
    # Determine color based on validation status
    if pd.isna(val_status):
        val_label = "Unbekannt"
        color = "#888888"
    else:
        is_valid = bool(int(val_status) != 0)
        val_label = "Ja" if is_valid else "Nein"
        color = "#1b9e77" if is_valid else "#d95f02"

    # Create map
    fig, ax = plt.subplots(figsize=(10, 8), dpi=150)
    
    # Load a basic world map for context
    try:
        world = gpd.read_file(gpd.datasets.get_path('naturalearth_lowres'))
        world.plot(ax=ax, color="#f0f0f0", edgecolor="#cccccc")
    except:
        # Fallback if dataset is missing
        ax.set_facecolor("#e0f2f1")

    # Plot the station
    ax.scatter(lon_pt, lat_pt, s=150, c=color, edgecolor="black", zorder=5, label=f"Status Validiert: {val_label}")
    
    # Set map limits around the point
    ax.set_xlim(lon_pt - 0.5, lon_pt + 0.5)
    ax.set_ylim(lat_pt - 0.4, lat_pt + 0.4)
    
    # Add annotation (Tooltip-style)
    info_text = f"Station: {name_val}\nID: {id_val}\nValidiert: {val_label}"
    ax.annotate(info_text, xy=(lon_pt, lat_pt), xytext=(10, 10),
                textcoords='offset points', fontsize=10, fontweight='bold',
                bbox=dict(boxstyle='round,pad=0.5', fc='white', ec='gray', alpha=0.9),
                arrowprops=dict(arrowstyle='->', connectionstyle='arc3,rad=0'))

    ax.set_title(f"Geografische Lage der Messstation: {name_val}", fontsize=14, pad=20)
    ax.set_xlabel("Längengrad")
    ax.set_ylabel("Breitengrad")
    
    plt.tight_layout()
    plt.savefig(plot_filename, dpi=300, bbox_inches="tight")
    plt.close(fig)

print(f"Visualisierung erfolgreich unter {plot_filename} gespeichert.")
