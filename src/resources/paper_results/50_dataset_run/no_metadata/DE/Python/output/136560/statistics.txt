import sys
import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

try:
    input_path = './src/resources/data/study\\136560/dataset.csv'
    output_dir = './src/resources/output\\136560'
    plot_name = '0_messzeit_nach_methode.png'
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(input_path, sep=',', encoding='utf-8', dtype=str)
    df.columns = [c.strip() for c in df.columns]
    for c in df.columns:
        if df[c].dtype == object:
            df[c] = df[c].astype(str).str.strip()
    df_eda = df.copy()
    df_eda['Ergebnis_clean'] = df_eda['Ergebnis'].str.replace('\u00a0', ' ', regex=False)
    df_eda['Ergebnis_clean'] = df_eda['Ergebnis_clean'].replace({'n.n.': np.nan, 'N.N.': np.nan, 'nn': np.nan, 'NN': np.nan})
    df_eda['Ergebnis_clean'] = df_eda['Ergebnis_clean'].str.replace(',', '.', regex=False)
    df_eda['Ergebnis_num'] = pd.to_numeric(df_eda['Ergebnis_clean'], errors='coerce')
    miss = df_eda.replace({'': np.nan}).isna().mean().rename('Anteil_fehlend')
    miss.to_frame().to_csv(os.path.join(output_dir, '0_eda_anteil_fehlender_werte.csv'), encoding='utf-8')
    numeric_summary = df_eda[['Ergebnis_num']].describe().T
    numeric_summary.to_csv(os.path.join(output_dir, '0_eda_numerische_zusammenfassung.csv'), encoding='utf-8')
    df_time = df_eda[df_eda['Pruefmerkmal'].str.lower() == 'messzeit'].copy()
    if 'Einheit' in df_time.columns:
        df_time = df_time[df_time['Einheit'].str.lower() == 's']
    df_time = df_time.dropna(subset=['Ergebnis_num'])
    if df_time.empty:
        raise ValueError('Keine Messzeit-Daten mit Einheit s gefunden.')
    times = df_time.groupby('Methode', as_index=False)['Ergebnis_num'].max()
    times = times.rename(columns={'Ergebnis_num': 'Messzeit_s'}).sort_values('Messzeit_s', ascending=False)
    probe_id = None
    if 'Hauptprobenummer' in df.columns and not df['Hauptprobenummer'].isna().all():
        probe_id = str(df['Hauptprobenummer'].dropna().iloc[0])
    sns.set_theme(style='whitegrid', context='talk')
    plt.figure(figsize=(9, 5.5))
    ax = sns.barplot(data=times, x='Methode', y='Messzeit_s', palette=['#4C78A8', '#F58518', '#54A24B', '#E45756'][:len(times)])
    ax.set_xlabel('Untersuchungsmethode')
    ax.set_ylabel('Messzeit (s)')
    if probe_id:
        ax.set_title(f'Vergleich der Messzeiten nach Untersuchungsmethode\nTrinkwasserprobe {probe_id}')
    else:
        ax.set_title('Vergleich der Messzeiten nach Untersuchungsmethode')
    for p in ax.patches:

    plt.savefig(os.path.join(output_dir, plot_name), dpi=300, bbox_inches='tight')

    def to_num_series(s):
        s_str = s.astype(str).str.replace(',', '.', regex=False)
        s_str = s_str.str.replace(r'[^0-9eE+\-\.]', '', regex=True)
        return pd.to_numeric(s_str, errors='coerce')
    df_eda['Ergebnis_num_all'] = to_num_series(df_eda['Ergebnis']) if 'Ergebnis' in df_columns else np.nan
    numeric_cols = list(df_eda.select_dtypes(include=[np.number]).columns)
    if 'Ergebnis_num_all' in df_eda.columns and 'Ergebnis_num_all' not in numeric_cols:
        numeric_cols.append('Ergebnis_num_all')
    if len(numeric_cols) > 0:
        desc = df_eda[numeric_cols].describe(include='all').T
        desc.to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'), encoding='utf-8')
        corr = df_eda[numeric_cols].corr(min_periods=1)
        corr.to_csv(os.path.join(output_dir, 'eda_correlation_matrix.csv'), encoding='utf-8')
    df_plot = df.copy()
    for col in ['Einheit', 'Pruefmerkmal', 'Methode']:
        if col in df_plot.columns:

    else:

    if 'Pruefmerkmal' in df_plot.columns:

    else:
