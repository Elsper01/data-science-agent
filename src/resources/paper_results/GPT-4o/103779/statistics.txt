library(ggplot2)
library(dplyr)
library(readr)

# Daten laden
# Pfad korrigiert (Backslashes in R Strings müssen oft escaped werden oder durch Forward-Slashes ersetzt werden)
file_path <- "./src/resources/data/study/103779/dataset.csv"

# Sicherstellen, dass die Datei existiert, bevor sie gelesen wird
if (file.exists(file_path)) {
  df <- read_csv(file_path, locale = locale(encoding = 'UTF-8'))

  # Daten aufbereiten: Anzahl der Kulturdenkmale pro Gemeinde
  # Wir filtern NA-Werte in der Gemeinde-Spalte aus, falls vorhanden
  kulturdenkmale_pro_gemeinde <- df %>% 
    filter(!is.na(Gemeinde)) %>%
    group_by(Gemeinde) %>% 
    summarise(Anzahl = n(), .groups = 'drop') %>% 
    arrange(desc(Anzahl))

  # Horizontales Balkendiagramm erstellen
  # Wir begrenzen ggf. auf die Top-Gemeinden, falls die Liste zu lang ist, 
  # aber hier zeigen wir alle wie im Ziel gefordert.
  plot <- ggplot(kulturdenkmale_pro_gemeinde, aes(x = Anzahl, y = reorder(Gemeinde, Anzahl))) +
    geom_col(fill = "steelblue") +
    labs(title = "Verteilung der Kulturdenkmale auf Gemeinden im Kreis Segeberg",
         subtitle = "Anzahl der Einträge pro Gemeinde",
         x = "Anzahl der Kulturdenkmale",
         y = "Gemeinde") +
    theme_minimal() +
    theme(axis.text.y = element_text(size = 8)) # Kleinere Schrift für bessere Lesbarkeit bei vielen Gemeinden

  # Verzeichnis erstellen falls nicht vorhanden (optional, ggsave macht das oft nicht automatisch)
  output_dir <- "./src/resources/output/103779"
  if (!dir.exists(output_dir)) {
    dir.create(output_dir, recursive = TRUE)
  }

  # Diagramm speichern
  ggsave(filename = paste0(output_dir, "/0_kulturdenkmale_pro_gemeinde.png"), 
         plot = plot, width = 10, height = 12, dpi = 300)
  
  print("Visualisierung erfolgreich erstellt.")
} else {
  stop(paste("Datei nicht gefunden:", file_path))
}

# Pakete laden (ohne install.packages, da dies in eingeschränkten Umgebungen oft fehlschlägt)
# Es wird davon ausgegangen, dass die Pakete in der Umgebung bereits vorhanden sind.
library(ggplot2)
library(dplyr)
library(readr)

# Daten laden
# Korrektur der Pfade: Verwendung von Vorwärts-Slashes für Kompatibilität
file_path <- './src/resources/data/study/103779/dataset.csv'

# Sicherstellen, dass die Datei existiert, bevor sie gelesen wird
if (file.exists(file_path)) {
  df <- read_csv(file_path, locale = locale(encoding = 'UTF-8'))

  # Datenvorbereitung
  # Wir gruppieren alles, was nicht "Alteintragung..." ist, als "Detaillierte Beschreibung"
  beschreibung_summary <- df %>% 
    mutate(Beschreibung_Typ = ifelse(Beschreibung == "Alteintragung (Aktualisierung vorgesehen)", 
                                      "Alteintragung", "Detaillierte Beschreibung")) %>%
    group_by(Beschreibung_Typ) %>%
    summarise(Anzahl = n()) %>%
    mutate(Prozent = round(Anzahl / sum(Anzahl) * 100, 1))

  # Visualisierung
  plot <- ggplot(beschreibung_summary, aes(x = Beschreibung_Typ, y = Anzahl, fill = Beschreibung_Typ)) +
    geom_bar(stat = "identity", width = 0.6) +
    geom_text(aes(label = paste0(Anzahl, " (", Prozent, "%)")), vyrust = -0.5, vjust = -0.5) +
    labs(title = "Datenqualität: Bearbeitungsstand der Beschreibungen",
         subtitle = "Verhältnis von Alteintragungen zu detaillierten Texten",
         x = "Kategorie",
         y = "Anzahl der Einträge",
         fill = "Typ") +
    theme_minimal() +
    scale_fill_manual(values = c("Alteintragung" = "#E69F00", "Detaillierte Beschreibung" = "#56B4E9")) +
    theme(legend.position = "none")

  # Output-Verzeichnis erstellen, falls es nicht existiert (optional, ggsave macht das oft)
  output_path <- './src/resources/output/103779/2_beschreibung_verhaeltnis.png'
  
  # Plot speichern
  ggsave(filename = output_path, plot = plot, width = 8, height = 6, dpi = 300)
  
  print("Visualisierung erfolgreich erstellt.")
} else {
  stop(paste("Datei nicht gefunden:", file_path))
}
