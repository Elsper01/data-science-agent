# Da es Versionskonflikte mit 'rlang' und 'ggplot2' gibt, 
# verwenden wir Basis-R für die Datenverarbeitung und die Visualisierung.

# Erstellen des Output-Verzeichnisses
output_dir <- './src/resources/output/51886'
if (!dir.exists(output_dir)) {
  dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)
}

# Pfad zur Datei
base_path <- "./src/resources/data/study 51886/"
file_name <- "dataset.csv"
full_path <- paste0(base_path, file_name)

# Überprüfung ob Datei existiert, ansonsten Suche im Verzeichnis
if (!file.exists(full_path)) {
  all_files <- list.files(path = "./src/resources/data", pattern = "dataset.csv", recursive = TRUE, full.names = TRUE)
  if (length(all_files) > 0) {
    full_path <- all_files[1]
  }
}

# Daten laden mit Basis-R (vermeidet readr/tidyverse Abhängigkeiten)
df <- read.csv(full_path, sep = ';', stringsAsFactors = FALSE, fileEncoding = 'UTF-8')

# Datenvorbereitung: Filtern nach 'Insgesamt'
# Spaltennamen in R werden oft automatisch bereinigt (z.B. X2_variable_attribute_label)
# Wir suchen die Spalte, die '2_variable_attribute_label' enthält
col_name <- grep("2_variable_attribute_label", names(df), value = TRUE)
if(length(col_name) == 0) col_name <- names(df)[2] # Fallback auf die zweite Spalte

df_gesamt <- df[df[[col_name]] == 'Insgesamt', ]

# Konvertierung der Werte
df_gesamt$time <- as.numeric(as.character(df_gesamt$time))
df_gesamt$value <- as.numeric(as.character(df_gesamt$value))

# Bereinigung von NAs und Sortierung
df_gesamt <- df_gesamt[!is.na(df_gesamt$time) & !is.na(df_gesamt$value), ]
df_gesamt <- df_gesamt[order(df_gesamt$time), ]

# Speichern der Visualisierung mit Basis-R Grafikgeräten
output_file <- paste0(output_dir, '/0_gesamtentwicklung_wohnungen.png')

png(filename = output_file, width = 1000, height = 600, res = 100)

if (nrow(df_gesamt) > 0) {
  # Plot erstellen
  plot(df_gesamt$time, df_gesamt$value, 
       type = "b", # "b" für both (Punkte und Linien)
       pch = 19, col = "#e74c3c", lwd = 2,
       xlab = "Jahr", ylab = "Anzahl Wohnungen",
       main = "Entwicklung der fertiggestellten Wohnungen in Oberschleißheim",
       sub = "Zeitraum: Gesamtzahl über die Jahre",
       axes = FALSE)
  
  # Achsen anpassen
  axis(1, at = seq(min(df_gesamt$time), max(df_gesamt$time), by = 1))
  axis(2)
  grid(nx = NA, ny = NULL, col = "lightgray", lty = "dotted")
  box()
} else {
  plot.new()
  text(0.5, 0.5, "Keine Daten für 'Insgesamt' gefunden")
}

dev.off()

cat(paste('Visualisierung erfolgreich erstellt und gespeichert unter:', output_file))

# Da es Probleme mit der rlang-Version in der Umgebung gibt, 
# laden wir die benötigten Pakete einzeln und vermeiden das Laden des gesamten tidyverse-Metapakets,
# um Abhängigkeitskonflikte zu minimieren.

library(ggplot2)
library(readr)
library(dplyr)
library(tidyr)

# Verzeichnis erstellen
dir.create('./src/resources/output/51886', recursive = TRUE, showWarnings = FALSE)

# Daten laden
df <- read_delim('./src/resources/data/study/51886/dataset.csv', 
                 delim = ';', 
                 locale = locale(encoding = 'UTF-8'),
                 show_col_types = FALSE)

# Datenvorbereitung
# Wir stellen sicher, dass 'time' chronologisch sortiert ist und 
# '2_variable_attribute_label' eine sinnvolle Reihenfolge für die Legende hat.
df_plot <- df %>% 
  filter(!is.na(value), value > 0) %>%
  # Extrahiere die Zahl aus dem Label für eine korrekte Sortierung (1 Raum, 2 Räume, etc.)
  mutate(time = as.factor(time),
         `2_variable_attribute_label` = factor(`2_variable_attribute_label`, 
                                               levels = sort(unique(`2_variable_attribute_label`))))

# Visualisierung: Gestapeltes Balkendiagramm
p <- ggplot(df_plot, aes(x = time, y = value, fill = `2_variable_attribute_label`)) +
  geom_col(position = 'stack', color = 'white', linewidth = 0.2) +
  scale_fill_viridis_d(option = 'viridis', name = 'Anzahl Räume') +
  labs(title = 'Baufertigstellungen nach Anzahl der Räume',
       subtitle = 'Oberschleißheim (Zeitverlauf)',
       x = 'Jahr',
       y = 'Anzahl der Wohnungen',
       caption = 'Quelle: Datensatz 51886') +
  theme_minimal(base_size = 12) +
  theme(axis.text.x = element_text(angle = 45, hjust = 1),
        plot.title = element_text(hjust = 0.5, face = 'bold'),
        plot.subtitle = element_text(hjust = 0.5),
        legend.position = 'bottom',
        panel.grid.minor = element_blank(),
        panel.grid.major.x = element_blank())

# Speichern des Diagramms
ggsave('./src/resources/output/51886/2_raumverteilung_zeitverlauf.png', 
       plot = p, 
       width = 12, 
       height = 7, 
       dpi = 300,
       bg = 'white')
