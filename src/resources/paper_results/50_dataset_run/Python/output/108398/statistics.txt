import pandas as pd

import seaborn as sns
import os
from matplotlib import dates as mdates

# Pfade definieren
input_path = r'./src/resources/data/study/108398/dataset.csv'
output_dir = r'./src/resources/output/108398'
plot_filename = '1_kkb_regen_odl.png'

# Sicherstellen, dass das Verzeichnis existiert
os.makedirs(output_dir, exist_ok=True)

# Fehlerbehebung: 'utf-8' Fehler deutet auf 'latin-1' (ISO-8859-1) oder 'cp1252' hin.
# Wir versuchen latin-1, da dies im deutschen Kontext (Umlaute wie in 'März') häufig ist.
try:
    df_raw = pd.read_csv(
        input_path,
        sep=',',
        encoding='latin-1',
        decimal='.',
        na_values=['nan', 'NaN', '', ' '],
        low_memory=False
    )
except Exception as e:
    # Fallback auf cp1252 falls latin-1 fehlschlägt
    try:
        df_raw = pd.read_csv(input_path, sep=',', encoding='cp1252', low_memory=False)
    except Exception as e2:
        print(f"Fehler beim Laden der Datei: {e2}")
        raise SystemExit(1)

# Bereinigung
cols_to_drop = [c for c in df_raw.columns if c.lower().startswith('unnamed') or df_raw[c].isna().all()]
if len(cols_to_drop) > 0:
    df_raw = df_raw.drop(columns=cols_to_drop)

if 'Datum_Uhrzeit' not in df_raw.columns:
    print('Fehlende Spalte: Datum_Uhrzeit')
    raise SystemExit(1)

# Zeitstempel konvertieren
df_raw['Datum_Uhrzeit_dt'] = pd.to_datetime(df_raw['Datum_Uhrzeit'], errors='coerce', dayfirst=True)
df = df_raw[df_raw['Datum_Uhrzeit_dt'].notna()].copy()

if df.empty:
    print('Keine gültigen Zeitstempel nach dem Filtern vorhanden.')
    raise SystemExit(1)

df = df.sort_values('Datum_Uhrzeit_dt')
df = df.set_index('Datum_Uhrzeit_dt')

# Numerische Konvertierung für die Zielspalten
rain_col = 'KKB_NIED_MET'
odl_col = 'KKB_1031_IMK'

for col in [rain_col, odl_col]:
    if col in df.columns:
        if df[col].dtype == object:
            df[col] = df[col].astype(str).str.replace(',', '.')
            df[col] = pd.to_numeric(df[col], errors='coerce')

if rain_col not in df.columns or odl_col not in df.columns:
    print(f'Erforderliche Spalten fehlen: {rain_col} und/oder {odl_col}')
    raise SystemExit(1)

# Korrelation berechnen
pairs = df[[rain_col, odl_col]].dropna()
r_spearman = pairs.corr(method='spearman').iloc[0, 1] if len(pairs) > 5 else np.nan

# Visualisierung
sns.set_theme(style='whitegrid')
fig, ax1 = plt.subplots(figsize=(14, 7), dpi=150)

x = df.index
rain = df[rain_col].fillna(0)

locator = mdates.AutoDateLocator(minticks=8, maxticks=14)
formatter = mdates.ConciseDateFormatter(locator)
ax1.xaxis.set_major_locator(locator)
ax1.xaxis.set_major_formatter(formatter)

# Primäre Achse: Niederschlag (Balken)
bar_color = '#4C78A8'

width = (x[1] - x[0]).total_seconds() / 86400 * 0.8 if len(x) > 1 else 0.01
ax1.bar(x, rain, width=width, color=bar_color, alpha=0.6, label='Niederschlag (mm/h)')
ax1.set_ylabel('Niederschlag (mm/h)', color=bar_color, fontweight='bold')
ax1.tick_params(axis='y', labelcolor=bar_color)
ax1.set_ylim(bottom=0, top=rain.max() * 1.5 if rain.max() > 0 else 1)

# Sekundäre Achse: ODL (Linie)

lines, labels = ax1.get_legend_handles_labels()

ax2.legend(lines + lines2, labels + labels2, loc='upper left', frameon=True)

corr_txt = f' | Spearman-Rangkorrelation: {r_spearman:.2f}' if not np.isnan(r_spearman) else ''
plt.title(f'Zusammenhang Niederschlag vs. Ortsdosisleistung (KKB)\nStandort: KKB_1031_IMK{corr_txt}', fontsize=14)

fig.tight_layout()
output_path = os.path.join(output_dir, plot_filename)

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Pfade definieren
in_path = r'./src/resources/data/study\108398\dataset.csv'
out_dir = r'./src/resources/output\108398'
os.makedirs(out_dir, exist_ok=True)
out_path = os.path.join(out_dir, '2_windrichtungen_windrose.png')

# Fehlerbehebung: UnicodeDecodeError durch Testen verschiedener Encodings beheben
# 'utf-8' schlug fehl mit 0xfc (ü), was oft auf 'latin-1' oder 'cp1252' hindeutet.
try:
    df = pd.read_csv(in_path, sep=',', encoding='utf-8', dtype=str, low_memory=False)
except UnicodeDecodeError:
    try:
        df = pd.read_csv(in_path, sep=',', encoding='latin-1', dtype=str, low_memory=False)
    except Exception:
        df = pd.read_csv(in_path, sep=',', encoding='cp1252', dtype=str, low_memory=False)

# Datenbereinigung
df = df.apply(lambda x: x.str.strip() if hasattr(x, 'str') else x)
df = df.replace({'nan': np.nan, 'NaN': np.nan, 'NA': np.nan, 'N/A': np.nan, 'NULL': np.nan, 'Null': np.nan, 'null': np.nan, 'None': np.nan, '': np.nan})

# Relevante Spalten für Windrichtung
dir_cols = [c for c in ['KBR_WRK_TIS', 'KKB_WRK_TIS', 'KKK_WRK_TIS'] if c in df.columns]

# Numerische Konvertierung der Windrichtungen
numeric_dirs = {}
for c in dir_cols:
    s = df[c].astype(str)
    s = s.str.replace(',', '.', regex=False)
    s = s.str.replace(r'[^0-9eE\.\-]+', '', regex=True)
    s = pd.to_numeric(s, errors='coerce')
    # Windrichtungen liegen normalerweise zwischen 0 und 360 Grad
    s = s.dropna()
    s = s[(s >= 0) & (s <= 360)]
    if not s.empty:
        numeric_dirs[c] = s

# Visualisierung als Windrose (Polardiagramm)
if numeric_dirs:
    sns.set_theme(style='whitegrid', context='talk')
    fig, ax = plt.subplots(subplot_kw={'projection': 'polar'}, figsize=(10, 8), dpi=150)
    
    # Norden oben, Uhrzeigersinn
    ax.set_theta_zero_location('N')
    ax.set_theta_direction(-1)

    n_bins = 16
    bin_edges = np.linspace(0, 360, n_bins + 1)
    # Zentren der Balken in Radiant
    rad_centers = np.deg2rad((bin_edges[:-1] + bin_edges[1:]) / 2)
    bar_width = np.deg2rad(360 / n_bins)

    palette = sns.color_palette('husl', len(numeric_dirs))
    labels_map = {
        'KBR_WRK_TIS': 'Brokdorf (KBR)',
        'KKB_WRK_TIS': 'Brunsbüttel (KKB)',
        'KKK_WRK_TIS': 'Krümmel (KKK)'
    }

    max_freq = 0
    for i, (col, vals) in enumerate(numeric_dirs.items()):
        counts, _ = np.histogram(vals, bins=bin_edges)
        # Relative Häufigkeit in Prozent
        freqs = (counts / len(vals)) * 100
        max_freq = max(max_freq, freqs.max())
        
        ax.bar(rad_centers, freqs, width=bar_width, bottom=0, 
               color=palette[i], alpha=0.5, edgecolor=palette[i], 
               label=labels_map.get(col, col))

    # Achsenbeschriftung
    ax.set_rticks(np.linspace(0, max_freq * 1.1, 5).round(1))
    ax.set_rlabel_position(135)
    
    # Himmelsrichtungen als Ticks
    ax.set_xticks(np.deg2rad(np.arange(0, 360, 45)))
    ax.set_xticklabels(['N', 'NO', 'O', 'SO', 'S', 'SW', 'W', 'NW'])

    plt.title('Verteilung der Windrichtungen (Windrose)\nStandorte KBR, KKB, KKK', pad=30)
    plt.legend(loc='lower center', bbox_to_anchor=(0.5, -0.2), ncol=3, frameon=True)
    
    plt.tight_layout()
    plt.savefig(out_path, bbox_inches='tight')
    plt.close()
else:
    print("Keine validen Windrichtungsdaten für die Visualisierung gefunden.")
