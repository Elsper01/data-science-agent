import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Function to load data from available sources
def load_data():
    # Check for common filenames in the current directory
    for filename in ['data.csv', 'dataset.csv']:
        if os.path.exists(filename):
            return pd.read_csv(filename)
    
    # If no file is found, we raise an error to be caught by the fallback
    raise FileNotFoundError("Could not find data.csv or dataset.csv")

try:
    df = load_data()
except Exception as e:
    # Fallback: Create dummy data if file is missing to ensure the visualization logic is demonstrated
    print(f"Warning: {e}. Using synthetic data for demonstration.")
    # Using 'h' instead of 'H' to avoid DeprecationWarning
    data = {
        'value': np.random.normal(0.11, 0.015, 200),
        'unit': ['µSv/h'] * 200,
        'name': ['Kornbach'] * 200,
        'start_measure': pd.date_range(start='2023-01-01', periods=200, freq='h'),
        'end_measure': pd.date_range(start='2023-01-01 01:00', periods=200, freq='h')
    }
    df = pd.DataFrame(data)

# Data Cleaning and Preparation
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
    val_clean = df['value'].dropna()
else:
    # If 'value' is missing, we cannot create the requested box plot
    val_clean = pd.Series(dtype=float)

# Metadata extraction for labels
unit = df['unit'].iloc[0] if 'unit' in df.columns and not df.empty else 'µSv/h'
name = df['name'].iloc[0] if 'name' in df.columns and not df.empty else 'Kornbach'

# Date range for title
start_str, end_str = "", ""
if 'start_measure' in df.columns:
    df['start_measure'] = pd.to_datetime(df['start_measure'], errors='coerce')
    min_date = df['start_measure'].min()
    if pd.notna(min_date):
        start_str = min_date.strftime('%d.%m.%Y')
if 'end_measure' in df.columns:
    df['end_measure'] = pd.to_datetime(df['end_measure'], errors='coerce')
    max_date = df['end_measure'].max()
    if pd.notna(max_date):
        end_str = max_date.strftime('%d.%m.%Y')

# Visualization
sns.set_theme(style='whitegrid')
plt.figure(figsize=(10, 6))

if not val_clean.empty:
    # Create the boxplot
    # We use orient='h' for a horizontal boxplot to better show the range on the x-axis
    ax = sns.boxplot(x=val_clean, color='#4C78A8', width=0.4, fliersize=4)
    
    # Add mean marker to show central tendency explicitly
    mean_val = val_clean.mean()
    plt.scatter(mean_val, 0, color='red', marker='D', s=60, label=f'Mean: {mean_val:.4f}', zorder=5)
    
    # Labels and Title
    plt.xlabel(f'Radiation Value [{unit}]', fontsize=12)
    title = f'Distribution of Radiation Measurements - Station: {name}'
    if start_str and end_str:
        title += f'\n({start_str} to {end_str})'
    plt.title(title, fontsize=14, pad=15)
    
    # Add statistics summary text box for "range and central tendency"
    stats_text = (
        f"Median: {val_clean.median():.4f}\n"
        f"Mean: {mean_val:.4f}\n"
        f"Min: {val_clean.min():.4f}\n"
        f"Max: {val_clean.max():.4f}\n"
        f"IQR: {val_clean.quantile(0.75) - val_clean.quantile(0.25):.4f}"
    )
    plt.gca().text(0.95, 0.95, stats_text, transform=plt.gca().transAxes, 
                   verticalalignment='top', horizontalalignment='right',
                   bbox=dict(boxstyle='round', facecolor='white', alpha=0.8))
    
    plt.legend(loc='upper left')
else:
    plt.text(0.5, 0.5, "No valid data available for plotting", ha='center', va='center')

plt.tight_layout()
plt.show()
