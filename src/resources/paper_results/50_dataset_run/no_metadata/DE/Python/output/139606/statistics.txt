import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].astype(str).str.strip()

missing_per_col = df.isna().sum()

df['Ergebnis_num'] = pd.to_numeric(df['Ergebnis'].str.replace('\u202f', '', regex=True).str.replace(' ', '', regex=False).str.replace(',', '.', regex=False), errors='coerce')

counts = df.assign(**{
    'Nicht nachweisbar': (df['Status'] == 'Nicht nachweisbar').astype(int),
    'Nachweisbar': (df['Status'] == 'Nachweisbar').astype(int)
})[['Pruefmerkmal', 'Nicht nachweisbar', 'Nachweisbar']]

order_df = df[['Pruefmerkmal', 'Status']].copy()
order_df['Status_num'] = order_df['Status'].cat.codes

counts = counts.groupby('Pruefmerkmal', as_index=True).sum().reindex(order)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Unterdrücke Warnungen für sauberere Ausgabe
pd.options.mode.chained_assignment = None

path = './src/resources/data/study/139606/dataset.csv'

    df = pd.read_csv(path, sep=',', encoding='utf-8')
except Exception as e:
    print(f'Fehler beim Laden der Datei: {e}')
    raise SystemExit(1)

# Hilfsfunktion zur Konvertierung deutscher/unsauberer Zahlenformate
def to_numeric_de(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    if s.lower() in ['n.n.', 'n.n', 'nn', 'na', 'nan', 'k.a.', '']:
        return np.nan
    # Entferne Tausendertrennzeichen (schmales Leerzeichen oder Leerzeichen) und ersetze Komma
    s = s.replace('\u202f', '').replace(' ', '').replace(',', '.')
    try:
        return float(s)
    except Exception:
        return pd.to_numeric(s, errors='coerce')

# Datenvorbereitung
# Ersetze applymap durch map (FutureWarning Fix)
mask_messzeit = df['Pruefmerkmal'].str.strip().str.lower() == 'messzeit'
df_mz = df[mask_messzeit]

if df_mz.empty:
    print('Fehler: Kein Eintrag für Pruefmerkmal "Messzeit" gefunden.')
    raise SystemExit(1)

# Extrahiere den Wert
val_sec = to_numeric_de(df_mz.iloc[0]['Ergebnis'])
if pd.isna(val_sec):
    print('Fehler: Der Messzeit-Wert konnte nicht als Zahl interpretiert werden.')
    raise SystemExit(1)

# Berechnungen für die Anzeige
hours = float(val_sec) / 3600.0
sec_str = f"{int(round(val_sec)):,}".replace(',', '.')
h_str = f"{hours:.1f}".replace('.', ',')

# Visualisierung
sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(10, 3.5), dpi=150)

color = '#2C7FB8'
ax.barh(['Messzeit'], [val_sec], color=color, height=0.5)

# Achsenbeschriftungen
ax.set_xlabel('Sekunden [s]', fontsize=12)
ax.set_ylabel('')
ax.set_title('Analyseaufwand: Messzeit der Gammaspektrometrie', fontsize=14, pad=20)

# Skalierung
max_x = val_sec * 1.25
ax.set_xlim(0, max_x)
