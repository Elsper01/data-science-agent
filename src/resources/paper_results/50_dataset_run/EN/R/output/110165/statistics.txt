make_na <- function(x) {
  if (is.character(x) || is.factor(x)) {
    y <- as.character(x)
    y <- stringr::str_replace_all(y, "\r\n|\n|\r", " ")
    y <- stringr::str_squish(y)
    y[stringr::str_to_lower(y) %in% c("", "na", "n/a", "nan", "null", "keine angabe")] <- NA_character_
    return(y)
  } else {
    return(x)
  }
}

clean_string <- function(x) {
  if (!is.character(x)) return(x)
  # Remove invalid UTF-8 characters
  x <- iconv(x, "UTF-8", "UTF-8", sub = "")
  x <- trimws(x)
  x[tolower(x) %in% c("nan", "na", "null", "")] <- NA_character_
  return(x)
}
