clean_num <- function(x){
  y <- ifelse(tolower(trimws(x)) %in% c("n.n.","n.n","<nachweisgrenze","< ld","<ld","na",""), NA, x)
  y <- stringr::str_replace_all(y, "\\.", ":")
  y <- stringr::str_replace_all(y, ",", ".")
  y <- stringr::str_replace_all(y, ":", "")
  suppressWarnings(as.numeric(y))
}
