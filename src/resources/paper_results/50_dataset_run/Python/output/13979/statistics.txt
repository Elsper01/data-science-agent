import seaborn as sns

    # Fallback auf cp1252 falls iso-8859-1 auch scheitert

df.columns = [c.strip() for c in df.columns]

for col in ['time', 'value']:

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

try:
    # Pfade definieren
    input_path = './src/resources/data/study/13979/dataset.csv'
    output_dir = './src/resources/output/13979'
    os.makedirs(output_dir, exist_ok=True)

    # Fehlerbehebung: 'utf-8' codec error deutet auf latin-1 oder cp1252 hin (oft bei deutschen CSVs)
    # Wir versuchen 'iso-8859-1' (latin-1), was oft bei Umlauten (Sünching) hilft
    try:
        df = pd.read_csv(input_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1')

    # Datenbereinigung
    for col in df.columns:
        if df[col].dtype == object:
            df[col] = df[col].replace(['', ' ', 'nan', 'NaN'], np.nan)
    
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')
    if 'time' in df.columns:
        df['time'] = pd.to_numeric(df['time'], errors='coerce')

    # EDA Exporte
    missing = df.isna().mean().to_frame('anteil_fehlender_werte')
    missing.to_csv(os.path.join(output_dir, 'eda_anteil_fehlender_werte.csv'), encoding='utf-8')
    
    num_df = df.select_dtypes(include=[np.number])
    if not num_df.empty:
        desc = num_df.describe().T
        desc.to_csv(os.path.join(output_dir, 'eda_kennzahlen_numerisch.csv'), encoding='utf-8')
        corr = num_df.corr(numeric_only=True)
        corr.to_csv(os.path.join(output_dir, 'eda_korrelationen.csv'), encoding='utf-8')

    # Filterung gemäß Visualisierungsziel
    # 1. Nur Sünching (falls Spalte vorhanden)
    if '1_variable_attribute_label' in df.columns:
        df = df[df['1_variable_attribute_label'].astype(str).str.strip().isin(['Sünching', 'Sünching'])]
    
    # 2. Filter auf Raumzahl-Kategorien, ohne 'Insgesamt'
    mask = pd.Series(True, index=df.index)
    if '2_variable_label' in df.columns:
        mask = mask & df['2_variable_label'].astype(str).str.contains('Raumzahl', case=False, na=False)
    
    if '2_variable_attribute_label' in df.columns:
        mask = mask & df['2_variable_attribute_label'].notna()
        mask = mask & (~df['2_variable_attribute_label'].astype(str).str.strip().str.contains('Insgesamt', case=False, na=False))

    plot_df = df.loc[mask].copy()
    
    if plot_df.empty:
        raise ValueError('Keine Daten nach Filterung für die Visualisierung verfügbar.')

    # Aggregation
    agg = plot_df.groupby('2_variable_attribute_label', dropna=False, as_index=False)['value'].sum()
    agg = agg.dropna(subset=['2_variable_attribute_label'])
    agg = agg.sort_values('value', ascending=False)

    # Zeitspanne für Titel ermitteln
    start, end = None, None
    if 'time' in df.columns and df['time'].notna().any():
        start = int(df['time'].min())
        end = int(df['time'].max())

    # Visualisierung
    sns.set_theme(style='whitegrid')
    plt.figure(figsize=(12, 7))
    
    ax = sns.barplot(data=agg, x='2_variable_attribute_label', y='value', palette='viridis')
    
    ax.set_xlabel('Wohnungsgröße (Anzahl der Räume)')
    
    title = 'Baufertigstellungen in Sünching nach Wohnungsgröße'
    ylabel = 'Summe der fertiggestellten Wohnungen'
    if start and end:
        title += f' ({start}–{end})'
        ylabel += f' ({start}–{end})'
    
    ax.set_ylabel(ylabel)
    ax.set_title(title, fontweight='bold', pad=20)
    
    # Beschriftungen rotieren
    plt.xticks(rotation=30, ha='right')

    # Werte an die Balken schreiben
    for p in ax.patches:
        h = p.get_height()
        if np.isfinite(h):
            ax.annotate(f'{int(round(h))}', 
                        (p.get_x() + p.get_width() / 2., h), 
                        ha='center', va='bottom', 
                        xytext=(0, 5), 
                        textcoords='offset points', 
                        fontsize=10, fontweight='bold')

    plt.tight_layout()
    out_path = os.path.join(output_dir, '1_baufertigstellungen_nach_raumzahl_summe.png')
    plt.savefig(out_path, dpi=200)
    plt.close()

except Exception as e:
    print(f"Fehler aufgetreten: {str(e)}")

import os
import sys
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Konfiguration für Visualisierung
sns.set_theme(style="whitegrid", context="talk")
plt.rcParams["figure.dpi"] = 120

# Datenpfad definieren
data_path = os.path.join(".", "src", "resources", "data", "study", "13979", "dataset.csv")

# Fehlerbehebung: 'utf-8' codec can't decode byte 0xfc (ü)
# Da die Fehlermeldung auf ein Encoding-Problem hinweist (0xfc ist 'ü' in ISO-8859-1/Latin-1),
# versuchen wir 'latin-1' oder 'iso-8859-1'.
try:
    df = pd.read_csv(data_path, sep=";", encoding="latin-1")
except Exception:
    try:
        df = pd.read_csv(data_path, sep=";", encoding="utf-8-sig")
    except Exception as e:
        print(f"Fehler beim Laden der Datei: {e}")
        sys.exit(1)

# Datenbereinigung
for col in df.columns:
    if df[col].dtype == object:
        df[col] = df[col].replace({"": np.nan})

# Konvertierung von Zeit und Werten
if "time" in df.columns:
    df["time"] = pd.to_numeric(df["time"], errors="coerce")
if "value" in df.columns:
    df["value"] = pd.to_numeric(df["value"], errors="coerce")

# Filterung gemäß Visualisierungsziel: '2_variable_attribute_label' ohne 'Insgesamt'
df_filtered = df.copy()

# Wir filtern auf die Variable für Raumzahl (RAUM07 laut vorherigem Code-Kontext)
if "2_variable_code" in df_filtered.columns:
    df_filtered = df_filtered[df_filtered["2_variable_code"].astype(str) == "RAUM07"]

# 'Insgesamt' ausschließen
if "2_variable_attribute_label" in df_filtered.columns:
    df_filtered = df_filtered[~df_filtered["2_variable_attribute_label"].astype(str).str.contains("insgesamt", case=False, na=False)]

# Fehlende Werte entfernen
df_filtered = df_filtered.dropna(subset=["time", "value", "2_variable_attribute_label"])

# Sortierung der Raumgrößen für eine logische Legende (1 Raum, 2 Räume, etc.)
def extract_number(label):
    import re
    nums = re.findall(r'\d+', str(label))
    return int(nums[0]) if nums else 999

if not df_filtered.empty:
    unique_labels = df_filtered["2_variable_attribute_label"].unique()
    sorted_labels = sorted(unique_labels, key=extract_number)
    
    # Pivot-Tabelle für gestapeltes Diagramm
    pivot_df = df_filtered.pivot_table(
        index="time", 
        columns="2_variable_attribute_label", 
        values="value", 
        aggfunc="sum"
    ).fillna(0)
    
    # Spalten nach der extrahierten Nummer sortieren
    existing_sorted = [l for l in sorted_labels if l in pivot_df.columns]
    pivot_df = pivot_df[existing_sorted]

    # Plot erstellen
    fig, ax = plt.subplots(figsize=(14, 8))
    
    pivot_df.plot(kind="bar", stacked=True, ax=ax, colormap="viridis", width=0.8)
    
    ax.set_title("Wohnraumstruktur in Sünching (Baufertigstellungen)", pad=20)
    ax.set_xlabel("Jahr")
    ax.set_ylabel("Anzahl fertiggestellter Wohnungen")
    ax.legend(title="Raumzahl", bbox_to_anchor=(1.02, 1), loc="upper left")
    
    plt.xticks(rotation=45)
    plt.tight_layout()

    # Speichern
    out_dir = os.path.join(".", "src", "resources", "output", "13979")
    os.makedirs(out_dir, exist_ok=True)
    out_path = os.path.join(out_dir, "2_gestapeltes_balkendiagramm_raumzahl_nach_jahr.png")
    plt.savefig(out_path, dpi=300, bbox_inches="tight")
    plt.close()
else:
    print("Keine Daten nach Filterung vorhanden.")
