import numpy as np
import pandas as pd

from matplotlib.ticker import FuncFormatter

sns.set_theme(style="whitegrid", context="talk")

output_dir = os.path.normpath('./src/resources/output\\104336')
os.makedirs(output_dir, exist_ok=True)

df_raw = pd.read_csv(input_path, sep=',', encoding='utf-8')

def is_missing_series(s):
    s_str = s.astype(str).str.strip()
    return s.isna() | s_str.isin(['', '-', 'nan', 'NaN', 'None'])

missing_prop = {col: is_missing_series(df_raw[col]).mean() for col in df_raw.columns}

df = df_raw.copy()
for c in ['Bund_in_Euro', 'Land_in_Euro', 'Gesamt_in_Euro']:
    if c in df.columns:
        df[c] = df[c].astype(str).str.strip().replace('-', 0)
        df[c] = df[c].str.replace('.', '', regex=False).str.replace(',', '', regex=False)
        df[c] = pd.to_numeric(df[c], errors='coerce').fillna(0).astype(int)

if 'F_rderung_in_2014' in df.columns:
    df['is_total_row'] = df['F_rderung_in_2014'].astype(str).str.strip().str.lower().eq('gesamt:')
else:
    df['is_total_row'] = False

calc_df = df.copy()
calc_df['Summe_aus_Teilen'] = calc_df.get('Bund_in_Euro', 0) + calc_df.get('Land_in_Euro', 0)
calc_df['Abweichung_zu_Gesamt'] = calc_df.get('Gesamt_in_Euro', 0) - calc_df['Summe_aus_Teilen']

plot_df = df.loc[~df['is_total_row']].copy()
plot_df['Projekt'] = plot_df['F_rderung_in_2014'].astype(str).str.replace(r'^\d+\.\s*', '', regex=True)
plot_df['Projekt_wrapped'] = plot_df['Projekt'].apply(lambda x: '\n'.join(textwrap.wrap(x, width=24)))

bund_total = int(plot_df['Bund_in_Euro'].sum()) if 'Bund_in_Euro' in plot_df else 0
land_total = int(plot_df['Land_in_Euro'].sum()) if 'Land_in_Euro' in plot_df else 0

anomaly_rows = calc_df.loc[calc_df['Abweichung_zu_Gesamt'] != 0]
contains_anomaly = not anomaly_rows.empty

plt.figure(figsize=(14, 8))
colors = sns.color_palette('Set2', 2)
x = np.arange(len(plot_df))
width = 0.6
bund_vals = plot_df['Bund_in_Euro'].values if 'Bund_in_Euro' in plot_df else np.zeros(len(plot_df))
land_vals = plot_df['Land_in_Euro'].values if 'Land_in_Euro' in plot_df else np.zeros(len(plot_df))

b1 = plt.bar(x, bund_vals, width=width, color=colors[0], label='Bund')
b2 = plt.bar(x, land_vals, bottom=bund_vals, width=width, color=colors[1], label='Land')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

dataset_path = './src/resources/data/study/104336/dataset.csv'
output_path = './src/resources/output/104336/2_gruppierte_balken_summe_vs_gesamt.png'

sns.set_theme(style='whitegrid', context='talk', font='DejaVu Sans')

df = pd.read_csv(dataset_path, sep=',', encoding='utf-8')

df_eda = df.replace(['-', '–', '—', '', ' '], np.nan)
missing_ratio = df_eda.isna().mean()

num_cols = ['Bund_in_Euro', 'Land_in_Euro', 'Gesamt_in_Euro']
df[num_cols] = df[num_cols].replace(['-', '–', '—', '', ' '], np.nan)
for c in num_cols:
    df[c] = pd.to_numeric(df[c], errors='coerce')

df['Bund_in_Euro'] = df['Bund_in_Euro'].fillna(0)
df['Land_in_Euro'] = df['Land_in_Euro'].fillna(0)

numeric_stats = df[num_cols].describe()
correlations = df[num_cols].corr()

df['Summe_Bund_Land'] = df['Bund_in_Euro'] + df['Land_in_Euro']
df['Einrichtung'] = df['F_rderung_in_2014'].astype(str).str.replace(r'^\s*\d+\.\s*', '', regex=True).str.replace(':', '').str.strip()

plot_df = df[['Einrichtung', 'Summe_Bund_Land', 'Gesamt_in_Euro']].copy()
plot_df['Abweichung'] = plot_df['Gesamt_in_Euro'] - plot_df['Summe_Bund_Land']
plot_df['Mismatch'] = plot_df['Abweichung'].abs() > 1

x = np.arange(len(plot_df))
width = 0.4
colors = sns.color_palette('colorblind', 2)

fig, ax = plt.subplots(figsize=(14, 8))
bar1 = ax.bar(x - width/2, plot_df['Summe_Bund_Land'].values, width=width, label='Summe Bund + Land', color=colors[0], edgecolor='black')
bar2 = ax.bar(x + width/2, plot_df['Gesamt_in_Euro'].values, width=width, label='Gemeldeter Gesamtbetrag', color=colors[1], edgecolor='black')

ax.set_title('Abgleich der Finanzierungsquellen mit gemeldetem Gesamtbetrag (2014)')
ax.set_xlabel('Einrichtung')
ax.set_ylabel('Betrag in Euro')
ax.set_xticks(x)
ax.set_xticklabels(plot_df['Einrichtung'].tolist(), rotation=25, ha='right')
ax.legend(loc='upper left')

for bars in [bar1, bar2]:
    for b in bars:
        h = b.get_height()
        ax.text(b.get_x() + b.get_width()/2, h, f"{int(round(h)):,}".replace(',', '.'), ha='center', va='bottom', fontsize=10, rotation=0)

for i, is_mismatch in enumerate(plot_df['Mismatch'].tolist()):
    if is_mismatch:
        y_top = max(plot_df['Summe_Bund_Land'].iloc[i], plot_df['Gesamt_in_Euro'].iloc[i])
        diff = int(plot_df['Abweichung'].iloc[i])
        sign = '+' if diff > 0 else ''
        ax.annotate(f'Abweichung: {sign}{diff:,} €'.replace(',', '.'), xy=(x[i], y_top), xytext=(x[i], y_top*1.08 if y_top > 0 else 1000), ha='center', arrowprops=dict(arrowstyle='->', color='red', lw=1.5), color='red', fontsize=11)

fig.tight_layout()
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close(fig)
