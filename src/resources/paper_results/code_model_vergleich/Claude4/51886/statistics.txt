import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xdf' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German datasets).
try:
    df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='latin-1')

# Filter out 'Insgesamt' as per the visualization goal
df_filtered = df[df['2_variable_attribute_label'] != 'Insgesamt'].copy()

# Ensure 'value' is numeric
df_filtered['value'] = pd.to_numeric(df_filtered['value'], errors='coerce')
df_filtered = df_filtered.dropna(subset=['value'])

# Group by room category and sum the values
room_distribution = df_filtered.groupby('2_variable_attribute_label')['value'].sum().reset_index()

# Sort by the label to ensure a logical order (e.g., 1 room, 2 rooms, etc.)
# If labels are strings like "1 Raum", "2 Räume", we sort them.
room_distribution = room_distribution.sort_values('2_variable_attribute_label')

# Visualization
plt.figure(figsize=(12, 7))
sns.set_style("whitegrid")

# Create the bar chart
bars = plt.bar(room_distribution['2_variable_attribute_label'], 
               room_distribution['value'], 
               color=sns.color_palette("Blues_d", len(room_distribution)),
               edgecolor='black')

# Add labels on top of bars
for bar in bars:
    height = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2., height + (height * 0.01),
             f'{int(height)}',
             ha='center', va='bottom', fontsize=10, fontweight='bold')

# Formatting
plt.xlabel('Anzahl der Räume (Wohnungsgröße)', fontsize=12, fontweight='bold')
plt.ylabel('Summe der fertiggestellten Wohnungen', fontsize=12, fontweight='bold')
plt.title('Verteilung der Wohnungsgrößen in Oberschleißheim\n(Summe über den gesamten Zeitraum)', 
          fontsize=14, fontweight='bold', pad=15)
plt.xticks(rotation=45, ha='right')

plt.tight_layout()

# Save the output
plt.savefig('./src/resources/output/51886/1_apartment_size_distribution.png', dpi=300)
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# The error 'UnicodeDecodeError' with byte 0xdf (ß) suggests 'latin-1' or 'iso-8859-1' encoding
# since the data likely contains German characters (e.g., 'Räume', 'Oberschleißheim').
try:
    df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='latin-1')

# Filter out 'Insgesamt' as requested
df_filtered = df[df['2_variable_attribute_label'] != 'Insgesamt'].copy()

# Ensure 'value' is numeric
df_filtered['value'] = pd.to_numeric(df_filtered['value'], errors='coerce').fillna(0)

# Pivot the data for the stacked bar chart
pivot_data = df_filtered.pivot_table(
    index='time',
    columns='2_variable_attribute_label',
    values='value',
    aggfunc='sum',
    fill_value=0
)

# Define a logical order for room categories if they exist in the data
room_order = ['1 Raum', '2 Räume', '3 Räume', '4 Räume', '5 Räume', '6 Räume', '7 oder mehr Räume']
available_rooms = [room for room in room_order if room in pivot_data.columns]
# Add any other categories that might be present but not in the predefined list
other_rooms = [col for col in pivot_data.columns if col not in available_rooms]
final_order = available_rooms + other_rooms
pivot_data = pivot_data[final_order]

# Plotting
fig, ax = plt.subplots(figsize=(12, 7))

# Using a stacked bar chart. 
# Note: If there are negative values, standard stacked bars in pandas might overlap 
# or behave unexpectedly. However, for construction net changes, this is the requested format.
pivot_data.plot(
    kind='bar',
    stacked=True,
    ax=ax,
    width=0.8,
    edgecolor='white',
    linewidth=0.5
)

# Customizing the chart
ax.set_xlabel('Jahr', fontsize=11)
ax.set_ylabel('Anzahl fertiggestellter Wohnungen (Netto)', fontsize=11)
ax.set_title('Zusammensetzung der Wohnungsgrößen im Zeitverlauf (Netto-Fertigstellungen)', 
             fontsize=13, fontweight='bold', pad=15)

ax.legend(title='Raumzahl', bbox_to_anchor=(1.05, 1), loc='upper left')
ax.axhline(0, color='black', linewidth=0.8) # Highlight the zero line for negative values
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.6)

plt.tight_layout()

# Save the output
plt.savefig('./src/resources/output/51886/2_stacked_bar_apartment_sizes.png', dpi=300, bbox_inches='tight')
