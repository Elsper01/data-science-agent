import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('./src/resources/data/study/32830/dataset.csv', sep=',', encoding='utf-8')

df_clean = df[df['Datum'].notna()].copy()

df_clean['Bounce_Rate_numeric'] = df_clean['Bounce_Rate'].str.replace('%', '').astype(float)

df_analysis = df_clean[['Seiten_pro_Besuch', 'Bounce_Rate_numeric']].dropna()

correlation = df_analysis['Seiten_pro_Besuch'].corr(df_analysis['Bounce_Rate_numeric'])

plt.figure(figsize=(10, 6))
sns.scatterplot(data=df_analysis, x='Seiten_pro_Besuch', y='Bounce_Rate_numeric', 
                s=100, alpha=0.6, color='#2E86AB')

z = np.polyfit(df_analysis['Seiten_pro_Besuch'], df_analysis['Bounce_Rate_numeric'], 1)
p = np.poly1d(z)
plt.plot(df_analysis['Seiten_pro_Besuch'], p(df_analysis['Seiten_pro_Besuch']), 
         "r--", alpha=0.8, linewidth=2, label=f'Trendlinie (r={correlation:.3f})')

plt.xlabel('Seiten pro Besuch', fontsize=12, fontweight='bold')
plt.ylabel('Absprungrate (%)', fontsize=12, fontweight='bold')
plt.title('Zusammenhang zwischen Seiten pro Besuch und Absprungrate', 
          fontsize=14, fontweight='bold', pad=20)
plt.legend(fontsize=10)
plt.grid(True, alpha=0.3, linestyle='--')

plt.tight_layout()
plt.savefig('./src/resources/output/32830/1_seiten_pro_besuch_vs_bounce_rate.png', 
            dpi=300, bbox_inches='tight')
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime

df = pd.read_csv('./src/resources/data/study/32830/dataset.csv', sep=',', encoding='utf-8')

df = df[df['Datum'].notna()]
df = df[df['Datum'] != 'nan']

df['Datum'] = pd.to_datetime(df['Datum'], format='%d.%m.%Y', errors='coerce')
df = df.dropna(subset=['Datum'])

df = df.sort_values('Datum')

def convert_time_to_seconds(time_str):
    if pd.isna(time_str) or time_str == '-':
        return np.nan
    try:
        parts = time_str.split(':')
        if len(parts) == 3:
            hours, minutes, seconds = map(int, parts)
            return hours * 3600 + minutes * 60 + seconds
        return np.nan
    except:
        return np.nan

df['Zeit_im_Besuch_Sekunden'] = df['Zeit_im_Besuch'].apply(convert_time_to_seconds)

print("Missing values per column:")
print(df[['Datum', 'Seitenaufrufe', 'Zeit_im_Besuch_Sekunden']].isnull().sum())
print("\nBasic statistics:")
print(df[['Seitenaufrufe', 'Zeit_im_Besuch_Sekunden']].describe())
print("\nCorrelation between Seitenaufrufe and Zeit_im_Besuch_Sekunden:")
print(df[['Seitenaufrufe', 'Zeit_im_Besuch_Sekunden']].corr())

fig, ax1 = plt.subplots(figsize=(14, 7))

color1 = '#2E86AB'
ax1.set_xlabel('Datum', fontsize=12, fontweight='bold')
ax1.set_ylabel('Seitenaufrufe (Anzahl)', color=color1, fontsize=12, fontweight='bold')
line1 = ax1.plot(df['Datum'], df['Seitenaufrufe'], color=color1, linewidth=2.5, marker='o', markersize=5, label='Seitenaufrufe')
ax1.tick_params(axis='y', labelcolor=color1)
ax1.grid(True, alpha=0.3, linestyle='--')

ax2 = ax1.twinx()
color2 = '#A23B72'
ax2.set_ylabel('Durchschnittliche Sitzungsdauer (Sekunden)', color=color2, fontsize=12, fontweight='bold')
line2 = ax2.plot(df['Datum'], df['Zeit_im_Besuch_Sekunden'], color=color2, linewidth=2.5, marker='s', markersize=5, label='Sitzungsdauer')
ax2.tick_params(axis='y', labelcolor=color2)

plt.title('Tägliche Seitenaufrufe vs. Durchschnittliche Sitzungsdauer\nSeptember 2024', fontsize=14, fontweight='bold', pad=20)

lines1, labels1 = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax1.legend(lines1 + lines2, labels1 + labels2, loc='upper left', fontsize=10, framealpha=0.9)

plt.xticks(rotation=45, ha='right')
fig.tight_layout()

plt.savefig('./src/resources/output/32830/2_seitenaufrufe_vs_sitzungsdauer.png', dpi=300, bbox_inches='tight')
plt.close()
