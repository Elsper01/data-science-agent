# Load necessary libraries
library(readr)
library(dplyr)
library(tidyr)
library(stringr)
library(lubridate)
library(ggplot2)
library(scales)
library(zoo)

# Set output directory and path
out_dir <- file.path(".", "src", "resources", "output", "16265")
dir.create(out_dir, recursive = TRUE, showWarnings = FALSE)
plot_path <- file.path(out_dir, "0_gamma_odl_zeitreihe.png")

# Load dataset
csv_path <- file.path(".", "src", "resources", "data", "study", "16265", "dataset.csv")
if (!file.exists(csv_path)) {
  stop("Dataset file not found.")
}

raw <- readr::read_csv(csv_path, locale = locale(encoding = "UTF-8"), show_col_types = FALSE)

# Data Cleaning and Preparation
# 1. Handle missing values and types
# 2. Convert start_measure to datetime
# 3. Filter for validated data if applicable
# 4. Aggregate to daily level to handle potential high-frequency noise or duplicates

clean <- raw %>%
  mutate(
    start_measure_dt = lubridate::as_datetime(start_measure),
    value_num = as.numeric(value)
  ) %>%
  filter(!is.na(start_measure_dt), !is.na(value_num)) %>%
  # Filter for validated data if the column exists and has values
  filter(if ("validated" %in% names(.)) is.na(validated) | validated == 1 else TRUE) %>%
  mutate(datum = as.Date(start_measure_dt)) %>%
  # Group by day and take the mean value for that day
  group_by(datum) %>%
  summarise(value = mean(value_num, na.rm = TRUE), .groups = "drop") %>%
  arrange(datum)

# Check if we have data after cleaning
if (nrow(clean) == 0) {
  stop("No valid data points found after cleaning.")
}

# Calculate statistics for the subtitle and plot
n_obs <- nrow(clean)
mean_val <- mean(clean$value, na.rm = TRUE)
sd_val <- sd(clean$value, na.rm = TRUE)
min_val <- min(clean$value, na.rm = TRUE)
max_val <- max(clean$value, na.rm = TRUE)

# Rolling average for trend line (k=7 days for weekly trend)
k_days <- 7
if (nrow(clean) >= k_days) {
  clean <- clean %>%
    mutate(roll_mean = zoo::rollmean(value, k = k_days, align = "center", fill = NA))
} else {
  clean$roll_mean <- NA
}

# Formatting for labels
fmt_num <- function(x) scales::number(x, accuracy = 0.001, decimal.mark = ",", big.mark = ".")

# Create the plot
p <- ggplot(clean, aes(x = datum, y = value)) +
  # Background band for +/- 1 SD
  annotate("rect", 
           xmin = min(clean$datum), xmax = max(clean$datum), 
           ymin = mean_val - sd_val, ymax = mean_val + sd_val, 
           fill = "#90CAF9", alpha = 0.2) +
  # Mean line
  geom_hline(yintercept = mean_val, linetype = "dashed", color = "#546E7A", linewidth = 0.5) +
  # Main data line
  geom_line(color = "#1976D2", linewidth = 0.6, alpha = 0.7) +
  # Rolling average line
  geom_line(aes(y = roll_mean), color = "#D32F2F", linewidth = 1, na.rm = TRUE) +
  # Scales
  scale_x_date(
    date_breaks = "2 months", 
    date_minor_breaks = "1 month",
    date_labels = "%b %Y"
  ) +
  scale_y_continuous(labels = fmt_num) +
  # Labels
  labs(
    title = "Zeitlicher Verlauf der Gamma-Ortsdosisleistung (ODL)",
    subtitle = paste0("Station: Dorfprozelten | Mittelwert: ", fmt_num(mean_val), 
                      " µSv/h | Min: ", fmt_num(min_val), " | Max: ", fmt_num(max_val)),
    x = "Zeitpunkt der Messung",
    y = "Dosisleistung [µSv/h]",
    caption = paste0("Blaue Fläche: ±1 Standardabweichung. Rote Linie: ", k_days, "-Tage-Mittelwert. n = ", n_obs, " Tage.")
  ) +
  # Theme
  theme_minimal(base_size = 12) +
  theme(
    plot.title = element_text(face = "bold", size = 14),
    plot.subtitle = element_text(size = 10, color = "grey30"),
    axis.text.x = element_text(angle = 45, hjust = 1),
    panel.grid.minor = element_blank(),
    plot.caption = element_text(size = 8, color = "grey50", margin = margin(t = 10))
  )

# Save the plot
ggsave(plot_path, p, width = 10, height = 6, dpi = 150, bg = "white")

library(readr)

# Define paths
input_path <- file.path(".", "src", "resources", "data", "study", "16265", "dataset.csv")

dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)
plot_path <- file.path(output_dir, "2_boxplot_wert.png")

# Read data
df <- readr::read_csv(input_path, locale = readr::locale(encoding = "UTF-8"), show_col_types = FALSE)
df <- janitor::clean_names(df)

# Data cleaning and transformation
# Ensure 'value' is numeric and dates are parsed correctly
df <- df %>% mutate(
  start_measure = suppressWarnings(lubridate::as_datetime(start_measure)),
  end_measure = suppressWarnings(lubridate::as_datetime(end_measure)),
  value = suppressWarnings(as.numeric(value)),
  validated = suppressWarnings(as.integer(validated))
)

# Filter for valid data (validated == 1 or NA if not specified)
df_valid <- df %>% filter(!is.na(value)) %>% filter(is.na(validated) | validated == 1)

# Check if we have data to plot
if (nrow(df_valid) > 0) {
  
  # Calculate statistics for the plot and summary files
  value_stats <- df_valid %>% summarise(
    n = n(),
    mittelwert = mean(value, na.rm = TRUE),
    sd = sd(value, na.rm = TRUE),
    minimum = min(value, na.rm = TRUE),
    q1 = quantile(value, 0.25, na.rm = TRUE),
    median = median(value, na.rm = TRUE),
    q3 = quantile(value, 0.75, na.rm = TRUE),
    maximum = max(value, na.rm = TRUE),
    iqr = IQR(value, na.rm = TRUE)
  )
  
  readr::write_csv(value_stats, file.path(output_dir, "eda_value_summary.csv"))
  
  # Determine time range for subtitle
  min_date <- min(df_valid$start_measure, na.rm = TRUE)
  max_date <- max(df_valid$end_measure, na.rm = TRUE)
  
  zeitraum <- if(!is.na(min_date) && !is.na(max_date)) {
    paste0(format(min_date, "%d.%m.%Y"), " – ", format(max_date, "%d.%m.%Y"))
  } else {
    "Unbekannter Zeitraum"
  }

  # Create Boxplot
  p <- ggplot(df_valid, aes(x = value, y = "Messwerte")) +

      fill = "#2C7FB8",

      title = "Gamma-Ortsdosisleistung (µSv/h) – Boxplot der Tagesmittelwerte",
