import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

pd.options.mode.chained_assignment = None

input_path = r'./src/resources/data/study\136560\dataset.csv'
output_dir = r'./src/resources/output\136560'
os.makedirs(output_dir, exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

for c in df.select_dtypes(include=['object']).columns:
    df[c] = df[c].astype(str).str.strip()

ergebnis_clean = df['Ergebnis'].astype(str).str.replace('\u202f', '', regex=True).str.replace('.', '.', regex=False).str.replace(',', '.', regex=False).str.strip()
df['Ergebnis_num'] = pd.to_numeric(ergebnis_clean, errors='coerce')

missing_overall = df.replace({'': np.nan, ' ': np.nan}).isna().mean()
missing_ergebnis_incl_nn = df['Ergebnis'].isin(['n.n.', 'n.n', 'nn', 'NN']).mean()
desc_ergebnis = df['Ergebnis_num'].describe()

activity_mask = df['Einheit'].astype(str).str.contains('Bq/l', na=False)
not_time_mask = ~df['Pruefmerkmal'].astype(str).str.fullmatch('Messzeit', na=False)
activity_df_raw = df[activity_mask & not_time_mask].copy()

activity_df = activity_df_raw.dropna(subset=['Ergebnis_num']).copy()
activity_df['Pruefmerkmal_kurz'] = activity_df['Pruefmerkmal'].replace({'Radon-222 Rn-222': 'Radon-222', 'Gesamt-Alpha-Aktivität': 'Gesamt-Alpha-Aktivität'})

probe_id = str(df['Hauptprobenummer'].iloc[0]) if 'Hauptprobenummer' in df.columns and len(df) > 0 else ''
probe_datum = str(df['Probenahme_Beginn_Datum'].iloc[0]) if 'Probenahme_Beginn_Datum' in df.columns and len(df) > 0 else ''

sns.set_theme(style='whitegrid')

fig, ax = plt.subplots(figsize=(8, 5))

if activity_df.empty:
    ax.axis('off')
    ax.text(0.5, 0.6, 'Keine numerischen Aktivitätskonzentrationen verfügbar', ha='center', va='center', fontsize=12)
else:
    order = ['Radon-222', 'Gesamt-Alpha-Aktivität']
    activity_df['Pruefmerkmal_kurz'] = pd.Categorical(activity_df['Pruefmerkmal_kurz'], categories=order, ordered=True)
    activity_df = activity_df.sort_values('Pruefmerkmal_kurz')
    x = np.arange(len(activity_df))
    y = activity_df['Ergebnis_num'].values
    colors = sns.color_palette('viridis', n_colors=len(activity_df))
    bars = ax.bar(x, y, color=colors, edgecolor='black', linewidth=0.6)
    ax.set_xticks(x)
    ax.set_xticklabels(activity_df['Pruefmerkmal_kurz'].astype(str).tolist())
    ax.set_ylabel('Aktivitätskonzentration (Bq/l)')
    ylim_top = max(y) * 1.3 if len(y) else 1
    ax.set_ylim(0, ylim_top)
    for i, v in enumerate(y):
        label = f"{v:.2f}".replace('.', ',')
        ax.text(x[i], v + (ylim_top * 0.02), label, ha='center', va='bottom', fontsize=10)
    ax.yaxis.grid(True, linestyle='--', alpha=0.4)
    for spine in ['top', 'right']:
        ax.spines[spine].set_visible(False)

title_main = 'Aktivitätskonzentrationen im Trinkwasser'
subtitle_parts = []

if alpha_non_detect:

    alpha_time_val = int(alpha_time.iloc[0]) if not alpha_time.empty and not np.isnan(alpha_time.iloc[0]) else None

plt.tight_layout()

import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

try:
    sns.set_theme(context='notebook', style='whitegrid', font_scale=1.1)
    plt.rcParams['figure.dpi'] = 120
    plt.rcParams['savefig.dpi'] = 200
    plt.rcParams['font.family'] = 'DejaVu Sans'

    input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '136560', 'dataset.csv')
    output_dir = os.path.join('.', 'src', 'resources', 'output', '136560')
    os.makedirs(output_dir, exist_ok=True)

    df = pd.read_csv(input_path, sep=',', encoding='utf-8')

    obj_cols = df.select_dtypes(include=['object']).columns.tolist()
    for c in obj_cols:
        df[c] = df[c].astype(str).str.strip()
    df = df.replace({'n.n.': np.nan, 'N.N.': np.nan, '': np.nan})

    def to_num_series(s):
        return pd.to_numeric(s.astype(str).str.replace(',', '.', regex=False), errors='coerce')

    df['Ergebnis_num'] = to_num_series(df['Ergebnis']) if 'Ergebnis' in df.columns else np.nan

    mv = df.isna().mean().rename('missing_rate')
    mv.to_csv(os.path.join(output_dir, 'eda_missing_values.csv'), encoding='utf-8')

    desc_by_feature = df.groupby('Pruefmerkmal', dropna=False)['Ergebnis_num'].agg(['count', 'mean', 'std', 'min', 'max']).reset_index()
    desc_by_feature.to_csv(os.path.join(output_dir, 'eda_ergebnis_by_pruefmerkmal.csv'), index=False, encoding='utf-8')

    numeric_candidates = {}
    for col in df.columns:
        if df[col].dtype.kind in 'biufc':
            numeric_candidates[col] = df[col]
        else:
            ser = to_num_series(df[col])
            if ser.notna().sum() > 0:
                numeric_candidates[col] = ser
    df_num = pd.DataFrame(numeric_candidates)
    cols_with_var = [c for c in df_num.columns if pd.Series(df_num[c].dropna().unique()).size > 1]
    corr = df_num[cols_with_var].corr(numeric_only=True) if len(cols_with_var) >= 2 else pd.DataFrame()
    corr.to_csv(os.path.join(output_dir, 'eda_numeric_correlation.csv'), encoding='utf-8')

    plot_df = df[df['Pruefmerkmal'] == 'Messzeit'].copy()
    plot_df['Messzeit_s'] = to_num_series(plot_df['Ergebnis'])
    agg = plot_df.groupby('Methode', dropna=False).agg(Messzeit_s_mean=('Messzeit_s', 'mean'), n=('Messzeit_s', 'count')).reset_index()
    agg = agg.dropna(subset=['Methode', 'Messzeit_s_mean'])
    agg = agg.sort_values('Messzeit_s_mean', ascending=False)

    fig_w = max(6, 1.2 * max(1, len(agg)))
    fig_h = 5
    plt.figure(figsize=(fig_w, fig_h))
    palette = sns.color_palette('viridis', n_colors=max(3, len(agg)))
    ax = sns.barplot(data=agg, x='Methode', y='Messzeit_s_mean', palette=palette)
    ax.set_title('Messzeiten der radiologischen Analysen nach Methode (Sekunden)')
    ax.set_xlabel('Analytische Methode')
    ax.set_ylabel('Messzeit (s)')
    for p in ax.patches:
        h = p.get_height()
        if not np.isnan(h):
            ax.annotate(f"{int(round(h, 0))} s", (p.get_x() + p.get_width() / 2., h), ha='center', va='bottom', xytext=(0, 4), textcoords='offset points', fontsize=10)
    plt.xticks(rotation=20, ha='right')
    plt.tight_layout()
    out_path = os.path.join(output_dir, '1_messzeit_nach_methode.png')
    plt.savefig(out_path, bbox_inches='tight')
    plt.close()
except Exception as e:
    print(str(e))
    sys.exit(1)
