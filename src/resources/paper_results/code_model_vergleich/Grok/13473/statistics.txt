import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Path to the dataset
path = r'./src/resources/data/study/13473/dataset.csv'

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xf6' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German data with umlauts like 'ö').
try:
    df = pd.read_csv(path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(path, sep=';', encoding='latin-1')

# Filter for gender distribution data
# Based on the previous code logic, we look for 'Geschlecht' and 'KREISE'
gender_df = df[(df['2_variable_label'] == 'Geschlecht') & (df['1_variable_code'] == 'KREISE')].copy()

# Ensure 'value' is numeric
gender_df['value'] = pd.to_numeric(gender_df['value'], errors='coerce')

# Define the order for the x-axis
order = ['männlich', 'weiblich', 'Insgesamt']
gender_df['2_variable_attribute_label'] = pd.Categorical(
    gender_df['2_variable_attribute_label'], 
    categories=order, 
    ordered=True
)

# Sort by the categorical order
gender_df = gender_df.sort_values('2_variable_attribute_label')

# Visualization
sns.set_style('whitegrid')
plt.figure(figsize=(10, 6))

# Create the bar chart
ax = sns.barplot(
    data=gender_df, 
    x='2_variable_attribute_label', 
    y='value', 
    palette='viridis'
)

# Add labels and title
plt.title('Geschlechterverteilung der Bevölkerung in Hof (31.12.2024)', fontsize=16, pad=20)
plt.xlabel('Geschlecht', fontsize=14)
plt.ylabel('Anzahl Personen', fontsize=14)

# Add value labels on top of bars for clarity
for p in ax.patches:
    ax.annotate(format(p.get_height(), '.0f'), 
                (p.get_x() + p.get_width() / 2., p.get_height()), 
                ha = 'center', va = 'center', 
                xytext = (0, 9), 
                textcoords = 'offset points')

plt.tight_layout()

# Ensure output directory exists
output_path = r'./src/resources/output/13473/0_geschlechterverteilung.png'
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# Save the plot
plt.savefig(output_path, dpi=300, bbox_inches='tight', facecolor='white')
plt.close()

print(f'Visualization saved successfully to {output_path}')

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# The error 'UnicodeDecodeError' with byte 0xf6 (ö) suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# which is common for German datasets containing umlauts.
path = r'./src/resources/data/study/13473/dataset.csv'

try:
    df = pd.read_csv(path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    # Fallback to latin-1 if utf-8 fails
    df = pd.read_csv(path, sep=';', encoding='latin-1')

# Data Cleaning: Ensure 'value' is numeric
df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for 'Insgesamt' category as per the visualization goal
# We use str.contains or exact match depending on the data content
filtered = df[df['2_variable_attribute_label'].str.contains('Insgesamt', na=False)].copy()

# Visualization
sns.set_style('whitegrid')
plt.figure(figsize=(10, 6))

# Create the grouped bar chart
# Goal: Compare 'value' across '1_variable_code' for the 'Insgesamt' category
ax = sns.barplot(
    data=filtered, 
    x='1_variable_code', 
    y='value', 
    palette='viridis'
)

# Add labels and title
plt.title('Population Consistency Check: KREISE vs GEMEINDEN (Category: Insgesamt)', fontsize=14, pad=20)
plt.xlabel('Administrative Classification Level (1_variable_code)', fontsize=12)
plt.ylabel('Population Count', fontsize=12)

# Add value labels on top of bars for clarity
for p in ax.patches:
    ax.annotate(format(p.get_height(), '.0f'), 
                (p.get_x() + p.get_width() / 2., p.get_height()), 
                ha = 'center', va = 'center', 
                xytext = (0, 9), 
                textcoords = 'offset points')

plt.tight_layout()

# Save the plot
output_dir = r'./src/resources/output/13473'
os.makedirs(output_dir, exist_ok=True)
plot_path = os.path.join(output_dir, '2_konsistenz_bevoelkerung_admin_ebenen.png')
plt.savefig(plot_path, dpi=300, bbox_inches='tight')
plt.close()

print(f'Plot successfully saved to {plot_path}')
print('Data Preview (Filtered):')
