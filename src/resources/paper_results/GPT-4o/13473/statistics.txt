# Benötigte Bibliotheken laden (ohne install.packages, da dies in eingeschränkten Umgebungen oft fehlschlägt)
library(ggplot2)
library(dplyr)
library(readr)

# Daten einlesen
# Korrektur der Pfade: Verwendung von Vorwärts-Slashes zur Vermeidung von Escape-Sequenz-Fehlern
file_path <- './src/resources/data/study/13473/dataset.csv'
data <- read_csv2(file_path, locale = locale(encoding = 'UTF-8'))

# Daten filtern und vorbereiten
# Wir filtern nach 'KREISE' und stellen sicher, dass wir nur die relevanten Geschlechterkategorien haben
data_kreise <- data %>%
  filter(`1_variable_code` == "KREISE") %>%
  select(`2_variable_attribute_label`, value) %>%
  # Optional: 'Insgesamt' entfernen, falls nur der direkte Vergleich m/w gewünscht ist, 
  # aber laut Rationale soll der Gesamtwert zum Vergleich bleiben.
  mutate(`2_variable_attribute_label` = factor(`2_variable_attribute_label`, 
                                               levels = c("männlich", "weiblich", "Insgesamt")))

# Visualisierung erstellen
plot <- ggplot(data_kreise, aes(x = `2_variable_attribute_label`, y = value, fill = `2_variable_attribute_label`)) +
  geom_bar(stat = "identity", width = 0.7) +
  geom_text(aes(label = format(value, big.mark = ".", decimal.mark = ",")), vjust = -0.5) +
  labs(title = "Bevölkerungsverteilung nach Geschlecht in Hof (Krfr.St)",
       subtitle = "Vergleich von männlicher, weiblicher und Gesamtbevölkerung",
       x = "Geschlecht",
       y = "Bevölkerungsanzahl",
       fill = "Kategorie") +
  theme_minimal() +
  scale_fill_manual(values = c("männlich" = "#3498db", "weiblich" = "#e74c3c", "Insgesamt" = "#95a5a6")) +
  theme(legend.position = "none")

# Grafik speichern
output_path <- './src/resources/output/13473/0_bevoelkerungsverteilung.png'
# Sicherstellen, dass das Verzeichnis existiert (in R meist nicht nötig, wenn Pfad korrekt)
ggsave(output_path, plot, width = 8, height = 6, dpi = 300)

# Benötigte Bibliotheken laden (ohne install.packages, da dies in eingeschränkten Umgebungen oft fehlschlägt)
library(ggplot2)
library(dplyr)
library(readr)

# Daten laden
# Hinweis: Backslashes in Pfaden müssen in R entweder verdoppelt (\\) oder durch Slashes (/) ersetzt werden.
file_path <- "./src/resources/data/study/13473/dataset.csv"

# Sicherstellen, dass die Datei existiert, bevor wir lesen
if (file.exists(file_path)) {
  df <- read_csv2(file_path, locale = locale(encoding = 'UTF-8'))

  # Daten filtern und vorbereiten
  # Wir filtern auf die Geschlechter, um die Zusammensetzung zu zeigen.
  # Laut Zielsetzung soll die Summe der Einzelwerte (männlich/weiblich) visualisiert werden.
  filtered_df <- df %>%
    filter(`2_variable_attribute_label` %in% c("männlich", "weiblich")) %>%
    mutate(value = as.numeric(value)) %>%
    filter(!is.na(value))

  # Plot erstellen
  # Ein gestapeltes Balkendiagramm zeigt die Einzelwerte, die zusammen die Gesamtbevölkerung ergeben.
  plot <- ggplot(filtered_df, aes(x = "Gesamtbevölkerung", y = value, fill = `2_variable_attribute_label`)) +
    geom_bar(stat = "identity", position = "stack", width = 0.5) +
    coord_flip() +
    scale_y_continuous(labels = scales::comma) +
    labs(title = "Bevölkerungszusammensetzung der Stadt Hof",
         subtitle = "Stichtag: 31.12.2024",
         x = "",
         y = "Anzahl Personen",
         fill = "Geschlecht",
         caption = "Datenquelle: Fortschreibung des Bevölkerungsstandes") +
    theme_minimal() +
    theme(legend.position = "bottom")

  # Verzeichnis erstellen, falls es nicht existiert
  output_dir <- "./src/resources/output/13473"
  if (!dir.exists(output_dir)) {
    dir.create(output_dir, recursive = TRUE)
  }

  # Plot speichern
  ggsave(filename = paste0(output_dir, "/2_bevoelkerung_gesamt_vs_einzel.png"), 
         plot = plot, width = 8, height = 4, dpi = 300)
  
  print("Visualisierung erfolgreich erstellt.")
} else {
  stop(paste("Datei nicht gefunden:", file_path))
}
