if (!require("ggplot2")) install.packages("ggplot2")
if (!require("dplyr")) install.packages("dplyr")
if (!require("readr")) install.packages("readr")

library(ggplot2)
library(dplyr)
library(readr)

df <- read_csv('./src/resources/data/study/32830/dataset.csv', locale = locale(encoding = "utf-8"))

df <- df %>%
  mutate(
    Bounce_Rate_numeric = as.numeric(gsub("%", "", Bounce_Rate)) / 100,
    Seiten_pro_Besuch_numeric = as.numeric(Seiten_pro_Besuch)
  )

df_clean <- df %>%
  filter(!is.na(Seiten_pro_Besuch_numeric) & !is.na(Bounce_Rate_numeric)) %>%
  filter(!is.na(Datum))

correlation <- cor(df_clean$Seiten_pro_Besuch_numeric, df_clean$Bounce_Rate_numeric, use = "complete.obs")

dir.create("./src/resources/output/32830", recursive = TRUE, showWarnings = FALSE)

png("./src/resources/output/32830/1_scatter_besuchstiefe_absprungrate.png", width = 1200, height = 800, res = 120)

ggplot(df_clean, aes(x = Seiten_pro_Besuch_numeric, y = Bounce_Rate_numeric)) +
  geom_point(color = "#2E86AB", size = 3, alpha = 0.7) +
  geom_smooth(method = "lm", se = TRUE, color = "#A23B72", linetype = "dashed", linewidth = 1) +
  labs(
    title = "Zusammenhang zwischen Besuchstiefe und Absprungrate",
    subtitle = paste0("Korrelation: r = ", round(correlation, 3)),
    x = "Seiten pro Besuch",
    y = "Absprungrate (Bounce Rate)"
  ) +
  scale_y_continuous(labels = scales::percent_format(accuracy = 1)) +
  theme_minimal(base_size = 14) +
  theme(
    plot.title = element_text(face = "bold", size = 16, hjust = 0.5),
    plot.subtitle = element_text(size = 12, hjust = 0.5, color = "gray30"),
    axis.title = element_text(face = "bold"),
    panel.grid.minor = element_blank(),
    panel.border = element_rect(color = "gray70", fill = NA, linewidth = 0.5)
  )

dev.off()

if (!require("ggplot2")) install.packages("ggplot2")

library(ggplot2)
library(dplyr)
library(tidyr)
library(lubridate)

data <- read.csv('./src/resources/data/study/32830/dataset.csv', 
                 sep = ',', 
                 encoding = 'UTF-8',
                 stringsAsFactors = FALSE)

data_clean <- data %>%
  filter(!is.na(Datum) & Datum != "" & Datum != "nan") %>%
  filter(row_number() != 1)

data_clean$Datum_parsed <- dmy(data_clean$Datum)

data_clean$Besuche <- as.numeric(data_clean$Besuche)
data_clean$Besucher <- as.numeric(data_clean$Besucher)

data_clean <- data_clean %>%
  filter(!is.na(Besuche) & !is.na(Besucher) & !is.na(Datum_parsed)) %>%
  arrange(Datum_parsed)

data_long <- data_clean %>%
  select(Datum_parsed, Besuche, Besucher) %>%
  pivot_longer(cols = c(Besuche, Besucher), 
               names_to = "Metrik", 
               values_to = "Anzahl")

data_long$Metrik <- factor(data_long$Metrik, 
                           levels = c("Besucher", "Besuche"),
                           labels = c("Besucher", "Besuche"))

dir.create('./src/resources/output/32830', showWarnings = FALSE, recursive = TRUE)

png('./src/resources/output/32830/2_besucher_vs_besuche.png', 
    width = 1400, 
    height = 800, 
    res = 100)

ggplot(data_long, aes(x = Datum_parsed, y = Anzahl, fill = Metrik)) +
  geom_bar(stat = "identity", position = position_dodge(width = 0.8), width = 0.7) +
  scale_fill_manual(values = c("Besucher" = "#2E86AB", "Besuche" = "#A23B72")) +
  labs(title = "Tägliches Aufkommen: Besucher vs. Besuche im September 2024",
       subtitle = "Vergleich der Unique Visitors (Besucher) mit der Gesamtzahl der Besuche",
       x = "Datum",
       y = "Anzahl",
       fill = "Metrik") +
  theme_minimal(base_size = 12) +
  theme(axis.text.x = element_text(angle = 45, hjust = 1),
        plot.title = element_text(face = "bold", size = 14),
        plot.subtitle = element_text(size = 11),
        legend.position = "top",
        panel.grid.minor = element_blank()) +
  scale_x_date(date_breaks = "2 days", date_labels = "%d.%m.%Y")

dev.off()
