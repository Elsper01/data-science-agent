import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

try:
    data_path = r'./src/resources/data/study\135982\dataset.csv'
    output_dir = r'./src/resources/output\135982'
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(data_path, sep=',', encoding='utf-8')
    df_columns_lower = {c: c for c in df.columns}
    if 'Ergebnis' not in df.columns:
        raise ValueError('Spalte "Ergebnis" nicht gefunden.')

    def parse_numeric(x):
        if pd.isna(x):
            return np.nan
        s = str(x).strip()
        if s == '':
            return np.nan
        s_low = s.lower().replace(' ', '')
        if s_low in {'n.n.', 'n.n', 'nn'}:
            return np.nan
        if s_low.startswith('<') or s_low.startswith('>'):
            s = s[1:]
        s = s.replace('−', '-').replace(',', '.')
        allowed = set('0123456789eE+-.')
        cleaned = ''.join(ch for ch in s if ch in allowed)
        try:
            if cleaned in {'', '.', '-.', '+.', 'e', 'E'}:
                return np.nan
            return float(cleaned)
        except Exception:
            return np.nan

    def categorize_detection(x):
        if pd.isna(x):
            return 'Nicht nachweisbar (n.n.)'
        s = str(x).strip()
        s_low = s.lower().replace(' ', '')
        if s_low in {'n.n.', 'n.n', 'nn'}:
            return 'Nicht nachweisbar (n.n.)'
        v = parse_numeric(x)
        if pd.isna(v):
            return 'Nicht nachweisbar (n.n.)'
        return 'Nachweisbar' if v > 0 else 'Nicht nachweisbar (n.n.)'

    df['Ergebnis_num'] = df['Ergebnis'].apply(parse_numeric)
    df['Ergebnis_Kategorie'] = df['Ergebnis'].apply(categorize_detection)

    missing = df.isna().mean().reset_index()
    missing.columns = ['Spalte', 'Anteil_Fehlend']
    missing.to_csv(os.path.join(output_dir, '2_missing_values.csv'), index=False, encoding='utf-8')

    desc_num = df[['Ergebnis_num']].describe(include='all').T
    desc_num.to_csv(os.path.join(output_dir, '2_numeric_summary_ergebnis.csv'), encoding='utf-8')

    num_df = df.select_dtypes(include=[np.number])
    if not num_df.empty:
        corr = num_df.corr(numeric_only=True)
        corr.to_csv(os.path.join(output_dir, '2_correlations_numeric.csv'), encoding='utf-8')

    order = ['Nicht nachweisbar (n.n.)', 'Nachweisbar']
    counts = df['Ergebnis_Kategorie'].value_counts().reindex(order).fillna(0).astype(int)
    total = int(counts.sum()) if counts.sum() > 0 else 1
    plot_df = counts.reset_index()
    plot_df.columns = ['Kategorie', 'Anzahl']
    plot_df['Anteil'] = plot_df['Anzahl'] / total

    sns.set_theme(style='whitegrid', context='talk')
    plt.figure(figsize=(9, 6))
    ax = sns.barplot(data=plot_df, x='Kategorie', y='Anzahl', palette=['#4C78A8', '#F58518'])
    ax.set_title('Verhältnis von nachweisbaren zu nicht nachweisbaren Ergebnissen', fontsize=16)
    ax.set_xlabel('Ergebniskategorie', fontsize=12)
    ax.set_ylabel('Anzahl der Messungen', fontsize=12)
    ax.set_ylim(0, max(1, plot_df['Anzahl'].max() * 1.15))
    for p, (_, row) in zip(ax.patches, plot_df.iterrows()):
        height = p.get_height()
        ax.annotate(f"{int(row['Anzahl'])} ({row['Anteil']:.1%})", 
                    (p.get_x() + p.get_width() / 2.0, height), 
                    ha='center', va='bottom', fontsize=11, xytext=(0, 6), textcoords='offset points')
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, '2_ergebniskategorien_barchart.png'), dpi=300)
    plt.close()
except Exception as e:
    print(str(e))
