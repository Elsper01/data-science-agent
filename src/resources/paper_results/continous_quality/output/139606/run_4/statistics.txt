import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os
import sys

input_path = './src/resources/data/study\\139606\\dataset.csv'
output_dir = os.path.normpath('./src/resources/output\\139606\\run_4')
output_path = os.path.join(output_dir, '0_verteilung_nachweisbar.png')

try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except Exception as e:
    print(str(e))
    sys.exit(1)

sns.set_theme(style='whitegrid', context='talk')

missing_ratio = df.isna().mean()

def parse_numeric(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    if s == '':
        return np.nan
    s_low = s.lower()
    if s_low.startswith('n.n'):
        return np.nan
    s = s.replace(' ', '')
    s = s.replace(',', '.')
    try:
        return float(s)
    except Exception:
        return pd.to_numeric(s, errors='coerce')

df['Ergebnis_str'] = df['Ergebnis'].astype(str).str.strip()
df['Ergebnis_num'] = df['Ergebnis_str'].apply(parse_numeric)

nn_mask = df['Ergebnis_str'].str.lower().str.startswith('n.n')
num_mask = df['Ergebnis_num'].notna()
cat = np.where(num_mask, 'nachweisbar', np.where(nn_mask, 'nicht nachweisbar', 'unklar/fehlend'))
df['Ergebnis_Kategorie'] = pd.Categorical(cat, categories=['nachweisbar', 'nicht nachweisbar', 'unklar/fehlend'], ordered=True)

counts = df.groupby('Ergebnis_Kategorie', dropna=False)['Pruefmerkmal'].count().reindex(['nachweisbar', 'nicht nachweisbar', 'unklar/fehlend'])
counts = counts.fillna(0).astype(int)

total = counts.sum()
perc = (counts / total * 100).round(1)
plot_df = pd.DataFrame({'Kategorie': counts.index, 'Anzahl': counts.values, 'Anteil_prozent': perc.values})

if plot_df['Anzahl'].sum() == 0:
    print('Keine Daten für die Visualisierung verfügbar.')
    sys.exit(1)

palette = {
    'nachweisbar': '#2ca02c',
    'nicht nachweisbar': '#7f7f7f',
    'unklar/fehlend': '#ff7f0e'
}

fig, ax = plt.subplots(figsize=(9, 6))
sns.barplot(data=plot_df, x='Kategorie', y='Anzahl', ax=ax, palette=palette)

for i, row in plot_df.iterrows():
    ax.text(i, row['Anzahl'] + max(1, total*0.01), f"{int(row['Anzahl'])} ({row['Anteil_prozent']:.1f}%)", ha='center', va='bottom', fontsize=12)

try:
    date_str = pd.to_datetime(df['Probenahme_Beginn_Datum'].dropna().astype(str).str.replace('.', '-', regex=False), errors='coerce').dt.strftime('%d.%m.%Y').dropna().iloc[0]
except Exception:
    date_str = ''

title = 'Verteilung nachweisbarer vs. nicht nachweisbarer radiologischer Prüfkriterien'
if date_str:
    title += f' (IMIS, {date_str})'
ax.set_title(title)
ax.set_xlabel('Ergebniskategorie')
ax.set_ylabel('Anzahl der Prüfkriterien')
ax.set_ylim(0, max(plot_df['Anzahl'].max() * 1.2, plot_df['Anzahl'].max() + 3))
ax.margins(x=0.1)
plt.tight_layout()

try:
    os.makedirs(output_dir, exist_ok=True)
    fig.savefig(output_path, dpi=300)
except Exception as e:
    print(str(e))
    sys.exit(1)
finally:
    plt.close(fig)

import os
import sys
import math
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

input_path = r'./src/resources/data/study\139606\dataset.csv'
output_dir = r'./src/resources/output\139606\run_4'
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '2_pruefmerkmale_tabelle.png')

try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except Exception as e:
    print(str(e))
    sys.exit(1)

for c in df.columns:
    if isinstance(c, str):
        df.rename(columns={c: c.strip()}, inplace=True)

if 'Pruefmerkmal' not in df.columns:
    print('Spalte Pruefmerkmal fehlt.')
    sys.exit(1)

if 'Ergebnis' not in df.columns:
    df['Ergebnis'] = np.nan

def parse_numeric(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip().lower()
    s = s.replace('−', '-')
    s_clean = s.replace(' ', '')
    s_norm = s_clean.replace('n.n.', 'nn').replace('n.n', 'nn')
    if s_norm in {'', 'nn', 'na', 'nan', 'None', 'none'}:
        return np.nan
    try:
        return float(s_norm.replace(',', '.'))
    except Exception:
        return np.nan

df['Ergebnis_num'] = df['Ergebnis'].apply(parse_numeric)

pruef_list = (
    df['Pruefmerkmal']
    .astype(str)
    .str.strip()
    .replace({'': np.nan})
    .dropna()
    .unique()
    .tolist()
)
pruef_list = sorted(pruef_list)

status_map = {}
for _, row in df.iterrows():
    pm = str(row['Pruefmerkmal']).strip()
    if pm == '' or pd.isna(pm):
        continue
    val = row['Ergebnis_num']
    if pm not in status_map:
        status_map[pm] = bool(pd.notna(val))
    else:
        status_map[pm] = status_map[pm] or bool(pd.notna(val))

n_total = len(pruef_list)
if n_total == 0:
    print('Keine Prüfmerkmale gefunden.')
    sys.exit(1)

n_detect = sum(1 for k in pruef_list if status_map.get(k, False))
n_nd = n_total - n_detect

missing_props = df.isna().mean().to_dict()
sel_cols = [c for c in ['Pruefmerkmal', 'Ergebnis', 'Einheit', 'Methode', 'Hauptprobenummer', 'Probenahme_Beginn_Datum'] if c in df.columns]
miss_text_parts = []
for c in sel_cols:
    mp = missing_props.get(c, 0.0)
    miss_text_parts.append(f"{c}: {mp:.0%} fehlend")
miss_text = ' | '.join(miss_text_parts)

sample_id = str(df['Hauptprobenummer'].iloc[0]) if 'Hauptprobenummer' in df.columns and pd.notna(df['Hauptprobenummer'].iloc[0]) else 'unbekannt'
method = str(df['Methode'].iloc[0]) if 'Methode' in df.columns and pd.notna(df['Methode'].iloc[0]) else 'unbekannt'
date_str = str(df['Probenahme_Beginn_Datum'].iloc[0]) if 'Probenahme_Beginn_Datum' in df.columns and pd.notna(df['Probenahme_Beginn_Datum'].iloc[0]) else ''
units = []
if 'Einheit' in df.columns:
    units = (
        df['Einheit']
        .astype(str)
        .str.strip()
        .replace({'': np.nan, 'nan': np.nan})
        .dropna()
        .unique()
        .tolist()
    )
unit_str = ', '.join(sorted(units)) if len(units) > 0 else ''

n_cols = 2 if n_total <= 60 else 3
n_rows = int(math.ceil(n_total / n_cols))
cell_text = [['' for _ in range(n_cols)] for _ in range(n_rows)]
cell_colors = [["#ffffff" for _ in range(n_cols)] for _ in range(n_rows)]

for idx, val in enumerate(pruef_list):
    r = idx % n_rows
    c = idx // n_rows
    cell_text[r][c] = val
    if status_map.get(val, False):
        cell_colors[r][c] = '#c7e9c0'

base_height = 1.0 + 0.35 * n_rows
fig_w = 12 if n_cols == 2 else 14
fig_h = max(6, min(24, base_height))
sns.set_theme(style='white')
fig, ax = plt.subplots(figsize=(fig_w, fig_h), dpi=200)
ax.axis('off')

col_labels = ['Prüfmerkmal'] * n_cols
tab = ax.table(cellText=cell_text, cellColours=cell_colors, colLabels=col_labels, loc='center')
tab.auto_set_font_size(False)
font_size = 10 if n_rows <= 25 else 9 if n_rows <= 35 else 8
header_font_size = font_size + 1
for key, cell in tab.get_celld().items():
    r, c = key
    if r == 0:
        cell.set_facecolor('#f0f0f0')
        cell.set_text_props(weight='bold', fontsize=header_font_size)
    else:
        cell.set_text_props(fontsize=font_size)
        cell.set_edgecolor('#dddddd')

tab.scale(1.0, 1.2)

title_main = f"Umfang der radiologischen Screeningparameter (IMIS Routinemessprogramm)"
subtitle = f"Alle geprüften Prüfermerkmale in Schweinefleisch – Probe {sample_id}{' | ' + date_str if date_str else ''}{' | ' + method if method else ''}"
ax.set_title(title_main + "\n" + subtitle, loc='center', fontsize=14, pad=20)

legend_y = 1.02
ax2 = fig.add_axes([0, 0, 1, 1])
ax2.axis('off')
ax2.add_patch(plt.Rectangle((0.06, legend_y), 0.02, 0.02, transform=ax2.transAxes, color='#c7e9c0', ec='k', lw=0.5))
ax2.text(0.085, legend_y, 'Nachweisbar/quantifizierbar', transform=ax2.transAxes, va='bottom', fontsize=10)
ax2.add_patch(plt.Rectangle((0.34, legend_y), 0.02, 0.02, transform=ax2.transAxes, color='#ffffff', ec='k', lw=0.5))
ax2.text(0.365, legend_y, 'Nicht nachweisbar (n.n.)', transform=ax2.transAxes, va='bottom', fontsize=10)

footnote = f"Geprüfte Parameter: {n_total} | Nachweisbar: {n_detect} | Nicht nachweisbar: {n_nd}{' | Einheit(en): ' + unit_str if unit_str else ''}"
ax.text(0.5, -0.06, footnote, ha='center', va='top', transform=ax.transAxes, fontsize=10)
if miss_text:
    ax.text(0.5, -0.10, f"Fehlende Werte – {miss_text}", ha='center', va='top', transform=ax.transAxes, fontsize=9, color='#555555')

plt.tight_layout()
try:
    plt.savefig(output_path, bbox_inches='tight')
except Exception as e:
    print(str(e))
finally:
    plt.close(fig)
