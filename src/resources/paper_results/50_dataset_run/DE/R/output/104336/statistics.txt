# Pfad zur Datei
file_path <- "./src/resources/data/study/104336/dataset.csv"
if(!file.exists(file_path)) stop(paste("Datendatei nicht gefunden:", file_path))

# Daten einlesen
raw <- readr::read_csv(
  file = file_path,
  col_types = cols(.default = col_character()),
  locale = locale(encoding = "UTF-8"),
  na = c("-", "", "NA", "na", "NaN")
)

# Spaltennamen bereinigen
df <- raw %>% janitor::clean_names()

# Dynamische Identifikation der Spalten für Institutionen und Beträge
# Da Spaltennamen variieren können (z.B. f_rderung_in_2014 vs foerderung_in_2014), 
# suchen wir nach Mustern.
inst_col <- names(df)[str_detect(names(df), "f[oe_]+rderung_in_2014")][1]
val_col <- names(df)[str_detect(names(df), "gesamt_in_euro")][1]

if(is.na(inst_col) || is.na(val_col)) {
  # Fallback: Falls Regex fehlschlägt, nimm die ersten plausiblen Spalten
  # In diesem Datensatz ist oft Spalte 1 die Institution und Spalte 4 der Gesamtwert
  inst_col <- names(df)[1]
  val_col <- names(df)[4]
}

# Datenaufbereitung
df_clean <- df %>%
  mutate(
    # Numerische Konvertierung
    wert_numerisch = readr::parse_number(
      as.character(get(val_col)), 
      locale = locale(decimal_mark = ",", grouping_mark = ".")
    ),
    # Bereinigung der Institutionsnamen
    institution = str_squish(as.character(get(inst_col)))
  ) %>%
  # Filter: Keine NAs, keine Summenzeilen (Gesamt), nur positive Werte
  filter(!is.na(wert_numerisch), !is.na(institution)) %>%
  filter(!str_detect(str_to_lower(institution), "^gesamt")) %>%
  filter(wert_numerisch > 0) %>%
  # Sortierung für das Diagramm
  mutate(institution = fct_reorder(institution, wert_numerisch))

# Visualisierung
plot_obj <- ggplot(df_clean, aes(x = wert_numerisch, y = institution)) +
  geom_col(fill = "#2C7FB8", width = 0.7) +
  geom_text(
    aes(label = scales::label_number(big.mark = ".", decimal.mark = ",", suffix = " €")(wert_numerisch)),
    hjust = -0.1, 
    size = 3.5, 
    color = "black"
  ) +
  scale_x_continuous(
    labels = scales::label_number(big.mark = ".", decimal.mark = ",", suffix = " €"),
    expand = expansion(mult = c(0, 0.2))
  ) +
  labs(
    title = "Höchste Gesamtförderung nach Institution (2014)",
    subtitle = "Investitionsförderung in Euro; Summenzeilen ausgeschlossen",
    x = "Gesamtbetrag in Euro",
    y = "Institution",
    caption = "Quelle: Minderheitenbericht 2021, Staatskanzlei Schleswig-Holstein"
  ) +
  theme_minimal(base_size = 12) +
  theme(
    panel.grid.minor = element_blank(),
    plot.title = element_text(face = "bold"),
    axis.text.y = element_text(size = 9)
  )
