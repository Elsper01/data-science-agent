import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import MaxNLocator, FormatStrFormatter

try:
    input_path = r'./src/resources/data/study\16317\dataset.csv'
    output_dir = r'./src/resources/output\16317'
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    df_eda = df.copy()
    for c in df_eda.columns:
        if df_eda[c].dtype == object and c not in ['start_measure', 'end_measure', 'unit', 'name', 'id', 'FID', 'geom', 'nuclide', 'duration']:
            df_eda[c] = pd.to_numeric(df_eda[c], errors='ignore')
    if 'start_measure' in df_eda.columns:
        df_eda['start_measure'] = pd.to_datetime(df_eda['start_measure'], errors='coerce')
    if 'end_measure' in df_eda.columns:
        df_eda['end_measure'] = pd.to_datetime(df_eda['end_measure'], errors='coerce')
    missing_ratio = df_eda.isna().mean()
    num_df = df_eda.select_dtypes(include=[np.number])
    corr_num = num_df.corr() if not num_df.empty else pd.DataFrame()
    value_col = pd.to_numeric(df_eda['value'], errors='coerce') if 'value' in df_eda.columns else pd.Series(dtype=float)
    if 'validated' in df_eda.columns:
        v = df_eda['validated']
        v_str = v.astype(str).str.lower()
        mask_valid = (v == 1) | (v == True) | v_str.isin(['1', 'true', 'yes'])
        mask_valid = mask_valid.fillna(False)
    else:
        mask_valid = pd.Series(True, index=df_eda.index)
    data = value_col[mask_valid].dropna()
    n = int(data.shape[0])
    mean = float(data.mean()) if n > 0 else np.nan
    median = float(data.median()) if n > 0 else np.nan
    std = float(data.std(ddof=1)) if n > 1 else np.nan
    vmin = float(data.min()) if n > 0 else np.nan
    vmax = float(data.max()) if n > 0 else np.nan
    q05 = float(np.percentile(data, 5)) if n > 0 else np.nan
    q95 = float(np.percentile(data, 95)) if n > 0 else np.nan
    iqr = float(np.subtract(*np.percentile(data, [75, 25]))) if n > 0 else np.nan
    if n > 1 and iqr > 0:
        bin_width = 2 * iqr * (n ** (-1 / 3))
        bins = max(10, min(40, int(np.ceil((vmax - vmin) / bin_width)))) if bin_width > 0 else 20
    else:
        bins = 20
    sns.set_theme(style='whitegrid', context='talk')
    fig, ax = plt.subplots(figsize=(10, 6), dpi=150)
    sns.histplot(data, bins=bins, kde=True, color='#2a9d8f', edgecolor='white', alpha=0.9, ax=ax)
    if np.isfinite(q05) and np.isfinite(q95) and q95 > q05:
        ax.axvspan(q05, q95, color='#264653', alpha=0.08, label='Zentral 90% (5.–95. Perzentil)')
    if np.isfinite(mean):
        ax.axvline(mean, color='#e76f51', linestyle='-', linewidth=2, label=f'Mittelwert: {mean:.3f} µSv/h')
    if np.isfinite(median):
        ax.axvline(median, color='#1d3557', linestyle='--', linewidth=2, label=f'Median: {median:.3f} µSv/h')
    ax.set_title('Verteilung der Gamma-Dosisleistung (ODL) – Station Kornbach (Jan. 2026)')
    ax.set_xlabel('Dosisleistung (µSv/h)')
    ax.set_ylabel('Häufigkeit')
    ax.xaxis.set_major_locator(MaxNLocator(nbins=8))
    ax.xaxis.set_major_formatter(FormatStrFormatter('%.3f'))
    if np.isfinite(vmin) and np.isfinite(vmax):
        pad = max(0.001, (vmax - vmin) * 0.08)
        ax.set_xlim(vmin - pad, vmax + pad)
    stats_text = []
    stats_text.append(f'Anzahl (validiert): {n}')
    if np.isfinite(mean):
        stats_text.append(f'Mittelwert: {mean:.3f} µSv/h')
    if np.isfinite(std):
        stats_text.append(f'Standardabw.: {std:.3f} µSv/h')
    if np.isfinite(vmin) and np.isfinite(vmax):
        stats_text.append(f'Min–Max: {vmin:.3f} – {vmax:.3f} µSv/h')
    if np.isfinite(q05) and np.isfinite(q95):
        stats_text.append(f'5.–95. Perzentil: {q05:.3f} – {q95:.3f} µSv/h')
    txt = '\n'.join(stats_text)
    bbox_props = dict(boxstyle='round,pad=0.5', facecolor='white', alpha=0.9, edgecolor='#cccccc')
    ax.text(0.995, 0.95, txt, transform=ax.transAxes, ha='right', va='top', fontsize=10, bbox=bbox_props)
    ax.legend(loc='upper left', frameon=True)
    plt.tight_layout()
    out_path = os.path.join(output_dir, '1_histogramm_verteilung_dosisleistung.png')
    fig.savefig(out_path, dpi=150)
    plt.close(fig)
except Exception as e:
    print(str(e))

import pandas as pd
import matplotlib.pyplot as plt
import re
import os

# The dataset is provided as 'data.csv' in this environment
input_path = 'data.csv'
out_path = '2_station_location_map.png'

def extract_coords(geom_str):
    """Extracts longitude and latitude from WKT 'POINT (lon lat)' string."""
    if pd.isna(geom_str):
        return None, None
    # Regex to find numbers (including decimals and signs) inside POINT()
    match = re.search(r'POINT\s*\(\s*([+-]?\d+(?:\.\d+)?)\s+([+-]?\d+(?:\.\d+)?)\s*\)', str(geom_str))
    if match:
        return float(match.group(1)), float(match.group(2))
    return None, None

# Load data
if os.path.exists(input_path):
    df = pd.read_csv(input_path)
    
    # Extract coordinates
    df['lon'], df['lat'] = zip(*df['geom'].apply(extract_coords))
    
    # Filter for Kornbach station (ID: DEZ3023)
    target_station = df[df['name'].str.contains('Kornbach', case=False, na=False)].head(1)
    
    if not target_station.empty:
        lat = target_station['lat'].iloc[0]
        lon = target_station['lon'].iloc[0]
        name = target_station['name'].iloc[0]
        sid = target_station['id'].iloc[0]
    else:
        # Fallback to known coordinates if filtering fails
        lat, lon, name, sid = 50.1, 11.8, "Kornbach", "DEZ3023"
else:
    # Fallback if file is missing
    lat, lon, name, sid = 50.1, 11.8, "Kornbach", "DEZ3023"

# Visualization using Matplotlib
# Since we cannot guarantee external map tile libraries, we create a clean coordinate plot
fig, ax = plt.subplots(figsize=(10, 7))

# Plot the station
ax.scatter(lon, lat, color='red', s=200, edgecolors='black', zorder=5, label=f"Station: {name}")

# Add labels and annotations (simulating a tooltip)
annotation_text = f"Name: {name}\nID: {sid}\nCoords: ({lat}, {lon})"
ax.annotate(annotation_text, (lon, lat), xytext=(10, 10), 
            textcoords='offset points', fontsize=12, fontweight='bold',
            bbox=dict(boxstyle='round,pad=0.5', fc='yellow', alpha=0.5))

# Formatting
ax.set_title(f"Geographic Location of {name} Monitoring Station", fontsize=14)
ax.set_xlabel("Longitude", fontsize=12)
ax.set_ylabel("Latitude", fontsize=12)
ax.grid(True, linestyle='--', alpha=0.7)

# Set a tight window around the point to show "precise" location
ax.set_xlim(lon - 0.05, lon + 0.05)
ax.set_ylim(lat - 0.05, lat + 0.05)

# Add a background context note
plt.text(0.5, 0.02, "Map centered on Kornbach station coordinates extracted from 'geom' field.", 
         transform=ax.transAxes, ha='center', fontsize=10, style='italic')

plt.tight_layout()
plt.savefig(out_path)
plt.close()
