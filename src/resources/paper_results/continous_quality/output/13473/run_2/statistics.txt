import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import os

# Set visual style
sns.set(context="talk", style="whitegrid", font_scale=1.05)

input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '13473', 'dataset.csv')
output_dir = os.path.join('.', 'src', 'resources', 'output', '13473', 'run_2')
plot_filename = '0_geschlechterverteilung.png'
output_path = os.path.join(output_dir, plot_filename)

# The error 'utf-8' codec can't decode byte 0xf6 (ö) suggests the file is likely encoded in ISO-8859-1 (Latin-1)
# which is common for German datasets containing umlauts.
try:
    df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1')
except Exception as e:
    # Fallback to latin-1 if iso-8859-1 fails, or try to detect
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Clean string columns
for c in df.columns:
    if df[c].dtype == 'object':
        df[c] = df[c].astype(str).replace({'nan': np.nan})

# Ensure numeric values
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Calculate missingness for footnote
missing_share = df.isna().mean()
value_missing_pct = float(missing_share.get('value', np.nan)) * 100

# Filter for gender data
# We look for 'Geschlecht' in the variable label and exclude 'Insgesamt' (Total)
mask = (
    df['2_variable_label'].str.contains('Geschlecht', na=False, case=False) &
    df['2_variable_attribute_label'].notna() &
    ~df['2_variable_attribute_label'].str.contains('Insgesamt', na=False, case=False) &
    df['value'].notna()
)

gender_df = df.loc[mask].copy()

if gender_df.empty:
    print('Keine geeigneten Daten für die Visualisierung gefunden.')
    # Try a broader search if the specific labels aren't found
    mask_alt = (df['2_variable_attribute_label'].isin(['männlich', 'weiblich']))
    gender_df = df.loc[mask_alt].copy()

if gender_df.empty:
    raise SystemExit("No gender data found after filtering.")

# Aggregate by gender (taking the mean or sum depending on the structure, 
# but usually these datasets have one value per year/category)
# We'll group by the attribute label and take the latest time point if available
if 'time' in gender_df.columns:
    latest_time = gender_df['time'].max()
    plot_df = gender_df[gender_df['time'] == latest_time].copy()
else:
    plot_df = gender_df.groupby('2_variable_attribute_label', as_index=False)['value'].sum()

# Formatting helpers
def fmt_de(x):
    return f"{int(x):,}".replace(",", ".")

# Plotting
fig, ax = plt.subplots(figsize=(10, 7))

# Add labels on top of bars

import os
import sys
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from matplotlib.ticker import FuncFormatter

try:
    # Set visual style
    sns.set_theme(style="whitegrid", context="talk")
    
    # Paths
    data_path = os.path.normpath(r"./src/resources/data/study/13473/dataset.csv")
    output_dir = os.path.normpath(r"./src/resources/output/13473/run_2")
    os.makedirs(output_dir, exist_ok=True)

    # FIX: Handle encoding error. German datasets often use 'latin-1' or 'iso-8859-1' 
    # if 'utf-8' fails (0xf6 is 'ö' in latin-1).
    try:
        df = pd.read_csv(data_path, sep=';', encoding='utf-8', dtype=str)
    except UnicodeDecodeError:
        df = pd.read_csv(data_path, sep=';', encoding='latin-1', dtype=str)

    # Data Cleaning
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')
    if 'time' in df.columns:
        df['time'] = pd.to_datetime(df['time'], errors='coerce')

    # Filter for 'Insgesamt' (Total population)
    if '2_variable_attribute_label' in df.columns:
        # Strip whitespace and filter
        df_filtered = df[df['2_variable_attribute_label'].str.strip() == 'Insgesamt'].copy()
    else:
        df_filtered = df.copy()

    # Use the latest available time point to avoid duplicates if time series exists
    if 'time' in df_filtered.columns and df_filtered['time'].notna().any():
        latest_time = df_filtered['time'].max()
        df_filtered = df_filtered[df_filtered['time'] == latest_time]
    else:
        latest_time = None

    # Group by administrative level
    if '1_variable_label' in df_filtered.columns and 'value' in df_filtered.columns:
        plot_df = df_filtered.groupby('1_variable_label', as_index=False)['value'].first()
    else:
        raise ValueError("Required columns '1_variable_label' or 'value' missing.")

    if plot_df.empty:
        raise ValueError("No data available for the requested visualization after filtering.")

    # Visualization
    fig, ax = plt.subplots(figsize=(10, 7), dpi=150)
    
    # German number formatting (1.000 instead of 1,000)
    def german_thousands(x, pos):
        return f"{int(x):,}".replace(',', '.')

    palette = sns.color_palette("viridis", n_colors=len(plot_df))
    bars = sns.barplot(data=plot_df, x='1_variable_label', y='value', ax=ax, palette=palette)
    
    # Formatting
    ax.yaxis.set_major_formatter(FuncFormatter(german_thousands))
    ax.set_xlabel('Verwaltungsebene (Administrative Level)', fontsize=12)
    ax.set_ylabel('Bevölkerung (Anzahl)', fontsize=12)
    
    title = "Gesamtbevölkerung nach Verwaltungsebene (Hof)"
    if latest_time:
        title += f"\nStichtag: {latest_time.strftime('%d.%m.%Y')}"
    ax.set_title(title, fontsize=14, pad=20)

    # Add value labels on top of bars
    for p in ax.patches:
        val = p.get_height()
        ax.annotate(f"{int(val):,}".replace(',', '.'), 
                    (p.get_x() + p.get_width() / 2., val), 
                    ha='center', va='bottom', 
                    fontsize=12, fontweight='bold', xytext=(0, 5), 
                    textcoords='offset points')

    # Add rationale note if values are identical
    if plot_df['value'].nunique() == 1 and len(plot_df) > 1:
        plt.figtext(0.5, 0.01, "Hinweis: Die Werte für Kreis und Gemeinde sind identisch (Hof ist eine kreisfreie Stadt).", 
                    ha="center", fontsize=10, bbox={"facecolor":"orange", "alpha":0.1, "pad":5})

    plt.tight_layout(rect=[0, 0.05, 1, 1])
    
    # Save outputs
    out_path = os.path.join(output_dir, 'bevoelkerung_vergleich_ebenen.png')
    fig.savefig(out_path, bbox_inches='tight')
    
    # Save summary data
    plot_df.to_csv(os.path.join(output_dir, 'bevoelkerung_ebenen_summary.csv'), index=False, encoding='utf-8')
    
    plt.close(fig)

except Exception as e:
    print(f"Error: {str(e)}")
    sys.exit(1)
