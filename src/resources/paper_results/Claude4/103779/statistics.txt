install.packages("ggplot2", repos = "http://cran.r-project.org")
install.packages("dplyr", repos = "http://cran.r-project.org")
install.packages("tidyr", repos = "http://cran.r-project.org")
install.packages("stringr", repos = "http://cran.r-project.org")

library(ggplot2)
library(dplyr)
library(tidyr)
library(stringr)

df <- read.csv('./src/resources/data/study/103779/dataset.csv', sep = ',', encoding = 'UTF-8', stringsAsFactors = FALSE)

df$Begr_ndung <- trimws(df$Begr_ndung)
df$Begr_ndung[df$Begr_ndung == "" | df$Begr_ndung == "nan"] <- NA

begr_clean <- df %>%
  filter(!is.na(Begr_ndung)) %>%
  pull(Begr_ndung)

begr_split <- unlist(strsplit(begr_clean, ",\\s*"))

begr_split <- trimws(begr_split)

begr_freq <- as.data.frame(table(begr_split))
colnames(begr_freq) <- c("Kriterium", "Häufigkeit")

begr_freq <- begr_freq %>%
  arrange(desc(Häufigkeit))

png('./src/resources/output/103779/2_rechtliche_kriterien_schutzstatus.png', width = 1200, height = 800, res = 120)

ggplot(begr_freq, aes(x = reorder(Kriterium, Häufigkeit), y = Häufigkeit)) +
  geom_bar(stat = "identity", fill = "#2E86AB", color = "black", width = 0.7) +
  coord_flip() +
  labs(
    title = "Häufigkeit der rechtlichen Kriterien für den Denkmalschutzstatus",
    subtitle = paste0("Basis: ", length(begr_clean), " Denkmale mit dokumentierter Begründung (ca. ", 
                      round(length(begr_clean)/nrow(df)*100, 1), "% des Datensatzes)"),
    x = "Rechtliches Kriterium",
    y = "Anzahl der Nennungen"
  ) +
  theme_minimal(base_size = 12) +
  theme(
    plot.title = element_text(face = "bold", size = 14, hjust = 0),
    plot.subtitle = element_text(size = 10, hjust = 0, color = "gray30"),
    axis.text.y = element_text(size = 11),
    axis.text.x = element_text(size = 11),
    axis.title = element_text(size = 12, face = "bold"),
    panel.grid.major.y = element_blank(),
    panel.grid.minor = element_blank()
  ) +
  geom_text(aes(label = Häufigkeit), hjust = -0.2, size = 3.5)

dev.off()
