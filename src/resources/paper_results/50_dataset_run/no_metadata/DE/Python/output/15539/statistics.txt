import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

# Pfade definieren
input_path = r'./src/resources/data/study/15539/dataset.csv'
output_dir = r'./src/resources/output/15539'
os.makedirs(output_dir, exist_ok=True)

# Fehlerbehebung: Encoding-Problem lösen
# Da 'utf-8' fehlschlug (0xe4 ist 'ä' in latin-1/iso-8859-1), probieren wir 'latin-1'
try:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1', dtype=str)
except Exception as e:
    # Falls latin-1 auch fehlschlägt, versuche cp1252 (Windows Standard)
    try:
        df = pd.read_csv(input_path, sep=';', encoding='cp1252', dtype=str)
    except Exception as e2:
        sys.stderr.write(f"Fehler beim Laden der Datei: {str(e2)}")
        sys.exit(1)

# Datenbereinigung und Konvertierung
for col in ['time', 'value']:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col].str.replace(',', '.'), errors='coerce')

# Spaltennamen und Inhalte säubern
df.columns = [c.strip() for c in df.columns]
for col in df.columns:
    if df[col].dtype == object:
        df[col] = df[col].astype(str).str.strip()

# Validierung der benötigten Spalten
required_cols = ['time', 'value', '2_variable_attribute_label', 'value_variable_label']
if not all(col in df.columns for col in required_cols):
    sys.stderr.write(f"Erforderliche Spalten fehlen. Vorhanden: {df.columns.tolist()}")
    sys.exit(1)

# Filterung auf Übernachtungen und die Kategorien Inland/Ausland
# Wir suchen nach 'übernacht' im Label, um sicherzustellen, dass wir die richtige Metrik haben
mask_metric = df['value_variable_label'].str.contains('übernacht', case=False, na=False)
mask_origin = df['2_variable_attribute_label'].isin(['Inland', 'Ausland'])

sub = df[mask_metric & mask_origin].copy()

# Entfernen von Zeilen mit fehlenden Werten in den Kernspalten
sub = sub.dropna(subset=['time', 'value', '2_variable_attribute_label'])

if sub.empty:
    sys.stderr.write('Keine Daten für Gästeübernachtungen nach Herkunft (Inland/Ausland) gefunden.')
    sys.exit(1)

# Aggregation (falls mehrere Einträge pro Jahr/Herkunft existieren)
agg = sub.groupby(['time', '2_variable_attribute_label'], as_index=False)['value'].sum()

# Sortierung für die Visualisierung
agg['time'] = agg['time'].astype(int)
agg = agg.sort_values(['time', '2_variable_attribute_label'])

# Visualisierung
sns.set_theme(style='whitegrid', context='talk')
plt.figure(figsize=(14, 8))

# Gruppiertes Balkendiagramm
ax = sns.barplot(

ax.set_xlabel('Jahr', fontsize=12, fontweight='bold')

ax.set_title('Vergleich der Übernachtungen: Inland vs. Ausland', fontsize=16, pad=20)

ax.yaxis.set_major_formatter(FuncFormatter(lambda x, p: f"{int(x):,}".replace(',', '.')))

# Legende anpassen
ax.legend(title='Herkunft', frameon=True, loc='upper left')

print(f"Visualisierung erfolgreich erstellt unter: {plot_path}")

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter
import os

try:
    input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '15539', 'dataset.csv')
    output_dir = os.path.join('.', 'src', 'resources', 'output', '15539')
    os.makedirs(output_dir, exist_ok=True)
    output_path = os.path.join(output_dir, '2_gaeste_ankuenfte_vs_uebernachtungen.png')

    # Behebung des Encoding-Fehlers: 'iso-8859-1' (oder 'latin1') wird oft für deutsche CSVs mit Umlauten verwendet, 
    # wenn 'utf-8' fehlschlägt.
    try:
        df = pd.read_csv(input_path, sep=';', encoding='utf-8', dtype=str)
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1', dtype=str)

    if 'time' in df.columns:
        df['time'] = pd.to_numeric(df['time'], errors='coerce')
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')

    # Filterung auf relevante Kennzahlen
    # Wir suchen nach Gästeankünften und Gästeübernachtungen
    if 'value_variable_label' in df.columns:
        # Suche nach Labels, die die Begriffe enthalten (unabhängig von Groß-/Kleinschreibung)
        mask_ank = df['value_variable_label'].str.contains('Gästeankünfte', na=False, case=False)
        mask_ueb = df['value_variable_label'].str.contains('Gästeübernachtungen', na=False, case=False)
        df_metric = df[mask_ank | mask_ueb].copy()
    else:
        # Fallback auf Codes falls Labels fehlen
        df_metric = df[df['value_variable_code'].isin(['GANK03', 'GUEB03'])].copy()

    if df_metric.empty:
        raise ValueError('Es wurden keine relevanten Kennzahlen (Ankünfte/Übernachtungen) gefunden.')

    # Filterung auf 'Insgesamt' Werte, um Doppelzählungen durch Unterkategorien zu vermeiden
    label_col = '2_variable_attribute_label' if '2_variable_attribute_label' in df_metric.columns else None
    code_col = '2_variable_attribute_code' if '2_variable_attribute_code' in df_metric.columns else None

    mask_total = pd.Series(False, index=df_metric.index)
    if code_col is not None:
        # Oft ist 'DG' oder NaN der Code für den Gesamtwert
        mask_total = mask_total | df_metric[code_col].isna() | df_metric[code_col].str.contains('DG', na=False)
    if label_col is not None:
        mask_total = mask_total | df_metric[label_col].astype(str).str.strip().str.lower().str.contains('insgesamt', na=False)

    # Falls wir 'Insgesamt' Zeilen gefunden haben, nutzen wir diese, sonst nehmen wir alle
    if mask_total.any():
        df_filtered = df_metric[mask_total].copy()
    else:
        df_filtered = df_metric.copy()

    df_filtered = df_filtered.dropna(subset=['time', 'value'])
    df_filtered['time'] = df_filtered['time'].astype(int)

    # Aggregation nach Jahr und Kennzahl
    df_agg = df_filtered.groupby(['time', 'value_variable_label'], as_index=False)['value'].sum()
    
    if df_agg.empty:
        raise ValueError('Nach Aggregation sind keine Daten für die Visualisierung vorhanden.')

    # Sortierung der Jahre für die X-Achse
    years = sorted(df_agg['time'].unique())
    df_agg['time_str'] = pd.Categorical(df_agg['time'].astype(str), categories=[str(y) for y in years], ordered=True)

    # Visualisierung
    sns.set_theme(style='whitegrid', context='talk')
    plt.figure(figsize=(14, 8))
    
    # Farbwahl: Blau für Ankünfte, Orange für Übernachtungen (klassisch)
    palette = {'Gästeankünfte': '#3498db', 'Gästeübernachtungen': '#e67e22'}
    # Dynamische Palette falls Labels leicht abweichen
    unique_labels = df_agg['value_variable_label'].unique()
    plot_palette = {}
    for label in unique_labels:
        if 'ankünfte' in label.lower():
            plot_palette[label] = '#3498db'
        elif 'übernachtungen' in label.lower():
            plot_palette[label] = '#e67e22'
        else:
            plot_palette[label] = 'gray'

    ax = sns.barplot(
        data=df_agg, 
        x='time_str', 
        y='value', 
        hue='value_variable_label', 
        palette=plot_palette,
        edgecolor='black'
    )

    ax.set_title('Verhältnis Gästeankünfte zu Übernachtungen in Bad Berneck i.Fichtelgebirge', pad=20)
    ax.set_xlabel('Jahr')
    ax.set_ylabel('Anzahl')
    
    # Tausendertrennzeichen für die Y-Achse
    ax.yaxis.set_major_formatter(FuncFormatter(lambda x, pos: f'{int(x):,}'.replace(',', '.')))
    
    plt.legend(title='Kennzahl', bbox_to_anchor=(1.05, 1), loc='upper left')
    plt.tight_layout()
    
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()

except Exception as e:
    print(f"Fehler aufgetreten: {str(e)}")
