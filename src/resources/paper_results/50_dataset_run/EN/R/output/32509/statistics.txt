dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)

raw <- readr::read_csv(data_file, locale = readr::locale(encoding = "UTF-8"), show_col_types = FALSE)

 df <- raw %>% 
  mutate(across(where(is.character), ~stringr::str_squish(.))) %>%
  mutate(across(c(E_Mail, Telefax), ~na_if(., ""))) %>%
  mutate(across(c(E_Mail, Telefax), ~na_if(., "-")))

n_obs <- nrow(df)
email_vorhanden <- sum(!is.na(df$E_Mail))
fax_vorhanden <- sum(!is.na(df$Telefax))
missing_props <- df %>% summarise(across(c(E_Mail, Telefax), ~mean(is.na(.))))

order_levels <- df %>% 
  mutate(avail_email = as.integer(!is.na(E_Mail)), avail_fax = as.integer(!is.na(Telefax))) %>%
  arrange(desc(avail_email), desc(avail_fax), Kommune_Stadt) %>%
  pull(Kommune_Stadt) %>% unique()

plot_df <- df %>%
  select(Kommune_Stadt, E_Mail, Telefax) %>%
  pivot_longer(cols = c(E_Mail, Telefax), names_to = "Kanal", values_to = "Wert") %>%
  mutate(Kanal = factor(Kanal, levels = c("E_Mail", "Telefax"), labels = c("E-Mail", "Telefax")),
         Kommune_Stadt = factor(Kommune_Stadt, levels = rev(order_levels)),
         Verfügbarkeit = if_else(is.na(Wert), "Fehlt", "Vorhanden"),
         Wert_anzeige = if_else(is.na(Wert), "–", stringr::str_trunc(Wert, 28)),
         Textfarbe = if_else(Verfügbarkeit == "Vorhanden", "white", "#525252"))

p <- ggplot(plot_df, aes(x = Kanal, y = Kommune_Stadt, fill = Verfügbarkeit)) +

  geom_text(aes(label = Wert_anzeige, color = Textfarbe), size = 3.6, lineheight = 0.95) +

  scale_color_identity() +
