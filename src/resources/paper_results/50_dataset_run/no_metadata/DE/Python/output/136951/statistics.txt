import os

import matplotlib.pyplot as plt

try:
    input_path = r'./src/resources/data/study/136951/dataset.csv'
    output_dir = r'./src/resources/output/136951'
    os.makedirs(output_dir, exist_ok=True)
    
    # The error 'utf-8' codec can't decode byte 0xfc indicates Latin-1 (ISO-8859-1) or cp1252 encoding (0xfc is 'ü')
    # We try 'latin-1' which is common for German datasets containing umlauts.
    try:
        df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=',', encoding='latin-1')

    df.columns = [str(c) for c in df.columns]
    
    # Data Preparation
    df_vis = df[['Pruefmerkmal', 'Ergebnis', 'Einheit']].copy()
    
    # Clean 'Ergebnis' for numeric plotting
    df_vis['Ergebnis_str'] = df_vis['Ergebnis'].astype(str)
    df_vis['censored'] = df_vis['Ergebnis_str'].str.contains('<', na=False)
    
    # Replace German decimal comma with dot and remove comparison operators
    df_vis['Wert'] = df_vis['Ergebnis_str'].str.replace(',', '.', regex=False)
    df_vis['Wert'] = df_vis['Wert'].str.replace('<', '', regex=False).str.replace('>', '', regex=False)
    df_vis['Wert'] = pd.to_numeric(df_vis['Wert'], errors='coerce')
    
    # Drop rows where we can't get a numeric value or label
    df_vis = df_vis.dropna(subset=['Pruefmerkmal', 'Wert'])
    
    # Create a label combining result and unit
    df_vis['Label'] = (df_vis['Ergebnis'].astype(str) + ' ' + df_vis['Einheit'].astype(str)).str.strip()
    
    # Visualization
    sns.set_theme(style='whitegrid', context='talk')
    plt.figure(figsize=(10, 7))
    
    # Use a barplot
    ax = sns.barplot(
        data=df_vis,

    if df_vis['Wert'].max() / (df_vis['Wert'].min() + 1e-9) > 100:

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import os

try:
    # Pfad zur Datei
    input_path = './src/resources/data/study/136951/dataset.csv'
    
    if not os.path.exists(input_path):
        input_path = 'dataset.csv'

    # BEHEBUNG DES FEHLERS: 'utf-8' codec can't decode byte 0xfc (ü)
    # Da die Daten deutsche Umlaute enthalten (Prüfmerkmal), ist 'latin-1' oder 'iso-8859-1' oft die richtige Wahl,
    # wenn utf-8 fehlschlägt.
    try:
        df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=',', encoding='latin-1')
    
    # Datenbereinigung: Whitespace-Strings zu NaN
    df = df.replace(r'^\s*$', np.nan, regex=True)
    
    # Sicherstellen, dass notwendige Spalten existieren (Case-insensitive Check)
    required_cols = ['Methode', 'Pruefmerkmal', 'Ergebnis', 'Einheit']
    # Mapping für flexiblere Spaltennamen (falls Pruefmerkmal vs Prüfmerkmal)
    col_map = {c.lower(): c for c in df.columns}
    
    final_cols = []
    for col in required_cols:
        found = False
        for actual_col in df.columns:
            if col.lower() == actual_col.lower():
                final_cols.append(actual_col)
                found = True
                break
        if not found:
            df[col] = 'k.A.'
            final_cols.append(col)

    # Ergebnis-Anzeige formatieren (Wert + Einheit)
    def format_result(row):
        res = str(row[final_cols[2]]).strip()
        einheit = str(row[final_cols[3]]).strip()
        if res.lower() in ['nan', '', 'none']:
            return 'k.A.'
        if einheit.lower() not in ['nan', '', 'none']:
            return f"{res} {einheit}"
        return res

    df['Ergebnis_Anzeige'] = df.apply(format_result, axis=1)
    
    # Relevante Daten für die Tabelle extrahieren
    tdf = df[[final_cols[0], final_cols[1], 'Ergebnis_Anzeige']].copy()
    tdf.columns = ['Methode', 'Prüfmerkmal', 'Ergebnis']
    
    # Fehlende Werte füllen
    tdf = tdf.fillna('k.A.')

    # Visualisierung als Tabelle
    rows_count = len(tdf)
    # Dynamische Höhe basierend auf Zeilenanzahl
    fig_h = max(2, 1.0 + 0.4 * rows_count)
    fig_w = 10
    
    fig, ax = plt.subplots(figsize=(fig_w, fig_h), dpi=200)
    ax.axis('off')

    # Tabelle erstellen
    table = ax.table(
        cellText=tdf.values, 
        colLabels=tdf.columns, 
        loc='center', 
        cellLoc='left'
    )

    # Tabellen-Styling
    table.auto_set_font_size(False)
    table.set_fontsize(10)
    table.scale(1, 1.5) 

    for (row, col), cell in table.get_celld().items():
        if row == 0:
            cell.set_text_props(weight='bold', color='white')
            cell.set_facecolor('#2c3e50') 
        else:
            cell.set_facecolor('#ffffff' if row % 2 != 0 else '#f9f9f9')
        cell.set_edgecolor('#dddddd')

    plt.title('Verteilung der Messwerte nach Methode und Prüfmerkmal', 
              fontsize=12, weight='bold', pad=10)

    # Speicherpfad
    save_dir = './src/resources/output/136951'

    plt.savefig(save_path, bbox_inches='tight', dpi=300)
    plt.close(fig)

    print(f"Fehler aufgetreten: {str(e)}")

import os
import sys
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Behebung des Encoding-Fehlers: 'latin-1' oder 'iso-8859-1' wird oft für deutsche Datensätze verwendet,
# wenn utf-8 fehlschlägt (wegen Umlauten wie ü, ä, ö).
input_path = './src/resources/data/study/136951/dataset.csv'
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(input_path, sep=',', encoding='latin-1')
    except Exception as e:
        sys.stderr.write(f"Fehler beim Laden der Datei: {str(e)}")
        sys.exit(1)
except Exception as e:
    sys.stderr.write(f"Fehler beim Laden der Datei: {str(e)}")
    sys.exit(1)

try:
    # Hilfsfunktion zur Extraktion des ersten gültigen Wertes
    def first_non_null(series):
        if series is None or series.empty:
            return None
        s = series.dropna()
        if s.empty:
            return None
        # Entferne leere Strings oder 'nan' Strings
        v = s.astype(str).str.strip()
        v = v[v != '']
        v = v[v.str.lower() != 'nan']
        if v.empty:
            return None
        return v.iloc[0]

    # Datenextraktion
    beg_date = first_non_null(df.get('Probenahme_Beginn_Datum', pd.Series(dtype=object)))
    beg_time = first_non_null(df.get('Probenahme_Beginn_Zeit', pd.Series(dtype=object)))
    end_date = first_non_null(df.get('Probenahme_Ende_Datum', pd.Series(dtype=object)))
    end_time = first_non_null(df.get('Probenahme_Ende_Zeit', pd.Series(dtype=object)))

    def combine_display(date_str, time_str):
        if not date_str:
            return None
        if not time_str:
            return str(date_str)
        return f"{date_str} {time_str}"

    start_disp = combine_display(beg_date, beg_time)
    end_disp = combine_display(end_date, end_time)

    # Zeitraum-Logik
    if start_disp and end_disp:
        zeitraum = start_disp if start_disp == end_disp else f"{start_disp} – {end_disp}"
    elif start_disp:
        zeitraum = start_disp
    elif end_disp:
        zeitraum = end_disp
    else:
        zeitraum = 'k.A.'

    entnahmestelle = first_non_null(df.get('Entnahmestelle', pd.Series(dtype=object))) or 'k.A.'
    umweltbereich = first_non_null(df.get('Umweltbereich', pd.Series(dtype=object))) or 'k.A.'

    # Erstellung des Info-DataFrames für die Tabelle
    info_df = pd.DataFrame([
        {
            'Entnahmestelle': entnahmestelle,
            'Probenahme_Zeitraum': zeitraum,
            'Umweltbereich': umweltbereich,
        }
    ])

    # Visualisierung
    sns.set_theme(context='talk', style='white')
    fig, ax = plt.subplots(figsize=(12, 3), dpi=150)
    ax.axis('off')

    header_colors = ['#1f77b4', '#1f77b4', '#1f77b4']
    table = ax.table(
        cellText=info_df.values,
        colLabels=info_df.columns,
        loc='center',
        cellLoc='center',
        colLoc='center',
        colColours=header_colors
    )

    table.auto_set_font_size(False)
    table.set_fontsize(12)
    table.scale(1, 2.5)

    # Styling der Zellen
    for (row, col), cell in table.get_celld().items():
        if row == 0:
            cell.set_text_props(color='white', weight='bold')
        else:
            cell.set_facecolor('#f9f9f9')
            cell.set_edgecolor('#dddddd')

    plt.title('Kontext der Probenahme: Ort, Zeitraum und Umweltbereich', 
              fontsize=16, fontweight='bold', pad=20)

    # Speicherpfad sicherstellen
    output_dir = os.path.join('.', 'src', 'resources', 'output', '136951')
    os.makedirs(output_dir, exist_ok=True)
    output_path = os.path.join(output_dir, '2_probenahme_uebersicht.png')
    
    plt.savefig(output_path, bbox_inches='tight')
    plt.close(fig)
    
    print(f"Visualisierung erfolgreich gespeichert unter: {output_path}")

except Exception as e:
    sys.stderr.write(f"Fehler bei der Verarbeitung: {str(e)}")
    sys.exit(1)
