# Daten einlesen
# check.names = FALSE ist wichtig, um die Spaltennamen exakt wie in der CSV zu erhalten
df <- read.csv("./src/resources/data/study/51886/dataset.csv", sep = ";", encoding = "UTF-8", stringsAsFactors = FALSE, check.names = FALSE)

# Fehlerbehebung: Der Fehler "Can't convert a character vector to a symbol" 
# tritt auf, wenn grep() mehrere Spalten findet oder target_col leer ist.
# Wir identifizieren die Spalte '2_variable_attribute_label' explizit.
target_col_name <- "2_variable_attribute_label"

# Falls die Spalte nicht exakt so heißt (z.B. durch Encoding-Probleme), 
# suchen wir nach dem Teilstring "variable_attribute_label"
if (!(target_col_name %in% colnames(df))) {
  found_cols <- grep("variable_attribute_label", colnames(df), value = TRUE)
  if (length(found_cols) > 0) {
    target_col_name <- found_cols[1] # Nimm den ersten Treffer
  } else {
    # Fallback auf die zweite Spalte, falls gar nichts gefunden wird
    target_col_name <- colnames(df)[2]
  }
}

# Datenaufbereitung
df_plot <- df %>%
  # Wir nutzen .data[[var]] für sicherere dynamische Spaltenreferenzierung
  filter(.data[[target_col_name]] == "Insgesamt") %>%
  mutate(
    # Sicherstellen, dass Zeit und Wert numerisch sind (Komma zu Punkt Umwandlung)
    time = as.numeric(gsub(",", ".", as.character(time))),
    value = as.numeric(gsub(",", ".", as.character(value)))
  ) %>%
  filter(!is.na(time) & !is.na(value)) %>%
  arrange(time)

# Verzeichnis erstellen
dir.create("./src/resources/output/51886", recursive = TRUE, showWarnings = FALSE)

# Visualisierung
if (nrow(df_plot) > 0) {
  p <- ggplot(df_plot, aes(x = time, y = value)) +
    geom_line(color = "#2c3e50", linewidth = 1.2) +
    geom_point(color = "#e74c3c", size = 3) +
    # Beschriftung der Datenpunkte
    geom_text(aes(label = value), vjust = -1.5, size = 3.5, color = "#34495e") +
    scale_x_continuous(breaks = seq(min(df_plot$time), max(df_plot$time), 1)) +
    # Mehr Platz nach oben für die Labels
    scale_y_continuous(expand = expansion(mult = c(0.1, 0.3))) +
    labs(
      title = "Gesamtzahl der fertiggestellten Wohnungen in Oberschleißheim",
      subtitle = paste0("Zeitliche Entwicklung (Filter: ", target_col_name, " = 'Insgesamt')"),
      x = "Berichtsjahr",
      y = "Anzahl der Wohnungen",
      caption = "Quelle: Statistik der Baufertigstellungen"
    ) +

      plot.title = element_text(face = "bold", size = 14),

# Benötigte Pakete laden
if (!require("tidyverse")) install.packages("tidyverse")
if (!require("janitor")) install.packages("janitor")

library(tidyverse)
library(janitor)

# Pfad zur Datei
file_path <- "./src/resources/data/study/51886/dataset.csv"

# Fehlerbehebung beim Einlesen:
# Die Fehlermeldung deutet auf Encoding-Probleme und unvollständige Zeilen hin.
# Wir nutzen read_csv2 (für Semikolon-getrennte Dateien mit Komma-Dezimal) 
# oder read_delim mit automatischer Erkennung.
# 'locale' mit encoding 'latin1' hilft oft bei deutschen Datensätzen.

df <- read_delim(
  file_path, 
  delim = ";", 
  locale = locale(encoding = "latin1", decimal_mark = ","),
  show_col_types = FALSE,
  trim_ws = TRUE
)

# Spaltennamen bereinigen
df <- df %>% clean_names()

# Zielspalte identifizieren (basierend auf der Anforderung '2_variable_attribute_label')
# clean_names macht daraus meist 'x2_variable_attribute_label'
target_col <- "x2_variable_attribute_label"

# Falls die Spalte anders heißt (z.B. durch clean_names), suchen wir sie
if (!target_col %in% names(df)) {
  target_col <- names(df)[str_detect(names(df), "variable_attribute_label")]
}

# Datenaufbereitung
df_clean <- df %>%
  # Sicherstellen, dass 'value' numerisch ist
  mutate(value = as.numeric(value)) %>%
  filter(!is.na(value)) %>%
  # "Insgesamt" ausschließen (case-insensitive)
  filter(!str_detect(tolower(!!sym(target_col)), "insgesamt"))

# Aggregation: Summe der Werte pro Kategorie
agg_data <- df_clean %>%
  group_by(!!sym(target_col)) %>%
  summarise(total_value = sum(value, na.rm = TRUE)) %>%
  ungroup() %>%
  arrange(total_value)

# Verzeichnis erstellen
dir.create("./src/resources/output/51886", recursive = TRUE, showWarnings = FALSE)

# Visualisierung: Horizontales Balkendiagramm
# Wir nutzen big.mark="." und decimal.mark="," für deutsche Formatierung
p <- ggplot(agg_data, aes(x = reorder(!!sym(target_col), total_value), y = total_value)) +
  geom_col(fill = "#2c7fb8", color = "white") +
  coord_flip() +
  # Korrektur der Warnung: big.mark und decimal.mark müssen unterschiedlich sein
  scale_y_continuous(labels = scales::label_number(big.mark = ".", decimal.mark = ",")) +
  labs(
    title = "Häufigkeit fertiggestellter Wohnungsgrößen",
    subtitle = "Summe der Fertigstellungen nach Raumzahl (Gesamtzeitraum)",
    x = "Wohnungsgröße (Raumzahl)",
    y = "Gesamtanzahl fertiggestellter Wohnungen",
    caption = "Quelle: Datensatz 51886"
  ) +
  theme_minimal(base_size = 12) +
  theme(
    plot.title = element_text(face = "bold", size = 14),
    axis.title = element_text(face = "bold"),
    panel.grid.major.y = element_blank(),
    plot.background = element_rect(fill = "white", color = NA)
  ) +
  # Text-Labels an den Balken
  geom_text(aes(label = format(total_value, big.mark = ".", decimal.mark = ",", scientific = FALSE)), 
            hjust = -0.1, size = 3.5)

# Speichern des Diagramms
ggsave("./src/resources/output/51886/1_wohnungsgrössen_vergleich.png", 
       plot = p, 
       width = 10, 
       height = 6, 
       dpi = 300)

# Erfolgskontrolle
print("Visualisierung erfolgreich erstellt und unter ./src/resources/output/51886/1_wohnungsgrössen_vergleich.png gespeichert.")

if (!require("pacman")) install.packages("pacman")
pacman::p_load(tidyverse, RColorBrewer)

file_path <- "./src/resources/data/study/51886/dataset.csv"
output_path <- "./src/resources/output/51886/2_baufertigstellungen_raumkategorien.png"

if (!dir.exists(dirname(output_path))) {
  dir.create(dirname(output_path), recursive = TRUE)
}

# Einlesen der Daten
# Da Spaltennamen oft durch Encoding oder Sonderzeichen (wie '2_...') Probleme machen, 
# lesen wir sie ein und bereinigen sie sofort.
df <- read.csv(file_path, sep = ";", encoding = "UTF-8", stringsAsFactors = FALSE, check.names = FALSE)

# Fehlerbehebung: Sicherstellen, dass die Spalte existiert. 
# Falls '2_variable_attribute_label' nicht exakt so heißt (z.B. durch Encoding-Fehler beim Einlesen),
# suchen wir nach der Spalte, die 'variable_attribute_label' im Namen trägt.
col_names <- colnames(df)
target_col <- col_names[grepl("variable_attribute_label", col_names)]

if(length(target_col) == 0) {
  stop("Die Spalte '2_variable_attribute_label' wurde nicht gefunden. Vorhandene Spalten: ", paste(col_names, collapse=", "))
}

# Umbenennen für einfacheres Handling
df <- df %>% rename(category = !!sym(target_col[1]))

# Datentypen konvertieren
df$value <- as.numeric(df$value)
df$time <- as.integer(df$time)

# Daten filtern: 'Insgesamt' entfernen und nur gültige Werte behalten
plot_data <- df %>%
  filter(!is.na(value)) %>%
  filter(!grepl("Insgesamt", category, ignore.case = TRUE))

# Sortierung der Kategorien
kategorien <- unique(plot_data$category)
plot_data$category <- factor(plot_data$category, levels = sort(kategorien))

# Erstellung des gestapelten Säulendiagramms
p <- ggplot(plot_data, aes(x = factor(time), y = value, fill = category)) +
  geom_bar(stat = "identity", position = "stack", color = "white", linewidth = 0.2) +
  scale_fill_brewer(palette = "Spectral", name = "Raumkategorie") +
  labs(
    title = "Baufertigstellungen von Wohnungen nach Raumkategorien",
    subtitle = "Jährliche Verteilung der fertiggestellten Einheiten",
    x = "Berichtsjahr",
    y = "Anzahl der Wohnungen",
    caption = "Quelle: Datensatz 51886. Hinweis: 'Insgesamt' wurde aus der Darstellung entfernt."
  ) +
  theme_minimal(base_size = 12) +
  theme(
    legend.position = "bottom",
    legend.title = element_text(size = 10),
    legend.text = element_text(size = 8),
    plot.title = element_text(face = "bold", size = 14),
    axis.text.x = element_text(angle = 45, vjust = 0.5),
    panel.grid.minor = element_blank()
  ) +
  guides(fill = guide_legend(ncol = 2))

# Speichern des Diagramms
ggsave(output_path, plot = p, width = 10, height = 8, dpi = 300)
