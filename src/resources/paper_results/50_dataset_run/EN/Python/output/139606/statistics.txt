import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

input_path = os.path.normpath('./src/resources/data/study\\139606\\dataset.csv')
output_path = os.path.normpath('./src/resources/output\\139606/0_verteilung_ergebnis.png')
os.makedirs(os.path.dirname(output_path), exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

missing_per_column = df.isna().mean()
overall_missing_pct = float(df.isna().sum().sum()) / float(df.shape[0] * df.shape[1]) * 100 if df.size > 0 else 0.0

erg_raw = df['Ergebnis'].astype(str).str.strip()
erg_std = erg_raw.str.replace(',', '.', regex=False)
erg_num = pd.to_numeric(erg_std, errors='coerce')

detected_flag = erg_num.notna()
category = np.where(detected_flag, 'Nachweis (numerischer Wert)', 'Nicht nachweisbar (n.n.)')
category = pd.Series(category, name='Ergebnis-Kategorie')

order = ['Nicht nachweisbar (n.n.)', 'Nachweis (numerischer Wert)']
counts = category.value_counts().reindex(order).fillna(0).astype(int)

total = int(counts.sum())
percent = (counts / total * 100.0).round(1) if total > 0 else counts.astype(float)
plot_df = pd.DataFrame({'Ergebnis-Kategorie': counts.index, 'Anzahl': counts.values, 'Anteil_%': percent.values})

sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(9, 6))
palette = ['#9aa0a6', '#2a9d8f']
sns.barplot(data=plot_df, x='Ergebnis-Kategorie', y='Anzahl', palette=palette, ax=ax)

ax.set_title(f'Verteilung der Nachweisergebnisse über alle Radionuklide (n={total})', pad=16)
ax.set_xlabel('Ergebnis-Kategorie')
ax.set_ylabel('Anzahl der untersuchten Radionuklide [Stück]')

ylim_top = max(plot_df['Anzahl'].max() * 1.2, 1)
ax.set_ylim(0, ylim_top)
for i, row in plot_df.iterrows():
    ax.text(i, row['Anzahl'] + ylim_top * 0.03, f"{int(row['Anzahl'])} ({row['Anteil_%']:.1f}%)", ha='center', va='bottom', fontsize=12, weight='bold')

ax.text(0.5, -0.18, f"Hinweis: 'n.n.' = nicht nachweisbar. Fehlende Werte (gesamt): {overall_missing_pct:.1f}%", transform=ax.transAxes, ha='center', va='top', fontsize=10, color='#555555')

for spine in ['top', 'right']:
    ax.spines[spine].set_visible(False)

plt.tight_layout()
fig.savefig(output_path, dpi=200, bbox_inches='tight', pad_inches=0.1)
plt.close(fig)

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(context='notebook', style='whitegrid', font_scale=1.0)

in_path = os.path.join('.', 'src', 'resources', 'data', 'study', '139606', 'dataset.csv')
out_dir = os.path.join('.', 'src', 'resources', 'output', '139606')
os.makedirs(out_dir, exist_ok=True)
out_path = os.path.join(out_dir, '1_pruefmerkmal_numerische_ergebnisse_tabelle.png')

df = pd.read_csv(in_path, sep=',', encoding='utf-8')

missing_pct = df.isna().mean()

e_str = df['Ergebnis'].astype(str).str.strip().str.lower()
is_nn = e_str.str.replace('.', '', regex=False).eq('nn')

def parse_de_number(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    if s == '':
        return np.nan
    sl = s.lower()
    if sl.replace('.', '') in ['nn'] or sl in ['n.n.', 'n.n', 'nn']:
        return np.nan
    s2 = s.replace(',', '.')
    try:
        return float(s2)
    except Exception:
        return np.nan

df['Ergebnis_num'] = df['Ergebnis'].apply(parse_de_number)

df_detected = df.loc[~is_nn & df['Ergebnis_num'].notna()].copy()

n_total = int(len(df))
n_detected = int(len(df_detected))
n_nn = int(n_total - n_detected)
share_nn = 0.0 if n_total == 0 else n_nn / n_total

if n_detected > 0:
    cols_show = ['Pruefmerkmal', 'Ergebnis', 'Einheit']
    for c in cols_show:
        if c not in df_detected.columns:
            df_detected[c] = ''
    df_detected = df_detected.sort_values(['Einheit', 'Pruefmerkmal', 'Ergebnis_num'], ascending=[True, True, False])
    data_rows = df_detected[cols_show].astype(str).values.tolist()
    col_labels = ['Prüfmerkmal', 'Ergebnis', 'Einheit']
    n_rows = len(data_rows)
    fig_h = max(3.0, 1.2 + 0.55 * n_rows)
    fig_w = 12.0
    fig, ax = plt.subplots(figsize=(fig_w, fig_h))
    ax.axis('off')
    tbl = ax.table(cellText=data_rows, colLabels=col_labels, loc='center', cellLoc='left', colLoc='center')
    tbl.auto_set_font_size(False)
    tbl.set_fontsize(10)
    tbl.scale(1, 1.4)
    total_cols = len(col_labels)
    col_widths = [0.6, 0.22, 0.18]
    for j in range(total_cols):
        key = (0, j)
        if key in tbl._cells:
            tbl._cells[key].set_facecolor('#234F6D')
            tbl._cells[key].set_text_props(color='white', weight='bold', ha='center', va='center')
            tbl._cells[key].set_edgecolor('#234F6D')
    for i in range(1, n_rows + 1):
        for j in range(total_cols):
            key = (i, j)
            if key in tbl._cells:
                face = '#F4F6F7' if i % 2 == 0 else '#FFFFFF'
                tbl._cells[key].set_facecolor(face)
                tbl._cells[key].PAD = 0.8
                if j == 0:
                    tbl._cells[key].set_width(fig_w * col_widths[0] / fig_w)
                elif j == 1:
                    tbl._cells[key].set_width(fig_w * col_widths[1] / fig_w)
                else:
                    tbl._cells[key].set_width(fig_w * col_widths[2] / fig_w)
    ax.set_title('Nachweisbare Ergebnisse (≠ n.n.) – Gammaspektrometrie, Probe 202000065', fontsize=14, fontweight='bold', pad=18)
    fig.text(0.01, 0.93, f'Anzahl Prüfmerkmale: {n_total} | Nachweisbar: {n_detected} | Anteil n.n.: {share_nn*100:.1f} %', ha='left', fontsize=10, color='dimgray')
    fig.text(0.01, 0.02, "Hinweis: 'n.n.' = nicht nachweisbar; Einheiten gemäß rechter Spalte.", ha='left', fontsize=9, color='dimgray')
    plt.subplots_adjust(top=0.86, bottom=0.08)
    plt.savefig(out_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
else:
    fig, ax = plt.subplots(figsize=(10, 4))
    ax.axis('off')
    ax.text(0.5, 0.6, 'Keine numerischen Ergebnisse gefunden (alle Werte n.n.)', ha='center', va='center', fontsize=13, weight='bold')

    plt.savefig(out_path, dpi=300, bbox_inches='tight')
    plt.close(fig)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style='whitegrid', context='talk')

fp = './src/resources/data/study\\139606\\dataset.csv'
df = pd.read_csv(fp, sep=',', encoding='utf-8')

_ = df.isna().mean()

def _to_num(x):
    if pd.isna(x):
        return np.nan
    if isinstance(x, (int, float)):
        return float(x)
    s = str(x).strip()
    if s.lower() in ['n.n.', 'n.n', 'nn', 'na', 'nan', '']:
        return np.nan
    s = s.replace(',', '.')
    return pd.to_numeric(s, errors='coerce')

df['Ergebnis_num'] = df['Ergebnis'].apply(_to_num)
_ = df['Ergebnis_num'].describe()

units = df['Einheit'].fillna('Unbekannt').astype(str).str.strip()
unit_counts = units.value_counts(dropna=False)
unit_perc = unit_counts / unit_counts.sum() * 100

plt.figure(figsize=(9, 5.5))
colors = sns.color_palette('viridis', n_colors=len(unit_counts))
ax = plt.gca()
ax.bar(unit_counts.index.astype(str), unit_counts.values, color=colors)
ax.set_title('Messeinheiten im Radiologie-Screening (Anzahl Datensätze)')
ax.set_xlabel('Einheit')
ax.set_ylabel('Anzahl Datensätze')
ax.set_ylim(0, max(unit_counts.values) * 1.2)
for i, (cnt, pct) in enumerate(zip(unit_counts.values, unit_perc.values)):
    ax.text(i, cnt + max(unit_counts.values) * 0.03, f"{int(cnt)} ({pct:.1f}%)", ha='center', va='bottom', fontsize=11)
ax.grid(axis='y', linestyle='--', alpha=0.4)
plt.tight_layout()

op = './src/resources/output\\139606/2_einheiten_verteilung.png'
plt.savefig(op, dpi=300)
plt.close()
