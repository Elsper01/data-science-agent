import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Konfiguration
pd.set_option('display.max_columns', None)

data_path = r'./src/resources/data/study/136560/dataset.csv'
output_dir = r'./src/resources/output/136560'
os.makedirs(os.path.normpath(output_dir), exist_ok=True)

# Daten laden
df = pd.read_csv(data_path, sep=',', encoding='utf-8')

# Datenvorbereitung
# Fehlerbehebung: datetime_is_numeric=True wurde in neueren Pandas-Versionen entfernt, da es nun Standard ist
_ = df.describe(include='all')

# Filtern auf Einheit Bq/l
sub = df[df['Einheit'].astype(str).str.strip().str.lower().eq('bq/l')].copy()

if not sub.empty:
    # Ergebnis bereinigen und konvertieren
    sub['Ergebnis_clean'] = sub['Ergebnis'].astype(str).str.strip()
    # 'n.n.' oder ähnliche Nicht-Nachweise identifizieren
    sub['is_nd'] = sub['Ergebnis_clean'].str.lower().str.contains('n')
    # Numerische Werte extrahieren
    sub['Ergebnis_num'] = pd.to_numeric(sub['Ergebnis_clean'].str.replace(',', '.', regex=False), errors='coerce')
    
    # Aggregation pro Prüfmerkmal
    plot_df = sub.groupby('Pruefmerkmal', as_index=False).apply(lambda g: pd.Series({
        'Aktivitaet_Bq_l': g['Ergebnis_num'].mean() if g['Ergebnis_num'].notna().any() else 0.0,
        'ist_nicht_nachweisbar': bool(g['is_nd'].any() and g['Ergebnis_num'].isna().all()),
        'anzahl_numeric': int(g['Ergebnis_num'].notna().sum())
    })).reset_index(drop=True)
else:
    plot_df = pd.DataFrame(columns=['Pruefmerkmal', 'Aktivitaet_Bq_l', 'ist_nicht_nachweisbar', 'anzahl_numeric'])

# Visualisierung
sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(10, 6), dpi=150)

if plot_df.empty:
    ax.axis('off')
    ax.text(0.5, 0.5, 'Keine Messwerte in Bq/l vorhanden.', ha='center', va='center', fontsize=14)
else:
    detect_color = '#2B8CBE'
    nd_color = '#9E9E9E'
    
    # Farben basierend auf Nachweisbarkeit
    colors = [nd_color if row['ist_nicht_nachweisbar'] else detect_color for _, row in plot_df.iterrows()]
    
    bars = ax.bar(plot_df['Pruefmerkmal'], plot_df['Aktivitaet_Bq_l'], color=colors, edgecolor='#333333', linewidth=0.8)
    
    ymax = float(plot_df['Aktivitaet_Bq_l'].max())
    # Padding für Labels
    padding = 0.05 * (ymax if ymax > 0 else 1)
    
    for bar, (_, row) in zip(bars, plot_df.iterrows()):
        y = float(row['Aktivitaet_Bq_l'])
        if row['ist_nicht_nachweisbar']:
            bar.set_hatch('//')
            ax.text(bar.get_x() + bar.get_width() / 2, y + padding, 'n.n.', ha='center', va='bottom', fontsize=10, fontweight='bold')
        else:
            label = f"{y:.3f}".replace('.', ',')
            ax.text(bar.get_x() + bar.get_width() / 2, y + padding, label, ha='center', va='bottom', fontsize=10)

    ax.set_ylabel('Aktivitätskonzentration (Bq/l)', fontsize=12)
    ax.set_xlabel('Prüfmerkmal', fontsize=12)
    ax.set_title('Gemessene Aktivitäten nach Prüfmerkmal (Einheit: Bq/l)', fontsize=14, pad=20)
    
    # Dynamisches Y-Limit
    ax.set_ylim(0, (ymax * 1.3) if ymax > 0 else 1)
    
    # X-Achsen Beschriftung rotieren
    plt.xticks(rotation=15, ha='right')
    
    # Fußnote
    ax.text(0.5, -0.25, "Hinweis: 'n.n.' markiert Werte unter der Nachweisgrenze (als 0 dargestellt).", 
            transform=ax.transAxes, ha='center', va='top', fontsize=9, color='#555555', style='italic')

out_path = os.path.normpath(os.path.join(output_dir, '0_aktivitaeten_vergleich.png'))
plt.tight_layout()
plt.savefig(out_path, dpi=200, bbox_inches='tight')
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

sns.set_theme(style='white', context='talk')
plt.rcParams['font.family'] = 'DejaVu Sans'

input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '136560', 'dataset.csv')
output_dir = os.path.join('.', 'src', 'resources', 'output', '136560')
os.makedirs(output_dir, exist_ok=True)

try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except Exception as e:
    print(e)
    raise

for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].astype(str).str.strip()

missing_token_patterns = ['n.n.', 'n.n', 'nn', 'NaN', 'nan', '']
df_mv = df.replace(missing_token_patterns, np.nan)
missing_ratio = df_mv.isna().mean()

if 'Ergebnis' in df.columns:
    ergebnis_clean = (
        df['Ergebnis']
        .astype(str)
        .str.strip()
        .replace({'n.n.': np.nan, 'n.n': np.nan, 'nn': np.nan, '': np.nan})
        .str.replace(',', '.', regex=False)
    )
    df['Ergebnis_num'] = pd.to_numeric(ergebnis_clean, errors='coerce')
else:
    df['Ergebnis_num'] = np.nan

numeric_cols = ['Ergebnis_num']
num_df = df[numeric_cols].select_dtypes(include=[np.number])
desc_stats = num_df.describe().T

if {'Methode', 'Einheit'}.issubset(df.columns):
    grouped_stats = df.groupby(['Methode', 'Einheit'], dropna=False)['Ergebnis_num'].agg(['count', 'mean', 'std', 'min', 'max'])
else:
    grouped_stats = pd.DataFrame()

plot_counts = pd.Series(dtype=int)
if 'Methode' in df.columns:
    method_series = df['Methode'].dropna()
    method_series = method_series[method_series.str.len() > 0]
    if not method_series.empty:
        plot_counts = method_series.value_counts().sort_values(ascending=False)

fig, ax = plt.subplots(figsize=(8, 8), dpi=150)
if plot_counts.empty:
    ax.axis('off')
    ax.text(0.5, 0.5, 'Keine Daten zur Spalte "Methode" verfügbar', ha='center', va='center', fontsize=16)
else:
    colors = sns.color_palette('deep', n_colors=len(plot_counts))
    def make_autopct(values):
        def autopct(pct):
            total = np.sum(values)
            val = int(round(pct * total / 100.0))
            return f'{pct:.1f}% ({val})'
        return autopct
    wedges, texts, autotexts = ax.pie(
        plot_counts.values,
        labels=plot_counts.index,
        autopct=make_autopct(plot_counts.values),
        startangle=90,
        colors=colors,
        pctdistance=0.75,
        labeldistance=1.05,
        wedgeprops={'linewidth': 1, 'edgecolor': 'white'}
    )
    ax.axis('equal')
    for t in texts:
        t.set_fontsize(12)
    for t in autotexts:
        t.set_fontsize(11)
    ax.set_title('Messmethoden der Trinkwasseruntersuchung – Häufigkeitsverteilung', pad=20)

save_path = os.path.join(output_dir, '1_methoden_kreisdiagramm.png')
plt.tight_layout()
plt.savefig(save_path, bbox_inches='tight')
plt.close(fig)
