import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np
import os

# Fix file path issues by using raw strings or forward slashes to avoid escape sequence errors
file_path = r'./src/resources/data/study/32830/dataset.csv'

# Ensure the directory exists for output
output_dir = r'./src/resources/output/32830'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

# Load the dataset
try:
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')
except FileNotFoundError:
    # Fallback for different environments if necessary
    df = pd.read_csv('dataset.csv', sep=',', encoding='utf-8')

# Data Cleaning
# Drop the first row if it's a summary/header-like row (based on previous code logic)
if len(df) > 0:
    df = df.drop(index=0)

# Drop columns that are likely empty or irrelevant
if 'Unnamed__14' in df.columns:
    df = df.drop(columns=['Unnamed__14'])

def convert_time_to_seconds(time_str):
    if pd.isnull(time_str) or not isinstance(time_str, str) or ':' not in time_str:
        return np.nan
    try:
        parts = list(map(int, time_str.split(':')))
        if len(parts) == 3:
            h, m, s = parts
            return h * 3600 + m * 60 + s
        elif len(parts) == 2:
            m, s = parts
            return m * 60 + s
    except ValueError:
        return np.nan
    return np.nan

# Process 'Zeit_im_Besuch' (Time in Visit)
if 'Zeit_im_Besuch' in df.columns:
    df['Zeit_im_Besuch_sec'] = df['Zeit_im_Besuch'].apply(convert_time_to_seconds)

# Process 'Seiten_pro_Besuch' (Pages per Visit)
if 'Seiten_pro_Besuch' in df.columns:
    df['Seiten_pro_Besuch'] = pd.to_numeric(df['Seiten_pro_Besuch'].astype(str).str.replace(',', '.'), errors='coerce')

# Drop rows with NaN in the columns we want to plot
plot_df = df.dropna(subset=['Seiten_pro_Besuch', 'Zeit_im_Besuch_sec'])

# Plotting
plt.figure(figsize=(10, 6))
sns.scatterplot(data=plot_df, x='Seiten_pro_Besuch', y='Zeit_im_Besuch_sec', alpha=0.6)

plt.title('Relationship between Pages per Visit and Time Spent', fontsize=14)
plt.xlabel('Pages per Visit (Seiten_pro_Besuch)', fontsize=12)
plt.ylabel('Time in Visit (Seconds)', fontsize=12)
plt.grid(True, linestyle='--', alpha=0.7)
plt.tight_layout()

# Save the plot using forward slashes for cross-platform compatibility
output_path = os.path.join(output_dir, '1_scatter_plot_seiten_vs_zeit.png')
plt.savefig(output_path)
plt.show()

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Use raw strings for file paths to avoid escape sequence errors (like \32830 being interpreted as octal)
file_path = r'./src/resources/data/study/32830/dataset.csv'

# Load the dataset
try:
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')
except FileNotFoundError:
    # Fallback for different environments if the relative path structure varies
    file_path = 'dataset.csv'
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')

# Data Cleaning
# 1. Remove the summary row (usually the first row in this dataset format)
# We check if the first row contains 'Summe' or similar in the 'Datum' column
if 'Summe' in str(df.iloc[0]['Datum']):
    df = df.drop(index=0)

# 2. Drop columns that are entirely empty or 'Unnamed'
df = df.loc[:, ~df.columns.str.contains('^Unnamed')]

# 3. Convert 'Datum' to datetime
df['Datum'] = pd.to_datetime(df['Datum'], format='%d.%m.%Y', errors='coerce')

# 4. Clean 'Bounce_Rate' - remove '%' and convert to float
if df['Bounce_Rate'].dtype == object:
    df['Bounce_Rate'] = df['Bounce_Rate'].str.replace('%', '').str.replace(',', '.').astype(float)

# Sort by date to ensure the bar chart follows chronological order
df = df.sort_values('Datum')

# Plotting
plt.figure(figsize=(12, 6))
# Format dates for better display on X-axis
df['Date_Str'] = df['Datum'].dt.strftime('%d.%m')

sns.barplot(data=df, x='Date_Str', y='Bounce_Rate', color='skyblue')

plt.title('Daily Bounce Rate Variation (September 2024)', fontsize=14)
plt.xlabel('Day of Month', fontsize=12)
plt.ylabel('Bounce Rate (%)', fontsize=12)
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.tight_layout()

# Ensure output directory exists
output_dir = r'./src/resources/output/32830'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

output_path = os.path.join(output_dir, '2_bounce_rate_variation.png')
plt.savefig(output_path)
print(f"Visualization saved to {output_path}")
