import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

def load_data(file_path):
    # Common encodings for European CSV files
    encodings = ['utf-8', 'latin1', 'iso-8859-1', 'cp1252']
    for enc in encodings:
        try:
            return pd.read_csv(file_path, sep=';', encoding=enc)
        except (UnicodeDecodeError, pd.errors.ParserError):
            continue
    # Fallback to default if all else fails
    return pd.read_csv(file_path, sep=';')

try:
    data_path = './src/resources/data/study/13979/dataset.csv'
    output_dir = './src/resources/output/13979/run_4'
    os.makedirs(output_dir, exist_ok=True)
    
    df = load_data(data_path)
    
    # Clean string columns
    for c in df.columns:
        if df[c].dtype == 'object':
            df[c] = df[c].astype(str).str.strip()
            
    # Convert time and value to numeric
    if 'time' in df.columns:
        df['time'] = pd.to_numeric(df['time'], errors='coerce')
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')
        
    # Check for required columns
    required_cols = ['2_variable_attribute_label', 'time', 'value']
    if not all(col in df.columns for col in required_cols):

    # Filter for 'Insgesamt' (Total)
    # Using case-insensitive match just in case
    mask_total = df['2_variable_attribute_label'].str.contains('Insgesamt', case=False, na=False)
    df_total = df.loc[mask_total, ['time', 'value']].copy()
    
    # Drop NaNs and group by year to ensure unique points
    df_total = df_total.dropna(subset=['time', 'value'])

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style='whitegrid')

input_path = r'./src/resources/data/study\13979\dataset.csv'
output_path = r'./src/resources/output\13979\run_4/1_baufertigstellungen_raumzahl_verteilung.png'

# The error 'utf-8' codec can't decode byte 0xfc (ü) suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# common for German datasets (e.g., Sünching, Baufertigstellungen).
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(input_path, sep=';', encoding='latin-1')
    except Exception as e:
        print(f"Failed to load with latin-1: {e}")
        raise
except Exception as e:
    print(f"An unexpected error occurred: {e}")
    raise

# Data Cleaning and Preprocessing
for c in ['time', 'value']:
    if c in df.columns:
        df[c] = pd.to_numeric(df[c], errors='coerce')

missing_value_count = int(df['value'].isna().sum()) if 'value' in df.columns else 0
missing_attr2_code_count = int(df['2_variable_attribute_code'].isna().sum()) if '2_variable_attribute_code' in df.columns else 0

# Filter for the requested time range 2014-2024
if 'time' in df.columns:
    df = df[df['time'].between(2014, 2024, inclusive='both')]

# Exclude 'Insgesamt' category
if '2_variable_attribute_label' in df.columns:
    df = df[df['2_variable_attribute_label'].astype(str).str.lower() != 'insgesamt']

# Drop rows with missing values for the visualization
if 'value' in df.columns:
    df = df[df['value'].notna()]

# Check for required columns
required_cols = {'2_variable_attribute_code', '2_variable_attribute_label', 'value'}
if not required_cols.issubset(df.columns):
    fig, ax = plt.subplots(figsize=(10, 6), dpi=150)
    ax.axis('off')
    ax.text(0.5, 0.5, 'Erforderliche Spalten fehlen. Keine Visualisierung möglich.', ha='center', va='center', fontsize=12)
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
else:
    # Grouping and sorting
    grouped = df.groupby(['2_variable_attribute_code', '2_variable_attribute_label'], dropna=False)['value'].sum().reset_index()

    def order_key(code):
        s = str(code)
        if 'RAUM-' in s:
            tail = s.split('-')[-1]
            if 'UM' in tail: # Handle "7 oder mehr" or similar
                return 7.5
            try:
                return float(tail)
            except:
                return 999.0
        return 999.0

    grouped['order'] = grouped['2_variable_attribute_code'].apply(order_key)
    grouped = grouped.sort_values(['order', '2_variable_attribute_label']).reset_index(drop=True)
    
    total_sum = float(grouped['value'].sum()) if not grouped.empty else 0.0
    grouped['pct'] = np.where(total_sum > 0, (grouped['value'] / total_sum) * 100.0, 0.0)

    # Visualization
    fig, ax = plt.subplots(figsize=(11, 7), dpi=150)
    if grouped.empty:
        ax.axis('off')

    else:
        palette = sns.color_palette('viridis', n_colors=len(grouped))
        sns.barplot(data=grouped, x='2_variable_attribute_label', y='value', palette=palette, ax=ax)
        
        ax.set_xlabel('Wohnungsgröße nach Raumzahl', fontsize=10)
        ax.set_ylabel('Fertiggestellte Wohnungen (Summe 2014–2024)', fontsize=10)
        title = f'Verteilung der Wohnungsgrößen bei Baufertigstellungen (2014–2024)\nOhne Kategorie "Insgesamt"; Gesamt: {int(total_sum)} Wohnungen'
        ax.set_title(title, fontsize=12, fontweight='bold', pad=20)
        
        ax.grid(axis='y', linestyle='--', alpha=0.4)
        
        # Add labels on top of bars
        max_val = grouped['value'].max()
        ax.set_ylim(0, max(max_val * 1.2, 1))
        for i, row in grouped.iterrows():
            ax.text(i, row['value'] + (max_val * 0.01), f"{int(row['value'])}\n({row['pct']:.1f}%)", 
                    ha='center', va='bottom', fontsize=9)
        
        plt.setp(ax.get_xticklabels(), rotation=15, ha='right')

    plt.subplots_adjust(bottom=0.2)
    footnote = f'Datenquelle: Statistik der Baufertigstellungen. Zeitraum: 2014–2024. Fehlende Werte: value={missing_value_count}.'
    fig.text(0.5, 0.02, footnote, ha='center', va='center', fontsize=8, alpha=0.7)
    
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Set visual style
sns.set_theme(style="whitegrid")

input_path = './src/resources/data/study/13979/dataset.csv'
output_path = './src/resources/output/13979/run_4/2_wohnungs_mix_raumzahl_gestapelt.png'

# Ensure output directory exists
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xfc' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German data with umlauts like 'ü')
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Data Cleaning
df['value'] = pd.to_numeric(df['value'], errors='coerce')
df['time'] = pd.to_numeric(df['time'], errors='coerce')

label_col = '2_variable_attribute_label'
df[label_col] = df[label_col].astype(str).str.strip()

# Filter out 'Insgesamt' and handle missing values
mask_valid = (df[label_col].str.lower() != 'insgesamt') & \
             (df[label_col] != 'nan') & \
             (df['time'].notna()) & \
             (df['value'].notna())

df_valid = df.loc[mask_valid, ['time', label_col, 'value']].copy()

# Custom sorting for the labels (e.g., "1 Raum", "2 Räume", ..., "7 oder mehr Räume")
def sort_key(label):
    digits = ''.join(filter(str.isdigit, label))
    if not digits:
        return 999
    return int(digits)

unique_labels = sorted(df_valid[label_col].unique(), key=sort_key)

# Pivot data for stacked bar chart
pivot = df_valid.pivot_table(index='time', columns=label_col, values='value', aggfunc='sum', fill_value=0)
pivot = pivot.reindex(columns=unique_labels)

# Plotting
fig, ax = plt.subplots(figsize=(12, 7))
pivot.plot(kind='bar', stacked=True, ax=ax, color=sns.color_palette('viridis', n_colors=len(unique_labels)), width=0.8)

# Formatting
ax.set_xlabel('Jahr', fontsize=12)
ax.set_ylabel('Anzahl fertiggestellter Wohnungen', fontsize=12)

years = pivot.index.astype(int)
title_range = f"{years.min()}–{years.max()}" if not years.empty else ""
ax.set_title(f'Entwicklung des Wohnungsmixes nach Raumzahl ({title_range})', fontsize=14, pad=20)

ax.legend(title='Raumzahl', bbox_to_anchor=(1.02, 1), loc='upper left')
plt.xticks(rotation=45)
plt.tight_layout()

# Save output
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close()
