import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv(r'./src/resources/data/study\104336\dataset.csv', sep=',', encoding='utf-8')
df.columns = [c.strip() for c in df.columns]
df = df.replace({'': np.nan, ' ': np.nan})
if 'F_rderung_in_2014' in df.columns:
    df['F_rderung_in_2014'] = df['F_rderung_in_2014'].astype(str).str.strip()
num_cols = [c for c in ['Bund_in_Euro', 'Land_in_Euro', 'Gesamt_in_Euro'] if c in df.columns]
for col in num_cols:
    s = df[col].astype(str)
    s = s.str.replace(',', '', regex=False)
    s = s.str.replace('.', '', regex=False)
    s = s.str.replace(r'[^0-9]', '', regex=True)
    s = s.replace('', np.nan)
    df[col] = pd.to_numeric(s, errors='coerce')
missing_pct = df.isna().mean(numeric_only=False) * 100
num_df = df[num_cols].copy()
desc = num_df.describe(include='all')
corr = num_df.corr(method='pearson', min_periods=1)
if 'F_rderung_in_2014' in df.columns:
    mask = ~df['F_rderung_in_2014'].str.contains('gesamt', case=False, na=False)
else:
    mask = pd.Series([True] * len(df))
df_viz = df.loc[mask].copy()
for col in ['Bund_in_Euro', 'Land_in_Euro']:
    if col in df_viz.columns:
        df_viz[col] = df_viz[col].fillna(0)
else:
    pass
df_viz['sum_BL'] = df_viz.get('Bund_in_Euro', 0) + df_viz.get('Land_in_Euro', 0)
df_viz['bund_share'] = np.where(df_viz['sum_BL'] > 0, df_viz.get('Bund_in_Euro', 0) / df_viz['sum_BL'], 0)
df_viz['land_share'] = np.where(df_viz['sum_BL'] > 0, df_viz.get('Land_in_Euro', 0) / df_viz['sum_BL'], 0)
sns.set_theme(style='whitegrid')
fig, ax = plt.subplots(figsize=(12, 6))
x = np.arange(len(df_viz))
width = 0.6
bund_pct = (df_viz['bund_share'] * 100).to_numpy()
land_pct = (df_viz['land_share'] * 100).to_numpy()
ax.bar(x, bund_pct, width=width, label='Bund', color='#1f77b4')
ax.bar(x, land_pct, bottom=bund_pct, width=width, label='Land', color='#ff7f0e')
labels = df_viz['F_rderung_in_2014'].astype(str).tolist() if 'F_rderung_in_2014' in df_viz.columns else [str(i) for i in range(len(df_viz))]
ax.set_xticks(x)
ax.set_xticklabels(labels, rotation=30, ha='right')
ax.set_ylim(0, 100)
ax.set_ylabel('Anteil in Prozent')
ax.set_xlabel('Einrichtung')
ax.set_title('Anteile von Bund und Land an der Förderung 2014 je Einrichtung (ohne Gesamtzeile)')
ax.legend(title='Finanzierungsquelle', loc='upper right')
for i, (b, l) in enumerate(zip(bund_pct, land_pct)):
    if b >= 8:
        ax.text(x[i], b / 2, f'{b:.0f}%', ha='center', va='center', color='white', fontsize=9)
    if l >= 8:
        ax.text(x[i], b + l / 2, f'{l:.0f}%', ha='center', va='center', color='white', fontsize=9)
plt.tight_layout()
plt.savefig(r'./src/resources/output\104336/1_anteile_bund_land_2014.png', dpi=300)
plt.close()

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

expected_cols = ['F_rderung_in_2014', 'Bund_in_Euro', 'Land_in_Euro', 'Gesamt_in_Euro']

for c in ['Bund_in_Euro', 'Land_in_Euro', 'Gesamt_in_Euro']:
