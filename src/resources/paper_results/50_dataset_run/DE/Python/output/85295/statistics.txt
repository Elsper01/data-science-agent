import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

try:
    input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '85295', 'dataset.csv')
    output_dir = os.path.join('.', 'src', 'resources', 'output', '85295')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    df.columns = [c.strip() for c in df.columns]
    for col in df.select_dtypes(include=['object']).columns:
        df[col] = df[col].astype(str).str.strip()
        df[col] = df[col].replace({'': np.nan, 'nan': np.nan, 'NaN': np.nan, 'None': np.nan, 'NULL': np.nan})
    dtypes_summary = pd.DataFrame({'spalte': df.columns, 'dtype': df.dtypes.astype(str).values})
    dtypes_summary.to_csv(os.path.join(output_dir, 'eda_dtypen.csv'), index=False, encoding='utf-8')
    missing_share = df.isna().mean().sort_values(ascending=False).rename('anteil_fehlend')
    missing_share.to_csv(os.path.join(output_dir, 'eda_anteil_fehlende_werte.csv'), encoding='utf-8')
    num_cols = df.select_dtypes(include=[np.number]).columns.tolist()
    if len(num_cols) > 0:
        desc = df[num_cols].describe().T
        desc.to_csv(os.path.join(output_dir, 'eda_numerische_kennwerte.csv'), encoding='utf-8')
        if len(num_cols) > 1:
            corr = df[num_cols].corr(method='pearson')
            corr.to_csv(os.path.join(output_dir, 'eda_korrelationen_pearson.csv'), encoding='utf-8')
    obj_cols = df.select_dtypes(include=['object']).columns.tolist()
    if len(obj_cols) > 0:
        cat_card = pd.Series({c: df[c].nunique(dropna=True) for c in obj_cols}).sort_values(ascending=False).rename('anzahl_eindeutige_werte')
        cat_card.to_csv(os.path.join(output_dir, 'eda_kategoriale_kardinalitaet.csv'), encoding='utf-8')
    required_cols = ['Gemeinde', 'Objektnummer']
    for rc in required_cols:
        if rc not in df.columns:
            raise KeyError(f"Erforderliche Spalte fehlt: {rc}")
    df['Objektnummer'] = pd.to_numeric(df['Objektnummer'], errors='coerce')
    gemeinde = df['Gemeinde'].copy()
    gemeinde = gemeinde.fillna('Unbekannt')
    tmp = pd.DataFrame({'Gemeinde': gemeinde, 'Objektnummer': df['Objektnummer']})
    counts = tmp.groupby('Gemeinde', dropna=False)['Objektnummer'].nunique().sort_values(ascending=False)
    sns.set_theme(style='whitegrid')
    n_cat = len(counts)
    fig_width = max(12, min(40, 0.6 * n_cat))
    fig, ax = plt.subplots(figsize=(fig_width, 8))
    palette = sns.color_palette('viridis', n_colors=n_cat)
    sns.barplot(x=counts.index, y=counts.values, palette=palette, ax=ax)
    ax.set_title('Verteilung der Kulturdenkmale nach Gemeinden im Kreis Stormarn (Stand Frühjahr 2021)', fontsize=14)
    ax.set_xlabel('Gemeinde', fontsize=12)
    ax.set_ylabel('Anzahl Kulturdenkmale (eindeutige Objektnummern)', fontsize=12)
    ax.set_xticklabels(ax.get_xticklabels(), rotation=70, ha='right')
    if n_cat <= 30:
        for i, v in enumerate(counts.values):
            ax.text(i, v + max(counts.values)*0.01, str(int(v)), ha='center', va='bottom', fontsize=9, rotation=0)
    plt.tight_layout()
    out_file = os.path.join(output_dir, '0_verteilung_kulturdenkmale_nach_gemeinden.png')
    fig.savefig(out_file, dpi=200)
    plt.close(fig)
except Exception as e:
    print(str(e))

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

try:
    df = pd.read_csv(r'./src/resources/data/study\85295\dataset.csv', sep=',', encoding='utf-8')
except Exception as e:
    print(str(e))
    raise SystemExit(1)

if 'Beschreibung' not in df.columns:
    print("Spalte 'Beschreibung' nicht gefunden.")
    raise SystemExit(1)

_ = df.isna().mean()
num_cols = df.select_dtypes(include=[np.number]).columns.tolist()
_ = df[num_cols].describe().T if len(num_cols) > 0 else None
_ = df[num_cols].corr(numeric_only=True) if len(num_cols) > 1 else None

def _normalize_desc(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    s = ' '.join(s.split())
    return s

beschr = df['Beschreibung'].map(_normalize_desc)
missing_mask = beschr.isna() | (beschr == '')
valid_mask = ~missing_mask
beschr_lower = beschr.str.lower()
legacy_mask = valid_mask & (beschr_lower == 'alteintragung (aktualisierung vorgesehen)')
valid_total = int(valid_mask.sum())
legacy_count = int(legacy_mask.sum())
modern_count = int(max(valid_total - legacy_count, 0))
missing_count = int(missing_mask.sum())

if valid_total == 0:
    plot_df = pd.DataFrame({
        'Kategorie': ['Alteintragung (Aktualisierung vorgesehen)', 'Andere Beschreibungen'],
        'Anzahl': [0, 0],
        'Prozent': [0.0, 0.0]
    })
else:
    plot_df = pd.DataFrame({
        'Kategorie': ['Alteintragung (Aktualisierung vorgesehen)', 'Andere Beschreibungen'],
        'Anzahl': [legacy_count, modern_count]
    })
    plot_df['Prozent'] = plot_df['Anzahl'] / valid_total

sns.set_theme(style='whitegrid')
colors = ['#d62728', '#1f77b4']
fig, ax = plt.subplots(figsize=(10, 6))
sns.barplot(data=plot_df, x='Kategorie', y='Anzahl', palette=colors, ax=ax)
ax.set_xlabel('Kategorie')
ax.set_ylabel('Anzahl Datensätze')
ax.set_title('Alteintragungen vs. modernisierte Beschreibungen in der Denkmalliste Stormarn')
max_y = plot_df['Anzahl'].max() if len(plot_df) > 0 else 0
ax.set_ylim(0, max(1, max_y * 1.15))
for i, row in plot_df.iterrows():
    val = int(row['Anzahl'])
    pct = float(row['Prozent']) if valid_total > 0 else 0.0
    ax.text(i, val + max(0.01 * max(1, max_y), 0.2), f"{val}\n({pct*100:.1f}%)", ha='center', va='bottom', fontsize=11)

caption = f"Datensätze gesamt: {len(df)} | gültige Beschreibungen: {valid_total} | fehlend/leer: {missing_count} ({(missing_count/len(df)*100 if len(df)>0 else 0):.1f}%)"
fig.text(0.5, 0.01, caption, ha='center', fontsize=10)
plt.tight_layout(rect=[0, 0.03, 1, 1])

out_path = r'./src/resources/output\85295/2_beschreibung_modernisierung.png'
try:
    fig.savefig(out_path, dpi=150)
except Exception as e:
    print(str(e))
    raise SystemExit(1)
plt.close(fig)
