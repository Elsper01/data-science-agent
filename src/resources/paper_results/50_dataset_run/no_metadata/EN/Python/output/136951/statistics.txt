import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xfc' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German datasets with umlauts like 'ü').
file_path = './src/resources/data/study/136951/dataset.csv'

try:
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    # Fallback to latin-1 if utf-8 fails
    df = pd.read_csv(file_path, sep=',', encoding='latin-1')

# Data cleaning and parsing
def parse_result(x):
    if pd.isna(x):
        return np.nan, False
    s = str(x).strip()
    # Handle common German decimal comma and comparison operators
    censored = s.startswith('<') or s.startswith('≤')
    s_clean = s.replace('<', '').replace('≤', '').replace('>', '').replace(',', '.').strip()
    try:
        val = float(s_clean)
    except Exception:
        val = np.nan
    return val, censored

if 'Ergebnis' in df.columns:
    parsed = df['Ergebnis'].apply(parse_result)
    df['Ergebnis_num'] = parsed.apply(lambda t: t[0])
    df['Ergebnis_zensiert'] = parsed.apply(lambda t: t[1])

# Count occurrences of each Pruefmerkmal
if 'Pruefmerkmal' in df.columns:
    counts = df['Pruefmerkmal'].astype(str).value_counts()
    # Sort by index to keep order consistent
    counts = counts.sort_index()
else:
    counts = pd.Series(dtype=int)

# Extract metadata for title
wasserwerk = df['Entnahmestelle'].dropna().unique()[0] if 'Entnahmestelle' in df.columns and not df['Entnahmestelle'].dropna().empty else "Großensee"
sample_id = str(df['Hauptprobenummer'].iloc[0]) if 'Hauptprobenummer' in df.columns and len(df) > 0 else ""
date_str = df['Probenahme_Beginn_Datum'].dropna().unique()[0] if 'Probenahme_Beginn_Datum' in df.columns and not df['Probenahme_Beginn_Datum'].dropna().empty else ""

# Visualization
sns.set_theme(style='whitegrid', font_scale=1.1)
plt.figure(figsize=(10, 6))

if not counts.empty:
    ax = sns.barplot(x=counts.index, y=counts.values, palette='viridis')
    
    # Add labels on top of bars
    for i, v in enumerate(counts.values):
        ax.text(i, v + (max(counts.values) * 0.02), str(int(v)), ha='center', va='bottom', fontweight='bold')

    ax.set_xlabel('Prüfmerkmal (Parameter)')
    ax.set_ylabel('Anzahl der Datensätze')
    
    # Construct Title
    title = f'Gemessene Parameter für WW {wasserwerk}'
    if sample_id or date_str:
        subtitle = f"Probe: {sample_id} | Datum: {date_str}".strip(" | ")
        plt.title(f"{title}\n({subtitle})", fontsize=14)
    else:
        plt.title(title, fontsize=14)

    # Adjust Y-axis to give space for labels
    plt.ylim(0, max(counts.values) * 1.2)
    plt.xticks(rotation=15, ha='right')
else:
    plt.text(0.5, 0.5, "Keine Daten für Prüfmerkmale gefunden", ha='center')

plt.tight_layout()

# Save output
output_path = './src/resources/output/136951/0_pruefmerkmal_counts.png'
os.makedirs(os.path.dirname(output_path), exist_ok=True)
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close()

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Set theme
sns.set_theme(style="whitegrid")

in_path = os.path.join('.', 'src', 'resources', 'data', 'study', '136951', 'dataset.csv')
out_dir = os.path.join('.', 'src', 'resources', 'output', '136951')
os.makedirs(out_dir, exist_ok=True)

# Fix for UnicodeDecodeError: Try 'latin-1' or 'iso-8859-1' if 'utf-8' fails
try:
    df = pd.read_csv(in_path, sep=',', encoding='utf-8', dtype=str)
except UnicodeDecodeError:
    df = pd.read_csv(in_path, sep=',', encoding='latin-1', dtype=str)
except Exception as e:
    print(f"Error loading CSV: {e}")
    raise

# Clean whitespace
for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].str.strip()

# Helper to convert strings to numbers (handling German comma and comparison operators)
def to_num(x):
    if pd.isna(x):
        return np.nan
    s = str(x).replace(',', '.')
    s = s.replace('<', '').replace('>', '').strip()
    try:
        return float(s)
    except:
        return np.nan

# Extract Messzeit
messzeit_df = df[df['Pruefmerkmal'].str.lower() == 'messzeit'] if 'Pruefmerkmal' in df.columns else pd.DataFrame()
messzeit_value = None
messzeit_unit = ''
if not messzeit_df.empty:
    mz_row = messzeit_df.iloc[0]
    messzeit_value = to_num(mz_row.get('Ergebnis', np.nan))
    messzeit_unit = mz_row.get('Einheit', '') if not pd.isna(mz_row.get('Einheit', '')) else ''

# EDA Metadata
n_rows = len(df)
unique_pruef = df['Pruefmerkmal'].nunique(dropna=True) if 'Pruefmerkmal' in df.columns else 0
missing_ratio = df.isna().mean().round(4)
missing_text_lines = []
missing_cols = missing_ratio[missing_ratio > 0]
if len(missing_cols) == 0:
    missing_text_lines.append("Keine fehlenden Werte")
else:
    for k, v in missing_cols.sort_values(ascending=False).items()[:5]: # Top 5 missing
        missing_text_lines.append(f"{k}: {int(v*100)}%")

# Sample Metadata for Header
probe = df.get('Bezeichnung', pd.Series(['N/A'])).iloc[0]
site = df.get('Entnahmestelle', pd.Series(['N/A'])).iloc[0]
num_id = df.get('Hauptprobenummer', pd.Series(['N/A'])).iloc[0]
date_beg = df.get('Probenahme_Beginn_Datum', pd.Series([''])).iloc[0]

# Visualization
fig = plt.figure(figsize=(10, 5))
axs = fig.add_gridspec(1, 2, width_ratios=[2, 3], wspace=0.1)
ax0 = fig.add_subplot(axs[0])
ax1 = fig.add_subplot(axs[1])

for ax in [ax0, ax1]:
    ax.set_axis_off()

h_title = "Messzeit der Radon-LSC-Analyse"
sub_title = f"{site} | Probe: {probe} | ID: {num_id} | Datum: {date_beg}"
fig.suptitle(h_title, fontsize=16, y=0.98, fontweight='bold')
fig.text(0.5, 0.92, sub_title, ha='center', va='center', fontsize=10, color='#555555')

# Indicator Box
if messzeit_value is not None and not np.isnan(messzeit_value):
    ax0.add_patch(plt.Rectangle((0.1, 0.1), 0.8, 0.8, color='#f0f7ff', transform=ax0.transAxes, zorder=-1))
    ax0.text(0.5, 0.65, "Messzeit", ha='center', va='center', fontsize=14, color='#333333')
    val_str = f"{int(messzeit_value) if float(messzeit_value).is_integer() else messzeit_value:g}"
    ax0.text(0.5, 0.40, val_str, ha='center', va='center', fontsize=44, fontweight='bold', color='#1f77b4')
    ax0.text(0.5, 0.20, messzeit_unit if messzeit_unit else "Sekunden", ha='center', va='center', fontsize=16, color='#1f77b4')
else:
    ax0.add_patch(plt.Rectangle((0.1, 0.1), 0.8, 0.8, color='#fff4f0', transform=ax0.transAxes, zorder=-1))
    ax0.text(0.5, 0.5, "Messzeit nicht\nverfügbar", ha='center', va='center', fontsize=14, color='#cc3300')

# EDA Text Block
eda_text = [
    f"Anzahl Datensätze: {n_rows}",

eda_text.extend([f"- {t}" for t in missing_text_lines])

ax1.text(0.05, 0.90, "Datensatz-Übersicht", ha='left', va='top', fontsize=12, fontweight='bold', color='#333333')

plt.tight_layout(rect=[0, 0, 1, 0.90])
plt.savefig(out_path, dpi=200, bbox_inches='tight')
plt.close(fig)
