import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

try:
    # Use raw strings (r'') for Windows paths to avoid escape sequence errors like \139 or \n
    input_path = os.path.normpath(r'./src/resources/data/study/139606/dataset.csv')
    output_dir = os.path.normpath(r'./src/resources/output/139606/run_3')
    
    os.makedirs(output_dir, exist_ok=True)
    
    # Load dataset
    df = pd.read_csv(input_path, sep=',', encoding='utf-8', low_memory=False)
    
    if 'Ergebnis' not in df.columns:
        print('Fehler: Die Spalte "Ergebnis" ist nicht im Datensatz vorhanden.')
        sys.exit(1)
    
    # Data Cleaning and Classification
    df_eda = df.copy()
    
    # Clean the 'Ergebnis' column for classification
    ergebnis_raw = df_eda['Ergebnis'].astype(str).str.strip()
    
    # Define masks for classification
    # 1. Non-detectable (n.n.)
    nn_mask = ergebnis_raw.str.lower().isin(['n.n.', 'n.n', 'nn', 'nicht nachweisbar'])
    
    # 2. Detectable (Numerical)
    # Remove spaces, handle commas, and remove comparison operators
    ergebnis_numeric_str = ergebnis_raw.str.replace(' ', '', regex=False)
    ergebnis_numeric_str = ergebnis_numeric_str.str.replace('<', '', regex=False).str.replace('>', '', regex=False)
    ergebnis_numeric_str = ergebnis_numeric_str.str.replace(',', '.', regex=False)
    
    # Try converting to numeric
    numeric_vals = pd.to_numeric(ergebnis_numeric_str, errors='coerce')
    detect_mask = numeric_vals.notna() & (~nn_mask)
    
    # Assign categories
    status = np.where(nn_mask, 'Nicht nachweisbar (n.n.)', 
             np.where(detect_mask, 'Nachweisbar (Zahl)', 'Fehlend/Sonstige'))
    df_eda['Nachweis_Status'] = status
    
    # Prepare plotting data
    counts = df_eda['Nachweis_Status'].value_counts()
    order = ['Nicht nachweisbar (n.n.)', 'Nachweisbar (Zahl)', 'Fehlend/Sonstige']
    # Filter order to only include categories present in data
    order = [o for o in order if o in counts.index]
    
    df_plot = counts.reindex(order).fillna(0).reset_index()
    df_plot.columns = ['Status', 'Anzahl']
    total = df_plot['Anzahl'].sum()
    df_plot['Anteil'] = (df_plot['Anzahl'] / total) * 100 if total > 0 else 0
    
    # Visualization
    sns.set_theme(style='whitegrid')
    fig, ax = plt.subplots(figsize=(10, 6))
    
    palette_map = {
        'Nicht nachweisbar (n.n.)': '#2E7D32', # Green
        'Nachweisbar (Zahl)': '#C62828',       # Red
        'Fehlend/Sonstige': '#9E9E9E'          # Grey
    }
    
    sns.barplot(data=df_plot, x='Status', y='Anzahl', order=order, 
                palette=[palette_map[s] for s in order], ax=ax)
    
    # Add labels on top of bars
    ymax = df_plot['Anzahl'].max() * 1.15 if not df_plot.empty and df_plot['Anzahl'].max() > 0 else 10
    ax.set_ylim(0, ymax)
    
    for i, row in df_plot.iterrows():
        ax.text(i, row['Anzahl'] + (ymax * 0.01), 
                f"{int(row['Anzahl'])}\n({row['Anteil']:.1f}%)", 
                ha='center', va='bottom', fontweight='bold')
    
    ax.set_title(f'Verteilung der Nachweisbarkeit radiologischer Ergebnisse (n = {int(total)})', fontsize=14, pad=20)
    ax.set_xlabel('Ergebnis-Kategorie', fontsize=12)
    ax.set_ylabel('Anzahl der Messungen', fontsize=12)
    
    plt.tight_layout()
    
    output_path = os.path.join(output_dir, '0_verteilung_nachweisbarkeit.png')
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
    
    print(f"Erfolgreich gespeichert unter: {output_path}")

except Exception as e:
    print(f'Fehler bei der Verarbeitung: {e}')
    sys.exit(1)

df = df.applymap(lambda x: np.nan if isinstance(x, str) and x.strip() == '' else x)

def parse_ergebnis(v):
    if pd.isna(v):
        return np.nan
    s = str(v).strip()
    sl = s.lower()
    if any(t in sl for t in ['n.n.', 'n.n', 'nn', '<', 'nachweis']):
        return np.nan
    s = s.replace(' ', '')
    s = s.replace(',', '.')
    try:
        return float(s)
    except Exception:
        return np.nan

df['Ergebnis_num'] = df['Ergebnis'].apply(parse_ergebnis)
df['Nachweis'] = df['Ergebnis_num'].notna()

import pandas as pd

df = pd.read_csv(file_path, sep=',', encoding='utf-8')
for c in df.select_dtypes(include=['object', 'string', 'category']).columns:
    df[c] = df[c].astype(str).str.strip()
missing_prop = df.isna().mean()
bezeichnung = None
if 'Bezeichnung' in df.columns and df['Bezeichnung'].notna().any():
    bezeichnung = df['Bezeichnung'].dropna().astype(str).str.strip().unique()[0]
else:
    bezeichnung = ''
if 'Einheit' not in df.columns or 'Ergebnis' not in df.columns or 'Pruefmerkmal' not in df.columns:
    fig, ax = plt.subplots(figsize=(8, 3))
    ax.axis('off')
    ax.text(0.5, 0.5, 'Erforderliche Spalten fehlen: Einheit/Ergebnis/Pruefmerkmal', ha='center', va='center', fontsize=12)
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()

    df['Ergebnis_num'] = pd.to_numeric(erg_str, errors='coerce')
    mask_unit = df['Einheit_clean'].str.lower().eq('bq/kg(fm)')
    mask_not_messzeit = ~df['Pruefmerkmal'].astype(str).str.contains('messzeit', case=False, na=False)
