import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Set theme for visualization
sns.set_theme(style="whitegrid")

# Fix path issues: Use forward slashes for cross-platform compatibility and raw strings or escaped backslashes
input_path = './src/resources/data/study/110165/dataset.csv'
output_dir = os.path.normpath('./src/resources/output/110165')
os.makedirs(output_dir, exist_ok=True)

try:
    # Read the dataset
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except Exception as e:
    # Fallback for different encodings if utf-8 fails
    try:
        df = pd.read_csv(input_path, sep=',', encoding='latin1')
    except Exception as e2:
        raise SystemExit(f"Failed to load data: {e2}")

# Basic EDA for missingness
eda_missing = df.isna().mean().sort_values(ascending=False)
eda_missing.to_csv(os.path.join(output_dir, 'eda_missingness.csv'), header=['missing_ratio'])

# Check for required columns
kreis_col = 'Kreis' if 'Kreis' in df.columns else None
obj_col = 'Objektnummer' if 'Objektnummer' in df.columns else None

if kreis_col is None or obj_col is None:
    # Try to find case-insensitive matches if exact match fails
    cols_lower = {c.lower(): c for c in df.columns}
    kreis_col = cols_lower.get('kreis')
    obj_col = cols_lower.get('objektnummer')

if kreis_col is None or obj_col is None:
    raise SystemExit(f'Required columns missing. Found: {df.columns.tolist()}')

# Data Cleaning
# 1. Clean Kreis names (remove extra whitespace, handle null-like strings)
df[kreis_col] = df[kreis_col].astype(str).str.strip()
null_values = ['nan', 'NaN', 'None', 'none', 'NULL', 'null', 'N/A', 'n/a', '']
df.loc[df[kreis_col].isin(null_values), kreis_col] = np.nan

# 2. Ensure Objektnummer is treated as a unique identifier (count unique occurrences)
# Drop rows where either Kreis or Objektnummer is missing for this specific analysis
clean_df = df.dropna(subset=[kreis_col, obj_col]).copy()

# Group and count unique monuments per district
counts_df = (
    clean_df.groupby(kreis_col)[obj_col]
    .nunique()
    .reset_index()
    .rename(columns={obj_col: 'AnzahlDenkmale', kreis_col: 'Kreis'})
    .sort_values('AnzahlDenkmale', ascending=False)
)

# Save the processed counts
counts_df.to_csv(os.path.join(output_dir, 'eda_kreis_counts.csv'), index=False)

# Visualization
plt.close('all')
if not counts_df.empty:
    num_kreise = len(counts_df)
    # Dynamic height based on number of categories
    fig_height = max(6, 0.5 * num_kreise)
    fig, ax = plt.subplots(figsize=(12, fig_height))

    # Create horizontal bar chart
    sns.barplot(
        data=counts_df,
        y='Kreis',
        x='AnzahlDenkmale',
        palette='viridis',
        ax=ax,
        edgecolor='0.2'
    )

    # Add value labels to the end of each bar
    max_val = counts_df['AnzahlDenkmale'].max()
    for i, p in enumerate(ax.patches):
        width = p.get_width()
        ax.text(
            width + (max_val * 0.01), 
            p.get_y() + p.get_height()/2, 
            f'{int(width)}', 
            va='center', 
            fontsize=10
        )

    ax.set_title('Verteilung der Kulturdenkmale nach Kreis in Schleswig-Holstein', fontsize=14, pad=20)
    ax.set_xlabel('Anzahl der Kulturdenkmale (Eindeutige Objektnummern)', fontsize=11)
    ax.set_ylabel('Kreis / Bezirk', fontsize=11)
    
    # Adjust layout and save
    plt.tight_layout()
    plot_path = os.path.join(output_dir, '0_verteilung_kulturdenkmale_nach_kreis.png')
    fig.savefig(plot_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
else:
    print("No data available to plot.")

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Define paths
input_path = os.path.normpath(r'./src/resources/data/study/110165/dataset.csv')
output_dir = os.path.normpath(r'./src/resources/output/110165')
output_path = os.path.join(output_dir, '1_kulturdenkmaltyp_haeufigkeit.png')

# The error 'utf-8' codec can't decode byte 0xfc suggests the file is likely encoded in ISO-8859-1 (Latin-1) 
# or cp1252, which is common for German datasets containing umlauts (ü = 0xfc).
try:
    # Attempting with 'latin-1' which handles the 0xfc byte correctly
    df = pd.read_csv(input_path, sep=',', encoding='latin-1', low_memory=False)
except Exception as e:
    # Fallback to cp1252 if latin-1 fails
    try:
        df = pd.read_csv(input_path, sep=',', encoding='cp1252', low_memory=False)
    except Exception as e2:
        print(f"Failed to read CSV: {e2}")
        raise SystemExit(1)

n_rows = int(df.shape[0])

if 'Kulturdenkmaltyp' not in df.columns:
    print("Die erforderliche Spalte 'Kulturdenkmaltyp' fehlt im Datensatz.")
    raise SystemExit(1)

# Data Cleaning and Aggregation
s = df['Kulturdenkmaltyp'].copy()
s = s.apply(lambda x: x.strip() if isinstance(x, str) else x)
counts = s.value_counts(dropna=True)
valid_total = int(counts.sum())
missing_pct_typ = float(s.isna().mean() * 100.0)

freq_df = (
    counts.rename_axis('Kulturdenkmaltyp')
    .reset_index(name='Anzahl')
    .sort_values('Anzahl', ascending=False)
)

if valid_total > 0:
    freq_df['Anteil_%'] = (freq_df['Anzahl'] / valid_total * 100.0).round(2)
else:
    freq_df['Anteil_%'] = 0.0

# Visualization
sns.set_theme(style='whitegrid', context='talk')
plt.rcParams['figure.dpi'] = 120
plt.rcParams['savefig.dpi'] = 200

# Dynamic height based on number of categories
fig_height = max(5, 0.7 * len(freq_df))
fig, ax = plt.subplots(figsize=(12, fig_height))

# Create barplot
sns.barplot(
    data=freq_df,
    y='Kulturdenkmaltyp',
    x='Anzahl',
    palette='viridis',
    ax=ax
)

# Add labels to bars
for i, (anz, pct) in enumerate(zip(freq_df['Anzahl'], freq_df['Anteil_%'])):
    ax.text(
        x=anz + (freq_df['Anzahl'].max() * 0.01),
        y=i,
        s=f"{int(anz):,} ({pct:.1f}%)",
        va='center',
        ha='left',
        fontsize=11,
        color='#333333'
    )

# Titles and Labels
title = 'Häufigkeit der Typen von Kulturdenkmalen in Schleswig-Holstein'
subtitle = f"Gesamtanzahl: {n_rows:,} | Fehlende Werte: {missing_pct_typ:.1f}%"
fig.suptitle(title, x=0.05, y=0.98, ha='left', fontsize=18, fontweight='bold')
ax.set_title(subtitle, loc='left', fontsize=12, pad=15, color='#555555')

ax.set_xlabel('Anzahl der Einträge', fontsize=12)
ax.set_ylabel('Kulturdenkmaltyp', fontsize=12)
ax.xaxis.set_major_locator(plt.MaxNLocator(integer=True))
ax.spines['right'].set_visible(False)
ax.spines['top'].set_visible(False)

plt.tight_layout(rect=[0, 0, 1, 0.95])

# Save output
try:
    os.makedirs(output_dir, exist_ok=True)
    plt.savefig(output_path, bbox_inches='tight')
    plt.close(fig)
    print(f"Successfully saved visualization to {output_path}")
except Exception as e:
    print(f"Error saving plot: {e}")
    raise SystemExit(1)

import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# File paths
file_path = os.path.normpath('./src/resources/data/study/110165/dataset.csv')
output_dir = os.path.normpath('./src/resources/output/110165')
os.makedirs(output_dir, exist_ok=True)

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xfc' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German datasets with umlauts like 'ü').
try:
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    # Fallback to latin-1 if utf-8 fails
    df = pd.read_csv(file_path, sep=',', encoding='latin-1')

# Column name handling (handling potential encoding artifacts in column names)
# The goal mentions 'Begr_ndung', which is likely 'Begründung'
target_col = 'Begr_ndung'
if target_col not in df.columns:
    # Try to find a similar column name if exact match fails
    potential_cols = [c for c in df.columns if 'Begr' in c]
    if potential_cols:
        target_col = potential_cols[0]
    else:
        raise KeyError(f"Column '{target_col}' not found in dataset. Available columns: {list(df.columns)}")

# Data Cleaning
# Replace common null-like strings with actual NaN
null_values = [None, np.nan, '', ' ', 'nan', 'NaN', 'NULL', 'null', 'None']
clean_series = df[target_col].replace(null_values, np.nan)

# Calculate missing ratio for the rationale
missing_share = float(clean_series.isna().mean())

# Process the justifications (they are often comma-separated strings)
# e.g., "Geschichtlich, Städtebaulich" -> ["Geschichtlich", "Städtebaulich"]
terms = clean_series.dropna().astype(str).str.split(',')
exploded = terms.explode().str.strip()
exploded = exploded[exploded.notna() & (exploded != '')]
counts = exploded.value_counts()

# Visualization
sns.set_theme(style='whitegrid')

if counts.empty:
    fig, ax = plt.subplots(figsize=(10, 6))
    ax.axis('off')
    ax.text(0.5, 0.5, f"Keine Daten für '{target_col}' verfügbar", ha='center', va='center', fontsize=14)
else:
    top_n = min(10, len(counts))
    top_counts = counts.head(top_n).sort_values(ascending=True)
    data_plot = top_counts.reset_index()
    data_plot.columns = ['Begründung', 'Anzahl']

    fig, ax = plt.subplots(figsize=(12, 8))
    # Use a color palette
    palette = sns.color_palette('viridis', n_colors=len(data_plot))
    sns.barplot(data=data_plot, x='Anzahl', y='Begründung', ax=ax, palette=palette, edgecolor='black')

    ax.set_title('Primary Legal Justifications for Monument Protection (Top 10)', fontsize=14, pad=15)
    ax.set_xlabel('Frequency of Mention', fontsize=12)
    ax.set_ylabel('Justification (Begründung)', fontsize=12)

    # Add value labels to the bars
    max_val = data_plot['Anzahl'].max() if not data_plot.empty else 0
    for i, v in enumerate(data_plot['Anzahl']):
        ax.text(v + max_val*0.01, i, f"{int(v):,}", va='center', ha='left', fontsize=10)

    # Add footnote about missing data
    fig.text(0.1, 0.01, f"Note: {missing_share:.1%} of entries have missing justifications.", 
             ha='left', va='bottom', fontsize=10, style='italic')

plt.tight_layout()
fig_path = os.path.join(output_dir, '2_begruendung_top10.png')
plt.savefig(fig_path, dpi=300, bbox_inches='tight')
plt.close()

print(f"Visualization saved to {fig_path}")
