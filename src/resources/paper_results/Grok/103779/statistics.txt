# Da es Probleme mit Paket-Abhängigkeiten (rlang, vroom) und Schreibrechten (LOCK-Files) gab,
# nutzen wir eine robustere Methode zum Laden der Daten und vermeiden Neuinstallationen,
# falls die Pakete bereits vorhanden sind. Wir nutzen base R für das Einlesen, falls readr scheitert.

# Funktion zum sicheren Laden von Paketen
load_pkg <- function(pkg) {
  if (!require(pkg, character.only = TRUE)) {
    # Falls Installation fehlschlägt (z.B. wegen Lock-Files), wird das Skript nicht hart abgebrochen
    try(install.packages(pkg, repos = 'https://cran.rstudio.com/'), silent = TRUE)
    library(pkg, character.only = TRUE)
  }
}

load_pkg('dplyr')
load_pkg('ggplot2')
load_pkg('scales')

# Daten einlesen mit Base R als Fallback für readr/vroom Probleme
file_path <- './src/resources/data/study/103779/dataset.csv'
df <- tryCatch({
  if (require('readr')) {
    readr::read_csv(file_path, locale = readr::locale(encoding = 'UTF-8'))
  } else {
    read.csv(file_path, fileEncoding = 'UTF-8', stringsAsFactors = FALSE)
  }
}, error = function(e) {
  read.csv(file_path, stringsAsFactors = FALSE)
})

# Verzeichnis erstellen
dir.create('./src/resources/output/103779', recursive = TRUE, showWarnings = FALSE)

# Daten transformieren
# Wir gruppieren in "Alteintragung" und "Spezifische Beschreibung"
beschreibung_counts <- df %>%
  mutate(
    Kategorie = ifelse(Beschreibung == 'Alteintragung (Aktualisierung vorgesehen)', 
                       'Alteintragung\n(Aktualisierung vorgesehen)', 
                       'Spezifische Beschreibung\n(Aktuell/Individuell)')
  ) %>%
  group_by(Kategorie) %>%
  summarise(Anzahl = n(), .groups = 'drop') %>%
  mutate(Anteil = Anzahl / sum(Anzahl))

# Visualisierung erstellen
p <- ggplot(beschreibung_counts, aes(x = Kategorie, y = Anzahl, fill = Kategorie)) +
  geom_col(alpha = 0.85, width = 0.6, color = "white", size = 0.5) +
  geom_text(aes(label = paste0(Anzahl, "\n(", percent(Anteil, accuracy = 0.1), ")")), 
            vjust = -0.2, size = 4.5, fontface = "bold") +
  scale_fill_manual(values = c(
    'Alteintragung\n(Aktualisierung vorgesehen)' = '#E74C3C', 
    'Spezifische Beschreibung\n(Aktuell/Individuell)' = '#2ECC71'
  )) +
  scale_y_continuous(expand = expansion(mult = c(0, 0.15)), labels = comma) +
  labs(
    title = 'Datenqualität: Anteil der Alteintragungen',
    subtitle = 'Vergleich von pauschalen Alteintragungen zu spezifischen Denkmalbeschreibungen',
    x = NULL,
    y = 'Anzahl der Denkmale',
    caption = 'Datenquelle: Denkmalliste'
  ) +
  theme_minimal(base_size = 14) +
  theme(
    plot.title = element_text(face = 'bold', size = 16),
    plot.subtitle = element_text(size = 12, color = "gray30"),
    axis.text.x = element_text(face = "bold"),
    legend.position = 'none',
    panel.grid.major.x = element_blank(),
    panel.grid.minor = element_blank()
  )

# Speichern des Diagramms
ggsave('./src/resources/output/103779/2_anteil_alteintragungen_beschreibung.png', 
       plot = p, width = 10, height = 7, dpi = 300, bg = 'white')
