# Funktion zur Klassifizierung
classify_email <- function(lp, name) {
  if (is.na(lp)) return("Fehlend")
  
  # 1. Check auf funktionale Keywords
  if (any(str_detect(lp, role_keywords))) return("Funktional")
  
  # 2. Check auf Namensbestandteile (einfacherer Ansatz ohne komplexe Regex-Escapes)
  if (!is.na(name)) {
    name_parts <- str_split(tolower(name), "[\\s\\-]+")[[1]]
    name_parts <- name_parts[nchar(name_parts) > 2] # Nur signifikante Namensbestandteile
    if (length(name_parts) > 0) {
      # Prüfe ob einer der Namensbestandteile im Local Part vorkommt
      for (part in name_parts) {
        if (str_detect(lp, fixed(part))) return("Personengebunden")
      }
    }
  }
  
  # 3. Check auf typische Namensmuster (Punkt oder Bindestrich zwischen zwei Wörtern)
  if (str_detect(lp, "^[a-zäöüß]{2,}[._-][a-zäöüß]{2,}")) return("Personengebunden")
  
  # Standardfall
  return("Funktional") # Wenn kein Name erkannt wird, oft Funktionspostfach
}
