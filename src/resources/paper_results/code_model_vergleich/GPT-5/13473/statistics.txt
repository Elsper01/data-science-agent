import os

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

def load_data(file_path):
    """Attempts to load CSV with multiple encodings to handle special characters like 'ö'."""
    encodings = ['utf-8', 'latin1', 'iso-8859-1', 'cp1252']
    for enc in encodings:
        try:
            return pd.read_csv(file_path, sep=";", encoding=enc)
        except (UnicodeDecodeError, pd.errors.ParserError):
            continue
    # Fallback to default if all fail
    return pd.read_csv(file_path, sep=";")

try:
    sns.set_theme(style="whitegrid", context="talk")
    
    data_path_raw = r"./src/resources/data/study/13473/dataset.csv"
    out_dir_raw = r"./src/resources/output/13473"
    data_path = data_path_raw.replace("\\", "/")
    out_dir = out_dir_raw.replace("\\", "/")
    os.makedirs(out_dir, exist_ok=True)

    # Load data with encoding fix
    df = load_data(data_path)

    # Data Cleaning
    if "time" in df.columns:
        df["time"] = pd.to_datetime(df["time"], errors="coerce")

        df["value"] = pd.to_numeric(df["value"], errors="coerce")

    # EDA: Missing Values
    mv = df.isna().sum()

        "spalte": mv.index,

    }).sort_values("anteil_missing", ascending=False)
    eda_missing.to_csv(os.path.join(out_dir, "eda_missing_values.csv"), index=False, encoding="utf-8")

    # Filter for the specific visualization goal: Hof and Gender (männlich/weiblich)
    # We look for 'Hof' in '1_variable_attribute_label' and gender in '2_variable_attribute_label'
    plot_df = df.copy()

    sub = plot_df[mask_hof & mask_gender].copy()

    # If specific filters return nothing, try a broader search for the gender labels

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import StrMethodFormatter
import os

# Set visual style
sns.set_theme(style="whitegrid", context="talk")

# File paths
input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '13473', 'dataset.csv')

os.makedirs(output_dir, exist_ok=True)

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xf6' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German datasets with umlauts like 'ö')

    df = pd.read_csv(input_path, sep=';', encoding='utf-8', na_values=na_vals, dtype=str)

    df = pd.read_csv(input_path, sep=';', encoding='latin-1', na_values=na_vals, dtype=str)

# Clean string columns
for c in df.columns:

# Convert value to numeric

# Filter for 'Insgesamt' (Total) to compare levels

# Group by regional classification level

# Check if values are identical across levels

# Formatting

import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

# Set visual style
sns.set_theme(style="whitegrid", context="talk")

# File paths
input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '13473', 'dataset.csv')
output_dir = os.path.join('.', 'src', 'resources', 'output', '13473')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '2_bevoelkerung_nach_geschlecht.png')

# Fix for UnicodeDecodeError: Try 'iso-8859-1' (Latin-1) which is common for German datasets
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1')

# Data Cleaning
for c in df.select_dtypes(include=['object']).columns:
    df[c] = df[c].astype(str).str.strip().replace({'nan': np.nan})

if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for gender-related data
# Based on the goal, we want 'männlich', 'weiblich', 'Insgesamt'
# Usually these are in '2_variable_attribute_label'
subset = df.copy()
if '2_variable_label' in subset.columns:
    # Filter for rows where the variable is 'Geschlecht'
    subset = subset[subset['2_variable_label'].str.contains('Geschlecht', case=False, na=False)]

# Group by attribute label to get the counts
group = subset.groupby('2_variable_attribute_label', dropna=False, as_index=False)['value'].sum()
group = group.dropna(subset=['value'])

# Ensure we have the categories we need
# If 'Insgesamt' is missing, calculate it
labels = group['2_variable_attribute_label'].str.lower().tolist()
if not any('insgesamt' in l for l in labels):
    total_val = group['value'].sum()
    group = pd.concat([group, pd.DataFrame({'2_variable_attribute_label': ['Insgesamt'], 'value': [total_val]})], ignore_index=True)

# Calculate percentages for labels
total_row = group[group['2_variable_attribute_label'].str.lower().str.contains('insgesamt', na=False)]
val_total = total_row['value'].iloc[0] if not total_row.empty else group['value'].sum()
group['anteil'] = (group['value'] / val_total) * 100

# Define order and colors
order = ['Insgesamt', 'männlich', 'weiblich']
# Check which labels actually exist in the data to avoid errors
existing_order = [o for o in order if any(group['2_variable_attribute_label'].str.contains(o, case=False, na=False))]
# If the specific German labels aren't found, just use what's there
if not existing_order:
    existing_order = group.sort_values('value', ascending=False)['2_variable_attribute_label'].tolist()

color_palette = {
    'männlich': '#4C78A8',
    'weiblich': '#F58518',
    'Insgesamt': '#7F7F7F'
}
# Map colors to the existing labels
colors = []
for lab in existing_order:
    found = False
    for key, color in color_palette.items():
        if key.lower() in str(lab).lower():
            colors.append(color)
            found = True
            break
    if not found:
        colors.append('#54A24B')

# Plotting
plt.figure(figsize=(12, 7))
ax = sns.barplot(data=group, y='2_variable_attribute_label', x='value', order=existing_order, palette=colors)

# Formatting
ax.set_xlabel('Bevölkerung (Anzahl Personen)')
ax.set_ylabel('')
ax.set_title('Bevölkerungszusammensetzung nach Geschlecht', loc='left', fontsize=18, pad=20)

# Thousand separator for x-axis
fmt_thousand = FuncFormatter(lambda x, p: f"{int(x):,}".replace(',', '.'))
ax.xaxis.set_major_formatter(fmt_thousand)

# Add value labels on bars
for i, label in enumerate(existing_order):
    row = group[group['2_variable_attribute_label'] == label]
    if not row.empty:
        val = row['value'].values[0]
        pct = row['anteil'].values[0]
        txt = f"{int(val):,}".replace(',', '.') + f" ({pct:.1f}%)"
        ax.text(val + (val_total * 0.01), i, txt, va='center', fontsize=12, fontweight='bold')

sns.despine()
plt.tight_layout()

# Save
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close()
