import os
import pandas as pd

import seaborn as sns

# Pfade definieren

    os.makedirs(output_dir, exist_ok=True)
except Exception as e:

    sys.exit(1)

# Daten laden
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except Exception as e:
    print(f"Fehler beim Laden der Daten: {e}")
    sys.exit(1)

if df.empty:
    print('Leerer Datensatz.')
    sys.exit(0)

# Datentypen konvertieren
for col in ['start_measure', 'end_measure']:
    if col in df.columns:
        df[col] = pd.to_datetime(df[col], errors='coerce')

if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Validierte Daten filtern (falls Spalte vorhanden)
if 'validated' in df.columns:
    df['validated'] = pd.to_numeric(df['validated'], errors='coerce')
    df = df[(df['validated'] == 1) | (df['validated'].isna())]

# Prüfung auf notwendige Spalten
if 'start_measure' not in df.columns or 'value' not in df.columns:
    print('Erforderliche Spalten (start_measure, value) fehlen.')
    sys.exit(1)

# Daten für EDA vorbereiten
eda_df = df.dropna(subset=['start_measure', 'value']).copy()

# Interpolation fix: Setze DatetimeIndex für zeitgewichtete Interpolation
eda_df = eda_df.set_index('start_measure')

eda_df['value_interp'] = eda_df['value'].interpolate(method='time')
# Index zurücksetzen für Plotting
eda_df = eda_df.reset_index()

eda_df['roll_mean_6h'] = eda_df['value_interp'].rolling(window=6, min_periods=1, center=False).mean()

# Statistiken für Titel/Labels
val_series = eda_df['value'].dropna()

unit_val = df['unit'].dropna().iloc[0] if 'unit' in df.columns and df['unit'].notna().any() else 'µSv/h'

import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

try:
    input_path = os.path.normpath('./src/resources/data/study\\16317\\dataset.csv')
    output_dir = os.path.normpath('./src/resources/output\\16317')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    if 'start_measure' in df.columns:
        df['start_measure'] = pd.to_datetime(df['start_measure'], errors='coerce')
    if 'end_measure' in df.columns:
        df['end_measure'] = pd.to_datetime(df['end_measure'], errors='coerce')
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')
    missing_counts = df.isna().sum()
    missing_ratios = df.isna().mean()
    missing_df = pd.DataFrame({'spalte': missing_counts.index, 'anzahl_fehlend': missing_counts.values, 'anteil_fehlend': missing_ratios.values})
    missing_df.to_csv(os.path.join(output_dir, 'eda_missing_values.csv'), index=False, encoding='utf-8')
    numeric_df = df.select_dtypes(include=[np.number])
    if not numeric_df.empty:
        desc = numeric_df.describe(percentiles=[0.25, 0.5, 0.75]).T
        desc.to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'), encoding='utf-8')
        constant_cols = [c for c in numeric_df.columns if numeric_df[c].nunique(dropna=True) <= 1]
        corr_df = numeric_df.drop(columns=constant_cols, errors='ignore')
        if corr_df.shape[1] >= 2:
            corr = corr_df.corr(numeric_only=True)
            corr.to_csv(os.path.join(output_dir, 'eda_numeric_correlations.csv'), encoding='utf-8')
        else:
            pd.DataFrame().to_csv(os.path.join(output_dir, 'eda_numeric_correlations.csv'), encoding='utf-8')
    s = df['value'] if 'value' in df.columns else pd.Series(dtype=float)
    s = pd.to_numeric(s, errors='coerce').dropna()
    if s.empty:
        raise ValueError('Spalte "value" ist nicht vorhanden oder enthält keine numerischen Werte.')
    n = len(s)
    iqr = np.subtract(*np.percentile(s, [75, 25]))
    if iqr > 0:
        bin_width = 2 * iqr * (n ** (-1 / 3))
        bins = int(np.ceil((s.max() - s.min()) / bin_width)) if bin_width > 0 else int(np.sqrt(n))
    else:
        bins = int(np.sqrt(n))
    bins = max(5, min(bins, 30))
    sns.set_theme(style='whitegrid', context='talk')
    fig, ax = plt.subplots(figsize=(10, 6))
    color = '#3b82f6'
    sns.histplot(s, bins=bins, kde=True, color=color, edgecolor='white', linewidth=0.6, ax=ax)
    mean_val = float(s.mean())
    ax.axvline(mean_val, color='#ef4444', linestyle='--', linewidth=1.5, label=f'Mittelwert = {mean_val:.3f} µSv/h')
    ax.set_title('Häufigkeitsverteilung der Gamma-Ortsdosisleistung (µSv/h) – Station Kornbach, Jan 2026')
    ax.set_xlabel('Gamma-Ortsdosisleistung (µSv/h)')
    ax.set_ylabel('Anzahl der Stunden')
    ax.legend(loc='upper right', frameon=True)
    plt.tight_layout()
    plot_path = os.path.normpath(os.path.join(output_dir, '1_hist_value.png'))
    plt.savefig(plot_path, dpi=200)
    plt.close(fig)
except Exception as e:
    print(str(e))

import numpy as np

import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid")

# Load dataset
file_path = './src/resources/data/study/16317/dataset.csv'
df = pd.read_csv(file_path, sep=',', encoding='utf-8')

# Basic EDA exports
for col in ['start_measure', 'end_measure']:
    if col in df.columns:
        df[col] = pd.to_datetime(df[col], errors='coerce')

numeric_cols = df.select_dtypes(include=[np.number]).columns.tolist()
missing_ratio = df.isna().mean().rename('missing_ratio').to_frame()
missing_ratio.to_csv(os.path.join(output_dir, 'eda_fehlende_werte.csv'), encoding='utf-8')

if len(numeric_cols) > 0:
    desc_numeric = df[numeric_cols].describe().T
    desc_numeric.to_csv(os.path.join(output_dir, 'eda_numerik_uebersicht.csv'), encoding='utf-8')

col_info = pd.DataFrame({
    'dtype': df.dtypes.astype(str),
    'n_unique': df.nunique(dropna=False),
    'n_missing': df.isna().sum()
})
col_info.to_csv(os.path.join(output_dir, 'eda_spalteninfo.csv'), encoding='utf-8')

# Geometry parsing
if 'geom' not in df.columns:
    raise ValueError('Spalte geom nicht gefunden.')

def parse_xy(wkt):
    if pd.isna(wkt):
        return np.nan, np.nan
    nums = re.findall(r'[-+]?\d*\.?\d+(?:[eE][-+]?\d+)?', str(wkt))
    if len(nums) >= 2:
        return float(nums[0]), float(nums[1])
    return np.nan, np.nan

xy = df['geom'].apply(parse_xy)
x = pd.Series([t[0] for t in xy], index=df.index)
y = pd.Series([t[1] for t in xy], index=df.index)

# Heuristic to handle potential lon/lat swaps (assuming Central Europe/Germany context)
mask1 = (x.between(5, 16)) & (y.between(47, 56))
mask2 = (y.between(5, 16)) & (x.between(47, 56))
lon = np.where(mask1, x, np.where(mask2, y, x))
lat = np.where(mask1, y, np.where(mask2, x, y))

gdf_points = gpd.GeoDataFrame(df.copy(), geometry=gpd.points_from_xy(lon, lat), crs='EPSG:4326')
valid_mask = (~gdf_points.geometry.is_empty) & gdf_points.geometry.x.between(-180, 180) & gdf_points.geometry.y.between(-90, 90)
gdf_points = gdf_points[valid_mask]

if gdf_points.empty:
    raise ValueError('Keine gültigen Geometrien nach Parsing gefunden.')

# Deduplicate to get the station location
cols_keep = [c for c in ['id', 'kenn', 'name'] if c in gdf_points.columns]
gdf_station = gdf_points.drop_duplicates(subset=cols_keep if cols_keep else None).head(1)

# FIX: Avoid deprecated gpd.datasets. Use a direct URL or a simple background if needed.
# Since we only need a context map, we can use a low-res world map from a stable URL.
world_url = "https://raw.githubusercontent.com/holtzy/The-Python-Graph-Gallery/master/static/data/world.geojson"
try:
    world = gpd.read_file(world_url)
except Exception:
    # Fallback: Create an empty GeoDataFrame if download fails, so the plot still works
    world = gpd.GeoDataFrame(geometry=gpd.GeoSeries(), crs='EPSG:4326')

station = gdf_station.iloc[0]
station_lon = float(station.geometry.x)
station_lat = float(station.geometry.y)

# Zoom window
pad_lon = 1.5
pad_lat = 1.0
minx, maxx = station_lon - pad_lon, station_lon + pad_lon
miny, maxy = station_lat - pad_lat, station_lat + pad_lat

fig, ax = plt.subplots(figsize=(10, 8))

# Plot background
if not world.empty:
    world.plot(ax=ax, color='#f2f2f2', edgecolor='#bdbdbd', linewidth=0.6)

# Plot station
point_color = '#d81b60'
gdf_station.plot(ax=ax, color=point_color, edgecolor='white', markersize=150, zorder=5, label='Messstation')

# Set limits
ax.set_xlim(minx, maxx)
ax.set_ylim(miny, maxy)

# Labeling (Tooltip equivalent in static plot)
label_parts = []
if 'name' in gdf_station.columns and pd.notna(station['name']):
    label_parts.append(str(station['name']))
if 'id' in gdf_station.columns and pd.notna(station['id']):
    label_parts.append(f"ID: {station['id']}")

label_text = " - ".join(label_parts)

if label_text:
    ax.annotate(label_text, xy=(station_lon, station_lat), xytext=(8, 8), 
                textcoords='offset points', fontsize=12, fontweight='bold',
                color=point_color, ha='left', va='bottom',
                bbox=dict(boxstyle='round,pad=0.3', fc='white', ec=point_color, alpha=0.8))

ax.set_title(f'Geografische Lage der Messstation: {label_text if label_text else "Unbekannt"}', fontsize=14, pad=20)
ax.set_xlabel('Längengrad (°)')

ax.grid(True, linestyle='--', linewidth=0.5, alpha=0.7)

plot_path = os.path.join(output_dir, '2_karte_messstation.png')
