# Ensure 'Objektnummer' is treated as a unique identifier
if 'Objektnummer' in df.columns:
    # We keep it as string or numeric to count unique occurrences

    df['Objektnummer'] = range(len(df))

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid")

# The error 'utf-8' codec can't decode byte 0xfc indicates the file is likely encoded in 'latin-1' or 'cp1252'
# (0xfc is 'ü' in Latin-1, common in German datasets).
file_path = r'./src/resources/data/study/110165/dataset.csv'

try:
    # Attempting to read with 'latin-1' which is common for German datasets containing umlauts
    df = pd.read_csv(file_path, sep=',', encoding='latin-1')
except Exception as e:
    # Fallback to 'cp1252' if latin-1 fails
    try:
        df = pd.read_csv(file_path, sep=',', encoding='cp1252')
    except Exception as e2:
        print(f"Failed to read CSV: {e2}")
        raise

# Basic data cleaning and validation
if 'Kulturdenkmaltyp' not in df.columns or 'Objektnummer' not in df.columns:
    print('Required columns missing: Kulturdenkmaltyp and/or Objektnummer.')
    raise SystemExit(1)

# Handle missing values and clean the category column
df['Kulturdenkmaltyp_clean'] = df['Kulturdenkmaltyp'].fillna('Unbekannt').astype(str).str.strip()
pct_missing_type = (df['Kulturdenkmaltyp'].isna().mean() * 100)

# Aggregate data: Count unique Objektnummer per type
counts = (
    df.groupby('Kulturdenkmaltyp_clean')['Objektnummer']
    .nunique()
    .reset_index(name='Anzahl')
    .sort_values('Anzahl', ascending=False)
)

n_total = int(df['Objektnummer'].nunique())

if counts.empty:
    print('No data available for visualization.')
    raise SystemExit(1)

# Visualization setup
ncat = len(counts)
width = max(10, min(2 + 1.2 * ncat, 20))
height = 7

fig, ax = plt.subplots(figsize=(width, height), dpi=120)
sns.barplot(data=counts, x='Kulturdenkmaltyp_clean', y='Anzahl', ax=ax, palette='viridis', edgecolor='0.3')

# Labels and Title
ax.set_xlabel('Kulturdenkmaltyp (Monument Type)', fontsize=11)
ax.set_ylabel('Anzahl der Objekte (Count)', fontsize=11)
ax.set_title(
    f'Häufigste Kulturdenkmaltypen in Schleswig-Holstein\n'
    f'(Gesamt n={n_total}, Fehlende Angaben: {pct_missing_type:.1f}%)',
    fontsize=14, pad=20
)

# Rotate x-labels for readability
plt.xticks(rotation=45, ha='right')

# Add value labels on top of bars
for p in ax.patches:
    h = p.get_height()
    if h > 0:
        ax.annotate(f'{int(h)}', 
                    (p.get_x() + p.get_width() / 2., h), 
                    ha='center', va='bottom', 
                    fontsize=10, xytext=(0, 5), 
                    textcoords='offset points')

# Clean up spines
sns.despine()
plt.tight_layout()

# Save the plot
output_path = r'./src/resources/output/110165/run_4/1_kulturdenkmaltypen_barchart.png'
import os
os.makedirs(os.path.dirname(output_path), exist_ok=True)

    fig.savefig(output_path, bbox_inches='tight')
    print(f"Successfully saved plot to {output_path}")
except Exception as e:
    print(f"Error saving plot: {e}")
    raise

plt.close(fig)

import os
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Paths
input_path = os.path.normpath('./src/resources/data/study/110165/dataset.csv')
output_dir = os.path.normpath('./src/resources/output/110165/run_4')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '2_top10_begruendung.png')

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xfc' 
# suggests the file is likely encoded in 'latin-1' (ISO-8859-1) or 'cp1252', 
# which is common for German datasets containing umlauts (ü = 0xfc).
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8', low_memory=False)
except UnicodeDecodeError:
    # Fallback to latin-1 if utf-8 fails
    df = pd.read_csv(input_path, sep=',', encoding='latin-1', low_memory=False)

# Column name handling
col_name = 'Begr_ndung'
if col_name not in df.columns:
    # Check if the encoding issue affected column names (e.g., 'Begründung')
    # Look for columns that start with 'Begr'
    potential_cols = [c for c in df.columns if c.startswith('Begr')]
    if potential_cols:
        col_name = potential_cols[0]
    else:
        print(f'Fehlende Spalte: {col_name}. Vorhandene Spalten: {df.columns.tolist()}')
        raise SystemExit(1)

# Data Cleaning
total_rows = len(df)
valid_series = df[col_name].dropna()
valid_series = valid_series.astype(str).str.strip()
# Replace multiple spaces with single space
valid_series = valid_series.str.replace(r'\s+', ' ', regex=True)
# Remove empty strings
valid_series = valid_series[valid_series != '']

valid_n = len(valid_series)
missing_n = total_rows - valid_n
missing_pct = (missing_n / total_rows * 100) if total_rows > 0 else 0

# Frequency Analysis
value_counts = valid_series.value_counts()

if value_counts.empty:
    fig, ax = plt.subplots(figsize=(10, 6), dpi=150)
    ax.text(0.5, 0.5, 'Keine gültigen Werte in der Spalte "Begründung" vorhanden.', ha='center', va='center', fontsize=14)
    ax.axis('off')
    plt.savefig(output_path, dpi=150, bbox_inches='tight')
    plt.close()
    raise SystemExit(0)

# Get Top 10
top10 = value_counts.head(10)
# Reverse for horizontal bar chart (highest at top)
top10 = top10.iloc[::-1]
percentages = (top10 / valid_n * 100)

# Visualization
sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(14, 8), dpi=150)

# Use a color palette
colors = sns.color_palette('viridis', n_colors=len(top10))
bars = ax.barh(top10.index, top10.values, color=colors)

# Labels and Title
ax.set_xlabel('Anzahl der Einträge', fontsize=12)
ax.set_ylabel('Begründung (Kriterienkombination)', fontsize=12)

# Add value labels on bars
max_val = top10.max()
for i, (val, pct) in enumerate(zip(top10.values, percentages.values)):
    ax.text(val + (max_val * 0.01), i, f'{int(val)} ({pct:.1f}%)', 
            va='center', ha='left', fontsize=11, fontweight='bold')

title_main = 'Top 10 häufigste Begründungen für den Denkmalschutz'
title_sub = f'Grundlage: {total_rows} Objekte | Gültige Einträge: {valid_n} | Fehlend/Leer: {missing_pct:.1f}%'
plt.title(f'{title_main}\n{title_sub}', fontsize=16, pad=20)

# Adjust layout to prevent clipping
plt.tight_layout()
plt.savefig(output_path, dpi=150, bbox_inches='tight')
plt.close()
