import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os
import json

plt.rcParams['font.family'] = 'DejaVu Sans'

def find_column(df, target_name):
    def norm(s):
        return ''.join(str(s).lower().replace('\n', ' ').split())
    for c in df.columns:
        if norm(c) == norm(target_name):
            return c
    return None

def ensure_dir(path):
    if not os.path.exists(path):
        os.makedirs(path, exist_ok=True)

def load_data():
    data_path = os.path.join('.', 'src', 'resources', 'data', 'study', '103779', 'dataset.csv')
    return pd.read_csv(data_path, sep=',', encoding='utf-8')

def eda_exports(df, out_dir):
    ensure_dir(out_dir)
    overview = {
        'anzahl_zeilen': int(df.shape[0]),
        'anzahl_spalten': int(df.shape[1]),
        'spalten': list(df.columns),
        'dtypen': {c: str(t) for c, t in df.dtypes.items()}
    }
    with open(os.path.join(out_dir, 'eda_overview.json'), 'w', encoding='utf-8') as f:
        json.dump(overview, f, ensure_ascii=False, indent=2)
    miss_count = df.isna().sum()
    miss_pct = df.isna().mean() * 100
    miss_df = (
        pd.DataFrame({'Spalte': df.columns, 'Fehlende_Werte': miss_count.values, 'Anteil_Fehlend_%': miss_pct.round(2).values})
        .sort_values('Anteil_Fehlend_%', ascending=False)
        .reset_index(drop=True)
    )
    miss_df.to_csv(os.path.join(out_dir, 'eda_missing_values.csv'), index=False, encoding='utf-8')
    num_df = df.select_dtypes(include=[np.number])
    if num_df.shape[1] > 0:
        desc = num_df.describe().T
        desc.to_csv(os.path.join(out_dir, 'eda_numeric_summary.csv'), encoding='utf-8')
        if num_df.shape[1] > 1:
            corr = num_df.corr(numeric_only=True)
            corr.to_csv(os.path.join(out_dir, 'eda_numeric_correlation.csv'), encoding='utf-8')

def normalize_kulturdenkmaltyp(s):
    if pd.isna(s):
        return 'Fehlend'
    val = str(s).strip()
    if val == '':
        return 'Fehlend'
    lower = val.lower()
    mapping = {
        'bauliche anlage': 'Bauliche Anlage',
        'sachgesamtheit': 'Sachgesamtheit',
        'gründenkmal': 'Gründenkmal',
        'gruendenkmal': 'Gründenkmal',
        'mehrheit von baulichen anlagen': 'Mehrheit von baulichen Anlagen',
        'mehrheit baulicher anlagen': 'Mehrheit von baulichen Anlagen',
        'mehrheit von baulichen anlage': 'Mehrheit von baulichen Anlagen'
    }
    if lower in mapping:
        return mapping[lower]
    return val.title()

def plot_kulturdenkmaltyp_counts(df, kult_col, out_dir):
    series = df[kult_col].apply(normalize_kulturdenkmaltyp)
    total = int(series.shape[0])
    counts = series.value_counts(dropna=False)
    counts_df = (
        counts.rename_axis('Kulturdenkmaltyp').reset_index(name='Anzahl')
    )
    counts_df['Anteil_%'] = (counts_df['Anzahl'] / total * 100).round(2)
    counts_df.to_csv(os.path.join(out_dir, 'kulturdenkmaltyp_value_counts.csv'), index=False, encoding='utf-8')
    sns.set_theme(style='whitegrid', context='talk')
    n = counts_df.shape[0]
    height = max(4.0, min(0.6 * n + 1.5, 12.0))
    fig, ax = plt.subplots(figsize=(10, height))
    palette = sns.color_palette('viridis', n_colors=n)
    counts_df_sorted = counts_df.sort_values('Anzahl', ascending=True)
    sns.barplot(data=counts_df_sorted, x='Anzahl', y='Kulturdenkmaltyp', palette=palette, ax=ax)
    for i, (an, pct) in enumerate(zip(counts_df_sorted['Anzahl'], counts_df_sorted['Anteil_%'])):
        ax.text(an + max(1, total * 0.005), i, f"{an} ({pct:.1f}%)", va='center', ha='left', fontsize=11)
    ax.set_title('Verteilung der Kulturdenkmaltypen im Kreis Segeberg', fontsize=16, pad=14)
    ax.set_xlabel('Anzahl')
    ax.set_ylabel('Kulturdenkmaltyp')
    ax.margins(x=0.05)
    plt.tight_layout()
    ensure_dir(out_dir)
    out_path = os.path.join(out_dir, '1_kulturdenkmaltyp_haeufigkeit.png')
    plt.savefig(out_path, dpi=200, bbox_inches='tight')
    plt.close(fig)

def main():
    try:
        out_dir = os.path.join('.', 'src', 'resources', 'output', '103779')
        df = load_data()
        df.columns = [str(c).strip() for c in df.columns]
        eda_exports(df, out_dir)
        kult_col = find_column(df, 'Kulturdenkmaltyp')
        if kult_col is None:
            print('Fehler: Die Spalte "Kulturdenkmaltyp" wurde im Datensatz nicht gefunden.')
            return
        plot_kulturdenkmaltyp_counts(df, kult_col, out_dir)
    except Exception as e:
        print(f'Fehler bei der Ausführung: {e}')

if __name__ == '__main__':
    main()

import os
import sys
import warnings
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

warnings.filterwarnings('ignore')

try:
    input_path = os.path.normpath('./src/resources/data/study\\103779\\dataset.csv')
    output_dir = os.path.normpath('./src/resources/output\\103779')
    os.makedirs(output_dir, exist_ok=True)

    df = pd.read_csv(input_path, sep=',', encoding='utf-8')

    for col in df.select_dtypes(include=['object']).columns:
        df[col] = df[col].replace(r'^\s*$', np.nan, regex=True)
        df[col] = df[col].replace({'nan': np.nan, 'NaN': np.nan, 'NONE': np.nan, 'None': np.nan, 'NULL': np.nan, 'null': np.nan})

    total_objects = len(df)

    missing_summary = pd.DataFrame({
        'Spalte': df.columns,
        'Anzahl_Fehlend': [df[c].isna().sum() for c in df.columns],
        'Anteil_Fehlend_%': [round(df[c].isna().mean()*100, 2) for c in df.columns]
    })
    missing_summary = missing_summary.sort_values('Anteil_Fehlend_%', ascending=False)
    missing_summary.to_csv(os.path.join(output_dir, 'eda_missing_values.csv'), index=False, encoding='utf-8')

    num_df = df.select_dtypes(include=[np.number])
    if not num_df.empty:
        desc_num = num_df.describe().T
        desc_num.to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'), encoding='utf-8')
        corr_num = num_df.corr(numeric_only=True)
        corr_num.to_csv(os.path.join(output_dir, 'eda_correlation_numeric.csv'), encoding='utf-8')

    col_reason = 'Begr_ndung'
    if col_reason not in df.columns:
        raise ValueError('Spalte "Begr_ndung" nicht gefunden.')

    reasons_series = df[col_reason].dropna().astype(str).str.split(',')
    exploded = reasons_series.explode().astype(str).str.strip().str.rstrip('.;:')
    exploded = exploded.replace('', np.nan).dropna()
    category_counts = exploded.value_counts().sort_values(ascending=False)

    missing_count = df[col_reason].isna().sum()

    plot_df = category_counts.reset_index()
    if not plot_df.empty:
        plot_df.columns = ['Kategorie', 'Anzahl']
    else:
        plot_df = pd.DataFrame(columns=['Kategorie', 'Anzahl'])
    plot_df = pd.concat([plot_df, pd.DataFrame([{'Kategorie': 'Fehlend', 'Anzahl': missing_count}])], ignore_index=True)

    order = plot_df.sort_values('Anzahl', ascending=False)['Kategorie'].tolist()
    non_missing_cats = [c for c in order if c != 'Fehlend']
    base_palette = sns.color_palette('viridis', n_colors=max(1, len(non_missing_cats)))
    palette_map = {cat: base_palette[i] for i, cat in enumerate(non_missing_cats)}
    palette_map['Fehlend'] = (0.7, 0.7, 0.7)

    sns.set_theme(style='whitegrid', context='talk')
    fig, ax = plt.subplots(figsize=(12, 7))
    ax = sns.barplot(data=plot_df, x='Kategorie', y='Anzahl', order=order, palette=palette_map, edgecolor='black')

    ax.set_title('Rechtliche Gründe für den Denkmalschutz im Kreis Segeberg')
    ax.set_xlabel('Kategorie der Begründung')
    ax.set_ylabel('Anzahl der Denkmale')
    ax.set_ylim(0, max(1, int(plot_df['Anzahl'].max()*1.15)))
    plt.xticks(rotation=20, ha='right')

    for p in ax.patches:
        height = p.get_height()
        if height is None or np.isnan(height):
            continue
        cat = p.get_x() + p.get_width()/2.0
        percent = 0.0 if total_objects == 0 else (height/total_objects)*100.0
        ax.annotate(f"{int(height)}\n{percent:.1f}%", (p.get_x() + p.get_width()/2., height), ha='center', va='bottom', fontsize=10)

    plt.tight_layout()

    output_path = os.path.join(output_dir, '2_begruendung_kategorien.png')
    plt.savefig(output_path, dpi=300)
    plt.close(fig)

    plot_df.sort_values('Anzahl', ascending=False).to_csv(os.path.join(output_dir, 'eda_begruendung_kategorien_counts.csv'), index=False, encoding='utf-8')

except Exception as e:
    sys.stderr.write(str(e))
    sys.exit(1)
