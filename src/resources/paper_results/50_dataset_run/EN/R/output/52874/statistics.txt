library(tidyverse)

dir.create(out_dir, showWarnings = FALSE, recursive = TRUE)

# Read the dataset
# Using semicolon as delimiter based on typical German CSV formats
# Using locale with latin1 or UTF-8 depending on the source, but we'll handle encoding carefully
raw <- readr::read_delim(in_path, delim = ";", show_col_types = FALSE)

# Data Preparation
# The error "Required variables ... not found" suggests encoding issues or exact string mismatch.
# We will use string detection to be more robust against encoding artifacts.

plot_data <- raw %>%
  # Filter using regex to handle potential encoding issues with 'ü' and 'ä'
  filter(str_detect(value_variable_label, "Sch.ler") | 
         str_detect(value_variable_label, "Voll- und teilzeitbesch.ftigte Lehrkr.fte")) %>%
  mutate(
    # Standardize labels for internal logic
    category = case_when(
      str_detect(value_variable_label, "Sch.ler") ~ "Schüler",
      str_detect(value_variable_label, "Voll- und teilzeitbesch.ftigte Lehrkr.fte") ~ "Lehrkräfte",
      TRUE ~ NA_character_
    ),
    # Ensure value is numeric (handling potential comma as decimal separator)
    value = as.numeric(gsub(",", ".", as.character(value))),
    year_label = as.character(time)
  ) %>%
  filter(!is.na(value), !is.na(category)) %>%
  group_by(year_label, category) %>%
  summarise(value = sum(value, na.rm = TRUE), .groups = "drop")

# Pivot wider to calculate scaling factors for the dual axis
wide_data <- plot_data %>%
  tidyr::pivot_wider(names_from = category, values_from = value)

# Check if we have both variables to plot
if ("Schüler" %in% names(wide_data) && "Lehrkräfte" %in% names(wide_data)) {
  
  # Calculate scaling parameters for dual axis
  s_min <- min(wide_data$Schüler, na.rm = TRUE)
  s_max <- max(wide_data$Schüler, na.rm = TRUE)
  l_min <- min(wide_data$Lehrkräfte, na.rm = TRUE)
  l_max <- max(wide_data$Lehrkräfte, na.rm = TRUE)
  
  # Avoid division by zero if range is 0
  l_range <- l_max - l_min
  if(l_range == 0) l_range <- 1
  s_range <- s_max - s_min
  if(s_range == 0) s_range <- 1
  
  # Linear transformation: y_scaled = (value - l_min) * (s_range / l_range) + s_min
  scale_factor <- s_range / l_range
  
  # Prepare data for plotting
  plot_df <- plot_data %>%
    mutate(y_scaled = ifelse(category == "Schüler", 
                             value, 
                             (value - l_min) * scale_factor + s_min))

  # Create the plot
  p <- ggplot(plot_df, aes(x = year_label, y = y_scaled, color = category, group = category)) +
    geom_line(linewidth = 1.2) +
    geom_point(size = 3) +
    scale_y_continuous(
      name = "Anzahl Schüler",
      sec.axis = sec_axis(~ (. - s_min) / scale_factor + l_min, 
                          name = "Anzahl Lehrkräfte",
                          labels = label_number(big.mark = ".", decimal.mark = ",")),
      labels = label_number(big.mark = ".", decimal.mark = ",")
    ) +
    scale_color_manual(
      values = c("Schüler" = "#2C3E50", "Lehrkräfte" = "#E74C3C")
    ) +
    labs(
      title = "Verhältnis von Schülern zu Lehrkräften",
      subtitle = "Entwicklung der Schülerzahlen und des Lehrpersonals (Voll- und Teilzeit)",
      x = "Schuljahr",
      color = "Kategorie",
      caption = "Datenquelle: Statistik der allgemeinbildenden Schulen"
    ) +
    theme_minimal(base_size = 11) +
    theme(
      legend.position = "bottom",
      plot.title = element_text(face = "bold", size = 14),
      axis.title.y.right = element_text(color = "#E74C3C", face = "bold"),
      axis.title.y.left = element_text(color = "#2C3E50", face = "bold"),
      axis.text.x = element_text(angle = 45, hjust = 1),
      panel.grid.minor = element_blank()
    )

  # Save the plot
  ggsave(file.path(out_dir, "schueler_lehrkraefte_verhaeltnis.png"), plot = p, width = 10, height = 6, dpi = 300)
  
} else {
  # Fallback if data is missing - print available labels to help debugging
  available_labels <- unique(raw$value_variable_label)
  cat("Available labels in dataset:\n")
  print(available_labels)
  stop("Required variables not found in dataset. Check encoding or column names.")
}
