import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Set plotting style
plt.switch_backend('Agg')
sns.set_style('whitegrid')

# Define paths
input_path = os.path.normpath('./src/resources/data/study/110165/dataset.csv')
output_dir = os.path.normpath('./src/resources/output/110165/run_1')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '0_verteilung_kulturdenkmale_nach_kreis.png')

# Load data with encoding handling
# The error 'utf-8' codec can't decode byte 0xfc (ü) suggests ISO-8859-1 or cp1252 encoding
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(input_path, sep=',', encoding='iso-8859-1')
    except Exception as e:
        df = pd.read_csv(input_path, sep=',', encoding='cp1252')

# Data Cleaning
try:
    # Clean string columns
    for c in df.select_dtypes(include=['object']).columns:
        df[c] = df[c].astype(str).str.strip()
        df[c] = df[c].replace({'': np.nan, 'nan': np.nan, 'NaN': np.nan, 'None': np.nan})
    
    # Deduplicate if possible
    if 'Objektnummer' in df.columns:
        df = df.drop_duplicates(subset=['Objektnummer'])
    
    # Prepare 'Kreis' data
    target_col = 'Kreis'
    if target_col not in df.columns:
        # Fallback if column name is slightly different or missing
        potential_cols = [c for c in df.columns if 'kreis' in c.lower()]
        if potential_cols:
            target_col = potential_cols[0]
        else:
            df[target_col] = 'Unbekannt/ohne Angabe'

    kreis_series = df[target_col].fillna('Unbekannt/ohne Angabe')
    kreis_series = kreis_series.replace({'nan': 'Unbekannt/ohne Angabe', '': 'Unbekannt/ohne Angabe'})
    
    # Count occurrences
    kreis_counts = kreis_series.value_counts(dropna=False).sort_values(ascending=True)
    
    # Visualization
    fig_height = max(6, 0.4 * len(kreis_counts))
    plt.figure(figsize=(12, fig_height))
    
    # Create horizontal bar chart
    colors = sns.color_palette("viridis", len(kreis_counts))
    bars = plt.barh(kreis_counts.index.astype(str), kreis_counts.values, color=colors)
    
    # Formatting
    plt.xlabel('Anzahl der Kulturdenkmale', fontsize=12)
    plt.ylabel('Kreis (Landkreis/Stadt)', fontsize=12)
    plt.title('Verteilung der Kulturdenkmale nach Kreis in Schleswig-Holstein', fontsize=14, pad=20)
    plt.grid(axis='x', linestyle='--', alpha=0.6)
    
    # Add value labels
    max_val = kreis_counts.values.max() if len(kreis_counts) > 0 else 0
    for rect in bars:
        width = rect.get_width()
        plt.text(width + (max_val * 0.01), rect.get_y() + rect.get_height()/2, 
                 f'{int(width):,}', va='center', ha='left', fontsize=10)
    
    plt.xlim(0, max_val * 1.12)
    plt.tight_layout()
    
    # Save plot
    plt.savefig(output_path, dpi=200, bbox_inches='tight')
    plt.close()
    
except Exception as e:
    print(f"An error occurred during processing: {e}")
    raise

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import os

# Define paths
input_path = './src/resources/data/study/110165/dataset.csv'
output_dir = './src/resources/output/110165/run_1'
plot_path = os.path.join(output_dir, '1_kulturdenkmaltyp_haeufigkeit.png')

if not os.path.exists(output_dir):
    os.makedirs(output_dir, exist_ok=True)

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xfc' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German data with umlauts like 'ü').
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=',', encoding='latin-1')

# Data Cleaning and Preparation
col = 'Kulturdenkmaltyp'
if col not in df.columns:
    # Fallback: check if column names have different casing or spaces
    df.columns = [c.strip() for c in df.columns]
    if col not in df.columns:
        raise ValueError(f"Column '{col}' not found in dataset. Available columns: {df.columns.tolist()}")

# Process the column for visualization
s = df[col].astype(str).str.strip()
s = s.replace({'': 'Unbekannt', 'None': 'Unbekannt', 'nan': 'Unbekannt', 'NaN': 'Unbekannt'})
counts = s.value_counts()

plot_df = pd.DataFrame({
    'Kulturdenkmaltyp': counts.index,
    'Anzahl': counts.values,
    'Anteil': counts.values / len(df)
}).sort_values('Anzahl', ascending=False)

# Visualization
sns.set_theme(style='whitegrid', font_scale=1.1)
cat_count = len(plot_df)
# Dynamic height based on number of categories
height = max(4, min(0.5 * cat_count + 2, 20))
fig, ax = plt.subplots(figsize=(12, height))

# Use a barplot
sns.barplot(
    data=plot_df, 
    x='Anzahl', 
    y='Kulturdenkmaltyp', 
    ax=ax, 
    palette='viridis', 
    hue='Kulturdenkmaltyp', 
    legend=False
)

ax.set_title('Häufigkeit der Kulturdenkmaltypen', fontsize=16, pad=20)
ax.set_xlabel('Anzahl der Objekte', fontsize=12)
ax.set_ylabel('Kulturdenkmaltyp', fontsize=12)

# Add labels to the bars
max_val = plot_df['Anzahl'].max() if not plot_df.empty else 0
for i, (val, pct) in enumerate(zip(plot_df['Anzahl'], plot_df['Anteil'])):
    ax.text(val + (max_val * 0.01), i, f"{int(val)} ({pct:.1%})", va='center', ha='left', fontsize=10)

plt.tight_layout()
fig.savefig(plot_path, dpi=300, bbox_inches='tight')
plt.close(fig)

import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# Set plotting parameters
plt.rcParams["figure.dpi"] = 150
plt.rcParams["savefig.dpi"] = 300
plt.rcParams["font.family"] = "DejaVu Sans"

input_path = './src/resources/data/study/110165/dataset.csv'
output_dir = os.path.normpath('./src/resources/output/110165/run_1')
plot_path = os.path.normpath('./src/resources/output/110165/run_1/2_top10_begruendung.png')
os.makedirs(output_dir, exist_ok=True)

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xfc' 
# usually indicates 'latin-1' or 'iso-8859-1' encoding (0xfc is 'ü' in latin-1).
# We try 'latin-1' which is common for German datasets.
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8', low_memory=False)
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=',', encoding='latin-1', low_memory=False)

# Basic EDA exports
missing_ratio = df.isna().mean().sort_values(ascending=False).to_frame('fehlend_anteil')
missing_ratio.to_csv(os.path.join(output_dir, 'eda_missing_ratio.csv'), encoding='utf-8')

num_df = df.select_dtypes(include=[np.number])
if num_df.shape[1] > 0:
    num_summary = num_df.describe().T
else:
    num_summary = pd.DataFrame()
num_summary.to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'), encoding='utf-8')

# Frequency counts for specific columns
for col in ['Kulturdenkmaltyp', 'Kreis']:
    if col in df.columns:
        vc = df[col].dropna().astype(str).str.strip().replace('', np.nan).dropna().value_counts()
        vc.to_csv(os.path.join(output_dir, f'eda_freq_{col}.csv'), encoding='utf-8')

# Visualization logic for 'Begr_ndung'
just_col = 'Begr_ndung'
if just_col not in df.columns:
    # Fallback check if encoding issues renamed the column slightly
    potential_cols = [c for c in df.columns if 'Begr' in c]
    if potential_cols:
        just_col = potential_cols[0]
    else:
        raise SystemExit(f'Spalte {just_col} nicht gefunden.')

total_records = len(df)
# Clean and split the justifications (often comma-separated)
just_series = df[just_col].dropna().astype(str)
just_exploded = just_series.str.split(',').explode().astype(str).str.strip()
just_exploded = just_exploded.replace({'': np.nan, 'nan': np.nan}).dropna()
# Remove trailing punctuation
just_exploded = just_exploded.str.replace(r'[.;:]+$', '', regex=True).str.strip()
just_counts = just_exploded.value_counts()
just_counts.to_csv(os.path.join(output_dir, 'eda_begruendung_counts_full.csv'), encoding='utf-8')

top_n = 10
top_counts = just_counts.head(top_n)

if top_counts.empty:
    raise SystemExit('Keine gültigen Werte in der Spalte Begründung gefunden.')

# Plotting
sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(12, 8))
order = top_counts.index.tolist()
palette = sns.color_palette('viridis', n_colors=len(order))

sns.barplot(x=top_counts.values, y=order, palette=palette, ax=ax)

ax.set_xlabel('Anzahl Denkmale')
ax.set_ylabel('Begründung (Kategorie)')

missing_begruendung = df[just_col].isna().sum()
missing_pct = 0.0 if total_records == 0 else (missing_begruendung / total_records) * 100

ax.set_title('Top 10 Schutzbegründungen von Kulturdenkmalen\nGrundlage: {} Datensätze; fehlend: {} ({:.1f}%)'.format(
    total_records, missing_begruendung, missing_pct), fontsize=14)

# Add value labels
for i, v in enumerate(top_counts.values):
    pct = (v / total_records) * 100
    ax.text(v + (max(top_counts.values) * 0.01), i, f"{int(v)} ({pct:.1f}%)", va='center', fontsize=11)

plt.tight_layout()
fig.savefig(plot_path, bbox_inches='tight')
plt.close(fig)
