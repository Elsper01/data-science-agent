import pandas as pd

    missing_rates = df.isna().mean()

    n = int(value_series.shape[0])

    std_v = float(value_series.std(ddof=1)) if n > 1 else np.nan
    median_v = float(value_series.median()) if n > 0 else np.nan

    outlier_mask = (value_series < lower) | (value_series > upper) if n > 0 else pd.Series(dtype=bool)
