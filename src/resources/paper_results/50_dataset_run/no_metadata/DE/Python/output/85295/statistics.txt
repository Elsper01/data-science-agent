import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import MaxNLocator

try:
    data_path = os.path.join('.', 'src', 'resources', 'data', 'study', '85295', 'dataset.csv')
    df = pd.read_csv(data_path, sep=',', encoding='utf-8')
except Exception as e:
    print(str(e))
    sys.exit(1)

output_dir = os.path.join('.', 'src', 'resources', 'output', '85295')
try:
    os.makedirs(output_dir, exist_ok=True)
except Exception as e:
    print(str(e))
    sys.exit(1)

try:
    df_columns = {c: c.strip() if isinstance(c, str) else c for c in df.columns}
    df.rename(columns=df_columns, inplace=True)

    missing_share = df.isna().mean().sort_values(ascending=False).rename('anteil_fehlend')
    missing_share.to_frame().reset_index(names='spalte').to_csv(os.path.join(output_dir, 'eda_missing_values.csv'), index=False, encoding='utf-8')

    numeric_cols = df.select_dtypes(include=[np.number]).columns.tolist()
    if len(numeric_cols) > 0:
        df[numeric_cols].describe().T.to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'), encoding='utf-8')
        corr = df[numeric_cols].corr(numeric_only=True)
        corr.to_csv(os.path.join(output_dir, 'eda_numeric_correlation.csv'), encoding='utf-8')
    else:
        pd.DataFrame().to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'), encoding='utf-8')
        pd.DataFrame().to_csv(os.path.join(output_dir, 'eda_numeric_correlation.csv'), encoding='utf-8')

    if 'Kulturdenkmaltyp' not in df.columns:
        raise KeyError("Spalte 'Kulturdenkmaltyp' nicht gefunden")

    s = df['Kulturdenkmaltyp']
    if s.dtype != object:
        s = s.astype('object')
    s = s.apply(lambda x: x.strip() if isinstance(x, str) else x)
    total_non_missing = s.notna().sum()
    counts = s.dropna().value_counts().sort_values(ascending=False)

    pd.DataFrame({
        'Kulturdenkmaltyp': counts.index,
        'anzahl': counts.values,
        'anteil_prozent': (counts.values / total_non_missing * 100).round(2)
    }).to_csv(os.path.join(output_dir, 'eda_kulturdenkmaltyp_counts.csv'), index=False, encoding='utf-8')

    if counts.empty:
        raise ValueError("Keine gültigen Werte in 'Kulturdenkmaltyp' vorhanden")

    sns.set_theme(style='whitegrid', context='talk')
    n = len(counts)
    width = max(8.0, 0.6 * n + 2.0)
    height = 6.0
    fig, ax = plt.subplots(figsize=(width, height))
    palette = sns.color_palette('colorblind', n_colors=n)
    bars = ax.bar(range(n), counts.values, color=palette)
    ax.set_xticks(range(n))
    ax.set_xticklabels(counts.index, rotation=20, ha='right')
    ax.yaxis.set_major_locator(MaxNLocator(integer=True))
    ax.set_xlabel('Kulturdenkmaltyp')
    ax.set_ylabel('Anzahl der Denkmale')
    title = 'Häufigkeit der Kulturdenkmaltypen im Kreis Stormarn'
    ax.set_title(title)

    for i, b in enumerate(bars):
        v = int(counts.values[i])
        p = counts.values[i] / total_non_missing * 100.0
        ax.text(b.get_x() + b.get_width() / 2.0, b.get_height() + max(counts.values) * 0.01, f"{v}\n{p:.1f}%", ha='center', va='bottom', fontsize=10)

    fehlend_anteil = 1.0 - (total_non_missing / len(df) if len(df) > 0 else 0)
    caption = f"Datenbasis: {int(total_non_missing)} gültige Einträge; Fehlend: {fehlend_anteil*100:.1f}%"
    fig.text(0.99, 0.01, caption, ha='right', va='bottom', fontsize=9, color='dimgray')
    fig.tight_layout(rect=[0, 0.03, 1, 1])

    plot_path = os.path.join(output_dir, '1_kulturdenkmaltyp_balkendiagramm.png')
    plt.savefig(plot_path, dpi=200, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
    print(str(e))
    sys.exit(1)

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

plt.rcParams.update({
    'figure.dpi': 100,
    'savefig.dpi': 300,
    'font.family': 'DejaVu Sans'
})

input_path = os.path.normpath('./src/resources/data/study\\85295\\dataset.csv')
output_dir = os.path.normpath('./src/resources/output\\85295')
plot_path = os.path.join(output_dir, '2_schutzgruende_balkendiagramm.png')
os.makedirs(output_dir, exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8', low_memory=False)

_ = df.isna().sum()
_ = df.isna().mean()
num_df = df.select_dtypes(include=[np.number])
_ = num_df.describe().T if not num_df.empty else pd.DataFrame()
_ = num_df.corr(numeric_only=True) if not num_df.empty else pd.DataFrame()

col = 'Begr_ndung'
s = df[col] if col in df.columns else pd.Series(dtype=object)
missing_mask = s.isna() | s.astype(str).str.strip().isin(['', 'nan', 'NaN', 'None', 'N/A'])
missing_count = int(missing_mask.sum())
s_clean = s[~missing_mask]
reasons = s_clean.astype(str).str.split(',').explode().astype(str).str.strip()
reasons = reasons[reasons.str.len() > 0]
reasons = reasons.str.replace(r'[\.;:()\[\]{}]', '', regex=True).str.replace(r'\s+', ' ', regex=True).str.strip()

def normalize_reason(x: str) -> str:
    t = x.lower().strip()
    t_key = (
        t.replace('ä', 'ae')
         .replace('ö', 'oe')
         .replace('ü', 'ue')
         .replace('ß', 'ss')
    )
    mapping = {
        'geschichtlich': 'Geschichtlich',
        'kuenstlerisch': 'Künstlerisch',
        'künstlerisch': 'Künstlerisch',
        'staedtebaulich': 'Städtebaulich',
        'städtebaulich': 'Städtebaulich',
        'kulturlandschaftlich': 'Kulturlandschaftlich',
        'wissenschaftlich': 'Wissenschaftlich',
        'technisch': 'Technisch',
        'volkskundlich': 'Volkskundlich',
        'archaologisch': 'Archäologisch',
        'archaeologisch': 'Archäologisch'
    }
    return mapping.get(t_key, t.capitalize())

if not reasons.empty:
    reasons_norm = reasons.map(normalize_reason)
    counts = reasons_norm.value_counts().sort_values(ascending=False)
else:
    counts = pd.Series(dtype=int)

combined = pd.concat([counts, pd.Series({'Fehlend': missing_count})])
combined = combined.sort_values(ascending=False)

sns.set_theme(style='whitegrid')
fig_height = max(3, 0.5 * len(combined) + 1.5)
fig, ax = plt.subplots(figsize=(10, fig_height))

categories = combined.index.tolist()
base_colors = sns.color_palette('viridis', max(1, len(categories)))
colors = list(base_colors)
if 'Fehlend' in categories:
    fehlend_idx = categories.index('Fehlend')
    colors[fehlend_idx] = (0.62, 0.62, 0.62)

ax.barh(categories, combined.values, color=colors, edgecolor='black', alpha=0.9)
ax.invert_yaxis()
ax.set_xlabel('Anzahl der Denkmale')
ax.set_ylabel('Schutzgründe (Begr_ndung)')
ax.set_title('Häufigste Schutzgründe der Kulturdenkmale und fehlende Angaben')

total_n = len(df)
if total_n > 0:
    for i, v in enumerate(combined.values):
        pct = 100.0 * v / total_n
        ax.text(v + max(0.01 * combined.max(), 0.5), i, f"{int(v)} ({pct:.1f}%)", va='center', ha='left', fontsize=9)

ax.xaxis.set_major_locator(plt.MaxNLocator(integer=True))
plt.tight_layout()
plt.savefig(plot_path, bbox_inches='tight')
plt.close(fig)
