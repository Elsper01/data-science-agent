import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import matplotlib.dates as mdates

data_path = os.path.normpath('./src/resources/data/study\\32830\\dataset.csv')
output_dir = os.path.normpath('./src/resources/output\\32830')
os.makedirs(output_dir, exist_ok=True)

sns.set_theme(style='whitegrid', context='talk')

df = pd.read_csv(data_path, sep=',', encoding='utf-8')

if 'Unnamed__14' in df.columns:
    if df['Unnamed__14'].isna().all() or (df['Unnamed__14'].astype(str).str.strip() == '').all():
        df = df.drop(columns=['Unnamed__14'])

percent_cols = [c for c in ['Anteil_Besuche_mit_Cookies','Anteil_Besucher_mit_Cookies','Bounce_Rate','Conversion_Rate__Last_Ad_'] if c in df.columns]
for c in percent_cols:
    df[c] = pd.to_numeric(df[c].astype(str).str.replace('%','', regex=False).str.replace(',','.', regex=False), errors='coerce')

if 'Besuchsh__ufigkeit' in df.columns:
    df['Besuchsh__ufigkeit'] = pd.to_numeric(df['Besuchsh__ufigkeit'].replace('-', np.nan), errors='coerce')

num_candidates = ['Besuche','Besucher','Seitenaufrufe','Seiten_pro_Besuch','Conversions__Last_Ad_','Conversion_Wert__Last_Ad_']
for c in num_candidates:
    if c in df.columns:
        df[c] = pd.to_numeric(df[c], errors='coerce')

if 'Datum' in df.columns:
    df['Datum'] = pd.to_datetime(df['Datum'], dayfirst=True, errors='coerce')

def parse_time_to_seconds(s):
    if pd.isna(s):
        return np.nan
    try:
        v = str(s).strip()
        if v == '' or v == '-':
            return np.nan
        if v.count(':') == 1:
            v = '00:' + v
        td = pd.to_timedelta(v)
        return float(td.total_seconds())
    except Exception:
        try:
            td = pd.to_timedelta(s)
            return float(td.total_seconds())
        except Exception:
            return np.nan

if 'Zeit_im_Besuch' in df.columns:
    df['Zeit_im_Besuch_Sek'] = df['Zeit_im_Besuch'].apply(parse_time_to_seconds)
if 'Bounce_Zeit' in df.columns:
    df['Bounce_Zeit_Sek'] = df['Bounce_Zeit'].apply(parse_time_to_seconds)

eda_df = df.copy()
if 'Datum' in eda_df.columns:
    eda_daily = eda_df[~eda_df['Datum'].isna()].copy()
else:
    eda_daily = eda_df.copy()

missing_prop = eda_daily.isna().mean().to_frame('Anteil_fehlend').reset_index().rename(columns={'index':'Spalte'})
missing_prop['Anteil_fehlend'] = (missing_prop['Anteil_fehlend'] * 100).round(2)
missing_path = os.path.join(output_dir, 'eda_missing_values_32830.csv')
missing_prop.to_csv(missing_path, index=False, encoding='utf-8')

num_df = eda_daily.select_dtypes(include=[np.number])
if num_df.shape[1] > 0:
    corr = num_df.corr(numeric_only=True)
    corr_path = os.path.join(output_dir, 'eda_correlations_32830.csv')
    corr.to_csv(corr_path, encoding='utf-8')

if {'Zeit_im_Besuch_Sek','Bounce_Zeit_Sek'}.issubset(eda_daily.columns):
    time_stats = pd.DataFrame({
        'Kennzahl':['Mittelwert','Median','Standardabweichung'],
        'Zeit_im_Besuch_Sek':[eda_daily['Zeit_im_Besuch_Sek'].mean(), eda_daily['Zeit_im_Besuch_Sek'].median(), eda_daily['Zeit_im_Besuch_Sek'].std()],
        'Bounce_Zeit_Sek':[eda_daily['Bounce_Zeit_Sek'].mean(), eda_daily['Bounce_Zeit_Sek'].median(), eda_daily['Bounce_Zeit_Sek'].std()]
    })
    time_stats[['Zeit_im_Besuch_Sek','Bounce_Zeit_Sek']] = time_stats[['Zeit_im_Besuch_Sek','Bounce_Zeit_Sek']].round(2)
    stats_path = os.path.join(output_dir, 'eda_time_stats_32830.csv')
    time_stats.to_csv(stats_path, index=False, encoding='utf-8')

plot_df = df.copy()
plot_df = plot_df[~plot_df['Datum'].isna()].copy()
plot_df = plot_df.sort_values('Datum')
plot_df = plot_df[['Datum','Zeit_im_Besuch_Sek','Bounce_Zeit_Sek']].dropna(subset=['Zeit_im_Besuch_Sek','Bounce_Zeit_Sek'])

plt.figure(figsize=(14, 7), dpi=150)
ax = plt.gca()
sns.lineplot(data=plot_df, x='Datum', y='Zeit_im_Besuch_Sek', marker='o', linewidth=2.2, color='#1f77b4', label='Ø Sitzungsdauer')
sns.lineplot(data=plot_df, x='Datum', y='Bounce_Zeit_Sek', marker='o', linewidth=2.2, color='#ff7f0e', label='Zeit bis Absprung')
ax.xaxis.set_major_locator(mdates.DayLocator(interval=2))
ax.xaxis.set_major_formatter(mdates.DateFormatter('%d.%m.%Y'))
plt.xticks(rotation=45, ha='right')
plt.ylabel('Dauer (Sekunden)')
plt.xlabel('Datum')
plt.title('Vergleich: Ø Sitzungsdauer vs. Zeit bis zum Absprung (September 2024)')
plt.legend(title='Metrik', loc='upper right', frameon=True)
plt.grid(True, which='major', axis='both', linestyle='--', alpha=0.3)
