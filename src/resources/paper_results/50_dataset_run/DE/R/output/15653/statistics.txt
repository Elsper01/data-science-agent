  df <- df %>% mutate(metric = case_when(
    str_detect(tolower(!!sym(label_col)), "ank") ~ "Gästeankünfte",
    str_detect(tolower(!!sym(label_col)), "übernacht|uebernacht") ~ "Gästeübernachtungen",
    TRUE ~ !!sym(label_col)
  ))
} else {
  df$metric <- "Wert"

    df_plot <- df

  df_plot <- df
