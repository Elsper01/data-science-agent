try:
    input_path = r'./src/resources/data/study/139606/dataset.csv'
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except Exception as e:
    print(f"Error loading data: {e}")
    sys.exit(1)

# Set visual theme
sns.set_theme(style='whitegrid', context='talk')

# Prepare the 'Ergebnis' column for visualization
if 'Ergebnis' in df.columns:
    ergebnis_series = df['Ergebnis'].copy()
else:
    # Fallback if column is missing
    ergebnis_series = pd.Series(['Missing'] * len(df))

# Clean data: fill NaNs and strip whitespace
ergebnis_clean = ergebnis_series.fillna('Fehlend').astype(str).str.strip()

# Calculate frequencies
counts = ergebnis_clean.value_counts(dropna=False)
counts_df = counts.rename_axis('Ergebnis').reset_index(name='Anzahl')
counts_df['Prozent'] = counts_df['Anzahl'] / counts_df['Anzahl'].sum()

# Define sorting order: 'n.n.' first, then others by frequency
order = counts_df.sort_values('Anzahl', ascending=False)['Ergebnis'].tolist()
if 'n.n.' in order:
    order = ['n.n.'] + [x for x in order if x != 'n.n.']

# Helper to identify numeric-like strings (detectable values)
def is_numeric_like(s):
    if s in ['Fehlend', 'n.n.']:
        return False
    try:
        float(str(s).replace(',', '.'))
        return True
    except Exception:
        return False

# Assign colors based on category
palette = []
for cat in order:
    if cat == 'n.n.':
        palette.append('#2ca25f')  # Green for non-detectable
    elif cat == 'Fehlend':
        palette.append('#9e9ac8')  # Purple for missing
    elif is_numeric_like(cat):
        palette.append('#fb6a4a')  # Red/Orange for detectable
    else:
        palette.append('#6baed6')  # Blue for others

# Create the plot
fig, ax = plt.subplots(figsize=(12, 7))
sns.barplot(data=counts_df, x='Ergebnis', y='Anzahl', order=order, palette=palette, ax=ax)

# Formatting
ax.set_title('Verteilung der Ergebnisse (Ergebnis-Feld)\nSchweinefiletköpfe, IMIS-Messung 26.02.2020', pad=20)
ax.set_xlabel('Ergebnis (Rohangabe)')
ax.set_ylabel('Anzahl der Prüfkriterien')

# Add labels on top of bars
for p in ax.patches:
    height = p.get_height()
    if height > 0:
        x = p.get_x() + p.get_width() / 2
        total = counts_df['Anzahl'].sum()
        perc = height / total if total else 0
        ax.text(x, height + (counts_df['Anzahl'].max() * 0.02), f'{int(height)}\n({perc:.0%})', 
                ha='center', va='bottom', fontsize=11, fontweight='bold')

plt.xticks(rotation=20, ha='right')

missing_mask = df['Ergebnis'].isna() if 'Ergebnis' in df.columns else pd.Series([False] * n_total)
detected_mask = ergebnis_num.notna()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

try:
    df = pd.read_csv('./src/resources/data/study\\139606\\dataset.csv', sep=',', encoding='utf-8', dtype=str)
    df = df.replace(r'^\s*$', np.nan, regex=True)
    total_rows = len(df)

    ergebnis_raw = df['Ergebnis'].astype(str).str.strip()
    ergebnis_norm = ergebnis_raw.str.lower().str.replace(r'\s+', '', regex=True)
    nn_mask = ergebnis_norm.isin(['n.n.', 'n.n', 'nn'])

    ergebnis_clean_str = ergebnis_raw.mask(nn_mask)
    ergebnis_clean_str = ergebnis_clean_str.str.replace(',', '.', regex=False)
    ergebnis_num = pd.to_numeric(ergebnis_clean_str, errors='coerce')
    df['Ergebnis_num'] = ergebnis_num

    detected = df[df['Ergebnis_num'].notna()].copy()
    detected = detected[['Pruefmerkmal', 'Ergebnis_num', 'Einheit']]

    unit = None
    if 'Einheit' in df.columns:
        u = detected['Einheit'].dropna().unique()
        if len(u) > 0:
            unit = u[0]
    if unit is None:
        unit = 'Bq/kg'

    na_rate = df.isna().mean()
    max_na_rate = float(na_rate.max()) if len(na_rate) > 0 else 0.0
    max_na_col = na_rate.idxmax() if len(na_rate) > 0 else ''

    n_detected = int(detected.shape[0])
    n_nn = int(nn_mask.sum())
    n_missing = int(df['Ergebnis'].isna().sum())
    detect_rate = (n_detected / total_rows) * 100 if total_rows > 0 else 0.0

    def fmt_de(v):
        if pd.isna(v):
            return ''
        if v == 0:
            return '0,00'
        if abs(v) >= 0.1 and abs(v) < 10000:
            s = f"{v:,.3g}" if abs(v) < 1 else f"{v:,.2f}"
            s = s.replace(',', 'X').replace('.', ',').replace('X', '.')
            return s
        s = f"{v:.2E}"
        s = s.replace('E', 'E').replace('.', ',')
        return s

    sns.set_theme(style='whitegrid', context='talk')

    fig, ax = plt.subplots(figsize=(10, 6))

    if n_detected == 0:
        ax.axis('off')
        title = 'Radionuklide mit messbarem Ergebnis in Schweinefiletköpfen'
        ax.set_title(title, loc='left', fontsize=16, weight='bold')
        note = (
            f"Keine messbaren Ergebnisse gefunden.\n"
            f"n = {total_rows}, Nachweisrate = {detect_rate:.1f} %.\n"
            f"'n.n.' Zählung = {n_nn}, Fehlende Werte in 'Ergebnis' = {n_missing}.\n"
            f"Größter Fehlwertanteil: {max_na_rate*100:.1f}% in Spalte '{max_na_col}'."
        )
        ax.text(0.01, 0.85, note, transform=ax.transAxes, fontsize=12, va='top')
    else:
        detected_sorted = detected.sort_values('Ergebnis_num', ascending=True)
        y = detected_sorted['Pruefmerkmal']
        x = detected_sorted['Ergebnis_num']
        palette = sns.color_palette('viridis', n_colors=len(detected_sorted))
        ax.barh(y=y, width=x, color=palette, edgecolor='black', linewidth=0.5)
        ax.set_xscale('log')
        ax.set_xlabel(f'Aktivitätskonzentration [{unit}] (log-Skala)')
        ax.set_ylabel('Radionuklid (Pruefmerkmal)')
        title = 'Radionuklide mit messbarem Ergebnis in Schweinefiletköpfen (Hamburg, 26.02.2020)'
        ax.set_title(title, loc='left', fontsize=16, weight='bold')
        ax.grid(True, which='both', axis='x', linestyle='--', alpha=0.3)
        ax.grid(False, axis='y')

        xmin = x.min() * 0.7
        xmax = x.max() * 1.5
        if xmin <= 0:
            xmin = max(x.min() * 0.5, 1e-6)
        ax.set_xlim(left=xmin, right=xmax)

        for i, (val, label) in enumerate(zip(x.values, y.values)):
            txt = fmt_de(val) + f" {unit}"
            xpos = val * 1.05
            ax.text(xpos, label, txt, va='center', ha='left', fontsize=11)

        summary_lines = []
        summary_lines.append(f"n = {total_rows}, Nachweisrate = {detect_rate:.1f} %")
        summary_lines.append(f"'n.n.' Zählung = {n_nn}, Fehlende Werte in 'Ergebnis' = {n_missing}")
        if detected_sorted['Ergebnis_num'].notna().any():
            vals = detected_sorted['Ergebnis_num'].dropna()
            smin = fmt_de(vals.min())
            smed = fmt_de(float(vals.median()))
            smean = fmt_de(float(vals.mean()))
            smax = fmt_de(vals.max())
            summary_lines.append(f"Statistik messbarer Werte: Min {smin}, Median {smed}, Mittel {smean}, Max {smax} {unit}")
        summary_lines.append(f"Größter Fehlwertanteil: {max_na_rate*100:.1f}% in Spalte '{max_na_col}'")
        fig.text(0.01, 0.02, '\n'.join(summary_lines), ha='left', va='bottom', fontsize=10)

    plt.tight_layout(rect=[0, 0.05, 1, 1])
    output_path = './src/resources/output\\139606\\1_radionuklide_mit_messwerten.png'
    plt.savefig(output_path, dpi=300)
    plt.close(fig)
except Exception as e:
    print(str(e))

import os
import sys
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from textwrap import fill

try:
    input_path = os.path.normpath('./src/resources/data/study\\139606\\dataset.csv')
    output_dir = os.path.normpath('./src/resources/output\\139606')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    df.columns = [c.strip() for c in df.columns]
    for col in df.select_dtypes(include=['object', 'string']).columns:
        df[col] = df[col].astype(str).str.strip()
    missing_ratio = df.replace({'': np.nan}).isna().mean()
    def parse_numeric(x):
        if pd.isna(x):
            return np.nan
        s = str(x).strip()
        if s.lower() in ['n.n.', 'nn', 'n.n', 'na', 'nan', '']:
            return np.nan
        s = s.replace(',', '.')
        try:
            return float(s)
        except Exception:
            return np.nan
    if 'Ergebnis' in df.columns:
        df['Ergebnis_num'] = df['Ergebnis'].apply(parse_numeric)
    numeric_df = df.select_dtypes(include=[np.number])
    _ = numeric_df.describe(include='all') if not numeric_df.empty else None
    _ = numeric_df.corr(numeric_only=True) if not numeric_df.empty else None
    map_df = df[['Pruefmerkmal', 'Einheit']].copy()
    map_df['Pruefmerkmal'] = map_df['Pruefmerkmal'].replace({'nan': np.nan}).fillna('Unbekannt')
    map_df['Einheit'] = map_df['Einheit'].replace({'nan': np.nan}).fillna('Unbekannt')
    map_df = map_df.drop_duplicates()
    if map_df.empty:
        raise ValueError('Die Zuordnung von Prüfmerkmal zu Einheit ist leer und kann nicht visualisiert werden.')
    unit_order = map_df['Einheit'].value_counts().sort_values(ascending=False).index.tolist()
    map_df['Einheit'] = pd.Categorical(map_df['Einheit'], categories=unit_order, ordered=True)
    map_df = map_df.sort_values(['Einheit', 'Pruefmerkmal'])
    map_df['Zuweisung'] = 1
    sns.set_theme(style='whitegrid', context='talk')
    n_items = len(map_df)
    fig_h = min(22, max(6, 0.45 * n_items))
    fig_w = 14
    fig, ax = plt.subplots(figsize=(fig_w, fig_h))
    palette = sns.color_palette('Set2', n_colors=len(unit_order))
    palette_map = {u: palette[i % len(palette)] for i, u in enumerate(unit_order)}
    sns.barplot(data=map_df, y='Pruefmerkmal', x='Zuweisung', hue='Einheit', dodge=False, ax=ax, palette=palette_map, edgecolor='black')
    ax.set_xlabel('Zuordnung (1 = Einheit zu Prüfmerkmal)')
    ax.set_ylabel('Prüfmerkmal')
    title = 'Messeinheiten je Prüfmerkmal\nIMIS Schweinefiletköpfe, Probenahme 26.02.2020'
    ax.set_title(title)
    ax.set_xlim(0, 1.2)
    ax.set_xticks([])
    ax.grid(axis='x', visible=False)
    ax.grid(axis='y', linestyle=':', alpha=0.2)
    ylabels = [fill(lbl.get_text(), width=40) for lbl in ax.get_yticklabels()]
    ax.set_yticklabels(ylabels)
    ax.legend(title='Einheit', loc='upper right', bbox_to_anchor=(1.0, 1.02))
    fig.tight_layout()
    fig.text(0.01, 0.01, 'Datenquelle: IMIS | Methode: Gammaspektrometrie (FM)', fontsize=9)
    output_path = os.path.join(output_dir, '2_pruefmerkmal_einheit.png')
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
    print(str(e))
    sys.exit(1)
