import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import os

# Set visual style
sns.set_theme(style="whitegrid", context="talk")

# File paths
input_path = r'./src/resources/data/study/142441/dataset.csv'
out_dir = r'./src/resources/output/142441'
os.makedirs(out_dir, exist_ok=True)
out_path = os.path.join(out_dir, '0_nachgewiesene_radionuklide.png')

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xdf' 
# usually indicates ISO-8859-1 (Latin-1) or cp1252 encoding (common in German datasets for 'ß').
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(input_path, sep=',', encoding='latin-1')
    except Exception:
        df = pd.read_csv(input_path, sep=',', encoding='cp1252')

# Data Cleaning
# Convert 'Ergebnis' to numeric, handling German decimal commas and 'n.n.' (not detectable)
if 'Ergebnis' in df.columns:
    # Remove whitespace and handle German comma
    erg_clean = df['Ergebnis'].astype(str).str.strip()
    # Replace 'n.n.' variants with NaN
    erg_clean = erg_clean.replace({'n.n.': np.nan, 'n.n': np.nan, 'N.N.': np.nan, 'N.N': np.nan}, regex=False)
    # Replace comma with dot for float conversion
    erg_clean = erg_clean.str.replace(',', '.', regex=False)
    df['Ergebnis_num'] = pd.to_numeric(erg_clean, errors='coerce')
else:
    df['Ergebnis_num'] = np.nan

# Filter for detected radionuclides (numeric and > 0)
subset = df.loc[df['Ergebnis_num'].notna() & (df['Ergebnis_num'] > 0)].copy()

produkt_label = prod_vals[0] if len(prod_vals) > 0 else 'Probe'

unit_vals = df.get('Einheit', pd.Series(['Bq/kg'])).dropna().unique()
unit_label = unit_vals[0] if len(unit_vals) > 0 else 'Bq/kg'

# Visualization

ax = plt.gca()

import os
import sys
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

def load_data(file_path):
    # Try common encodings for German datasets if utf-8 fails
    encodings = ['utf-8', 'latin1', 'iso-8859-1', 'cp1252']
    for enc in encodings:
        try:
            return pd.read_csv(file_path, sep=',', encoding=enc)
        except (UnicodeDecodeError, pd.errors.ParserError):
            continue
    # If all fail, try with engine='python' and no encoding specified
    return pd.read_csv(file_path, sep=',', engine='python')

try:
    input_path = os.path.normpath('./src/resources/data/study/142441/dataset.csv')
    output_dir = os.path.normpath('./src/resources/output/142441')
    os.makedirs(output_dir, exist_ok=True)
    
    df = load_data(input_path)
    
    # Clean object columns
    obj_cols = df.select_dtypes(include=['object']).columns.tolist()
    for c in obj_cols:
        df[c] = df[c].astype(str).str.strip()
        df.loc[df[c].isin(['', 'nan', 'NaN', 'None']), c] = np.nan

    # Function to identify numeric results vs non-detected
    def classify_result(val):
        if pd.isna(val):
            return 'Missing'
        s = str(val).strip().lower()
        # Common German indicators for "not detected"
        not_detected_terms = {'n.n.', 'n.n', 'n. n.', 'nn', 'n n', 'nicht nachgewiesen', 'unter nachweisgrenze'}
        if s in not_detected_terms:
            return 'Nicht nachgewiesen'
        
        # Try to convert to float to see if it's a numeric detection
        try:
            # Handle German decimal commas and spaces
            clean_val = s.replace(' ', '').replace(',', '.')
            float(clean_val)
            return 'Nachgewiesen'
        except ValueError:
            return 'Sonstige/Fehler'

    df['Nachweisstatus'] = df['Ergebnis'].apply(classify_result)
    
    # Filter for the two main categories for the plot
    plot_df = df[df['Nachweisstatus'].isin(['Nicht nachgewiesen', 'Nachgewiesen'])].copy()
    categories = ['Nicht nachgewiesen', 'Nachgewiesen']
    counts = plot_df['Nachweisstatus'].value_counts().reindex(categories).fillna(0).astype(int)
    total_n = int(counts.sum())
    
    # Visualization
    sns.set_theme(style='whitegrid', font_scale=1.1)
    fig, ax = plt.subplots(figsize=(9, 6))
    
    # Colors: Grey for non-detected, Green for detected
    palette = {'Nicht nachgewiesen': '#9aa0a6', 'Nachgewiesen': '#2ca02c'}
    sns.barplot(x=counts.index, y=counts.values, ax=ax, palette=palette, hue=counts.index, legend=False)
    
    # Title and Labels
    ax.set_xlabel('Nachweisstatus', fontweight='bold')
    ax.set_ylabel('Anzahl der Messungen', fontweight='bold')
    
    title = 'Nachweisstatus der Radionuklide'
    subtitle = 'Kabeljaufilet (IMIS Routinemessprogramm)'
    if 'Probenahme_Beginn_Datum' in df.columns and df['Probenahme_Beginn_Datum'].notna().any():
        date_val = df['Probenahme_Beginn_Datum'].dropna().iloc[0]
        subtitle += f' | Stand: {date_val}'
    
    plt.suptitle(title, fontsize=16, fontweight='bold', y=0.98)
    ax.set_title(subtitle, fontsize=12, pad=10)

    # Annotate bars with counts and percentages
    for i, p in enumerate(ax.patches):
        h = p.get_height()
        perc = (h / total_n * 100) if total_n > 0 else 0
        ax.annotate(f'{int(h)}\n({perc:.1f}%)', 
                    (p.get_x() + p.get_width() / 2., h), 
                    ha='center', va='bottom', 
                    xytext=(0, 5), 
                    textcoords='offset points',
                    fontsize=11, fontweight='bold')

    # Add metadata footer
    det_rate = (counts['Nachgewiesen'] / total_n * 100) if total_n > 0 else 0
    footer_text = f'Gesamtanzahl Messungen (n) = {total_n} | Detektionsrate = {det_rate:.1f}%'
    plt.figtext(0.5, 0.01, footer_text, ha='center', fontsize=10, bbox={"facecolor":"orange", "alpha":0.1, "pad":5})

    plt.tight_layout(rect=[0, 0.05, 1, 0.95])
    
    output_path = os.path.join(output_dir, 'nachweisstatus_radionuklide.png')
    plt.savefig(output_path, dpi=200, bbox_inches='tight')
    plt.close(fig)

except Exception as e:
    print(f"Error: {str(e)}")
    sys.exit(1)

import os
import sys
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

def load_data(file_path):
    """Try multiple encodings to handle potential special characters like 'ß' or 'ü'."""
    encodings = ['utf-8', 'latin1', 'iso-8859-1', 'cp1252']
    for enc in encodings:
        try:
            return pd.read_csv(file_path, sep=',', encoding=enc, dtype=str)
        except (UnicodeDecodeError, pd.errors.ParserError):
            continue
    raise ValueError(f"Could not decode file {file_path} with common encodings.")

try:
    pd.set_option('display.max_columns', None)
    sns.set_theme(style='whitegrid', context='talk')
    plt.rcParams['font.family'] = 'DejaVu Sans'

    data_path = os.path.join('.', 'src', 'resources', 'data', 'study', '142441', 'dataset.csv')
    out_dir = os.path.join('.', 'src', 'resources', 'output', '142441')
    os.makedirs(out_dir, exist_ok=True)
    out_path = os.path.join(out_dir, '2_radionuklide_pruefmerkmal.png')

    df = load_data(data_path)

    # Clean whitespace
    df = df.apply(lambda x: x.str.strip() if x.dtype == "object" else x)

    if 'Pruefmerkmal' not in df.columns:
        raise ValueError("Column 'Pruefmerkmal' missing from dataset.")

    # Helper to check if a result is a valid detection
    def is_detected(x):
        if pd.isna(x): return False
        s = str(x).strip().lower()
        if s in {'n.n.', 'n.n', 'nn', 'na', 'k.a.', '', '0', '0.0'}:
            return False
        # Try to see if it's a number > 0
        try:
            val = float(s.replace(',', '.'))
            return val > 0
        except:
            return True # If it's a string that isn't 'n.n.', assume it's a descriptive detection

    if 'Ergebnis' in df.columns:
        df['detected'] = df['Ergebnis'].apply(is_detected)
    else:
        df['detected'] = False

    # Aggregate to get unique radionuclides and their detection status
    grp = df.groupby('Pruefmerkmal', dropna=False).agg(
        Anzahl=('Pruefmerkmal', 'size'),
        HatNachweis=('detected', 'max')
    ).reset_index()

    grp['Pruefmerkmal'] = grp['Pruefmerkmal'].fillna('Unbekannt')
    grp['Status'] = np.where(grp['HatNachweis'], 'Nachgewiesen', 'Nicht nachweisbar (n.n.)')

    # Sort: Detected first, then alphabetically
    grp = grp.sort_values(['HatNachweis', 'Pruefmerkmal'], ascending=[False, True])

    # Visualization
    height = max(6, 0.4 * len(grp))
    plt.figure(figsize=(12, height))
    
    ax = sns.barplot(
        data=grp,
        y='Pruefmerkmal',
        x='Anzahl',
        hue='Status',
        dodge=False,
        palette={'Nachgewiesen': '#2E7D32', 'Nicht nachweisbar (n.n.)': '#B0BEC5'}
    )

    plt.title('Im Gammaspektrometrie-Screening untersuchte Radionuklide', pad=20)
    plt.xlabel('Anzahl der Messungen')
    plt.ylabel('Radionuklid (Prüfmerkmal)')
    
    # Adjust x-axis to show integers if counts are small
    max_val = grp['Anzahl'].max()
    if max_val < 10:
        plt.xticks(range(int(max_val) + 2))

    plt.legend(title='Nachweisstatus', loc='lower right')
    plt.tight_layout()
    
    # Ensure labels aren't cut off
    plt.subplots_adjust(left=0.35)
    
    plt.savefig(out_path, dpi=300)
    plt.close()

except Exception as e:
    print(f"Error: {str(e)}")
    sys.exit(1)
