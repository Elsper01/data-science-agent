import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid", context="talk")

file_path = './src/resources/data/study\\136560\\dataset.csv'
output_path = './src/resources/output\\136560/1_messzeit_nach_methode.png'

df = pd.read_csv(file_path, sep=',', encoding='utf-8')

for c in df.select_dtypes(include=['object']).columns:
    df[c] = df[c].apply(lambda x: x.strip() if isinstance(x, str) else x)

def to_numeric_de(x):
    if pd.isna(x):
        return np.nan
    if isinstance(x, (int, float, np.number)):
        return float(x)
    s = str(x).strip().lower()
    if s in ['n.n.', 'n.n', 'nn', 'nicht nachweisbar', 'nan', '']:
        return np.nan
    s = s.replace('.', '') if s.count(',') == 0 and s.count('.') > 1 else s
    s = s.replace(',', '.')
    try:
        return float(s)
    except Exception:
        return np.nan

if 'Ergebnis' in df.columns:
    df['Ergebnis_num'] = df['Ergebnis'].apply(to_numeric_de)
else:
    df['Ergebnis_num'] = np.nan

eda_df = df.copy()
for c in eda_df.select_dtypes(include=['object']).columns:
    eda_df[c] = eda_df[c].replace({'': np.nan, 'n.n.': np.nan, 'n.n': np.nan, 'Nicht nachweisbar': np.nan})
missing_ratio = eda_df.isna().mean()

num_df = pd.DataFrame()
for c in eda_df.columns:
    if c == 'Ergebnis_num':
        num_df[c] = eda_df[c]
    else:
        if eda_df[c].dtype.kind in 'biufc':
            num_df[c] = eda_df[c]
        else:
            coerced = pd.to_numeric(eda_df[c], errors='coerce')
            if coerced.notna().any():
                num_df[c] = coerced
num_df = num_df.dropna(how='all', axis=1)
_ = num_df.describe().T if not num_df.empty else None
_ = num_df.corr(numeric_only=True) if not num_df.empty else None

if 'Pruefmerkmal' in df.columns and 'Methode' in df.columns:
    messzeit_df = df[df['Pruefmerkmal'].astype(str).str.strip().str.lower() == 'messzeit'].copy()
else:
    messzeit_df = pd.DataFrame(columns=['Methode', 'Ergebnis_num'])

messzeit_df = messzeit_df[['Methode', 'Ergebnis_num']].dropna(subset=['Ergebnis_num'])

if not messzeit_df.empty:
    order = messzeit_df.sort_values('Ergebnis_num', ascending=False)['Methode']
    plt.figure(figsize=(10, 6))
    ax = sns.barplot(data=messzeit_df, x='Methode', y='Ergebnis_num', order=order, palette='Blues_d', ci=None)
    ax.set_xlabel('Messmethode')
    ax.set_ylabel('Messzeit (s)')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

try:
    sns.set_theme(style="white", context="talk")
    data_path = os.path.normpath(r"./src/resources/data/study/136560/dataset.csv")
    
    # Load data
    df = pd.read_csv(data_path, sep=",", encoding="utf-8", dtype=str)
    
    # Clean data
    for c in df.columns:
        df[c] = df[c].astype(str).str.strip()
        df[c] = df[c].replace({"": np.nan, "None": np.nan, "nan": np.nan})
    
    # Check required columns
    required_cols = ["Methode", "Pruefmerkmal", "Einheit"]
    for col in required_cols:
        if col not in df.columns:
            df[col] = "Unbekannt"
            
    # Calculate missing proportions for footer
    missing_prop = {c: float(df[c].isna().mean()) for c in required_cols}
    size_info = int(df.shape[0])
    
    # Group data
    grp = (
        df.groupby(["Methode", "Pruefmerkmal", "Einheit"], dropna=False)
          .size()
          .reset_index(name="Anzahl")
    )
    
    # Ensure numeric type for heatmap values to avoid 'd' format error
    grp["Anzahl"] = grp["Anzahl"].fillna(0).astype(int)
    
    methods = sorted(df["Methode"].unique().tolist())
    pruefs = sorted(df["Pruefmerkmal"].unique().tolist())
    einheiten = sorted(df["Einheit"].unique().tolist())
    
    vmax = int(grp["Anzahl"].max()) if not grp.empty else 1
    n_methods = len(methods)
    
    # Dynamic sizing
    fig_height = max(5, 0.6 * len(pruefs))
    fig_width = max(5 * n_methods, 6)
    
    fig, axes = plt.subplots(1, n_methods, figsize=(fig_width, fig_height), squeeze=False)
    axes = axes.flatten()
    
    for i, m in enumerate(methods):
        ax = axes[i]
        sub = grp[grp["Methode"] == m].copy()
        
        # Pivot for heatmap
        pivot = sub.pivot_table(index="Pruefmerkmal", columns="Einheit", values="Anzahl", aggfunc="sum", fill_value=0)
        
        # Reindex to ensure all methods share the same axes for comparison
        pivot = pivot.reindex(index=pruefs, columns=einheiten, fill_value=0)
        
        # Convert to int to ensure 'd' format works
        pivot = pivot.astype(int)
        
        sns.heatmap(
            pivot, 
            ax=ax, 
            annot=True, 
            fmt="d", 
            cmap="YlGnBu", 
            vmin=0, 
            vmax=vmax, 
            cbar=(i == n_methods - 1), 
            linewidths=0.5, 
            linecolor="#eeeeee"
        )
        
        ax.set_title(f"Methode: {m}", fontsize=14, pad=10)
        ax.set_xlabel("Einheit", fontsize=12)
        ax.set_ylabel("Prüfmerkmal" if i == 0 else "", fontsize=12)
        ax.tick_params(axis='x', rotation=45)
        ax.tick_params(axis='y', rotation=0)

    plt.suptitle("Verteilung von Prüfmerkmalen und Einheiten nach Methode", fontsize=18, y=1.05)
    
    # Footer
    miss_m = f"{missing_prop.get('Methode', 0):.1%}"
    miss_p = f"{missing_prop.get('Pruefmerkmal', 0):.1%}"
    miss_e = f"{missing_prop.get('Einheit', 0):.1%}"
    footer_text = f"Gesamtzeilen: {size_info} | Fehlende Werte - Methode: {miss_m}, Prüfmerkmal: {miss_p}, Einheit: {miss_e}"
    fig.text(0.01, -0.05, footer_text, ha='left', va='top', fontsize=10, color='gray')
    
    output_dir = os.path.normpath(r"./src/resources/output/136560")
    os.makedirs(output_dir, exist_ok=True)
    out_path = os.path.join(output_dir, "2_pruefmerkmal_methode_einheit_heatmap.png")
    
    plt.tight_layout()
    plt.savefig(out_path, dpi=200, bbox_inches='tight')
    plt.close(fig)

except Exception as e:
    print(f"Error: {str(e)}")
