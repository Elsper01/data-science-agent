import os
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

input_path = r'./src/resources/data/study\32743\dataset.csv'
output_dir = r'./src/resources/output\32743'
os.makedirs(output_dir, exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8', dtype=str)
df = df.replace(r'^\s*$', np.nan, regex=True)
if 'Unnamed__11' in df.columns:
    df = df.drop(columns=['Unnamed__11'])
if 'Datum' in df.columns:
    df['Datum'] = df['Datum'].replace(['nan', 'NaN'], np.nan)
    df = df[df['Datum'].notna()].copy()
    df['Datum'] = pd.to_datetime(df['Datum'], dayfirst=True, errors='coerce')
    df = df[df['Datum'].notna()].copy()

percent_cols = [c for c in df.columns if isinstance(c, str) and df[c].astype(str).str.contains('%', na=False).any()]
for c in percent_cols:
    s = df[c].astype(str).str.replace('%', '', regex=False).str.replace(',', '.', regex=False)
    df[c + '_pct'] = pd.to_numeric(s, errors='coerce')

time_cols = [c for c in df.columns if isinstance(c, str) and df[c].astype(str).str.contains(':', na=False).any()]
for c in time_cols:
    td = pd.to_timedelta(df[c], errors='coerce')
    df[c + '_sek'] = td.dt.total_seconds()

num_candidate_cols = ['Besuche', 'Besucher', 'Besuchsh_ufigkeit', 'Seitenaufrufe', 'Seiten_pro_Besuch']
for c in [col for col in num_candidate_cols if col in df.columns]:
    df[c] = pd.to_numeric(df[c].astype(str).str.replace(',', '.', regex=False), errors='coerce')

eda_df = df.copy()
missing = eda_df.isna().mean().to_frame('anteil_fehlend')
missing.to_csv(os.path.join(output_dir, 'eda_anteil_fehlender_werte.csv'), encoding='utf-8')
numeric_cols = eda_df.select_dtypes(include=['number']).columns.tolist()
summary = eda_df[numeric_cols].describe().T
summary.to_csv(os.path.join(output_dir, 'eda_kennwerte.csv'), encoding='utf-8')
corr = eda_df[numeric_cols].corr(method='pearson')
corr.to_csv(os.path.join(output_dir, 'eda_korrelationen.csv'), encoding='utf-8')

x_col = 'Zeit_im_Besuch_sek'
y_col = 'Bounce_Rate_pct'
plot_df = eda_df[[x_col, y_col]].dropna().copy()

sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(10, 6))
if len(plot_df) > 0:
    sns.regplot(data=plot_df, x=x_col, y=y_col, ax=ax, scatter_kws={'s': 60, 'alpha': 0.7, 'color': '#1f77b4'}, line_kws={'color': '#d62728', 'lw': 2}, ci=None)
    ax.set_title('Zusammenhang zwischen Besuchsdauer und Bounce Rate (Dezember 2021)')
    ax.set_xlabel('Zeit im Besuch (Sekunden)')
    ax.set_ylabel('Bounce Rate (%)')
    r = np.nan
    if len(plot_df) > 1:
        x = plot_df[x_col].to_numpy(dtype=float)
        y = plot_df[y_col].to_numpy(dtype=float)
        if np.isfinite(x).sum() > 1 and np.isfinite(y).sum() > 1:
            r = np.corrcoef(x, y)[0, 1]
    if np.isfinite(r):
        txt = f'Pearson r = {r:.2f}'
        ax.text(0.02, 0.95, txt, transform=ax.transAxes, ha='left', va='top', fontsize=12, bbox=dict(boxstyle='round,pad=0.3', fc='white', ec='gray', alpha=0.8))
    ax.margins(0.05)
else:
    ax.set_title('Keine ausreichenden Daten für die Visualisierung vorhanden')
    ax.set_xlabel('Zeit im Besuch (Sekunden)')
    ax.set_ylabel('Bounce Rate (%)')

plt.tight_layout()
plot_path = os.path.join(output_dir, '1_scatter_besuchsdauer_vs_bounce_rate.png')
fig.savefig(plot_path, dpi=150)
plt.close(fig)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid")
plt.rcParams.update({"figure.dpi": 120, "savefig.dpi": 120, "font.size": 11})

path = r"./src/resources/data/study\32743\dataset.csv"
df = pd.read_csv(path, sep=",", encoding="utf-8", low_memory=False)

if "Unnamed__11" in df.columns:
    df = df.drop(columns=["Unnamed__11"]) 

if "Datum" in df.columns:
    df["Datum_parsed"] = pd.to_datetime(df["Datum"], dayfirst=True, errors="coerce")
else:
    df["Datum_parsed"] = pd.NaT

df = df[~df["Datum_parsed"].isna()].copy()

for col in ["Anteil_Besuche_mit_Cookies", "Anteil_Besucher_mit_Cookies", "Bounce_Rate"]:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col].astype(str).str.replace("%", "", regex=False).str.replace(",", ".", regex=False), errors="coerce")

for col in ["Besuche", "Besucher", "Besuchsh_ufigkeit", "Seitenaufrufe", "Seiten_pro_Besuch"]:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors="coerce")

if "Besuchsh_ufigkeit" not in df.columns and {"Besuche", "Besucher"}.issubset(df.columns):
    df["Besuchsh_ufigkeit"] = np.where(df["Besucher"] > 0, df["Besuche"] / df["Besucher"], np.nan)

df = df.sort_values("Datum_parsed").reset_index(drop=True)

missing_ratio = df.isna().mean()
num_df = df.select_dtypes(include=[np.number]).copy()
descr = num_df.describe().T
corr_full = num_df.corr(numeric_only=True)

x = df["Besuchsh_ufigkeit"].copy()
y = df["Anteil_Besucher_mit_Cookies"].copy()
valid = x.notna() & y.notna()
if valid.sum() >= 2:
    r = np.corrcoef(x[valid], y[valid])[0, 1]
else:
    r = np.nan

fig, ax = plt.subplots(figsize=(12, 6.5))
idx = np.arange(len(df))
bar_color = sns.color_palette("Blues", 6)[4]
line_color = sns.color_palette("Reds", 6)[4]
ax.bar(idx, df["Besuchsh_ufigkeit"], color=bar_color, alpha=0.85, label="Besuchshäufigkeit (Besuche pro Besucher)")
ax.set_ylabel("Besuchshäufigkeit (Besuche pro Besucher)")
ax.set_xlabel("Datum")
ax.set_xticks(idx)
ax.set_xticklabels(df["Datum_parsed"].dt.strftime("%d.%m.%Y"), rotation=45, ha="right")
ax2 = ax.twinx()
ax2.plot(idx, df["Anteil_Besucher_mit_Cookies"], color=line_color, marker="o", linewidth=2.0, markersize=4, label="Cookie-Zustimmung (% der Besucher)")
ax2.set_ylabel("Cookie-Zustimmung (% der Besucher)")

yl = df["Besuchsh_ufigkeit"].dropna()
if not yl.empty:
    ymin, ymax = yl.min(), yl.max()
    ax.set_ylim(max(0, ymin - 0.2), ymax + 0.2)
pl = df["Anteil_Besucher_mit_Cookies"].dropna()
if not pl.empty:
    pmin, pmax = pl.min(), pl.max()
    ax2.set_ylim(max(0, pmin - max(0.05, 0.1 * (pmax - pmin + 1e-9))), pmax + max(0.05, 0.1 * (pmax - pmin + 1e-9)))

handles1, labels1 = ax.get_legend_handles_labels()
handles2, labels2 = ax2.get_legend_handles_labels()
ax.legend(handles1 + handles2, labels1 + labels2, loc="upper left", frameon=True)

corr_text = "Korrelation r = {:.2f}".format(r) if pd.notna(r) else "Korrelation r = n.v."
fig.suptitle("Nutzerbindung vs. Cookie-Zustimmung im Zeitverlauf (Dezember 2021)\n" + corr_text, fontsize=14, y=0.98)
plt.tight_layout()

output_path = r"./src/resources/output\32743/2_nutzerbindung_vs_cookie_zustimmung.png"
plt.savefig(output_path, bbox_inches="tight")
plt.close(fig)
