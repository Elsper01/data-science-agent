# Function to parse 'Ergebnis' column

    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    # Handle 'n.n.' (nicht nachweisbar)
    s_compact = s.lower().replace(' ', '')
    if s_compact in {'n.n.', 'nn', 'n.n'}:
        return np.nan

    s = s.replace(' ', '').replace(',', '.')
    try:
        return pd.to_numeric(s, errors='coerce')
    except Exception:
        return np.nan

if 'Ergebnis' in df.columns:

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Set theme
sns.set_theme(style="whitegrid", context="talk")

# Paths
input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '142441', 'dataset.csv')
output_dir = os.path.join('.', 'src', 'resources', 'output', '142441')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '1_verteilung_nachweisbarkeit.png')

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xdf' 
# suggests the file is likely encoded in ISO-8859-1 (Latin-1) or Windows-1252.
# 0xdf is the German 'ß' in Latin-1.
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=',', encoding='iso-8859-1')

row_count = len(df)

# Clean 'Ergebnis' column
# 'n.n.' usually stands for 'nicht nachweisbar' (not detectable)
# We treat numeric values as detectable and 'n.n.' or non-numeric as non-detectable
df['Ergebnis_str'] = df['Ergebnis'].astype(str).str.strip().str.lower()

# Define detection status
# If it's 'n.n.' or empty, it's non-detectable. 
# Otherwise, we try to see if it's a number.
def check_detection(val):
    if val in ['n.n.', 'nan', '', 'none']:
        return 'Nicht nachweisbar'
    # Replace German decimal comma with dot for numeric check
    clean_val = val.replace(',', '.')
    try:
        float(clean_val)
        return 'Nachweisbar'
    except ValueError:
        return 'Nicht nachweisbar'

df['Nachweisstatus'] = df['Ergebnis_str'].apply(check_detection)

# Prepare data for plotting
cats = ['Nicht nachweisbar', 'Nachweisbar']
counts = [int((df['Nachweisstatus'] == c).sum()) for c in cats]
percentages = [c / row_count * 100 if row_count > 0 else 0 for c in counts]

# Plotting
fig, ax = plt.subplots(figsize=(9, 6))
colors = ['#8da0cb', '#66c2a5']
sns.barplot(x=cats, y=counts, palette=colors, ax=ax, hue=cats, legend=False)

# Adjust y-axis
if row_count > 0:
    step = max(1, row_count // 10)
    ax.set_yticks(np.arange(0, row_count + step, step))

ax.set_xlabel('Nachweisstatus')
ax.set_ylabel('Anzahl Datensätze')
ax.set_title(f'Verteilung der Messergebnisse (n={row_count})')

# Add labels on top of bars
for i, (c, p) in enumerate(zip(counts, percentages)):
    label = f"{c}\n({str(round(p, 1)).replace('.', ',')}%)"
    ax.text(i, c + (row_count * 0.01), label, ha='center', va='bottom', fontsize=12, fontweight='bold')

# Clean up spines
sns.despine()

plt.tight_layout()
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close(fig)

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Define paths
input_path = os.path.normpath('./src/resources/data/study/142441/dataset.csv')
output_dir = os.path.normpath('./src/resources/output/142441')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '2_radionuklide_pruefmerkmal_balken.png')

# Handle encoding error: 'utf-8' failed with 0xdf (ß in latin-1/cp1252). 
# Trying 'cp1252' which is common for German datasets.
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=',', encoding='cp1252')

# Clean string columns
for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].astype(str).str.strip()

# Identify the column for radionuclides
pm_col = 'Pruefmerkmal' if 'Pruefmerkmal' in df.columns else None
if pm_col is None:
    # Fallback if column name is slightly different
    pm_col = [col for col in df.columns if 'merkmal' in col.lower()]
    pm_col = pm_col[0] if pm_col else None

if pm_col is None:
    raise ValueError("Required column 'Pruefmerkmal' not found in dataset.")

# Count occurrences of each radionuclide
df_counts = df[pm_col].value_counts().reset_index()
df_counts.columns = [pm_col, 'Anzahl']
df_counts = df_counts.sort_values(by=pm_col)

# Visualization
sns.set_theme(style='whitegrid')
# Adjust height based on number of unique entries (34 expected)
height = max(8, 0.25 * len(df_counts))
fig, ax = plt.subplots(figsize=(10, height))

sns.barplot(
    data=df_counts, 
    y=pm_col, 
    x='Anzahl', 
    ax=ax, 
    color='skyblue', 
    edgecolor='0.3'
)

# Add labels and title
ax.set_title('Übersicht der untersuchten Radionuklide (Prüfmerkmale)', fontsize=14, pad=20)
ax.set_xlabel('Anzahl der Messungen', fontsize=12)
ax.set_ylabel('Radionuklid', fontsize=12)

# Add count labels to the end of bars
for i, v in enumerate(df_counts['Anzahl']):
    ax.text(v + 0.1, i, str(int(v)), color='black', va='center', fontsize=10)

plt.tight_layout()

# Save the plot
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close(fig)
