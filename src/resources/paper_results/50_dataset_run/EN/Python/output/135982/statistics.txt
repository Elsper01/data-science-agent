import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid")
plt.rcParams["axes.titlesize"] = 12
plt.rcParams["axes.labelsize"] = 10
plt.rcParams["xtick.labelsize"] = 9
plt.rcParams["ytick.labelsize"] = 8
plt.rcParams["legend.fontsize"] = 9

in_path = os.path.normpath(r"./src/resources/data/study\135982\dataset.csv")
out_dir = os.path.normpath(r"./src/resources/output\135982")
os.makedirs(out_dir, exist_ok=True)
plot_name = "verteilung_nachweisbarkeit_nach_pruefmerkmal"
out_path = os.path.join(out_dir, f"1_{plot_name}.png")

df = pd.read_csv(in_path, sep=",", encoding="utf-8")

missing_props = df.isna().mean()

if "Ergebnis" not in df.columns or "Pruefmerkmal" not in df.columns:
    raise ValueError("Erforderliche Spalten 'Ergebnis' oder 'Pruefmerkmal' fehlen im Datensatz.")

df["Ergebnis_str"] = df["Ergebnis"].astype(str).str.strip()
low = df["Ergebnis_str"].str.lower().str.replace("\u202f", "", regex=False).str.replace("\xa0", "", regex=False).str.replace(" ", "", regex=False)
mask_nn = low.isin({"n.n.", "n.n", "nn"})

def to_float(v):
    s = str(v).strip()
    if s == "":
        return np.nan
    sl = s.lower().replace("\u202f", "").replace("\xa0", "").strip()
    if sl.replace(" ", "") in {"n.n.", "n.n", "nn"}:
        return np.nan
    s2 = s.replace(" ", "").replace(",", ".")
    try:
        return float(s2)
    except Exception:
        return np.nan

df["Ergebnis_num"] = df["Ergebnis_str"].apply(to_float)

df["Ergebnis_Typ"] = np.where(mask_nn, "Nicht nachweisbar (n.n.)", np.where(df["Ergebnis_num"].notna(), "Messwert", "Fehlend/Unklar"))

eda_counts_typ = df["Ergebnis_Typ"].value_counts(dropna=False)
eda_numeric_stats = df.loc[df["Ergebnis_Typ"] == "Messwert", "Ergebnis_num"].describe()

counts = df.groupby(["Pruefmerkmal", "Ergebnis_Typ"], dropna=False).size().unstack(fill_value=0)
for col in ["Nicht nachweisbar (n.n.)", "Messwert", "Fehlend/Unklar"]:
    if col not in counts.columns:
        counts[col] = 0
counts = counts[["Nicht nachweisbar (n.n.)", "Messwert", "Fehlend/Unklar"]]

if "Messwert" in counts.columns:
    counts = counts.sort_values(by=["Messwert", "Nicht nachweisbar (n.n.)"], ascending=[False, True])

color_map = {
    "Nicht nachweisbar (n.n.)": "#8fbcd4",
    "Messwert": "#e07a5f",
    "Fehlend/Unklar": "#a8a8a8",
}

height = max(8, 0.28 * len(counts))
fig, ax = plt.subplots(figsize=(12, height))

y_pos = np.arange(len(counts))
left = np.zeros(len(counts))
handles = []
labels = []
for cat in ["Nicht nachweisbar (n.n.)", "Messwert", "Fehlend/Unklar"]:
    if cat in counts.columns:
        vals = counts[cat].values
        if np.any(vals > 0):
            bars = ax.barh(y_pos, vals, left=left, height=0.8, color=color_map.get(cat, "#cccccc"), edgecolor="white", label=cat)
            left = left + vals
            handles.append(bars)
            labels.append(cat)

unit_vals = df["Einheit"].dropna().astype(str).unique().tolist()
unit = unit_vals[0] if len(unit_vals) > 0 else ""

messwert_map = df.loc[df["Ergebnis_Typ"] == "Messwert", ["Pruefmerkmal", "Ergebnis_num"]].dropna().set_index("Pruefmerkmal")["Ergebnis_num"].to_dict()

for i, prm in enumerate(counts.index.tolist()):
    mv = counts.loc[prm, "Messwert"] if "Messwert" in counts.columns else 0
    if mv > 0 and prm in messwert_map:
        val = messwert_map[prm]
        sval = ("{:.2e}".format(val)).replace("e", "E").replace(".", ",")
        xpos = counts.loc[prm, "Nicht nachweisbar (n.n.)"] + mv / 2.0
        ax.text(xpos, i, f"Wert: {sval} {unit}", va="center", ha="center", fontsize=7, color="#222222")

ax.set_yticks(y_pos)
ax.set_yticklabels(counts.index.tolist())
ax.set_xlabel("Anzahl der Ergebnisse")
ax.set_ylabel("Radionuklid (Prüfmerkmal)")
ax.set_title("Nachweisstatus je Radionuklid: n.n. vs. Messwert (Reinwasserprobe 24.11.2015, WW Curslack Nord)")
ax.legend(title="Ergebnistyp", loc="lower right")
ax.set_xlim(0, 1 if counts.values.max() <= 1 else counts.values.max())
plt.tight_layout()
plt.subplots_adjust(left=0.38)
plt.savefig(out_path, dpi=300, bbox_inches="tight")
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('./src/resources/data/study\\135982\\dataset.csv', sep=',', encoding='utf-8')
for c in df.select_dtypes(include=['object', 'string']).columns:
    df[c] = df[c].astype(str).str.strip()
missing_prop = df.replace({'': np.nan}).isna().mean()
df['Ergebnis_num'] = pd.to_numeric(df['Ergebnis'].str.replace(',', '.', regex=False), errors='coerce')
detection_flag = df['Ergebnis_num'].notna()
detection_rate = detection_flag.mean()
method_counts = df[['Methode', 'Pruefmerkmal']].drop_duplicates()['Methode'].value_counts()
desc_stats = df['Ergebnis_num'].describe()
map_df = df[['Methode', 'Pruefmerkmal']].drop_duplicates()
map_df = map_df.dropna(subset=['Methode', 'Pruefmerkmal'])
map_df['val'] = 1
pivot = map_df.pivot_table(index='Methode', columns='Pruefmerkmal', values='val', aggfunc='max', fill_value=0)
method_order = map_df['Methode'].value_counts().index.tolist()
col_order = []
for m in method_order:
    col_order.extend(sorted(map_df.loc[map_df['Methode'] == m, 'Pruefmerkmal'].unique().tolist()))
col_order = [c for c in col_order if c in pivot.columns]
pivot = pivot.reindex(index=method_order)
pivot = pivot.reindex(columns=col_order)
sns.set_theme(style='white')
fig_w = min(24, 0.35 * max(1, pivot.shape[1]) + 2)
fig_h = min(12, 0.6 * max(1, pivot.shape[0]) + 1.5)
fig, ax = plt.subplots(figsize=(fig_w, fig_h))
sns.heatmap(pivot, ax=ax, cmap='YlGnBu', vmin=0, vmax=1, cbar=True, linewidths=0.5, linecolor='white')
ax.set_xlabel('Prüfmerkmal (Radionuklide)')
ax.set_ylabel('Analytische Methode')
ax.set_title('Eingesetzte Analysemethoden je Radionuklid (Prüfmerkmal)\nReinwasserprobe WW Curslack Nord, 24.11.2015')
ax.tick_params(axis='x', labelrotation=90, labelsize=8)
ax.tick_params(axis='y', labelrotation=0, labelsize=10)
cb = ax.collections[0].colorbar
cb.set_ticks([0, 1])
cb.set_ticklabels(['nicht verwendet', 'verwendet'])
cb.set_label('Methodenverwendung')
plt.tight_layout()
plt.savefig('./src/resources/output\\135982/2_methoden_vs_radionuklide_heatmap.png', dpi=300, bbox_inches='tight')
plt.close()
