df = pd.read_csv(data_path, sep=',', encoding='utf-8')

missing_pct = df.isna().mean().reset_index()

missing_pct.to_csv(missing_pct_path, index=False, encoding='utf-8')

def parse_ergebnis_to_float(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    if s == '':
        return np.nan
    low = s.lower()
    if low in ['n.n.', 'n.n', 'nicht nachweisbar', 'nn']:
        return np.nan
    s = s.replace(' ', '')
    s = s.replace('<', '')
    s = s.replace('>', '')
    s = s.replace(',', '.')
    try:
        return float(s)
    except Exception:
        try:
            return float(s)
        except Exception:
            return np.nan

if 'Ergebnis' in df.columns:
    df['Ergebnis_num'] = df['Ergebnis'].apply(parse_ergebnis_to_float)
    erz_desc = df['Ergebnis_num'].describe().to_frame(name='Ergebnis_num_Statistik')
    erz_desc_path = os.path.join(output_dir, 'eda_ergebnis_num_beschreibung.csv')
    erz_desc.to_csv(erz_desc_path, encoding='utf-8')
    erz_raw = df['Ergebnis'].astype(str).str.strip().str.lower()
    censored_mask = erz_raw.str.startswith('<') | erz_raw.isin(['n.n.', 'n.n', 'nicht nachweisbar'])
    censored_summary = pd.DataFrame({'Kategorie': ['unter_Nachweisgrenze', 'quantifizierbar_oder_sonstiges'], 'Anzahl': [int(censored_mask.sum()), int((~censored_mask).sum())]})
    censored_path = os.path.join(output_dir, 'eda_ergebnis_zensierung.csv')
    censored_summary.to_csv(censored_path, index=False, encoding='utf-8')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv(r'./src/resources/data/study\137336\dataset.csv', sep=',', encoding='utf-8')
_ = df.isna().mean()
num_df = df.select_dtypes(include=[np.number])
_ = num_df.corr() if num_df.shape[1] > 1 else None
s = df['Ergebnis'].copy()
mask_na = s.isna()
s_nonnull = s[~mask_na].astype(str).str.strip()
s_norm = s_nonnull.str.lower()
s_no_punct = s_norm.str.replace(r'[\s\.]', '', regex=True)
mask_lt = s_nonnull.str.startswith('<', na=False)
mask_nn = s_no_punct.eq('nn')
mask_empty = s_nonnull.eq('')
cat = pd.Series(index=df.index, dtype='object')
cat.loc[mask_na | mask_empty.reindex(df.index, fill_value=False)] = 'Fehlend'
nd_mask = pd.Series(False, index=df.index)
nd_mask.loc[s_nonnull.index] = (mask_lt | mask_nn)
cat.loc[nd_mask] = 'Nicht nachweisbar'
cat.loc[cat.isna()] = 'Nachweisbar'
counts_full = cat.value_counts()
counts = counts_full.reindex(['Nachweisbar', 'Nicht nachweisbar']).fillna(0).astype(int)
valid_total = int(counts.sum())
perc = (counts / valid_total * 100) if valid_total > 0 else counts.astype(float)
plot_df = pd.DataFrame({'Kategorie': counts.index, 'Anzahl': counts.values, 'Prozent': perc.values})
missing_count = int(counts_full.get('Fehlend', 0))
sns.set_theme(style='whitegrid', font_scale=1.1)
plt.figure(figsize=(8, 5))
palette = {'Nachweisbar': '#1f77b4', 'Nicht nachweisbar': '#7f7f7f'}
ax = sns.barplot(data=plot_df, x='Kategorie', y='Anzahl', palette=palette, order=['Nachweisbar', 'Nicht nachweisbar'])
for i, row in plot_df.iterrows():
    ax.text(i, row['Anzahl'] + max(valid_total * 0.01, 0.2), f"{int(row['Anzahl'])} ({row['Prozent']:.1f}%)" if valid_total > 0 else f"{int(row['Anzahl'])}", ha='center', va='bottom', fontsize=11)
ax.set_xlabel('Kategorie des Ergebnisses')
ax.set_ylabel('Anzahl der Prüfergebnisse [n]')
ax.set_title('Nachweisbare vs. nicht nachweisbare Ergebnisse\nFeinstes Wiesenstreu, Probenahme 22.11.2016')
ax.set_ylim(0, max(plot_df['Anzahl'].max() * 1.15, 1))
ax.set_xticklabels(['Nachweisbar', 'Nicht nachweisbar (n.n./<)'])
caption = f"Definition: Nicht nachweisbar = Angabe 'n.n.' oder Werte mit '<'. Fehlende Ergebnisse: {missing_count}. Basis n={valid_total}."
plt.figtext(0.5, -0.02, caption, ha='center', va='top', fontsize=9)
plt.tight_layout()
plt.savefig(r'./src/resources/output\137336\run_2/2_nachweis_vs_nicht.png', dpi=300, bbox_inches='tight')
plt.close()
