parse_ergebnis_to_numeric <- function(x){
  y <- trimws(as.character(x))
  nd_flag <- grepl("^(?i)\\s*n\\.?\\s*n\\.?\\s*$", y, perl=TRUE) | grepl("nicht\\s*nachweisbar", tolower(y))
  y_clean <- gsub("\\s+", "", y)
  y_clean <- gsub(",", ".", y_clean, fixed = FALSE)
  suppressWarnings({
    y_num <- as.numeric(y_clean)
  })
  y_num[nd_flag | y == "" | is.na(y)] <- NA_real_
  y_num
}
