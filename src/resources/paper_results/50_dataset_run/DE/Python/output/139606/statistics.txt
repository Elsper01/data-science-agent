def _parse_numeric(v):
    if pd.isna(v):
        return np.nan
    s = str(v).strip()
    if s == '':
        return np.nan
    sl = s.lower()
    if sl in {'n.n.', 'n.n', 'nn', 'na', 'nan', 'n/a', '-'}:
        return np.nan
    s = s.replace(' ', '')
    if ',' in s:
        s2 = s.replace('.', '').replace(',', '.')

            return float(s2)
        except Exception:
            pass

            pass

                pass
