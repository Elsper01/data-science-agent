import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

# File paths
input_path = r'./src/resources/data/study\15539\dataset.csv'
output_dir = r'./src/resources/output\15539'
os.makedirs(os.path.normpath(output_dir), exist_ok=True)

# Set visual style
sns.set_theme(style='whitegrid', context='talk')

# Handle potential encoding issues (latin-1 is common for German datasets with umlauts)
na_vals = ['', 'NA', 'NaN', 'nan', None]
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8', na_values=na_vals)
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1', na_values=na_vals)

# Clean string columns
for c in ['2_variable_attribute_label', 'value_variable_label', 'value_unit', 'value_variable_code']:
    if c in df.columns:
        df[c] = df[c].astype(str).str.strip()

# Convert numeric columns
if 'time' in df.columns:
    df['time'] = pd.to_numeric(df['time'], errors='coerce')
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for relevant data
# We want Inland vs Ausland, excluding 'Insgesamt'
allowed_origins = ['Inland', 'Ausland']
df_vis = df[df['2_variable_attribute_label'].isin(allowed_origins)].copy()
df_vis = df_vis.dropna(subset=['value', 'value_variable_label'])

# Aggregate data over the time period
min_year = int(df_vis['time'].min()) if df_vis['time'].notna().any() else None
max_year = int(df_vis['time'].max()) if df_vis['time'].notna().any() else None

# Group by metric and origin
agg = df_vis.groupby(['value_variable_label', '2_variable_attribute_label'], as_index=False)['value'].sum(min_count=1)
agg['value'] = agg['value'].fillna(0)

# Define order for metrics (Arrivals vs Overnight stays)
# Mapping codes to labels if possible for consistent ordering
code_to_label = {}
if 'value_variable_code' in df_vis.columns:
    tmp = df_vis[['value_variable_code', 'value_variable_label']].dropna().drop_duplicates()
    code_to_label = dict(zip(tmp['value_variable_code'], tmp['value_variable_label']))

# GANK03: Gästeankünfte, GUEB03: Gästeübernachtungen
desired_code_order = ['GANK03', 'GUEB03']
x_order = [code_to_label[c] for c in desired_code_order if c in code_to_label]
if not x_order:
    x_order = sorted(agg['value_variable_label'].unique().tolist())

hue_order = [o for o in allowed_origins if o in agg['2_variable_attribute_label'].unique()]

# Calculate shares for annotation
if not agg.empty:
    totals = agg.groupby('value_variable_label')['value'].transform('sum')
    agg['share'] = np.where(totals > 0, agg['value'] / totals, 0.0)

# Plotting
plt.figure(figsize=(12, 7))
ax = sns.barplot(
    data=agg,
    x='value_variable_label',
    y='value',
    hue='2_variable_attribute_label',
    order=x_order,
    hue_order=hue_order,
    palette=['#2b7bba', '#f57f20']
)

# Formatting
ax.yaxis.set_major_formatter(FuncFormatter(lambda x, p: f"{int(x):,}".replace(',', ' ')))
ax.set_xlabel('Kennzahl')
ax.set_ylabel('Anzahl Personen')

# Title logic
title_period = f'{min_year}–{max_year}' if min_year and max_year and min_year != max_year else (str(max_year) if max_year else "")
title_text = 'Inland vs. Ausland nach Kennzahl'
if title_period:
    title_text += f' ({title_period})'
ax.set_title(title_text, pad=20)

ax.legend(title='Herkunft', loc='upper left', bbox_to_anchor=(1, 1))
ax.grid(axis='y', linestyle='--', alpha=0.4)

# Add percentage labels on bars
for p in ax.patches:
    if p.get_height() > 0:
        # Find the corresponding share from the dataframe
        # This is a simplified approach for bar labels
        ax.annotate(f'{p.get_height():,.0f}'.replace(',', ' '), 
                    (p.get_x() + p.get_width() / 2., p.get_height()), 
                    ha='center', va='center', 
                    xytext=(0, 9), 
                    textcoords='offset points',
                    fontsize=10)

plt.tight_layout()

# Save output
output_path = os.path.normpath(os.path.join(output_dir, '1_anteil_inland_ausland.png'))
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close()

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import os

# Set visual style
sns.set_theme(style="whitegrid", context="talk")

# The environment provides the data in 'data.csv'
# To handle the FileNotFoundError, we ensure the path is correct or provide a fallback
input_path = 'data.csv'
output_dir = './'
plot_path = os.path.join(output_dir, 'gaeste_ankuenfte_vs_uebernachtungen.png')

def load_data(path):
    if not os.path.exists(path):
        # If file is missing, we'll try to find any csv in the current directory
        csv_files = [f for f in os.listdir('.') if f.endswith('.csv')]
        if csv_files:
            path = csv_files[0]
        else:
            raise FileNotFoundError(f"No CSV file found. Expected {path}")
    
    # Try different separators
    for sep in [';', ',']:
        try:
            df = pd.read_csv(path, sep=sep, encoding='utf-8')
            if df.shape[1] > 1:
                return df
        except:
            continue
    return pd.read_csv(path, sep=None, engine='python', encoding='utf-8')

try:
    df = load_data(input_path)

    # Data Cleaning & Preparation
    # 1. Filter for 'Insgesamt' (Total) in '2_variable_attribute_label'
    label_col = '2_variable_attribute_label'
    if label_col in df.columns:
        # Use case-insensitive and stripped comparison
        df_filtered = df[df[label_col].astype(str).str.strip().str.lower() == 'insgesamt'].copy()
    else:
        df_filtered = df.copy()

    # 2. Convert time and value to numeric
    df_filtered['time'] = pd.to_numeric(df_filtered['time'], errors='coerce')
    df_filtered['value'] = pd.to_numeric(df_filtered['value'], errors='coerce')

    # 3. Identify the specific variables for Arrivals and Overnight Stays
    # We look for labels containing 'ankunft' and 'übernachtung'
    unique_labels = df_filtered['value_variable_label'].unique()
    target_labels = [l for l in unique_labels if any(x in str(l).lower() for x in ['ankunft', 'ankünf', 'übernacht', 'uebernacht'])]

    df_plot = df_filtered[df_filtered['value_variable_label'].isin(target_labels)].copy()
    df_plot = df_plot.dropna(subset=['time', 'value'])
    df_plot = df_plot.sort_values('time')

    # Visualization
    if not df_plot.empty:
        plt.figure(figsize=(12, 7))
        
        # Create the multi-line chart
        ax = sns.lineplot(
            data=df_plot, 
            x='time', 
            y='value', 
            hue='value_variable_label', 
            marker='o', 
            linewidth=2.5
        )

        # Formatting
        plt.title('Gästeankünfte vs. Übernachtungen (Insgesamt)', fontsize=16, pad=20)
        plt.xlabel('Jahr', fontsize=12)
        plt.ylabel('Anzahl', fontsize=12)
        
        # Ensure x-axis shows years clearly
        years = sorted(df_plot['time'].unique().astype(int))
        if len(years) > 0:
            plt.xticks(years)
        
        # Improve legend
        plt.legend(title='Kennzahl', bbox_to_anchor=(1.05, 1), loc='upper left', fontsize='small')
        
        plt.tight_layout()
        plt.savefig(plot_path, dpi=150, bbox_inches='tight')
        print(f"Plot successfully saved to {plot_path}")
    else:
        print("No data found matching the criteria 'Insgesamt' and relevant labels.")

except Exception as e:
    print(f"An error occurred: {e}")
