import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from matplotlib.ticker import FuncFormatter
import os

try:
    data_path = os.path.normpath('./src/resources/data/study\\103779\\dataset.csv')
    output_dir = os.path.normpath('./src/resources/output\\103779')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(
        data_path,
        sep=',',
        encoding='utf-8',
        na_values=['nan', 'NaN', 'NULL', 'None', '']
    )
    missing_pct = df.isna().mean().sort_values(ascending=False)
    numeric_cols = df.select_dtypes(include=[np.number]).columns.tolist()
    numeric_summary = df[numeric_cols].describe().T if len(numeric_cols) > 0 else pd.DataFrame()
    def clean_typ(s):
        if pd.isna(s):
            return 'Unbekannt'
        v = str(s).strip()
        if v == '':
            return 'Unbekannt'
        return v
    if 'Kulturdenkmaltyp' in df.columns:
        typ_series = df['Kulturdenkmaltyp'].apply(clean_typ)
    else:
        raise KeyError('Spalte "Kulturdenkmaltyp" nicht gefunden.')
    typ_counts = typ_series.value_counts(dropna=False)
    total = int(typ_counts.sum())
    typ_share = (typ_counts / total * 100.0).round(2)
    plot_df = pd.DataFrame({
        'Kulturdenkmaltyp': typ_counts.index,
        'Anzahl': typ_counts.values,
        'Anteil_%': typ_share.values
    })
    def wrap_label(text, width=16):
        s = str(text)
        words = s.split(' ')
        lines = []
        cur = ''
        for w in words:
            if len(cur) + len(w) + (1 if cur else 0) <= width:
                cur = (cur + ' ' + w).strip() if cur else w
            else:
                lines.append(cur)
                cur = w
        if cur:
            lines.append(cur)
        return '\n'.join(lines)
    plot_df = plot_df.sort_values('Anzahl', ascending=False).reset_index(drop=True)
    plot_df['Kulturdenkmaltyp_wrapped'] = plot_df['Kulturdenkmaltyp'].apply(lambda x: wrap_label(x, 18))
    sns.set_theme(style='whitegrid')
    plt.rcParams['axes.titlepad'] = 12
    plt.rcParams['figure.dpi'] = 150
    fig, ax = plt.subplots(figsize=(10, 6))
    palette = sns.color_palette('Set2', n_colors=len(plot_df))
    sns.barplot(
        data=plot_df,
        x='Kulturdenkmaltyp_wrapped',
        y='Anzahl',
        palette=palette,
        ax=ax
    )
    for i, row in plot_df.iterrows():
        ax.text(
            i,
            row['Anzahl'] + max(plot_df['Anzahl']) * 0.01,
            f"{int(row['Anzahl'])} ({row['Anteil_%']:.1f}%)",
            ha='center', va='bottom', fontsize=10
        )
    ax.set_xlabel('Typ des Kulturdenkmals')
    ax.set_ylabel(f'Anzahl der Einträge (n={total})')
    ax.set_title('Häufigste Typen von Kulturdenkmalen im Kreis Segeberg')
    ax.yaxis.set_major_formatter(FuncFormatter(lambda x, pos: f"{int(x):d}"))
    plt.xticks(rotation=0)
    plt.tight_layout()
    out_path = os.path.join(output_dir, '0_kulturdenkmaltyp_barchart.png')

import os
import re
import numpy as np
import pandas as pd

import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

sns.set_theme(style="whitegrid", context="talk")

data_path = r'./src/resources/data/study\103779\dataset.csv'
output_dir = r'./src/resources/output\103779'
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '2_begruendung_haeufigkeiten.png')

df = pd.read_csv(data_path, sep=',', encoding='utf-8')

missing_ratio = df.isna().mean()
num_desc = df.select_dtypes(include=[np.number]).describe().T
obj_counts = {col: df[col].nunique(dropna=True) for col in df.select_dtypes(include=['object', 'bool']).columns}

def split_reasons(x):
    if pd.isna(x):
        return []
    s = str(x).strip()
    if s == '':
        return []
    s_low = s.lower()
    if s_low in ['nan', 'none', 'alteintragung (aktualisierung vorgesehen)']:
        return []
    parts = re.split(r'[;,]', s)
    cleaned = []
    for p in parts:
        t = p.strip()
        if t == '':
            continue
        tl = t.lower()
        if tl in ['nan', 'none', 'alteintragung (aktualisierung vorgesehen)']:
            continue
        tl = re.sub(r'\s+', ' ', tl)
        tl = tl.strip(' .;:,')
        if tl == '':
            continue
        cleaned.append(tl.title())
    return list(dict.fromkeys(cleaned))

col_name = 'Begr_ndung'
if col_name not in df.columns:
    fig, ax = plt.subplots(figsize=(8, 4), dpi=150)
    ax.axis('off')
    ax.text(0.5, 0.5, 'Spalte "Begr_ndung" nicht gefunden.', ha='center', va='center', fontsize=12)
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
else:
    df['__reasons_list__'] = df[col_name].apply(split_reasons)
    n_mon_with_reason = df.loc[df['__reasons_list__'].apply(lambda v: isinstance(v, list) and len(v) > 0), 'Objektnummer'].nunique()
    exploded = df[['Objektnummer', '__reasons_list__']].explode('__reasons_list__')
    exploded = exploded[exploded['__reasons_list__'].notna()]

        plt.savefig(output_path, dpi=300, bbox_inches='tight')
        plt.close()
    else:
        counts = exploded.groupby('Begründung', as_index=False)['Objektnummer'].nunique()
        counts = counts.rename(columns={'Objektnummer': 'Anzahl'})
        counts['Prozent'] = counts['Anzahl'] / n_mon_with_reason * 100.0
