if (!require("tidyverse")) install.packages("tidyverse")
if (!require("scales")) install.packages("scales")

library(tidyverse)
library(scales)

df <- read.csv("./src/resources/data/study/103779/dataset.csv", sep = ",", encoding = "UTF-8", stringsAsFactors = FALSE)

missing_stats <- colSums(is.na(df) | df == "" | df == "nan") / nrow(df) * 100

df_clean <- df %>%
  mutate(Begr_ndung = ifelse(is.na(Begr_ndung) | Begr_ndung == "" | Begr_ndung == "nan", "Keine Angabe / Alteintragung", Begr_ndung))

reasons_split <- df_clean %>%
  separate_rows(Begr_ndung, sep = ",\\s*") %>%
  mutate(Begr_ndung = trimws(Begr_ndung)) %>%
  group_by(Begr_ndung) %>%
  summarise(Anzahl = n()) %>%
  arrange(desc(Anzahl))

dir.create("./src/resources/output/103779", recursive = TRUE, showWarnings = FALSE)

ggplot(reasons_split, aes(x = reorder(Begr_ndung, Anzahl), y = Anzahl, fill = (Begr_ndung == "Keine Angabe / Alteintragung"))) +
  geom_bar(stat = "identity", show.legend = FALSE) +
  geom_text(aes(label = Anzahl), hjust = -0.2, size = 3.5) +
  coord_flip() +
  scale_fill_manual(values = c("skyblue3", "tomato3")) +
  labs(
    title = "Häufigste Begründungen für den Denkmalwert",
    subtitle = "Kreis Segeberg (Stand 2021) - Mehrfachnennungen möglich",
    x = "Begründungskriterium",
    y = "Anzahl der Nennungen",
    caption = "Hinweis: 'Keine Angabe' umfasst oft Alteintragungen ohne explizite Neukatalogisierung."
  ) +
  theme_minimal(base_size = 12) +
  theme(
    plot.title = element_text(face = "bold", size = 14),
    axis.title = element_text(face = "italic"),
    panel.grid.minor = element_blank()
  ) +
  scale_y_continuous(expand = expansion(mult = c(0, 0.15)))

ggsave("./src/resources/output/103779/2_haeufigste_begruendungen.png", width = 10, height = 6, dpi = 300)
