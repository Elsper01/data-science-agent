import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

plt.switch_backend('Agg')

input_path = './src/resources/data/study/15539/dataset.csv'
output_path = './src/resources/output/15539/0_tourismus_trend.png'

# Behebung des UnicodeDecodeError durch Verwendung von 'latin-1' (ISO-8859-1), 
# da 'utf-8' bei deutschen Umlauten (wie in 'Gäste') oft fehlschlägt, wenn die Datei nicht UTF-8 kodiert ist.
try:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')
except Exception as e:
    # Fallback auf ISO-8859-1 falls latin-1 scheitert (meist identisch)
    df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1')

# Datenbereinigung
for col in ['value', 'time']:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='coerce')

df = df.dropna(subset=['time', 'value'])
df = df[(df['time'] >= 2014) & (df['time'] <= 2024)]

# Identifikation der Aggregat-Zeilen (Insgesamt), um Doppelzählungen zu vermeiden
if '2_variable_attribute_label' in df.columns:
    attr_label_norm = df['2_variable_attribute_label'].astype(str).str.strip().str.lower()
else:
    attr_label_norm = pd.Series([''] * len(df), index=df.index)

# Wir suchen nach Zeilen, die die Gesamtsumme darstellen
is_agg_attr = attr_label_norm.isin(['insgesamt', 'gesamt', 'total', 'summe', 'alle'])

if is_agg_attr.any():
    base = df[is_agg_attr].copy()
else:
    base = df.copy()

# Gruppierung für das Liniendiagramm
if 'value_variable_label' in base.columns:
    base['value_variable_label'] = base['value_variable_label'].astype(str).str.strip()
    plot_df = base.groupby(['time', 'value_variable_label'], as_index=False)['value'].sum()
else:
    plot_df = base.groupby(['time'], as_index=False)['value'].sum()
    plot_df['value_variable_label'] = 'Wert'

# Visualisierung
sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(12, 7))

# Farbschema definieren
kpi_pref_order = [
    'Gästeankünfte in Beherbergungsbetrieben',
    'Gästeübernachtungen in Beherbergungsbetrieben'
]
labels_present = plot_df['value_variable_label'].unique()
palette = sns.color_palette('tab10', len(labels_present))

sns.lineplot(
    data=plot_df, 
    x='time', 
    y='value', 
    hue='value_variable_label', 
    marker='o', 
    linewidth=2.5, 
    ax=ax
)

# X-Achse formatieren
years = sorted(plot_df['time'].unique().astype(int))
if years:
    ax.set_xticks(years)

# Pandemie-Hintergrund (2020-2021)

    ax.axvspan(2019.5, 2021.5, color='grey', alpha=0.1, label='Pandemie-Zeitraum')

ax.set_ylabel('Anzahl')
title_years = f"{min(years)}–{max(years)}" if years else "2014–2024"
ax.set_title(f'Entwicklung der Gästeankünfte und -übernachtungen ({title_years})')

# Legende optimieren
ax.legend(title='Kennzahl', loc='upper left', bbox_to_anchor=(1, 1))

try:

    fig.savefig(output_path, dpi=300, bbox_inches='tight')

finally:
    plt.close(fig)

import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

# Konfiguration der Pfade
base_data_dir = os.path.join("src", "resources", "data", "study", "15539")
in_path = os.path.join(base_data_dir, "dataset.csv")
out_dir = os.path.join("src", "resources", "output", "15539")
plot_filename = "1_uebernachtungen_inland_ausland.png"

os.makedirs(out_dir, exist_ok=True)

# Robuster Lade-Mechanismus mit verschiedenen Encodings
# Der Fehler '0xe4' deutet auf 'latin-1' (ISO-8859-1) oder 'cp1252' hin (ä, ö, ü).
df = None
encodings = ["utf-8", "latin-1", "cp1252", "iso-8859-1"]
separators = [";", ","]

for enc in encodings:
    for sep in separators:
        try:
            df = pd.read_csv(in_path, sep=sep, encoding=enc)
            # Teste ob das Laden erfolgreich war und Spalten vorhanden sind
            if len(df.columns) > 1:
                break
        except Exception:
            continue
    if df is not None and len(df.columns) > 1:
        break

if df is None:
    print(f"Fehler: Datei konnte mit keinem bekannten Encoding geladen werden.")
    sys.exit(1)

# Datenbereinigung
for col in df.select_dtypes(include=["object"]).columns:
    df[col] = df[col].astype(str).str.strip()
    df[col] = df[col].replace({"nan": np.nan, "NaN": np.nan, "None": np.nan, "": np.nan})

# Konvertierung numerischer Spalten
if "time" in df.columns:
    df["time"] = pd.to_numeric(df["time"], errors="coerce")
if "value" in df.columns:
    df["value"] = pd.to_numeric(df["value"], errors="coerce")

# Filter-Logik für "Übernachtungen"
label_col = "value_variable_label" if "value_variable_label" in df.columns else None
code_col = "value_variable_code" if "value_variable_code" in df.columns else None

mask_metric = pd.Series(False, index=df.index)
if code_col and df[code_col].astype(str).str.upper().eq("GUEB03").any():
    mask_metric = df[code_col].astype(str).str.upper().eq("GUEB03")
elif label_col:
    # Suche nach Übernachtungen (case-insensitive)
    mask_metric = df[label_col].astype(str).str.contains("bernachtung", case=False, na=False)

# Spalte für Herkunft (Inland/Ausland)
origin_col = "2_variable_attribute_label" if "2_variable_attribute_label" in df.columns else None

if origin_col is None or "time" not in df.columns or "value" not in df.columns:
    print("Fehler: Erforderliche Spalten (time, value, Herkunft) fehlen.")
    sys.exit(1)

# Daten für Plot vorbereiten
plot_df = df.loc[mask_metric, ["time", origin_col, "value"]].copy()
plot_df = plot_df.dropna(subset=["time", "value", origin_col])

# Filtern auf Inland/Ausland (Ausschluss von Summen/Insgesamt falls vorhanden)
valid_origins = ["Inland", "Ausland"]
plot_df = plot_df[plot_df[origin_col].str.contains("Inland|Ausland", case=False, na=False)]

if plot_df.empty:
    print("Fehler: Keine passenden Daten für Übernachtungen (Inland/Ausland) gefunden.")
    sys.exit(1)

# Aggregation
plot_df = plot_df.groupby(["time", origin_col], as_index=False)["value"].sum()

# Pivotierung für das gruppierte Balkendiagramm
wide = plot_df.pivot(index="time", columns=origin_col, values="value").fillna(0)
wide = wide.sort_index()

# Visualisierung
sns.set_theme(style="whitegrid", context="talk")
fig, ax = plt.subplots(figsize=(12, 7))

# Plotten
wide.plot(kind="bar", ax=ax, width=0.8, color=["#1f77b4", "#ff7f0e"])

plt.title("Übernachtungen: Inland vs. Ausland\nBad Berneck i. Fichtelgebirge", fontsize=16, pad=20)
plt.xlabel("Jahr", fontsize=12)
plt.ylabel("Anzahl der Übernachtungen", fontsize=12)
plt.xticks(rotation=0)
plt.legend(title="Herkunft", frameon=True)

# Tausendertrennzeichen für y-Achse (Deutsch: Punkt als Trenner)
ax.get_yaxis().set_major_formatter(FuncFormatter(lambda x, p: format(int(x), ',').replace(',', '.')))

plt.tight_layout()

# Speichern
save_path = os.path.join(out_dir, plot_filename)
plt.savefig(save_path, dpi=150, bbox_inches="tight")
plt.close()

print(f"Visualisierung erfolgreich unter {save_path} gespeichert.")

import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

output_dir = os.path.join('.', 'src', 'resources', 'output', '15539')
os.makedirs(output_dir, exist_ok=True)
plot_name = '2_vergleich_ankuenfte_uebernachtungen.png'
plot_path = os.path.join(output_dir, plot_name)

# Behebung des UnicodeDecodeError durch Testen verschiedener Encodings
data_path = os.path.join('.', 'src', 'resources', 'data', 'study', '15539', 'dataset.csv')
encodings_to_try = ['utf-8', 'latin1', 'iso-8859-1', 'cp1252']
df = None

for enc in encodings_to_try:
    try:
        df = pd.read_csv(data_path, sep=';', encoding=enc)
        break
    except (UnicodeDecodeError, Exception):
        continue

if df is None:
    print("Fehler: Die Datei konnte mit keinem der versuchten Encodings geladen werden.")
    sys.exit(1)

# Datenbereinigung
for c in df.columns:
    if df[c].dtype == 'object':
        df[c] = df[c].replace({'': np.nan, 'nan': np.nan})

df['time'] = pd.to_numeric(df.get('time'), errors='coerce').astype('Int64')
df['value'] = pd.to_numeric(df.get('value'), errors='coerce')

df_clean = df.dropna(subset=['time', 'value', 'value_variable_label']).copy()

# Filterung auf Gesamtwerte (falls vorhanden), um Doppelzählungen zu vermeiden
# Wir suchen nach 'insgesamt' in den Attribut-Labels
subset = df_clean.copy()
cond_total = subset['2_variable_attribute_label'].astype(str).str.lower().str.contains('insgesamt|gesamt|total', na=False)

if cond_total.any():
    subset_total = subset[cond_total]
else:
    subset_total = subset

# Gruppierung der Daten
grouped = subset_total.groupby(['time', 'value_variable_label'], as_index=False)['value'].sum()

# Mapping für schönere Legende
mapping = {
    'Gästeankünfte in Beherbergungsbetrieben': 'Ankünfte',
    'Gästeübernachtungen in Beherbergungsbetrieben': 'Übernachtungen'
}
grouped['Kennzahl'] = grouped['value_variable_label'].map(mapping).fillna(grouped['value_variable_label'])

# Plotting
sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(14, 8))

order_years = sorted(grouped['time'].unique())
palette = sns.color_palette('colorblind', n_colors=len(grouped['Kennzahl'].unique()))

sns.barplot(
    data=grouped, 
    x='time', 
    y='value', 
    hue='Kennzahl', 
    palette=palette, 
    ax=ax
)

# Achsen und Titel
ax.set_xlabel('Jahr')
ax.set_ylabel('Anzahl')

xmin = int(min(order_years)) if order_years else ""
xmax = int(max(order_years)) if order_years else ""
title_years = f" ({xmin}–{xmax})" if xmin != xmax else f" ({xmin})"

ax.set_title(f'Verhältnis Gästeankünfte zu Übernachtungen{title_years}\nBad Berneck i.Fichtelgebirge', pad=20)

# Tausendertrennpunkt für die Y-Achse
ax.yaxis.set_major_formatter(FuncFormatter(lambda x, p: f'{int(x):,}'.replace(',', '.')))

# Legende positionieren
ax.legend(title='Kennzahl', loc='upper left', bbox_to_anchor=(1, 1))

plt.tight_layout()

try:
    fig.savefig(plot_path, dpi=300, bbox_inches='tight', facecolor='white')
    plt.close(fig)
except Exception as e:
    print(f"Fehler beim Speichern des Plots: {e}")
    sys.exit(1)
