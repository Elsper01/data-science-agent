# Bibliotheken laden
library(ggplot2)
library(dplyr)
library(readr)

# Da der Pfad in der lokalen Umgebung des Nutzers nicht gefunden wurde, 
# nutzen wir den Standardnamen 'dataset.csv', der in dieser Umgebung erwartet wird.
file_path <- "dataset.csv"

# Überprüfung, ob die Datei existiert, bevor wir fortfahren
if (!file.exists(file_path)) {
  stop(paste("Die Datei", file_path, "wurde im aktuellen Verzeichnis nicht gefunden."))
}

# Daten laden
# Wir nutzen read_csv, da die meisten Standard-Datasets in dieser Umgebung Komma-separiert sind.
# Falls es Semikolons sind, fängt read_delim das ab.
df <- read_delim(file_path, delim = NULL, locale = locale(encoding = "UTF-8", decimal_mark = ","), show_col_types = FALSE)

# Spaltennamen bereinigen (Leerzeichen durch Punkte ersetzen etc.)
colnames(df) <- make.names(colnames(df))

# Identifikation der Zielspalten (flexibel gegenüber Namensänderungen durch make.names)
# Wir suchen nach 'Seiten_pro_Besuch' und 'Bounce_Rate'
x_col <- grep("Seiten_pro_Besuch", colnames(df), value = TRUE)[1]
y_col <- grep("Bounce_Rate", colnames(df), value = TRUE)[1]

# Falls die exakten Namen nicht gefunden werden, suchen wir nach Teilbegriffen
if (is.na(x_col)) x_col <- grep("Seiten", colnames(df), ignore.case = TRUE, value = TRUE)[1]
if (is.na(y_col)) y_col <- grep("Bounce", colnames(df), ignore.case = TRUE, value = TRUE)[1]

# Datenaufbereitung: Sicherstellen, dass die Werte numerisch sind
df_plot <- df %>%
  filter(!is.na(!!sym(x_col)), !is.na(!!sym(y_col))) %>%
  mutate(
    x_val = as.numeric(as.character(!!sym(x_col))),
    y_val = as.numeric(as.character(!!sym(y_col)))
  ) %>%
  filter(!is.na(x_val), !is.na(y_val))

# Streudiagramm erstellen
plot <- ggplot(df_plot, aes(x = x_val, y = y_val)) +
  geom_point(color = "#3498db", alpha = 0.7, size = 3) +
  geom_smooth(method = "lm", color = "#e74c3c", fill = "#ecf0f1", se = TRUE) +
  labs(
    title = "Zusammenhang: Seitenaufrufe vs. Absprungrate",
    subtitle = "Analyse der Nutzerinteraktion (Korrelation)",
    x = "Seiten pro Besuch",
    y = "Absprungrate (%)",
    caption = "Datenquelle: dataset.csv"
  ) +
  theme_minimal(base_size = 14) +
  theme(
    plot.title = element_text(face = "bold", size = 16),
    panel.grid.minor = element_blank()
  )

# Plot speichern
ggsave("1_streudiagramm.png", plot = plot, width = 10, height = 7, dpi = 300)

# Erfolgskontrolle
print("Visualisierung '1_streudiagramm.png' wurde erfolgreich erstellt.")

# Daten einlesen
# Korrektur des Pfades (Backslashes in R müssen oft escaped werden oder durch Forward-Slashes ersetzt werden)
file_path <- './src/resources/data/study/32830/dataset.csv'
data <- read.csv(file_path, sep = ',', encoding = 'UTF-8', stringsAsFactors = FALSE)

# Datenvorbereitung
data_clean <- data %>%
  filter(!is.na(Datum) & Datum != "") %>%
  mutate(
    # Datum konvertieren (Annahme: Format ist Tag.Monat.Jahr oder ähnlich, dmy ist robust)
    Datum = dmy(Datum),
    # Bereinigung der Besuche (Entfernen von Tausendertrennzeichen falls vorhanden)
    Besuche = as.numeric(gsub("[^0-9]", "", as.character(Besuche))),
    # Anteil_Besuche_mit_Cookies von Prozent-String in Dezimalzahl umwandeln
    Anteil_Besuche_mit_Cookies = as.numeric(gsub(",", ".", gsub("%", "", Anteil_Besuche_mit_Cookies))) / 100
  ) %>%
  filter(!is.na(Datum)) %>%
  arrange(Datum)

# Skalierungsfaktor für die sekundäre Achse berechnen
max_besuche <- max(data_clean$Besuche, na.rm = TRUE)
if(is.infinite(max_besuche) || max_besuche == 0) max_besuche <- 1

# Visualisierung erstellen
p <- ggplot(data_clean, aes(x = Datum)) +
  # Balken für die absolute Anzahl der Besuche
  geom_bar(aes(y = Besuche), stat = "identity", fill = "#3498db", alpha = 0.7) +
  # Linie für den Anteil der Cookies (skaliert auf die primäre Y-Achse)
  geom_line(aes(y = Anteil_Besuche_mit_Cookies * max_besuche, group = 1), 
            color = "#e74c3c", size = 1.2) +
  geom_point(aes(y = Anteil_Besuche_mit_Cookies * max_besuche), 
             color = "#e74c3c", size = 2) +
  # Primäre und sekundäre Achse definieren
  scale_y_continuous(
    name = "Anzahl der Besuche",
    labels = comma_format(big.mark = "."),
    sec.axis = sec_axis(~ . / max_besuche, 
                        name = "Anteil Besuche mit Cookies (%)", 
                        labels = percent_format(accuracy = 1))
  ) +
  scale_x_date(date_labels = "%d.%m.", date_breaks = "1 day") +
  labs(
    title = "Besuche vs. Cookie-Einwilligung pro Tag",
    subtitle = "Vergleich der Gesamtzahl der Besuche mit der Akzeptanzrate",
    x = "Datum",
    caption = "Datenquelle: kreis-kleve.de Web-Analytics September 2024",
    color = "Legende"
  ) +
  theme_minimal() +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1),
    panel.grid.minor = element_blank(),
    plot.title = element_text(face = "bold", size = 14),
    axis.title.y.right = element_text(color = "#e74c3c"),
    axis.text.y.right = element_text(color = "#e74c3c")
  )

# Speicherpfad sicherstellen und Diagramm speichern

output_path <- paste0(output_dir, '2_besuche_und_cookies.png')

ggsave(output_path, plot = p, width = 10, height = 6, dpi = 300)
