import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os
import sys
from matplotlib.ticker import FuncFormatter

def solve():
    try:
        # Pfade definieren
        data_path = os.path.normpath(r"./src/resources/data/study/15653/dataset.csv")
        output_dir = os.path.normpath(r"./src/resources/output/15653")
        os.makedirs(output_dir, exist_ok=True)

        # Fehlerbehebung: 'utf-8' codec error. 
        # Deutsche CSV-Dateien nutzen oft 'latin-1' oder 'iso-8859-1' (wegen Umlauten wie 'ü').
        # Wir versuchen latin-1, da utf-8 laut Fehlermeldung fehlschlug.
        try:
            df = pd.read_csv(data_path, sep=';', encoding='utf-8', dtype=str)
        except UnicodeDecodeError:
            df = pd.read_csv(data_path, sep=';', encoding='latin-1', dtype=str)

        # Whitespace entfernen
        for c in df.columns:
            df[c] = df[c].str.strip()

        # Datentypen konvertieren
        if 'time' in df.columns:
            df['time'] = pd.to_numeric(df['time'], errors='coerce')

        if 'value' in df.columns:
            # Komma durch Punkt ersetzen für deutsche Zahlenformate
            df['value'] = pd.to_numeric(df['value'].str.replace(',', '.'), errors='coerce')

        # Filterung auf Übernachtungen (GUEB03 oder Label-Suche)
        mask_uebernachtungen = pd.Series(False, index=df.index)
        if 'value_variable_code' in df.columns:
            mask_uebernachtungen = mask_uebernachtungen | (df['value_variable_code'].str.upper() == 'GUEB03')
        if 'value_variable_label' in df.columns:
            mask_uebernachtungen = mask_uebernachtungen | df['value_variable_label'].str.contains('übernachtung', case=False, na=False)

        # Validierung der benötigten Spalten
        required_cols = ['2_variable_attribute_label', 'time', 'value']
        for col in required_cols:
            if col not in df.columns:
                raise ValueError(f"Spalte '{col}' fehlt im Datensatz.")

        # Daten filtern
        df_ue = df[mask_uebernachtungen].copy()
        # Normalisierung der Labels für den Vergleich
        df_ue['2_variable_attribute_label'] = df_ue['2_variable_attribute_label'].str.title()
        df_ue = df_ue[df_ue['2_variable_attribute_label'].isin(['Inland', 'Ausland'])]
        df_ue = df_ue.dropna(subset=['time', 'value'])

        if df_ue.empty:
            raise ValueError('Keine Daten für Übernachtungen mit Inland/Ausland gefunden.')

        # Pivotierung für das Diagramm
        pivot = df_ue.pivot_table(
            index='time', 
            columns='2_variable_attribute_label', 
            values='value', 
            aggfunc='sum'
        ).sort_index()

        # Sicherstellen, dass beide Spalten existieren
        for col in ['Inland', 'Ausland']:
            if col not in pivot.columns:
                pivot[col] = 0
        
        pivot = pivot[['Inland', 'Ausland']]

        # Visualisierung
        sns.set_theme(style='whitegrid', context='talk')
        years = pivot.index.to_numpy()
        inland = pivot['Inland'].to_numpy()
        ausland = pivot['Ausland'].to_numpy()
        
        x = np.arange(len(years))
        width = 0.35

        fig, ax = plt.subplots(figsize=(12, 7))
        ax.bar(x - width/2, inland, width, label='Inland', color='#1f77b4')
        ax.bar(x + width/2, ausland, width, label='Ausland', color='#ff7f0e')

        # Achsenbeschriftung
        ax.set_xlabel('Jahr')
        ax.set_ylabel('Anzahl der Übernachtungen')
        ax.set_title('Übernachtungen nach Herkunft: Inland vs. Ausland')
        ax.set_xticks(x)
        ax.set_xticklabels([str(int(y)) for y in years])
        ax.legend(title='Herkunft')

        # Tausendertrennzeichen (Punkt)
        def de_format(x, pos):
            return f"{int(x):,}".replace(',', '.')
        ax.yaxis.set_major_formatter(FuncFormatter(de_format))

        plt.tight_layout()
        
        # Speichern
        plot_path = os.path.join(output_dir, 'uebernachtungen_inland_ausland.png')
        fig.savefig(plot_path, dpi=150)
        plt.close(fig)

        # CSV Export der Anteile für EDA Zwecke
        totals = pivot.sum(axis=1).replace(0, np.nan)
        shares = pivot.div(totals, axis=0)
        shares.to_csv(os.path.join(output_dir, 'anteile_herkunft.csv'), encoding='utf-8')

    except Exception as e:
        print(f"Fehler: {str(e)}")
        sys.exit(1)

if __name__ == "__main__":
    solve()
