    datum = ''

def to_numeric_de(series):
    s = series.astype(str).str.strip()
    s = s.str.replace('\u00a0', '', regex=False)
    s = s.str.replace('−', '-', regex=False)
    s = s.str.replace('[<>~≤≥*]', '', regex=True)
    s = s.str.replace('n.n.', '', regex=False)
    s = s.str.replace('n.n', '', regex=False)
    s = s.str.replace(',', '.', regex=False)
    s = s.replace({'': np.nan})
    return pd.to_numeric(s, errors='coerce')

df['Ergebnis_num'] = to_numeric_de(df['Ergebnis'])

eda_counts = pd.DataFrame({
    'gesamt': [len(df)],

    'anzahl_nicht_numerisch': [df['Ergebnis_num'].isna().sum()]
})
eda_counts.to_csv(os.path.join(eda_dir, 'anzahl_numerisch.csv'), index=False, encoding='utf-8')
