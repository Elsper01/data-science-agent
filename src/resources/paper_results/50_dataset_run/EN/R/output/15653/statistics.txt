# Read data with robust handling of delimiters and encoding
raw <- readr::read_delim(
  file = in_path,
  delim = ";",
  locale = locale(encoding = "UTF-8"),
  na = c("", "NA", "NaN"),
  show_col_types = FALSE
)

# Data Cleaning
df <- raw %>%
  mutate(
    time = suppressWarnings(as.numeric(time)),
    value = suppressWarnings(as.numeric(value)),
    `2_variable_attribute_label` = as.character(`2_variable_attribute_label`),
    value_variable_label = as.character(value_variable_label)
  ) %>%
  filter(!is.na(time), !is.na(value))

# EDA: Missingness
missing_tbl <- tibble(
  column = names(raw),
  missing_count = colSums(is.na(raw)),
  total = nrow(raw),
  missing_prop = colMeans(is.na(raw))
) %>% arrange(desc(missing_prop))
readr::write_csv(missing_tbl, file.path(out_dir, "eda_missingness.csv"))

# Filter for the specific metrics requested
metrics <- c("Gästeankünfte in Beherbergungsbetrieben", "Gästeübernachtungen in Beherbergungsbetrieben")
plot_df <- df %>% 
  filter(value_variable_label %in% metrics)

# Filter for "Insgesamt" (Total) if available to avoid double counting
if ("2_variable_attribute_label" %in% names(plot_df)) {
  if (any(plot_df$`2_variable_attribute_label` == "Insgesamt", na.rm = TRUE)) {
    plot_df <- plot_df %>% filter(`2_variable_attribute_label` == "Insgesamt")
  }
}

# Final check: if plot_df is empty, the script should not crash on seq()
if (nrow(plot_df) > 0) {
  
  plot_df <- plot_df %>% mutate(
    value_variable_label = factor(
      value_variable_label,
      levels = metrics,
      labels = c("Gästeankünfte", "Gästeübernachtungen")
    )
  )

  yr_min <- min(plot_df$time, na.rm = TRUE)
  yr_max <- max(plot_df$time, na.rm = TRUE)
  
  # Ensure yr_min and yr_max are finite for seq()
  if (is.finite(yr_min) && is.finite(yr_max)) {
    x_breaks <- seq(yr_min, yr_max, by = 1)
  } else {
    x_breaks <- waiver()
  }

  p <- ggplot(plot_df, aes(x = time, y = value, color = value_variable_label)) +
    geom_line(linewidth = 1.2, alpha = 0.8) +
    geom_point(size = 2.5) +
    scale_x_continuous(breaks = x_breaks) +
    scale_y_continuous(labels = scales::label_number(big.mark = ".", decimal.mark = ",")) +
    scale_color_manual(values = c("#2E86AB", "#D62828"), name = "Kennzahl") +
    labs(
      title = "Entwicklung der Gästeankünfte und -übernachtungen in Rettenberg",
      subtitle = paste0("Zeitraum: ", yr_min, " bis ", yr_max),
      x = "Jahr",
      y = "Anzahl",
      caption = "Quelle: Monatserhebung im Tourismus; Aggregat: Insgesamt"
    ) +
    theme_minimal(base_size = 12) +
    theme(
      legend.position = "bottom",
      plot.title = element_text(face = "bold", size = 14),
      panel.grid.minor = element_blank()
    )

  outfile <- file.path(out_dir, "0_trend_gaeste_rettenberg.png")
  ggplot2::ggsave(outfile, plot = p, width = 10, height = 6, dpi = 300)
  
  # Summary statistics for output
  summary_tbl <- plot_df %>%
    group_by(value_variable_label) %>%

    )
  readr::write_csv(summary_tbl, file.path(out_dir, "eda_value_summary.csv"))

  warning("No data available for the specified metrics and filters.")
