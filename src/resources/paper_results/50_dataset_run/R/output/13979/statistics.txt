# Hilfsfunktion zum sicheren Trimmen (vermeidet UTF-8 Fehler bei binären/unbekannten Zeichen)
safe_trim <- function(x) {
  if (!is.character(x)) return(x)
  # Entferne nur Standard-Whitespace und ignoriere Fehler bei ungültigen Sequenzen
  iconv(x, "UTF-8", "UTF-8", sub = "") |> stringr::str_trim()
}

    total_n = nrow(df),

    missing_n = sum(is.na(df[[column]]) | (is.character(df[[column]]) & safe_trim(df[[column]]) == "")),
