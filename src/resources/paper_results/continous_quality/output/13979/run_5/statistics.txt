import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import MaxNLocator
from pathlib import Path
import sys

try:
    data_path = Path('./src/resources/data/study') / '13979' / 'dataset.csv'
    
    # The error 'utf-8' codec can't decode byte 0xfc (ü) suggests the file is likely encoded in ISO-8859-1 or cp1252.
    # We try 'latin-1' which is a common encoding for German datasets containing umlauts.
    try:
        df = pd.read_csv(data_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(data_path, sep=';', encoding='latin-1')

    # Clean column names and data
    for c in ['time', 'value']:
        if c in df.columns:
            df[c] = pd.to_numeric(df[c], errors='coerce')
    
    label_col = '2_variable_attribute_label'
    time_col = 'time'
    value_col = 'value'
    
    # Filter for 'Insgesamt' (Total)
    # Using case-insensitive and strip to be robust
    if label_col in df.columns:
        df_tot = df[df[label_col].astype(str).str.strip().str.lower() == 'insgesamt'].copy()
    else:
        df_tot = pd.DataFrame(columns=df.columns)

    # Group by time to ensure unique points per year
    totals = df_tot.groupby(time_col, as_index=False)[value_col].sum(min_count=1)
    
    # Fallback if filter returned nothing
    if totals.empty or totals[value_col].isna().all():
        totals = df.groupby(time_col, as_index=False)[value_col].sum(min_count=1)
    
    totals = totals.dropna(subset=[time_col, value_col]).sort_values(time_col).reset_index(drop=True)
    
    if totals.empty:
        print('Fehler: Keine gültigen Zeit- und Wertdaten für die Visualisierung verfügbar.')
        sys.exit(1)

    years = totals[time_col].astype(int).values
    vals = totals[value_col].values

    # Visualization
    sns.set_theme(style='whitegrid', context='talk')
    fig, ax = plt.subplots(figsize=(11, 6))
    
    color_line = sns.color_palette('deep')[0]
    color_trend = sns.color_palette('deep')[3]
    
    ax.plot(years, vals, marker='o', linewidth=2.5, color=color_line, label='Insgesamt')
    
    # Add trend line if enough data points
    if len(years) >= 2 and np.isfinite(vals).all():
        z = np.polyfit(years, vals, 1)
        p = np.poly1d(z)
        ax.plot(years, p(years), linestyle='--', linewidth=1.8, color=color_trend, label='Linearer Trend')

    ax.set_xlabel('Jahr')
    ax.set_ylabel('Anzahl fertiggestellter Wohnungen')
    ax.set_title('Fertiggestellte Wohnungen in Sünching: Entwicklung insgesamt (2014–2024)')
    
    ax.xaxis.set_major_locator(MaxNLocator(integer=True))
    ax.yaxis.set_major_locator(MaxNLocator(integer=True))
    ax.set_xticks(years)

    # Statistics for the text box
    start_year, end_year = int(years[0]), int(years[-1])
    start_val, end_val = float(vals[0]), float(vals[-1])
    n_years = end_year - start_year
    change_abs = end_val - start_val
    
    # Correlation
    r = np.corrcoef(years, vals)[0, 1] if len(years) >= 2 else np.nan
    
    stats_text = (
        f'Zeitraum: {start_year}–{end_year}\n'
        f'Startwert: {int(start_val)}\n'
        f'Endwert: {int(end_val)}\n'
        f'Gesamtänderung: {int(change_abs):+d}\n'
        f'Korrelation (r): {r:.2f}' if np.isfinite(r) else 'Korrelation: n/a'
    )
    
    ax.text(0.02, 0.95, stats_text, transform=ax.transAxes, ha='left', va='top', 
            fontsize=10, bbox=dict(boxstyle='round,pad=0.5', facecolor='white', alpha=0.8, edgecolor='#cccccc'))

    # Annotate last point
    ax.annotate(f'{int(vals[-1])}', xy=(years[-1], vals[-1]), xytext=(5, 5), 
                textcoords='offset points', fontsize=11, color=color_line, weight='bold')

    ax.legend(frameon=True, loc='upper right')
    fig.text(0.01, 0.01, 'Quelle: Statistik der Baufertigstellungen; Darstellung: Eigene Berechnung', fontsize=9, alpha=0.7)
    
    plt.tight_layout()
    
    out_dir = Path('./src/resources/output') / '13979' / 'run_5'
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / '0_zeitreihe_wohnungen_gesamt.png'
    plt.savefig(out_path, dpi=300, bbox_inches='tight')
    plt.close(fig)

except Exception as e:

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# File paths
file_path = r'./src/resources/data/study\13979\dataset.csv'
output_path = r'./src/resources/output\13979\run_5/1_verteilung_raumzahl.png'

# Ensure output directory exists
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# The error was a UnicodeDecodeError with utf-8. 
# Common encodings for German/European CSVs are 'latin-1' or 'iso-8859-1'.
try:
    df = pd.read_csv(file_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(file_path, sep=';', encoding='latin-1')

# Clean and prepare data
df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for room count variables, excluding 'Insgesamt' (Total)
# Based on the goal: '2_variable_attribute_label' (excluding 'Insgesamt')
# We look for labels related to "Raum" (Rooms)
filt = (
    (df['2_variable_label'].astype(str).str.contains('Raumzahl', case=False, na=False)) &
    (~df['2_variable_attribute_label'].astype(str).str.contains('Insgesamt', case=False, na=False)) &
    (~df['value'].isna())
)

df_f = df.loc[filt].copy()

# Helper to extract numeric room count for sorting
def extract_room_num(s):
    s = str(s)
    digits = ''.join([c for c in s if c.isdigit()])
    return int(digits) if digits != '' else 999

df_f['rooms_num'] = df_f['2_variable_attribute_label'].apply(extract_room_num)

# Aggregate sum of values per room category
agg = df_f.groupby(['2_variable_attribute_label', 'rooms_num'], as_index=False)['value'].sum()
agg = agg.sort_values('rooms_num')

# Visualization
sns.set_theme(style='whitegrid')
plt.figure(figsize=(10, 6))

palette = sns.color_palette('viridis', n_colors=len(agg))
ax = sns.barplot(data=agg, x='2_variable_attribute_label', y='value', palette=palette)

# Add labels and title
ax.set_xlabel('Anzahl der Räume')
ax.set_ylabel('Summe der fertiggestellten Wohnungen')
ax.set_title('Verteilung der Wohnungsgrößen nach Raumzahl (Gesamtzeitraum)')

# Add data labels on top of bars
total_val = agg['value'].sum()
for i, p in enumerate(ax.patches):
    height = p.get_height()
    if not np.isnan(height) and height > 0:
        percentage = (height / total_val) * 100 if total_val > 0 else 0
        ax.annotate(f'{int(height)}\n({percentage:.1f}%)', 
                    (p.get_x() + p.get_width() / 2., height), 
                    ha='center', va='bottom', fontsize=9, xytext=(0, 5),
                    textcoords='offset points')

# Adjust layout and save
plt.xticks(rotation=15)
plt.tight_layout()
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close()

import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# Set plotting style
plt.rcParams["axes.unicode_minus"] = False
sns.set_theme(style="whitegrid", context="talk")

# File paths
raw_input_path = r"./src/resources/data/study/13979/dataset.csv"
input_path = raw_input_path.replace("\\", os.sep)
output_dir = os.path.join(".", "src", "resources", "output", "13979", "run_5")
os.makedirs(output_dir, exist_ok=True)

# The error 'utf-8' codec can't decode byte 0xfc suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# (0xfc is 'ü' in latin-1, common in German datasets).
try:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')
except Exception as e:
    # Fallback to utf-8 if latin-1 fails
    try:
        df = pd.read_csv(input_path, sep=';', encoding='utf-8')
    except Exception as e2:
        print(f"Failed to read CSV: {e2}")
        sys.exit(1)

# Data Cleaning
for col in df.columns:
    if df[col].dtype == object:
        df[col] = df[col].replace({"": np.nan})

# Convert numeric columns
if 'time' in df.columns:
    df['time'] = pd.to_numeric(df['time'], errors='coerce')
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# EDA Exports
missing_ratio = df.isna().mean().to_frame(name='anteil_missing')
num_df = df.select_dtypes(include=[np.number])
num_desc = num_df.describe().T
num_corr = num_df.corr(numeric_only=True)

try:
    missing_ratio.to_csv(os.path.join(output_dir, '2_eda_missing_values.csv'), encoding='utf-8')
    num_desc.to_csv(os.path.join(output_dir, '2_eda_numeric_summary.csv'), encoding='utf-8')
    num_corr.to_csv(os.path.join(output_dir, '2_eda_numeric_correlation.csv'), encoding='utf-8')
except Exception:
    pass

# Visualization Logic
label_col = '2_variable_attribute_label'
code_col = '2_variable_attribute_code'

# Filter data for the stacked bar chart
# Exclude 'Insgesamt' (Total) to show the composition
cat_df = df.copy()
cat_df = cat_df[cat_df[label_col].notna()]
cat_df = cat_df[cat_df[label_col].str.contains('Insgesamt') == False]
cat_df = cat_df[['time', label_col, code_col, 'value']].copy()
cat_df = cat_df.dropna(subset=['time', 'value'])

# Group and pivot for plotting
cat_df_grouped = cat_df.groupby(['time', label_col], dropna=False, as_index=False)['value'].sum()
pivot = cat_df_grouped.pivot_table(index='time', columns=label_col, values='value', aggfunc='sum').fillna(0)

# Sort columns based on the code (RAUM-1 to RAUM-7UM)
code_map = df[[label_col, code_col]].dropna().drop_duplicates().set_index(label_col)[code_col].to_dict()
def get_sort_key(label):
    code = str(code_map.get(label, ''))
    if 'RAUM-7UM' in code: return 7
    if 'RAUM-' in code:
        try: return int(code.split('-')[1])
        except: return 99
    return 99

ordered_cols = sorted(pivot.columns.tolist(), key=get_sort_key)
pivot = pivot[ordered_cols]

# Ensure time is sorted
pivot = pivot.sort_index()

# Plotting
fig, ax = plt.subplots(figsize=(14, 8))
colors = sns.color_palette('viridis', n_colors=len(pivot.columns))

pivot.plot(kind='bar', stacked=True, ax=ax, color=colors, edgecolor='white', width=0.8)

ax.set_xlabel('Jahr', fontsize=14)
ax.set_ylabel('Anzahl fertiggestellter Wohnungen', fontsize=14)
ax.set_title('Entwicklung der Wohnungsgrößen (Zimmeranzahl) über die Zeit', fontsize=16, pad=20)
ax.legend(title='Raumzahl', bbox_to_anchor=(1.02, 1), loc='upper left', borderaxespad=0.)
ax.grid(axis='y', linestyle='--', alpha=0.6)

plt.xticks(rotation=0)
fig.tight_layout()

# Save plot
plot_path = os.path.join(output_dir, '2_wohnungsraum_stacked_barchart.png')
plt.savefig(plot_path, dpi=300, bbox_inches='tight')
plt.close(fig)

# Totals check for EDA
try:
    total_rows = df[df[label_col].str.contains('Insgesamt', na=False)][['time', 'value']].dropna(subset=['time'])
    cat_sums = cat_df_grouped.groupby('time', as_index=False)['value'].sum().rename(columns={'value': 'sum_kategorien'})
    totals_check = pd.merge(total_rows, cat_sums, on='time', how='outer')
    totals_check.sort_values('time').to_csv(os.path.join(output_dir, '2_eda_totals_check.csv'), index=False, encoding='utf-8')
except Exception:
    pass
