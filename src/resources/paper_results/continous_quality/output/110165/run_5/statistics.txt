import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# The error log indicates two issues:
# 1. FileNotFoundError: 'data.csv' was not found.
# 2. UnicodeDecodeError: 'utf-8' failed to decode byte 0xfc (likely 'ü' in German).
# We will try 'latin-1' or 'iso-8859-1' which are common for German datasets.

file_path = 'data.csv'

def load_data(path):
    encodings = ['utf-8', 'latin-1', 'iso-8859-1', 'cp1252']
    for enc in encodings:
        try:
            return pd.read_csv(path, sep=',', encoding=enc)
        except (UnicodeDecodeError, FileNotFoundError):
            continue
    # If comma fails, try semicolon (common in European CSVs)
    for enc in encodings:
        try:
            return pd.read_csv(path, sep=';', encoding=enc)
        except:
            continue
    return None

df = load_data(file_path)

if df is None:
    # Fallback path from previous attempt
    fallback_path = './src/resources/data/study/110165/dataset.csv'
    df = load_data(fallback_path)

if df is None:
    raise FileNotFoundError("Could not load the dataset with available encodings or paths.")

# Data Cleaning
# Standardize column names to handle potential encoding issues in headers
df.columns = [c.strip() for c in df.columns]

col = 'Kulturdenkmaltyp'
if col not in df.columns:
    # Case-insensitive search
    potential_cols = [c for c in df.columns if col.lower() in c.lower()]
    if potential_cols:
        col = potential_cols[0]
    else:
        raise ValueError(f"Column '{col}' not found. Available: {df.columns.tolist()}")

# Clean the target column
df[col] = df[col].astype(str).str.strip()
# Replace common null-like strings
null_values = ['', 'None', 'NaN', 'nan', 'nan', 'NULL', 'n/a']
df.loc[df[col].isin(null_values), col] = 'Unbekannt'

# Prepare data for visualization
counts = df[col].value_counts().sort_values(ascending=True)

# Visualization
sns.set_theme(style='whitegrid')

# Dynamic figure size
n_cat = len(counts)
fig_height = max(6, n_cat * 0.4)
plt.figure(figsize=(12, fig_height))

# Create horizontal bar chart
colors = sns.color_palette('viridis', n_colors=n_cat)
ax = counts.plot(kind='barh', color=colors)

# Add labels and title
plt.title(f'Häufigkeit der Kulturdenkmaltypen (Gesamt: {len(df)})', fontsize=16, pad=20)
plt.xlabel('Anzahl der Denkmale', fontsize=12)
plt.ylabel('Kulturdenkmaltyp', fontsize=12)

# Add value labels on bars
for i, v in enumerate(counts.values):
    ax.text(v + (counts.max() * 0.005), i, str(int(v)), va='center', fontsize=10)

plt.tight_layout()

# Save output
out_dir = './src/resources/output/110165/run_5'
os.makedirs(out_dir, exist_ok=True)
out_path = os.path.join(out_dir, '1_kulturdenkmaltyp_haeufigkeit.png')
plt.savefig(out_path, dpi=300, bbox_inches='tight')
plt.show()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Set visual style
sns.set_theme(style='whitegrid', context='talk', font='DejaVu Sans')

# Paths
input_path = os.path.normpath('./src/resources/data/study/110165/dataset.csv')
output_dir = os.path.normpath('./src/resources/output/110165/run_5')
os.makedirs(output_dir, exist_ok=True)

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xfc' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German data with umlauts like 'ü').
try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8', low_memory=False)
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=',', encoding='latin-1', low_memory=False)

# Basic EDA exports
missing_rates = df.isna().mean().reset_index()
missing_rates.columns = ['Spalte', 'Anteil_Fehlend']
missing_rates = missing_rates.sort_values('Anteil_Fehlend', ascending=False)
missing_rates.to_csv(os.path.join(output_dir, 'eda_missing_rates.csv'), index=False, encoding='utf-8')

numeric_df = df.select_dtypes(include=[np.number])
if numeric_df.shape[1] > 0:
    numeric_summary = numeric_df.describe().T.reset_index()
    numeric_summary = numeric_summary.rename(columns={'index': 'Spalte'})
    numeric_summary.to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'), index=False, encoding='utf-8')
    corr_df = numeric_df.corr(numeric_only=True)
    corr_df.to_csv(os.path.join(output_dir, 'eda_numeric_correlations.csv'), encoding='utf-8')

# Visualization: Primary legal justifications ('Begr_ndung')
col = 'Begr_ndung'
if col in df.columns:
    # Clean and split the justifications (often semicolon or comma separated)
    s = df[col].dropna().astype(str)
    # Split by common delimiters and clean whitespace
    tokens = s.str.split(r'[;,]')
    tokens = tokens.explode().str.strip()
    tokens = tokens[tokens.ne('')]
    
    # Standardize casing (Title Case) for better grouping
    tokens = tokens.str.replace(r'\s+', ' ', regex=True).str.title()
    
    # Get top 10
    value_counts = tokens.value_counts()
    top10 = value_counts.head(10).reset_index()
    top10.columns = ['Begründung', 'Anzahl']
    top10 = top10.sort_values('Anzahl', ascending=True)

    # Plotting
    plt.figure(figsize=(12, 8))
    ax = sns.barplot(data=top10, x='Anzahl', y='Begründung', palette='viridis')
    
    ax.set_title('Top 10 Legal Justifications for Monument Protection', pad=20)
    ax.set_xlabel('Frequency of Mention')
    ax.set_ylabel('Justification Type')

    # Add value labels
    max_val = top10['Anzahl'].max() if not top10.empty else 0
    for i, v in enumerate(top10['Anzahl']):
        ax.text(v + (max_val * 0.01), i, str(int(v)), va='center', fontweight='bold')

    # Add context caption
    total_records = len(df)
    valid_records = df[col].notna().sum()
    caption = f'Data source: {total_records} total records ({valid_records} with justifications). Multiple reasons per site are counted individually.'
    plt.figtext(0.5, 0.01, caption, ha='center', fontsize=10, style='italic')

    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, '2_begruendung_top10.png'), dpi=300, bbox_inches='tight')
    plt.close()
