try:
    df = pd.read_csv(data_path, sep=',', encoding='utf-8')
    if 'value' not in df.columns:
        raise ValueError("Spalte 'value' nicht im Datensatz gefunden.")
    
    # Datentypen bereinigen
    for c in ['start_measure', 'end_measure']:
        if c in df.columns:
            df[c] = pd.to_datetime(df[c], errors='coerce')
    
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
    val_series = df['value'].dropna()
    n = int(val_series.shape[0])
    
    # Statistiken berechnen
    mean = float(val_series.mean()) if n else np.nan
    std = float(val_series.std(ddof=1)) if n > 1 else np.nan
    desc = val_series.describe(percentiles=[.25, .5, .75]) if n else pd.Series(dtype=float)
    q1, med, q3 = desc.get('25%', np.nan), desc.get('50%', np.nan), desc.get('75%', np.nan)
    vmin, vmax = desc.get('min', np.nan), desc.get('max', np.nan)
    skew = float(val_series.skew()) if n else np.nan
    miss_pct_val = float(df['value'].isna().mean() * 100)

    # Einheit bestimmen
    unit = 'µSv/h'
    if 'unit' in df.columns:
        mode_unit = df['unit'].dropna().astype(str).mode()
        if not mode_unit.empty:
            unit = mode_unit.iat[0]

    # Zeitraum bestimmen
    date_range_txt = ''
    if 'start_measure' in df.columns and df['start_measure'].notna().any():
        date_min = df['start_measure'].min()
        date_max = df['end_measure'].max() if 'end_measure' in df.columns else df['start_measure'].max()
        if pd.notna(date_min) and pd.notna(date_max):
            date_range_txt = f"{date_min.strftime('%d.%m.%Y')} – {date_max.strftime('%d.%m.%Y')}"

    # Bin-Berechnung (Freedman-Diaconis Regel)
    def compute_bins(x):
        x = x[np.isfinite(x)]
        if x.size < 2: return 10
        x_range = x.max() - x.min()
        if x_range == 0: return 1
        q25, q75 = np.percentile(x, [25, 75])
        iqr = q75 - q25
        bw = (2 * iqr / np.cbrt(x.size)) if iqr > 0 else (x_range / np.sqrt(x.size))
        k = int(np.ceil(x_range / bw)) if bw > 0 else 10
        return int(np.clip(k, 10, 50))

    bins = compute_bins(val_series.values)
    os.makedirs(out_dir, exist_ok=True)
    
    # Plotting
    sns.set_theme(context='notebook', style='whitegrid')
    fig, ax = plt.subplots(figsize=(10, 6))

    color_main = sns.color_palette('Blues_d')[0]
    
    sns.histplot(
        val_series, 
        bins=bins, 
        kde=True, 
        color=color_main, 
        edgecolor='white', 
        ax=ax,
        line_kws={'color': 'darkblue', 'linewidth': 2.5} # KDE Linie separat stylen
    )

    if np.isfinite(mean):

        ax.axvline(med, color='#2E8B57', linestyle=':', linewidth=1.8, label=f"Median: {med:.3f}")

    title_main = "Verteilung der Gamma-Ortsdosisleistung (ODL) - Station 'Kornbach'"
    subtitle = f"Zeitraum: {date_range_txt}" if date_range_txt else ""
    ax.set_title(f"{title_main}\n{subtitle}", loc='left', fontsize=14, pad=15)

        f"Ø = {mean:.3f} {unit}\n"
        f"σ = {std:.3f}\n"
        f"Min/Max = {vmin:.3f}/{vmax:.3f}\n"
        f"Schiefe = {skew:.2f}\n"
        f"Fehlend = {miss_pct_val:.1f}%"
    )
    ax.text(0.97, 0.85, stats_text, transform=ax.transAxes, ha='right', va='top', 
            fontsize=10, bbox=dict(boxstyle='round', facecolor='white', alpha=0.8, edgecolor='#cccccc'))

    ax.legend(loc='upper right', frameon=True)
    plt.tight_layout()
    
    fig.savefig(plot_fp, dpi=300)
    plt.close(fig)

def parse_point_wkt(s):
    if not isinstance(s, str): return None
    m = re.search(r'POINT\s*\(\s*([+-]?[0-9]*\.?[0-9]+)\s+([+-]?[0-9]*\.?[0-9]+)\s*\)', s)
    if not m: return None
    return float(m.group(1)), float(m.group(2))

def choose_lon_lat(a, b):
    # Heuristik zur Bestimmung von Lon/Lat (Deutschland Fokus)
    def in_germany(lon, lat): return (5.0 <= lon <= 16.0) and (47.0 <= lat <= 55.5)
    if in_germany(a, b): return a, b
    if in_germany(b, a): return b, a
    return a, b

geom_str = str(valid_df.loc[valid_df['geom'].notna(), 'geom'].iloc[0]) if 'geom' in valid_df.columns and valid_df['geom'].notna().any() else None
