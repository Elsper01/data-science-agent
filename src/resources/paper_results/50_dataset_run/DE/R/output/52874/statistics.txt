  # Wir lesen die Datei ein. Da Encoding-Probleme bei Umlauten auftraten, 
  # versuchen wir latin1 oder lassen readr die Erkennung übernehmen.
  df <- read_delim(input_path, delim = ";", show_col_types = FALSE, locale = locale(encoding = "ISO-8859-1"))
  
  # Um Encoding-Probleme beim Filtern zu vermeiden (z.B. "Lehrkräfte" vs "Lehrkr\xe4fte"),

    filter(
      grepl("Sch.ler", value_variable_label, ignore.case = TRUE) | 
      grepl("Lehrkr", value_variable_label, ignore.case = TRUE)
    ) %>%
    mutate(
      time = as.numeric(as.character(time)),
      value = as.numeric(as.character(value))
    ) %>%
    filter(!is.na(value), !is.na(time))

  if (nrow(plot_data) > 0) {
    # Visualisierung erstellen
    p <- ggplot(plot_data, aes(x = time, y = value, color = value_variable_label, group = value_variable_label)) +
      geom_line(linewidth = 1.2) +
      geom_point(size = 2) +
      scale_y_continuous(labels = label_number(big.mark = ".", decimal.mark = ",")) +
      scale_x_continuous(breaks = function(x) seq(floor(min(x)), ceiling(max(x)), by = 1)) +
      labs(

        subtitle = "Zeitverlauf in Stadtlauringen",
        x = "Jahr",

        color = "Kategorie"

      theme_minimal() +

        legend.direction = "vertical",

        axis.title = element_text(size = 12),
        panel.grid.minor = element_blank()
      )
    
    # Speichern des Diagramms

    cat("Visualisierung erfolgreich erstellt unter:", output_path)
