df = df.loc[:, ~df.columns.str.contains('^Unnamed')]

for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].replace({'-': np.nan, '': np.nan})

def parse_percent(s):
    if pd.isna(s):
        return np.nan
    if isinstance(s, (int, float)):
        return float(s)
    s = str(s).strip().replace('%', '').replace(' ', '').replace(',', '.')
    try:
        return float(s) / 100.0
    except Exception:
        return np.nan


def parse_hms(s):
    if pd.isna(s):
        return np.nan
    s = str(s).strip()
    try:
        parts = s.split(':')
        parts = [int(float(p)) for p in parts]
        if len(parts) == 3:
            h, m, sec = parts
            return h * 3600 + m * 60 + sec
        if len(parts) == 2:
            m, sec = parts
            return m * 60 + sec
        return np.nan
    except Exception:
        return np.nan


if 'Datum' in df.columns:
    df['Datum'] = pd.to_datetime(df['Datum'], format='%d.%m.%Y', dayfirst=True, errors='coerce')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

pd.set_option('display.max_columns', None)

file_path = r'./src/resources/data/study\32830\dataset.csv'
output_path = r'./src/resources/output\32830/2_bounce_vs_verweildauer.png'

def to_percent_numeric(s):
    if pd.isna(s):
        return np.nan
    if isinstance(s, (int, float)):
        return float(s)
    s = str(s).strip().replace('%', '').replace(',', '.')
    if s in ['-', '']:
        return np.nan
    try:
        return float(s)
    except Exception:
        return np.nan

def to_minutes_from_time(s):
    if pd.isna(s):
        return np.nan
    try:
        td = pd.to_timedelta(s, errors='coerce')
        if pd.isna(td):
            return np.nan
        return td.total_seconds() / 60.0
    except Exception:
        return np.nan

try:
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')
    df = df.loc[:, ~df.columns.astype(str).str.contains('^Unnamed', case=False, regex=True)]
    if 'Datum' in df.columns:
        df['Datum'] = pd.to_datetime(df['Datum'], dayfirst=True, errors='coerce')
        df = df[~df['Datum'].isna()].copy()
    for c in ['Besuche', 'Besucher', 'Seitenaufrufe']:
        if c in df.columns:
            df[c] = pd.to_numeric(df[c], errors='coerce')
    for c in ['Seiten_pro_Besuch', 'Besuchsh__ufigkeit']:
        if c in df.columns:
            df[c] = pd.to_numeric(df[c], errors='coerce')
    if 'Bounce_Rate' in df.columns:
        df['Bounce_Rate_num'] = df['Bounce_Rate'].apply(to_percent_numeric)
    else:
        df['Bounce_Rate_num'] = np.nan
    if 'Zeit_im_Besuch' in df.columns:
        df['Zeit_im_Besuch_min'] = df['Zeit_im_Besuch'].apply(to_minutes_from_time)
    else:
        df['Zeit_im_Besuch_min'] = np.nan
    num_cols = df.select_dtypes(include=[np.number]).columns
    _ = df.isna().mean()
    _ = df[num_cols].describe().T if len(num_cols) > 0 else pd.DataFrame()
    _ = df[['Bounce_Rate_num', 'Zeit_im_Besuch_min']].corr(method='pearson', min_periods=2)
    plot_df = df[['Datum', 'Bounce_Rate_num', 'Zeit_im_Besuch_min']].dropna(subset=['Bounce_Rate_num', 'Zeit_im_Besuch_min']).sort_values('Datum')
    if plot_df.empty:
        raise ValueError('Keine gültigen Datenpunkte für die Visualisierung vorhanden.')
    vals = plot_df[['Bounce_Rate_num', 'Zeit_im_Besuch_min']].dropna()
    r = np.nan
    if len(vals) > 1:
        r = np.corrcoef(vals['Bounce_Rate_num'].to_numpy(), vals['Zeit_im_Besuch_min'].to_numpy())[0, 1]
    sns.set_theme(style='whitegrid', context='talk')
    fig, ax1 = plt.subplots(figsize=(12, 6))
    ax1p = sns.lineplot(data=plot_df, x='Datum', y='Bounce_Rate_num', ax=ax1, color='#D62728', linewidth=2)
    ax2 = ax1.twinx()
    ax2p = sns.lineplot(data=plot_df, x='Datum', y='Zeit_im_Besuch_min', ax=ax2, color='#1F77B4', linewidth=2)
    l1 = ax1p.lines[0] if ax1p.lines else None
    l2 = ax2p.lines[0] if ax2p.lines else None
    handles = []
    labels = []
    if l1 is not None:
        handles.append(l1)
        labels.append('Bounce Rate')
    if l2 is not None:
        handles.append(l2)
        labels.append('Ø Verweildauer (Minuten)')
    if handles:
        ax1.legend(handles, labels, loc='upper right', frameon=True)
    ax1.set_title('Bounce Rate und durchschnittliche Verweildauer über die Zeit')
    subtitle = ''
    if not np.isnan(r):
        subtitle = f'Zusammenhang (Pearson r): {r:.2f}'
    if subtitle:
        ax1.text(0.01, 1.04, subtitle, transform=ax1.transAxes, ha='left', va='bottom', fontsize=11, color='#555555')
    ax1.set_xlabel('Datum')

    ax2.set_ylabel('Durchschnittliche Verweildauer (in Minuten)')
    for label in ax1.get_xticklabels():
        label.set_rotation(45)
        label.set_horizontalalignment('right')
    ax1.margins(x=0.01)
    ax2.margins(x=0.01)
    ax1.grid(True, which='major', axis='both', linestyle='--', linewidth=0.5, alpha=0.6)
    fig.tight_layout()
    fig.savefig(output_path, dpi=200, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
