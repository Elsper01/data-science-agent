import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

# Paths
input_path = './src/resources/data/study/13473/dataset.csv'
output_dir = os.path.normpath('./src/resources/output/13473/run_4')
plot_path = os.path.join(output_dir, '0_geschlechterverteilung_hof.png')
os.makedirs(output_dir, exist_ok=True)

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xf6' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German data with umlauts like 'ö').
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Basic cleaning
for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].replace({'': np.nan, 'nan': np.nan})

# EDA Exports
missing_prop = df.isna().mean()
missing_prop.to_csv(os.path.join(output_dir, 'eda_missingness.csv'), header=['missing_share'])
num_df = df.select_dtypes(include=[np.number])
if not num_df.empty:
    num_summary = num_df.describe().T
    num_summary.to_csv(os.path.join(output_dir, 'eda_numeric_summary.csv'))

# Ensure 'value' is numeric
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for Gender distribution
# Based on the goal: filter '2_variable_label' for 'Geschlecht' and exclude 'Gesamt'
df_gender = df.copy()
if '2_variable_label' in df_gender.columns:
    # Case-insensitive filter for 'Geschlecht'
    df_gender = df_gender[df_gender['2_variable_label'].astype(str).str.strip().str.lower() == 'geschlecht']

labels_col = '2_variable_attribute_label'
value_col = 'value'

# Filter out 'Gesamt' (Total) and keep only 'männlich' and 'weiblich'
valid_labels = ['männlich', 'weiblich']
df_gender = df_gender[df_gender[labels_col].astype(str).str.strip().str.lower().isin(valid_labels)]

# Aggregate to ensure we have unique values for the plot
df_gender = df_gender.groupby(labels_col, as_index=False)[value_col].sum()

# Formatting and Plotting
def fmt_thousands(x, pos=None):
    return f"{int(x):,}".replace(',', '.')

sns.set_theme(style='whitegrid', context='talk')
colors = ['#4C78A8', '#F58518']
fig, ax = plt.subplots(figsize=(10, 6), dpi=150)

# Sort order for consistency
order = ['männlich', 'weiblich']
order = [o for o in order if o in df_gender[labels_col].values]

sns.barplot(data=df_gender, x=labels_col, y=value_col, order=order, palette=colors, ax=ax, edgecolor='#333333')

ax.set_xlabel('Geschlecht')
ax.set_ylabel('Bevölkerung (Anzahl)')
ax.yaxis.set_major_formatter(FuncFormatter(fmt_thousands))
ax.set_title('Bevölkerungsverteilung nach Geschlecht in Hof')

                label, ha='center', va='bottom', fontsize=12, fontweight='bold')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

data_path = './src/resources/data/study/13473/dataset.csv'
output_path = './src/resources/output/13473/run_4/1_regionalebenen_vergleich.png'

sns.set_theme(style='whitegrid', context='talk')

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xf6' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German datasets with umlauts like ö).
try:
    df = pd.read_csv(data_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(data_path, sep=';', encoding='latin-1')

# Data Cleaning
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for total population
# Using case-insensitive search and stripping whitespace to be robust
mask_total = df['2_variable_attribute_label'].astype(str).str.strip().str.lower() == 'insgesamt'
df_total = df.loc[mask_total].copy()

# Group by regional classification level
plot_df = df_total.groupby('1_variable_code', as_index=False, dropna=False)['value'].sum()

# Define order for x-axis
unique_codes = plot_df['1_variable_code'].astype(str).unique().tolist()
order = [x for x in ['KREISE', 'GEMEIN'] if x in unique_codes]
if not order:
    order = sorted(unique_codes)

plot_df['1_variable_code'] = pd.Categorical(plot_df['1_variable_code'].astype(str), categories=order, ordered=True)
plot_df = plot_df.sort_values('1_variable_code')

# Visualization
plt.figure(figsize=(10, 7))
ax = plt.gca()

# Use a distinct palette
palette_list = sns.color_palette("viridis", len(order))
sns.barplot(data=plot_df, x='1_variable_code', y='value', order=order, palette=palette_list, ax=ax, edgecolor='black')

ax.set_xlabel('Regionale Klassifikationsebene (1_variable_code)')
ax.set_ylabel('Bevölkerung (Anzahl)')
ax.set_title('Vergleich der Bevölkerungszahl nach regionaler Klassifikation\n(Filter: 2_variable_attribute_label = "Insgesamt")')

# Add value labels on top of bars
max_val = plot_df['value'].max()
if pd.notna(max_val) and max_val > 0:
    ax.set_ylim(0, max_val * 1.2)
    for p in ax.patches:
        h = p.get_height()
        if pd.notna(h):
            # Format number with thousands separator
            label = f"{int(h):,}".replace(',', '.')
            ax.annotate(label, 
                        (p.get_x() + p.get_width() / 2.0, h), 
                        ha='center', va='bottom', 
                        fontsize=14, fontweight='bold',
                        xytext=(0, 5), textcoords='offset points')

plt.tight_layout()
plt.savefig(output_path, dpi=300, bbox_inches='tight', facecolor='white')
plt.close()

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Set visual style
sns.set_theme(style='white')

input_path = os.path.normpath('./src/resources/data/study/13473/dataset.csv')
output_path = os.path.normpath('./src/resources/output/13473/run_4/2_bevoelkerung_indicator.png')
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# The error was a UnicodeDecodeError with utf-8. 
# Common encodings for German datasets are 'latin1' (ISO-8859-1) or 'cp1252'.
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin1')

# Clean string columns
for c in df.columns:
    if df[c].dtype == 'object':
        df[c] = df[c].astype(str).str.strip()

# Convert time to datetime for sorting/filtering
if 'time' in df.columns:
    df['time_dt'] = pd.to_datetime(df['time'], errors='coerce')
else:
    df['time_dt'] = pd.NaT

# Filter for the total population
# Based on the goal: '2_variable_attribute_label' represents the total
mask_total = df['2_variable_attribute_label'].str.lower().isin(['insgesamt', 'total', 'gesamt'])
totals = df[mask_total].copy()

if totals.empty:
    # Fallback if no explicit 'total' label is found
    latest_data = df.sort_values('time_dt', ascending=False)
    if not latest_data.empty:
        row = latest_data.iloc[0]
    else:
        row = None
else:
    # Get the latest record
    totals = totals.sort_values('time_dt', ascending=False)
    row = totals.iloc[0]

if row is not None:
    total_value = pd.to_numeric(row['value'], errors='coerce')
    time_label = str(row['time'])
    area = row['1_variable_attribute_label'] if '1_variable_attribute_label' in df.columns else 'Hof'
else:
    total_value = 0
    time_label = "N/A"
    area = "Hof"

# Helper functions for formatting
def fmt_int_de(n):
    try:
        if np.isnan(n): return "N/A"
        return f"{int(round(n)):,}".replace(',', '.')
    except:
        return str(n)

# Visualization
fig, ax = plt.subplots(figsize=(8, 5), dpi=200)
fig.patch.set_facecolor('#FFFFFF')
ax.set_xlim(0, 1)
ax.set_ylim(0, 1)
ax.axis('off')

# Title
title = f"Gesamtbevölkerung in {area}"
plt.text(0.5, 0.85, title, ha='center', va='center', fontsize=20, fontweight='bold', color='#333333')

# Main Metric
plt.text(0.5, 0.50, fmt_int_de(total_value), ha='center', va='center', fontsize=64, fontweight='bold', color='#004c8c')
plt.text(0.5, 0.35, 'Einwohner insgesamt', ha='center', va='center', fontsize=16, color='#666666')

# Reference Date
plt.text(0.5, 0.15, f"Stand: {time_label}", ha='center', va='center', fontsize=14, color='#888888', style='italic')

# Footer note
plt.text(0.05, 0.02, 'Quelle: Bevölkerungsstatistik', ha='left', va='bottom', fontsize=8, color='#aaaaaa')

plt.savefig(output_path, dpi=200, bbox_inches='tight', facecolor=fig.get_facecolor())
plt.close(fig)
