import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Load the dataset
# The error 'UnicodeDecodeError' with byte 0xf6 (ö) suggests 'latin-1' or 'iso-8859-1' encoding
file_path = './src/resources/data/study/13473/dataset.csv'

try:
    df = pd.read_csv(file_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    # Fallback to latin-1 if utf-8 fails
    df = pd.read_csv(file_path, sep=';', encoding='latin-1')

# Filter the dataset for the visualization goal
# Goal: Bar chart showing '2_variable_attribute_label' on the x-axis and 'value' on the y-axis, 
# filtered to exclude the 'Total' (Insgesamt) category.
# We also filter for 'KREISE' to avoid double counting with 'GEMEINDEN' if both exist for the same region.
plot_df = df[(df['2_variable_attribute_label'] != 'Insgesamt')].copy()

# If there are multiple regional levels, we take the highest level (KREISE) to represent the city/district
if '1_variable_code' in plot_df.columns and 'KREISE' in plot_df['1_variable_code'].values:
    plot_df = plot_df[plot_df['1_variable_code'] == 'KREISE']

# Set aesthetic style
sns.set_theme(style="whitegrid")
plt.figure(figsize=(10, 6))

# Create the bar chart
# Mapping colors: Blue for Männlich, Pink/Red for Weiblich
palette = {'männlich': '#3498db', 'weiblich': '#e74c3c'}

ax = sns.barplot(
    data=plot_df, 
    x='2_variable_attribute_label', 
    y='value', 
    palette='viridis', # Defaulting to viridis if specific labels don't match
    hue='2_variable_attribute_label',
    legend=False
)

# Add labels and title
plt.title('Geschlechterverteilung der Bevölkerung in Hof', fontsize=14, fontweight='bold', pad=20)
plt.xlabel('Geschlecht', fontsize=12)
plt.ylabel('Anzahl der Personen', fontsize=12)

# Add data labels on top of bars
for p in ax.patches:
    if p.get_height() > 0:
        ax.annotate(format(p.get_height(), ',.0f').replace(',', '.'), 
                    (p.get_x() + p.get_width() / 2., p.get_height()), 
                    ha = 'center', va = 'center', 
                    xytext = (0, 9), 
                    textcoords = 'offset points',
                    fontsize=11,
                    fontweight='bold')

# Adjust layout

# Ensure output directory exists

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

def create_visualization():
    file_path = './src/resources/data/study/13473/dataset.csv'
    output_dir = './src/resources/output/13473'
    output_file = os.path.join(output_dir, '1_population_comparison_levels.png')
    
    if not os.path.exists(output_dir):
        os.makedirs(output_dir)

    # The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xf6' 
    # suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (common for German data with umlauts like 'ö').
    try:
        df = pd.read_csv(file_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(file_path, sep=';', encoding='latin-1')

    # Filter for 'Total' category in '2_variable_attribute_label'
    # 'Insgesamt' is the German label for 'Total'
    df_total = df[df['2_variable_attribute_label'] == 'Insgesamt'].copy()

    # Ensure we have data to plot
    if df_total.empty:
        print("No data found for 'Insgesamt' in '2_variable_attribute_label'.")
        return

    # Prepare data for plotting
    # We want to compare KREISE vs GEMEIN
    plot_data = df_total[df_total['1_variable_code'].isin(['KREISE', 'GEMEIN'])].copy()
    
    # Set aesthetic style
    sns.set_theme(style="whitegrid")
    plt.figure(figsize=(10, 6))

    # Create bar chart
    ax = sns.barplot(
        data=plot_data, 
        x='1_variable_label', 
        y='value', 
        palette='viridis',
        hue='1_variable_label',
        legend=False
    )

    # Add data labels on top of bars
    for p in ax.patches:
        height = p.get_height()
        ax.annotate(f'{height:,.0f}'.replace(',', '.'), 
                   (p.get_x() + p.get_width() / 2., height), 
                   ha = 'center', va = 'center', 
                   xytext = (0, 9), 
                   textcoords = 'offset points',
                   fontsize=12,
                   fontweight='bold')

    # Labels and Title
    plt.title('Vergleich der Gesamtbevölkerung: Kreisebene vs. Gemeindeebene\n(Stadt Hof)', fontsize=14, pad=20)
    plt.xlabel('Administrative Ebene', fontsize=12)
    plt.ylabel('Gesamtbevölkerung (Anzahl)', fontsize=12)
    
    # Adjust y-axis to give space for labels
    plt.ylim(0, plot_data['value'].max() * 1.2)

    plt.tight_layout()
    plt.savefig(output_file)
    plt.close()

if __name__ == "__main__":
    create_visualization()

import pandas as pd
import matplotlib.pyplot as plt
import os

# Load the dataset
file_path = './src/resources/data/study/13473/dataset.csv'

# The error 'UnicodeDecodeError' with byte 0xf6 (ö) suggests 'latin-1' or 'iso-8859-1' encoding
# common in German datasets.
try:
    df = pd.read_csv(file_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(file_path, sep=';', encoding='latin-1')

# Exploratory Data Analysis (EDA)
# The goal is to find the total population for the latest reference date.
# Based on the previous code, the label is 'Insgesamt' (Total) and date is '2024-12-31'.

# Filter for the specific KPI: Total population
# We look for 'Insgesamt' in '2_variable_attribute_label'
kpi_data = df[df['2_variable_attribute_label'].str.contains('Insgesamt', na=False, case=False)]

# Sort by time to get the latest date if 2024-12-31 is not available or to be dynamic
kpi_data = kpi_data.sort_values(by='time', ascending=False)

if not kpi_data.empty:
    latest_row = kpi_data.iloc[0]
    total_population = latest_row['value']
    reference_date_raw = latest_row['time']
    # Format date for display (assuming YYYY-MM-DD)
    try:
        reference_date = pd.to_datetime(reference_date_raw).strftime('%d.%m.%Y')
    except:
        reference_date = reference_date_raw
else:
    total_population = 0
    reference_date = "N/A"

# Create the visualization (KPI Card)
fig, ax = plt.subplots(figsize=(8, 5))

# Remove axes for a clean card look
ax.axis('off')

# Add a background rectangle for the card
rect = plt.Rectangle((0.1, 0.2), 0.8, 0.6, color='#f8f9fa', ec='#dee2e6', lw=2, transform=ax.transAxes)
ax.add_patch(rect)

# Add Title
plt.text(0.5, 0.72, 'Gesamtbevölkerung Hof', 
         horizontalalignment='center', verticalalignment='center', 
         fontsize=22, fontweight='bold', color='#212529', transform=ax.transAxes)

# Add the KPI Value
# Format with thousands separator
formatted_val = f"{int(total_population):,}".replace(',', '.') if not pd.isna(total_population) else "k.A."
plt.text(0.5, 0.52, formatted_val, 
         horizontalalignment='center', verticalalignment='center', 
         fontsize=55, fontweight='bold', color='#007bff', transform=ax.transAxes)

# Add Subtitle/Reference Date
plt.text(0.5, 0.38, f'Stand: {reference_date}', 
         horizontalalignment='center', verticalalignment='center', 
         fontsize=16, color='#6c757d', transform=ax.transAxes)

# Add Unit
plt.text(0.5, 0.28, 'Einwohner', 
         horizontalalignment='center', verticalalignment='center', 
         fontsize=14, color='#6c757d', transform=ax.transAxes)

# Ensure output directory exists
output_dir = './src/resources/output/13473'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

# Save the plot
plt.tight_layout()
plt.savefig(f'{output_dir}/2_total_population_kpi.png', dpi=300, bbox_inches='tight')
plt.close()
