import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import os

# The error was a UnicodeDecodeError with 'utf-8'. 
# Common encodings for German/European datasets are 'latin1' (ISO-8859-1) or 'cp1252'.
data_path = r'./src/resources/data/study/136951/dataset.csv'

try:
    df = pd.read_csv(data_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(data_path, sep=',', encoding='latin1')
    except Exception:
        df = pd.read_csv(data_path, sep=',', encoding='cp1252')

# Clean string columns
for c in df.select_dtypes(include=['object']).columns:
    df[c] = df[c].astype(str).str.strip()

# Prepare data for visualization
# We want to show Pruefmerkmal and Ergebnis (with Einheit if available)
viz_cols = ['Pruefmerkmal', 'Ergebnis', 'Einheit']
existing_cols = [c for c in viz_cols if c in df.columns]
viz_df = df[existing_cols].copy()

# Combine Ergebnis and Einheit for better display
if 'Ergebnis' in viz_df.columns:
    if 'Einheit' in viz_df.columns:
        viz_df['Ergebnis_Display'] = viz_df.apply(
            lambda row: f"{row['Ergebnis']} {row['Einheit']}" if pd.notna(row['Einheit']) and str(row['Einheit']).lower() != 'nan' else str(row['Ergebnis']),
            axis=1
        )
    else:
        viz_df['Ergebnis_Display'] = viz_df['Ergebnis'].astype(str)
else:
    viz_df['Ergebnis_Display'] = "N/A"

if 'Pruefmerkmal' not in viz_df.columns:
    viz_df['Pruefmerkmal'] = 'Unbekannt'

# Create a table visualization
fig, ax = plt.subplots(figsize=(10, 2 + 0.6 * len(viz_df)), dpi=150)
ax.axis('off')

col_labels = ['Prüfmerkmal', 'Ergebnis']
cell_text = viz_df[['Pruefmerkmal', 'Ergebnis_Display']].values.tolist()

table = ax.table(
    cellText=cell_text, 
    colLabels=col_labels, 
    loc='center', 
    cellLoc='left',
    colWidths=[0.5, 0.4]
)

table.auto_set_font_size(False)
table.set_fontsize(11)
table.scale(1.2, 1.8)

# Style the header
for (row, col), cell in table.get_celld().items():
    if row == 0:
        cell.set_text_props(weight='bold', color='white')
        cell.set_facecolor('#40466e')

plt.title('Gemessene Prüfmerkmale und Ergebnisse\n(Hauptprobenummer 201700572)', pad=20, fontsize=14, fontweight='bold')

# Save the output
out_dir = r'./src/resources/output/136951'
os.makedirs(out_dir, exist_ok=True)
out_path = os.path.join(out_dir, '0_pruefmerkmal_ergebnisse.png')
plt.savefig(out_path, bbox_inches='tight')
plt.close(fig)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# The error 'utf-8' codec can't decode byte 0xfc suggests the file is likely encoded in 'latin-1' or 'cp1252'
# (0xfc is 'ü' in latin-1, common in German datasets).
file_path = r'./src/resources/data/study/136951/dataset.csv'

try:
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(file_path, sep=',', encoding='latin-1')
    except Exception:
        df = pd.read_csv(file_path, sep=',', encoding='cp1252')

# Clean string columns
for col in df.select_dtypes(include=['object']).columns:
    df[col] = df[col].astype(str).str.strip()

# Helper to convert results to numeric if needed (though we use strings for the table)
def _to_num(v):
    if pd.isna(v) or v == 'nan':
        return np.nan
    if isinstance(v, str):
        s = v.replace(' ', '').replace('<', '').replace('>', '').replace('≤', '').replace('≥', '').replace(',', '.')
        try:
            return float(s)
        except:
            return np.nan
    return float(v)

# Filter for 'Messzeit'
mask_messzeit = df['Pruefmerkmal'].str.contains('Messzeit', case=False, na=False)
view_cols = ['Entnahmestelle', 'Pruefmerkmal', 'Ergebnis', 'Einheit']

if mask_messzeit.any():
    tbl_df = df.loc[mask_messzeit, view_cols].copy()
else:
    # Fallback if no Messzeit found, show first few rows or empty
    tbl_df = df[view_cols].head(5).copy()

# Formatting for display
if 'Ergebnis' in tbl_df.columns:
    # Ensure numeric-looking strings use comma for German context if desired, 
    # but here we keep it clean for the table.
    tbl_df['Ergebnis'] = tbl_df['Ergebnis'].astype(str)

# Metadata for titles
method_txt = str(df['Methode'].iloc[0]) if 'Methode' in df.columns and not df['Methode'].empty else "Radon-LSC-Analyse"
entnahmestelle_txt = str(tbl_df['Entnahmestelle'].iloc[0]) if not tbl_df.empty else ""
date_txt = str(df['Probenahme_Beginn_Datum'].iloc[0]) if 'Probenahme_Beginn_Datum' in df.columns and not df['Probenahme_Beginn_Datum'].empty else ""

# Visualization
sns.set_theme(style='white')
rows = tbl_df.shape[0]
fig_h = 2.0 + 0.6 * max(1, rows)
fig, ax = plt.subplots(figsize=(10, fig_h))
ax.axis('off')

col_labels = ['Entnahmestelle', 'Prüfmerkmal', 'Ergebnis', 'Einheit']
cell_text = tbl_df.values.tolist()
col_widths = [0.35, 0.25, 0.20, 0.20]

table = ax.table(
    cellText=cell_text, 
    colLabels=col_labels, 
    cellLoc='center', 
    loc='center', 
    colWidths=col_widths
)

table.auto_set_font_size(False)
table.set_fontsize(11)

# Style the table
for (row, col), cell in table.get_celld().items():
    cell.set_edgecolor('#dddddd')
    if row == 0:
        cell.set_text_props(color='white', weight='bold')
        cell.set_facecolor('#2f5597')
    else:
        if row % 2 == 1:
            cell.set_facecolor('#f9f9f9')
        if col == 2: # Highlight the 'Ergebnis' column
            cell.set_text_props(weight='bold')

# Titles
main_title = 'Technische Messzeit der Radon-LSC-Analyse'
subtitle = f"{entnahmestelle_txt} | {date_txt}" if entnahmestelle_txt and date_txt else entnahmestelle_txt

ax.set_title(main_title, fontsize=16, pad=25, weight='bold', loc='center')
if subtitle:
    ax.text(0.5, 0.92, subtitle, transform=ax.transAxes, fontsize=12, ha='center', color='#555555')

overall_missing_pct = (df.isna().sum().sum() / df.size) * 100 if df.size > 0 else 0

ax.text(0.0, 0.05, foot_left, transform=ax.transAxes, fontsize=9, ha='left', color='#777777')
