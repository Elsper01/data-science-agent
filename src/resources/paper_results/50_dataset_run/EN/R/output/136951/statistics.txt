library(tidyverse)
library(scales)

# Define paths
input_path <- file.path(".", "src", "resources", "data", "study", "136951", "dataset.csv")
output_dir <- file.path(".", "src", "resources", "output", "136951")
dir.create(output_dir, showWarnings = FALSE, recursive = TRUE)

# Read data with explicit encoding and handling potential UTF-8 issues
# Using read_csv with guess_max to ensure column types are handled well
raw <- readr::read_csv(input_path, 
                       locale = readr::locale(encoding = "UTF-8"), 
                       show_col_types = FALSE)

# Fix the UTF-8 error in trimws/sub by using stringr::str_trim which is more robust
# and avoiding purrr::map_dbl with trimws on raw character vectors that might contain invalid bytes
clean_raw <- raw %>%
  mutate(across(where(is.character), ~stringr::str_trim(.)))

# Calculate missingness safely
missingness <- tibble(
  Spalte = names(clean_raw),
  Fehlendenanteil = map_dbl(names(clean_raw), function(nm) {
    x <- clean_raw[[nm]]
    if (is.character(x)) {
      sum(is.na(x) | x == "", na.rm = TRUE) / length(x)
    } else {
      sum(is.na(x)) / length(x)
    }
  })
)
readr::write_csv(missingness, file.path(output_dir, "eda_missingness.csv"))

# Process 'Ergebnis' column: handle commas, units, and censored values (<)
df <- clean_raw %>%
  mutate(
    Ergebnis_clean = stringr::str_replace_all(Ergebnis, ",", "."),
    Zensiert = stringr::str_detect(Ergebnis_clean, "<"),
    Ergebnis_num = as.numeric(stringr::str_extract(Ergebnis_clean, "[-+]?[0-9]*\\.?[0-9]+"))
  )

# Summary statistics
summary_by_pm <- df %>%
  group_by(Pruefmerkmal) %>%
  summarise(
    anzahl = n(),
    anzahl_nicht_na = sum(!is.na(Ergebnis_num)),
    minimum = min(Ergebnis_num, na.rm = TRUE),
    maximum = max(Ergebnis_num, na.rm = TRUE),
    mittelwert = mean(Ergebnis_num, na.rm = TRUE),
    .groups = "drop"
  )
readr::write_csv(summary_by_pm, file.path(output_dir, "eda_erg_summary_by_pruefmerkmal.csv"))

# Visualization: Bar chart of Pruefmerkmal vs Ergebnis
# Since units differ (Bq/l vs s), we use facets with free scales
plot_df <- df %>%
  mutate(
    Label = paste(Ergebnis, Einheit)
  )
