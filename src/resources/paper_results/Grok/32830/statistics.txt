# Die Fehlermeldung deutet auf einen Konflikt mit dem Paket 'rlang' hin, das bereits geladen ist.
# In vielen Umgebungen ist es besser, auf die explizite Installation zu verzichten, 
# wenn die Pakete bereits vorhanden sind, oder die Bibliotheken direkt zu laden.
# Wir versuchen, die Bibliotheken zu laden, ohne eine Neuinstallation zu erzwingen, 
# da dies oft zu 'Permission Denied' Fehlern führt, wenn Dateien gesperrt sind.

library(ggplot2)
library(dplyr)
library(lubridate)
library(scales)

# Daten laden
df <- read.csv('./src/resources/data/study/32830/dataset.csv', sep = ',', fileEncoding = 'UTF-8', stringsAsFactors = FALSE, na.strings = c('', 'NA', 'nan'))

# Vorverarbeitung
# Die erste Zeile enthält oft Summen/Gesamtwerte bei etracker-Exports, daher entfernen wir sie
df_daily <- df[-1, ] %>% filter(!is.na(Datum))

# Datum konvertieren (Format im Datensatz ist meist DD.MM.YYYY)
df_daily$Datum <- dmy(df_daily$Datum)
df_daily <- df_daily %>% arrange(Datum)

# Numerische Konvertierung der Seitenaufrufe
df_daily$Seitenaufrufe <- as.numeric(gsub('\\.', '', as.character(df_daily$Seitenaufrufe)))

# Funktion zur Umwandlung von HH:MM:SS in Sekunden
hms_to_seconds <- function(time_str) {
  if (is.na(time_str) || time_str == '' || !grepl(':', time_str)) return(NA)
  parts <- as.numeric(unlist(strsplit(time_str, ':')))
  if (length(parts) == 3) {
    return(parts[1] * 3600 + parts[2] * 60 + parts[3])
  } else if (length(parts) == 2) {
    return(parts[1] * 60 + parts[2])
  } else {
    return(NA)
  }
}

df_daily$Zeit_im_Besuch_sec <- sapply(df_daily$Zeit_im_Besuch, hms_to_seconds)

# Skalierungsfaktor für die sekundäre Achse berechnen
max_seiten <- max(df_daily$Seitenaufrufe, na.rm = TRUE)
max_zeit <- max(df_daily$Zeit_im_Besuch_sec, na.rm = TRUE)

# Falls max_zeit 0 ist, um Division durch Null zu vermeiden
if(max_zeit == 0 || is.na(max_zeit)) { scale_factor <- 1 } else { scale_factor <- max_seiten / max_zeit }

df_daily$Zeit_scaled <- df_daily$Zeit_im_Besuch_sec * scale_factor

# Visualisierung erstellen
p <- ggplot(df_daily, aes(x = Datum)) +
  # Balken für Seitenaufrufe
  geom_col(aes(y = Seitenaufrufe, fill = 'Seitenaufrufe'), alpha = 0.7) +
  # Linie für Zeit im Besuch
  geom_line(aes(y = Zeit_scaled, color = 'Zeit im Besuch (Sek.)'), size = 1.2) +
  geom_point(aes(y = Zeit_scaled, color = 'Zeit im Besuch (Sek.)'), size = 2.5) +
  # Achsen-Konfiguration
  scale_x_date(date_labels = '%d.%m.', date_breaks = '2 days') +
  scale_y_continuous(
    name = 'Seitenaufrufe',
    sec.axis = sec_axis(~ . / scale_factor, name = 'Verweildauer (Sekunden)')
  ) +
  # Farben und Legende
  scale_fill_manual(name = '', values = c('Seitenaufrufe' = '#2E86AB')) +
  scale_color_manual(name = '', values = c('Zeit im Besuch (Sek.)' = '#A23B72')) +
  labs(
    title = 'Tägliches Engagement: Seitenaufrufe vs. Verweildauer',
    subtitle = 'Vergleich von Quantität (Aufrufe) und Qualität (Zeit) im Zeitverlauf',
    x = 'Datum',
    caption = 'Datenquelle: etracker Export'
  ) +
  theme_minimal(base_size = 12) +
  theme(
    legend.position = 'bottom',
    plot.title = element_text(face = 'bold', size = 14),
    axis.text.x = element_text(angle = 45, hjust = 1),
    axis.title.y.right = element_text(color = '#A23B72'),
    axis.text.y.right = element_text(color = '#A23B72'),
    axis.title.y.left = element_text(color = '#2E86AB'),
    axis.text.y.left = element_text(color = '#2E86AB')
  )

# Verzeichnis erstellen und Speichern
dir.create('./src/resources/output/32830', recursive = TRUE, showWarnings = FALSE)
ggsave('./src/resources/output/32830/2_taegliches_engagement.png', plot = p, width = 12, height = 7, dpi = 300, bg = 'white')
