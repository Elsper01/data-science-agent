col_is_missing <- function(x) {
  if (is.character(x)) {
    return(is.na(x) | trimws(x) == "")
  } else {
    return(is.na(x))
  }
}
missing_prop <- sapply(df, function(x) mean(col_is_missing(x)))
missing_df <- data.frame(Spalte = names(missing_prop), Anteil_NA_oder_Leer = as.numeric(missing_prop), stringsAsFactors = FALSE)
if ("Ergebnis" %in% names(df)) {
  ergebnis_chr <- if (is.character(df$Ergebnis)) df$Ergebnis else as.character(df$Ergebnis)
  ergebnis_chr <- trimws(ergebnis_chr)
  ergebnis_num <- suppressWarnings(as.numeric(gsub(",", ".", ergebnis_chr)))
  nn_key <- gsub("[\\s\\.]", "", tolower(ergebnis_chr))
  is_nn <- nn_key %in% c("nn","nichtnachweisbar")
  status <- ifelse(is_nn, "Nicht nachweisbar (n.n.)", ifelse(!is.na(ergebnis_num), "Messwert vorhanden", NA))
  df$Nachweisstatus <- status
  df$Ergebnis_num <- ergebnis_num
} else {
  df$Nachweisstatus <- NA_character_
  df$Ergebnis_num <- NA_real_
}
counts <- df %>% dplyr::filter(!is.na(Nachweisstatus)) %>% dplyr::count(Nachweisstatus, name = "Anzahl") %>% dplyr::mutate(Prozent = Anzahl/sum(Anzahl)) %>% dplyr::mutate(Nachweisstatus = factor(Nachweisstatus, levels = c("Nicht nachweisbar (n.n.)","Messwert vorhanden"))) %>% dplyr::arrange(Nachweisstatus)
stats_ergebnis <- {
  x <- df$Ergebnis_num[!is.na(df$Ergebnis_num)]
  einheit <- if ("Einheit" %in% names(df)) paste(unique(na.omit(df$Einheit)), collapse = ", ") else NA_character_
  data.frame(
    Anzahl_Messwerte = length(x),
    Minimum = ifelse(length(x) > 0, min(x, na.rm = TRUE), NA_real_),
    Median = ifelse(length(x) > 0, median(x, na.rm = TRUE), NA_real_),
    Mittelwert = ifelse(length(x) > 0, mean(x, na.rm = TRUE), NA_real_),
    Standardabweichung = ifelse(length(x) > 1, sd(x, na.rm = TRUE), NA_real_),
    Maximum = ifelse(length(x) > 0, max(x, na.rm = TRUE), NA_real_),

  )
}
