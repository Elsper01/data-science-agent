import os
import sys
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from matplotlib.ticker import FuncFormatter, MaxNLocator

try:
    input_path = './src/resources/data/study/15653/dataset.csv'
    output_dir = os.path.normpath('./src/resources/output/15653')
    os.makedirs(output_dir, exist_ok=True)
    plot_path = os.path.join(output_dir, '0_entwicklung_tourismus_rettenberg.png')

    # Behebung des UnicodeDecodeError: Versuche 'latin-1' (ISO-8859-1), da 'utf-8' fehlgeschlagen ist.
    # Das Byte 0xe4 entspricht 'ä' in latin-1, was in deutschen Datensätzen häufig vorkommt.
    try:
        df = pd.read_csv(input_path, sep=';', encoding='utf-8', low_memory=False)
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=';', encoding='latin-1', low_memory=False)

    df = df.drop_duplicates()

    # Datenbereinigung und Konvertierung
    if 'time' in df.columns:
        df['time'] = pd.to_numeric(df['time'], errors='coerce')
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')
    if 'value_variable_label' in df.columns:
        df['value_variable_label'] = df['value_variable_label'].astype(str)

    # Filtern auf den relevanten Zeitraum und Entfernen von NaNs in Schlüsselspalten
    df = df.dropna(subset=['time', 'value', 'value_variable_label'])
    df = df[(df['time'] >= 2014) & (df['time'] <= 2024)]

    # Logik zur Identifizierung von Gesamtwerten (um Doppelzählungen zu vermeiden, falls Unterkategorien existieren)
    # Wir suchen nach 'Gesamt' oder 'Insgesamt' in den Attributspalten
    mask_total = pd.Series(False, index=df.index)
    attr_cols = ['2_variable_attribute_label', '2_variable_attribute_code']
    
    found_total_logic = False
    for col in attr_cols:
        if col in df.columns:
            mask_total = mask_total | df[col].astype(str).str.contains('Insgesamt|Gesamt|Total', case=False, na=False)
            found_total_logic = True
            
    if found_total_logic and mask_total.any():
        plot_df = df[mask_total]
    else:
        plot_df = df

    # Gruppierung der Daten für das Liniendiagramm
    plot_df = plot_df.groupby(['time', 'value_variable_label'], as_index=False)['value'].sum()

    if plot_df.empty:
        raise ValueError('Keine Daten gefunden, um die Visualisierung zu erstellen.')

    # Visualisierung
    sns.set_theme(style='whitegrid', context='talk')
    plt.figure(figsize=(12, 7))
    
    # Sortierung nach Zeit sicherstellen
    plot_df = plot_df.sort_values('time')
    
    ax = sns.lineplot(
        data=plot_df, 
        x='time', 
        y='value', 
        hue='value_variable_label', 
        marker='o', 
        linewidth=2.5, 
        palette='viridis'
    )

    ax.set_title('Entwicklung der Gästeankünfte und -übernachtungen in Rettenberg (2014–2024)', pad=20)
    ax.set_xlabel('Jahr')
    ax.set_ylabel('Anzahl')
    
    # X-Achse auf ganze Jahre formatieren
    ax.xaxis.set_major_locator(MaxNLocator(integer=True))
    
    # Y-Achse mit Tausendertrennzeichen formatieren
    ax.yaxis.set_major_formatter(FuncFormatter(lambda x, p: f"{int(x):,}".replace(',', '.')))
    
    plt.legend(title='Kennzahl', bbox_to_anchor=(1.05, 1), loc='upper left')
    plt.tight_layout()
    
    plt.savefig(plot_path, dpi=300, bbox_inches='tight')
    plt.close()

except Exception as e:
    print(f"Fehler aufgetreten: {str(e)}")

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Set visual style
sns.set_theme(style='whitegrid', context='talk')

# File paths
input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '15653', 'dataset.csv')
output_dir = os.path.join('.', 'src', 'resources', 'output', '15653')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '1_uebernachtungen_inland_ausland_gruppiert.png')

# Handle encoding error: 'utf-8' failed with 0xe4 (ä), which suggests 'latin-1' or 'iso-8859-1'
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Clean string columns
for col in df.select_dtypes(include=['object', 'string', 'category']).columns:
    df[col] = df[col].astype('string').str.strip()

# Convert numeric columns
df['value'] = pd.to_numeric(df['value'], errors='coerce')
df['time'] = pd.to_numeric(df['time'], errors='coerce').astype('Int64')

# Filter for relevant data
# Filter 1: Only 'Gästeübernachtungen in Beherbergungsbetrieben'
mask_ov = df['value_variable_label'].astype('string') == 'Gästeübernachtungen in Beherbergungsbetrieben'
df_ov = df.loc[mask_ov].copy()

# Filter 2: Only 'Inland' and 'Ausland'
# We use case-insensitive matching to be robust
df_ov['origin_clean'] = df_ov['2_variable_attribute_label'].astype('string').str.lower()
df_ov = df_ov[df_ov['origin_clean'].isin(['inland', 'ausland'])].copy()

# Drop rows with missing values in key columns
df_ov = df_ov.dropna(subset=['value', 'time'])

# Group by year and origin
grp = df_ov.groupby(['time', '2_variable_attribute_label'], as_index=False)['value'].sum()

# Pivot for plotting
pv = grp.pivot(index='time', columns='2_variable_attribute_label', values='value')

# Ensure both columns exist
for col in ['Inland', 'Ausland']:
    if col not in pv.columns:
        pv[col] = 0

pv = pv[['Inland', 'Ausland']].fillna(0).sort_index()

# Calculate shares for the title/labels
total_inland = pv['Inland'].sum()
total_ausland = pv['Ausland'].sum()
total_sum = total_inland + total_ausland
share_inland = (total_inland / total_sum * 100) if total_sum > 0 else 0
share_ausland = (total_ausland / total_sum * 100) if total_sum > 0 else 0

# Plotting
years = pv.index.astype(int)
x = np.arange(len(years))
width = 0.35

fig, ax = plt.subplots(figsize=(12, 7), dpi=150)

# Scale values to thousands for better readability
scaled_inland = pv['Inland'] / 1000.0
scaled_ausland = pv['Ausland'] / 1000.0

rects1 = ax.bar(x - width/2, scaled_inland, width, label='Inland', color='#1f77b4', edgecolor='white')
rects2 = ax.bar(x + width/2, scaled_ausland, width, label='Ausland', color='#ff7f0e', edgecolor='white')

# Labels and Title
ax.set_xlabel('Jahr')
ax.set_ylabel('Übernachtungen (in Tausend)')
ax.set_title(f'Übernachtungen nach Herkunft (Inland vs. Ausland)\n'
             f'Gesamtanteil: Inland {share_inland:.1f}%, Ausland {share_ausland:.1f}%', pad=20)
ax.set_xticks(x)
ax.set_xticklabels(years)
ax.legend(title='Herkunft')

# Add value labels on top of bars
def autolabel(rects):
    for rect in rects:
        height = rect.get_height()
        ax.annotate(f'{height:.1f}',
                    xy=(rect.get_x() + rect.get_width() / 2, height),
                    xytext=(0, 3),  # 3 points vertical offset
                    textcoords="offset points",
                    ha='center', va='bottom', fontsize=10)

autolabel(rects1)
autolabel(rects2)

sns.despine()
plt.tight_layout()

# Save the plot
plt.savefig(output_path, bbox_inches='tight')
plt.close(fig)

# -*- coding: utf-8 -*-
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# Backend für Headless-Umgebungen setzen
plt.switch_backend('Agg')

input_path = r'./src/resources/data/study/15653/dataset.csv'
output_path = r'./src/resources/output/15653/2_ankuenfte_uebernachtungen.png'

# Fehlerbehebung: UnicodeDecodeError beheben durch Verwendung von 'latin-1'
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Datenbereinigung
for col in df.columns:
    if df[col].dtype == object:
        df[col] = df[col].replace({'': np.nan, 'NA': np.nan, 'NaN': np.nan})

# Konvertierung numerischer Spalten
if 'time' in df.columns:
    df['time'] = pd.to_numeric(df['time'], errors='coerce')
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Sicherstellen, dass benötigte Spalten existieren
needed_cols = ['time', 'value', 'value_variable_label', 'value_variable_code', '2_variable_attribute_code', '2_variable_attribute_label']
for c in needed_cols:
    if c not in df.columns:
        df[c] = np.nan

# FEHLERBEHEBUNG: np.where mit gemischten Typen (String und NaN-Float) vermeiden.
# Wir initialisieren die Spalte als Objekt-Typ (Strings).
df['metric'] = None

# Identifikation der Metriken (Ankünfte vs. Übernachtungen)
# Wir nutzen .loc für sichere Zuweisung
mask_gank = df['value_variable_code'].astype(str).str.contains('GANK', na=False)
mask_gueb = df['value_variable_code'].astype(str).str.contains('GUEB', na=False)

df.loc[mask_gank, 'metric'] = 'Gästeankünfte'
df.loc[mask_gueb, 'metric'] = 'Gästeübernachtungen'

# Fallback-Suche über Labels
mask_ank = df['metric'].isna() & df['value_variable_label'].astype(str).str.contains('ankünf', case=False, na=False)
mask_ueb = df['metric'].isna() & df['value_variable_label'].astype(str).str.contains('übernacht', case=False, na=False)
df.loc[mask_ank, 'metric'] = 'Gästeankünfte'
df.loc[mask_ueb, 'metric'] = 'Gästeübernachtungen'

# Falls immer noch leer, nutzen wir das Original-Label als Fallback
df['metric'] = df['metric'].fillna(df['value_variable_label'])

# Filtern auf valide Daten
df_filt = df.dropna(subset=['time', 'value', 'metric'])

# Filtern auf "Insgesamt"-Werte, um Doppelzählungen durch Unterkategorien zu vermeiden
mask_total = df_filt['2_variable_attribute_label'].astype(str).str.lower().str.contains('insgesamt', na=False) | \
             df_filt['2_variable_attribute_code'].isna()

df_total = df_filt[mask_total]

# Falls durch das Filtern alles verloren ging, nehmen wir die Rohdaten
if df_total.empty:
    df_total = df_filt.copy()

# Aggregation der Daten pro Jahr und Metrik
plot_df = df_total.groupby(['time', 'metric'], as_index=False)['value'].sum()

if plot_df.empty:
    fig, ax = plt.subplots(figsize=(10, 4), dpi=150)
    ax.text(0.5, 0.5, 'Keine Daten für Ankünfte/Übernachtungen verfügbar', ha='center', va='center', fontsize=12)
    ax.axis('off')
    plt.tight_layout()
    plt.savefig(output_path)
    plt.close()
else:
    # Visualisierung
    sns.set_theme(style='whitegrid')
    fig, ax = plt.subplots(figsize=(12, 6), dpi=150)
    
    # Gruppiertes Balkendiagramm
    sns.barplot(data=plot_df, x='time', y='value', hue='metric', ax=ax)

    years_sorted = np.sort(plot_df['time'].unique())
    year_min = int(years_sorted[0])
    year_max = int(years_sorted[-1])
    ax.set_title(f'Verhältnis Ankünfte zu Übernachtungen pro Jahr ({year_min}–{year_max})')

    # Formatierung der Y-Achse (Tausendertrennzeichen)
    ax.get_yaxis().set_major_formatter(plt.matplotlib.ticker.FuncFormatter(lambda x, p: format(int(x), ',').replace(',', '.')))
    
    ax.legend(title='Kategorie', loc='upper left', bbox_to_anchor=(1, 1))
    
    plt.tight_layout()
    plt.savefig(output_path, bbox_inches='tight')
    plt.close()
