import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# The error 'UnicodeDecodeError' with byte 0xf6 (ö) suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# common for German datasets (e.g., "Bevölkerung").
try:
    df = pd.read_csv('./src/resources/data/study/13473/dataset.csv', sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv('./src/resources/data/study/13473/dataset.csv', sep=';', encoding='latin-1')

# Filter data: exclude 'Insgesamt' to focus on gender distribution
# Based on the previous code, '2_variable_attribute_label' contains the gender categories
gender_data = df[df['2_variable_attribute_label'] != 'Insgesamt'].copy()

# Ensure 'value' is numeric
gender_data['value'] = pd.to_numeric(gender_data['value'], errors='coerce')

# Grouping to ensure we have unique values for the bar chart
gender_summary = gender_data.groupby('2_variable_attribute_label')['value'].sum().reset_index()

# Visualization
plt.figure(figsize=(10, 7))
sns.set_style("whitegrid")

# Create the bar chart
colors = ['#3498db', '#e74c3c'] # Blue for Male, Red/Pink for Female (typical)
bars = plt.bar(gender_summary['2_variable_attribute_label'], 
               gender_summary['value'], 
               color=colors[:len(gender_summary)], 
               edgecolor='black', 
               alpha=0.85)

# Add labels and title
plt.xlabel('Geschlecht (Gender)', fontsize=12, fontweight='bold')
plt.ylabel('Bevölkerung (Population Count)', fontsize=12, fontweight='bold')
plt.title('Gender Distribution of the Population in Hof', fontsize=14, fontweight='bold', pad=15)

# Add value labels on top of bars
for bar in bars:
    height = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2., height + (height * 0.01),
             f'{int(height):,}',
             ha='center', va='bottom', fontsize=11, fontweight='bold')

# Ensure output directory exists
output_dir = './src/resources/output/13473/'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

plt.tight_layout()
plt.savefig(os.path.join(output_dir, '0_geschlechterverteilung.png'), dpi=300)
plt.show()

print("Visualization generated and saved successfully.")

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# The error 'UnicodeDecodeError' with byte 0xf6 (ö) suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# common in German datasets where 'ö' is used.
file_path = './src/resources/data/study/13473/dataset.csv'
output_path = './src/resources/output/13473/1_bevoelkerung_geschlecht_vergleich.png'

try:
    df = pd.read_csv(file_path, sep=';', encoding='latin-1')
except UnicodeDecodeError:
    df = pd.read_csv(file_path, sep=';', encoding='iso-8859-1')

# Ensure output directory exists
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# Filter for the relevant labels to ensure we only plot what's requested
# Labels: 'männlich', 'weiblich', 'Insgesamt'
target_labels = ['männlich', 'weiblich', 'Insgesamt']
plot_df = df[df['2_variable_attribute_label'].isin(target_labels)].copy()

# Convert value to numeric just in case
plot_df['value'] = pd.to_numeric(plot_df['value'], errors='coerce')

# Sort to have a consistent order: männlich, weiblich, Insgesamt
plot_df['2_variable_attribute_label'] = pd.Categorical(
    plot_df['2_variable_attribute_label'], 
    categories=['männlich', 'weiblich', 'Insgesamt'], 
    ordered=True
)
plot_df = plot_df.sort_values('2_variable_attribute_label')

# Visualization
plt.figure(figsize=(10, 6))
sns.set_style("whitegrid")

# Define colors
colors = ['#3498db', '#e74c3c', '#2ecc71']

# Create bar chart
ax = sns.barplot(
    data=plot_df, 
    x='2_variable_attribute_label', 
    y='value', 
    palette=colors,
    hue='2_variable_attribute_label',
    legend=False
)

# Add data labels on top of bars
for p in ax.patches:
    ax.annotate(f'{int(p.get_height()):,}', 
                (p.get_x() + p.get_width() / 2., p.get_height()), 
                ha='center', va='center', 
                xytext=(0, 9), 
                textcoords='offset points',
                fontsize=11, fontweight='bold')

# Labels and Title
plt.xlabel('Geschlecht (Gender Category)', fontsize=12)
plt.ylabel('Bevölkerung (Population Count)', fontsize=12)
plt.title('Comparison of Total Population vs. Gender-Specific Subgroups\n(Data Integrity Check)', 
          fontsize=14, fontweight='bold', pad=20)

# Adjust y-axis to give space for labels
plt.ylim(0, plot_df['value'].max() * 1.15)

plt.tight_layout()
plt.savefig(output_path, dpi=300)
plt.close()

print(f"Visualization saved to {output_path}")

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# The error 'UnicodeDecodeError: utf-8' suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# which is common for German datasets containing umlauts (like 'ö' in 'Bevölkerung').
try:
    df = pd.read_csv('./src/resources/data/study/13473/dataset.csv', sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv('./src/resources/data/study/13473/dataset.csv', sep=';', encoding='latin-1')

# Filter the data as requested: '2_variable_attribute_label' == 'Insgesamt'
df_filtered = df[df['2_variable_attribute_label'] == 'Insgesamt'].copy()

# Ensure 'value' is numeric
df_filtered['value'] = pd.to_numeric(df_filtered['value'], errors='coerce')

# Prepare data for plotting
df_plot = df_filtered[['1_variable_code', 'value']].copy()

# Set visual style
sns.set_style("whitegrid")
plt.figure(figsize=(10, 6))

# Create the bar chart
x_positions = np.arange(len(df_plot))
bars = plt.bar(x_positions, df_plot['value'], color=['#2E86AB', '#A23B72'], alpha=0.8, edgecolor='black', linewidth=1.2)

# Labels and Title
plt.xlabel('Administrative Level (1_variable_code)', fontsize=12, fontweight='bold')
plt.ylabel('Population Count', fontsize=12, fontweight='bold')
plt.title('Population of Hof by Administrative Level\n(Filtered for "Insgesamt")', fontsize=14, fontweight='bold', pad=20)

# Set x-axis ticks
plt.xticks(x_positions, df_plot['1_variable_code'], fontsize=11)

# Add value labels on top of bars
for bar in bars:
    height = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2., height + (max(df_plot['value']) * 0.01),
             f'{int(height):,}', ha='center', va='bottom', fontweight='bold')

# Adjust y-axis limit to provide space for labels
plt.ylim(0, max(df_plot['value']) * 1.15)

plt.tight_layout()

# Save the output
import os
output_dir = './src/resources/output/13473/'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

plt.savefig(os.path.join(output_dir, '2_population_by_administrative_level.png'), dpi=300, bbox_inches='tight')
plt.close()

print("Visualization created successfully using latin-1 encoding.")
