library(readr)
library(dplyr)
library(stringr)
library(tidyr)
library(ggplot2)
library(scales)

# Set encoding options
options(encoding = "UTF-8")

# Define paths
input_path <- file.path(".", "src", "resources", "data", "study", "103779", "dataset.csv")
output_dir <- file.path(".", "src", "resources", "output", "103779")
dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)

# Load data
raw <- readr::read_csv(input_path, locale = locale(encoding = "UTF-8"), show_col_types = FALSE)

# Data Cleaning
df <- raw %>%
  mutate(across(where(is.character), function(x) {
    y <- x
    y <- stringr::str_squish(y)
    y[!is.na(y) & tolower(y) %in% c("nan", "na", "null", "none")] <- NA_character_
    y
  }))

n_total <- nrow(df)

# Logic for "Available"
# 1. FotoURL: must be non-NA and look like a URL
valid_url <- !is.na(df$FotoURL) & stringr::str_detect(df$FotoURL, "^https?://")

# 2. Begründung: must be non-NA and not the specific placeholder
placeholder <- tolower("Alteintragung (Aktualisierung vorgesehen)")
# Note: The column name in the CSV might be 'Begr_ndung' due to encoding/import
begr_col <- if ("Begr_ndung" %in% names(df)) "Begr_ndung" else "Begründung"
valid_begr <- !is.na(df[[begr_col]]) & tolower(stringr::str_squish(df[[begr_col]])) != placeholder

n_foto <- sum(valid_url, na.rm = TRUE)
n_begr <- sum(valid_begr, na.rm = TRUE)

# Prepare data for plotting
summary_long <- tibble(
  Feld = rep(c("FotoURL", "Begründung"), each = 2),
  Status = rep(c("vorhanden", "fehlend"), times = 2),
  Anzahl = c(n_foto, n_total - n_foto, n_begr, n_total - n_begr)
) %>%
  group_by(Feld) %>%
  mutate(Prozent = if (sum(Anzahl) > 0) Anzahl / sum(Anzahl) else 0) %>%
  ungroup() %>%
  mutate(Feld = factor(Feld, levels = c("FotoURL", "Begründung")),
         Status = factor(Status, levels = c("vorhanden", "fehlend")))

# Create labels for the 'vorhanden' segments
# Using big.mark = "'" to avoid confusion with decimal.mark = ","
labels_avail <- summary_long %>%
  filter(Status == "vorhanden") %>%
  mutate(Label = paste0(format(Anzahl, big.mark = "'", scientific = FALSE),
                        " (", scales::percent(Prozent, accuracy = 0.1, decimal.mark = ","), ")"))

# Create total labels for the top of the bars
totals_df <- summary_long %>%
  group_by(Feld) %>%
  summarise(Total = sum(Anzahl), .groups = "drop") %>%
  mutate(Total_lab = paste0("N=", format(Total, big.mark = "'", scientific = FALSE)))

# Visualization
title_txt <- "Verfügbarkeit von Foto-URLs und Begründungen"
subtitle_txt <- paste0("Gesamtzahl Objekte: ", format(n_total, big.mark = "'", scientific = FALSE))

p <- ggplot(summary_long, aes(x = Feld, y = Anzahl, fill = Status)) +
  geom_col(width = 0.6, color = "grey25", linewidth = 0.3) +
  # Explicitly map Status in geom_text to avoid 'Status not found' error if inherited incorrectly
  geom_text(data = labels_avail, 
            aes(x = Feld, y = Anzahl, label = Label, fill = Status), 
            position = position_stack(vjust = 0.5), 
            color = "white", size = 4.2, fontface = "bold", inherit.aes = FALSE) +
  geom_text(data = totals_df, 
            aes(x = Feld, y = Total, label = Total_lab), 
            vjust = -0.5, size = 4, inherit.aes = FALSE) +
  scale_fill_manual(values = c("vorhanden" = "#2E7D32", "fehlend" = "#D3D3D3"), 
                    labels = c("Vorhanden", "Fehlend/Platzhalter")) +
  scale_y_continuous(labels = label_number(big.mark = "'", decimal.mark = ","), 
                     expand = expansion(mult = c(0, 0.15))) +
  labs(title = title_txt, 
       subtitle = subtitle_txt, 
       x = "Datenfeld", 
       y = "Anzahl der Denkmale", 
       fill = "Status", 
       caption = "Quelle: Denkmalliste Kreis Segeberg") +
  theme_minimal(base_size = 13) +
  theme(legend.position = "top", 
        plot.title = element_text(face = "bold", size = 16),
        axis.title.x = element_text(margin = margin(t = 10)),
        axis.title.y = element_text(margin = margin(r = 10)))

# Save output
plot_path <- file.path(output_dir, "2_verfuegbarkeit_foto_begruendung.png")
ggsave(filename = plot_path, plot = p, width = 10, height = 6, dpi = 300)

# Save CSV summaries
readr::write_csv(summary_long, file.path(output_dir, "eda_availability_summary.csv"))
