question='Wie effizient finden Nutzer Informationen auf der Website?' visualization="Kombiniertes Diagramm (Linie/Balken) mit 'Datum' auf der x-Achse und 'Seiten_pro_Besuch' sowie 'Bounce_Rate' auf den y-Achsen." rationale="Durch die Gegenüberstellung von 'Seiten_pro_Besuch' und 'Bounce_Rate' über den Zeitraum ('Datum') lässt sich analysieren, ob Nutzer bei kurzen Besuchen (hohe Bounce-Rate) dennoch die gesuchten Informationen finden. -> Code:
if (!require("pacman")) install.packages("pacman")
pacman::p_load(tidyverse, lubridate, scales)

df <- read.csv("./src/resources/data/study/32830/dataset.csv", sep = ",", encoding = "UTF-8", stringsAsFactors = FALSE)

df_clean <- df %>%
  filter(!is.na(Datum) & Datum != "" & Datum != "nan") %>%
  select(-matches("Unnamed"))

df_clean$Datum <- dmy(df_clean$Datum)

clean_percent <- function(x) {
  as.numeric(gsub("%", "", x)) / 100
}

df_clean$Bounce_Rate <- clean_percent(df_clean$Bounce_Rate)
df_clean$Seiten_pro_Besuch <- as.numeric(df_clean$Seiten_pro_Besuch)

df_clean <- df_clean %>% arrange(Datum)

dir.create("./src/resources/output/32830", recursive = TRUE, showWarnings = FALSE)

coeff <- max(df_clean$Seiten_pro_Besuch, na.rm = TRUE) / max(df_clean$Bounce_Rate, na.rm = TRUE)

p <- ggplot(df_clean, aes(x = Datum)) +
  geom_bar(aes(y = Seiten_pro_Besuch, fill = "Seiten pro Besuch"), stat = "identity", alpha = 0.7) +
  geom_line(aes(y = Bounce_Rate * coeff, color = "Absprungrate (Bounce Rate)"), size = 1.2, group = 1) +
  geom_point(aes(y = Bounce_Rate * coeff, color = "Absprungrate (Bounce Rate)"), size = 2) +
  scale_y_continuous(
    name = "Seiten pro Besuch",
    sec.axis = sec_axis(~./coeff, name = "Absprungrate", labels = percent_format(accuracy = 1))
  ) +
  scale_x_date(date_labels = "%d.%m.", date_breaks = "2 days") +
  scale_fill_manual(values = c("Seiten pro Besuch" = "steelblue")) +
  scale_color_manual(values = c("Absprungrate (Bounce Rate)" = "firebrick")) +
  labs(
    title = "Effizienz der Informationssuche: Seiten pro Besuch vs. Absprungrate",
    subtitle = "Kreis Kleve Web-Analytics - September 2024",
    x = "Datum",
    fill = "Metrik (Balken)",
    color = "Metrik (Linie)"
  ) +
  theme_minimal() +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1),
    legend.position = "bottom",
    plot.title = element_text(face = "bold", size = 14),
    panel.grid.minor = element_blank()
  )

ggsave("./src/resources/output/32830/1_effizienz_suche_kombidiagramm.png", plot = p, width = 10, height = 6, dpi = 300)

if (!require("pacman")) install.packages("pacman")
pacman::p_load(tidyverse, lubridate, scales)

df <- read.csv("./src/resources/data/study/32830/dataset.csv", 
               sep = ",", 
               encoding = "UTF-8", 
               stringsAsFactors = FALSE)

df_clean <- df %>%
  filter(!is.na(Datum)) %>%
  mutate(
    Datum = dmy(Datum),
    Anteil_Besuche_mit_Cookies = as.numeric(gsub("%", "", Anteil_Besuche_mit_Cookies)) / 100
  ) %>%
  arrange(Datum)

dir.create("./src/resources/output/32830/", recursive = TRUE, showWarnings = FALSE)

png("./src/resources/output/32830/2_cookie_akzeptanz_verlauf.png", width = 1000, height = 600, res = 120)

  geom_area(fill = "steelblue", alpha = 0.4) +
  geom_line(color = "steelblue", size = 1.2) +
  geom_point(color = "darkblue", size = 2) +
  scale_y_continuous(labels = percent_format(accuracy = 0.01), 
                     limits = c(0, max(df_clean$Anteil_Besuche_mit_Cookies, na.rm = TRUE) * 1.2)) +
  scale_x_date(date_labels = "%d.%m.", date_breaks = "2 days") +
  labs(title = "Zeitlicher Verlauf der Cookie-Akzeptanz (September 2024)",
       subtitle = "Anteil der Besuche mit Cookies auf kreis-kleve.de",
       x = "Datum",
       y = "Anteil Besuche mit Cookies (%)",

  theme_minimal() +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1),
