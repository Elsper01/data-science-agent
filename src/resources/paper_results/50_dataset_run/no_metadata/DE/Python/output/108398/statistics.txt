all_cols = list(df.columns)

def find_precip_col(site, columns):
    exact_map = {'KBR': 'KBR_NID1_MET', 'KKB': 'KKB_NIED_MET', 'KKK': 'KKK_NIED_MET'}
    if exact_map[site] in columns:
        return exact_map[site]
    cands = [c for c in columns if c.startswith(site + '_') and c.endswith('_MET') and (('NIED' in c) or ('NID' in c))]
    return cands[0] if len(cands) > 0 else None

precip_cols = {s: find_precip_col(s, all_cols) for s in site_prefixes}

# Convert numeric columns
def to_float_series(s):

import os
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

try:
    # Pfade definieren
    input_path = './src/resources/data/study/108398/dataset.csv'
    output_dir = os.path.join('.', 'src', 'resources', 'output', '108398')
    os.makedirs(output_dir, exist_ok=True)

    # Fehlerbehebung: Encoding auf 'latin-1' (ISO-8859-1) ändern, da 'utf-8' fehlschlug (0xfc deutet auf ein 'ü' in latin-1 hin)
    # Zudem wird 'engine="python"' genutzt, um robuster gegenüber Formatierungsproblemen zu sein
    df = pd.read_csv(input_path, sep=',', encoding='latin-1', dtype=str, na_values=['', ' ', 'nan', 'NaN', 'NA', None])

    # Spaltenbereinigung
    if 'Unnamed__87' in df.columns:
        df = df.drop(columns=['Unnamed__87'])

    # Zeitstempel verarbeiten
    if 'Datum_Uhrzeit' in df.columns:
        df['Datum_Uhrzeit'] = pd.to_datetime(df['Datum_Uhrzeit'], format='%d.%m.%Y %H:%M:%S', errors='coerce', dayfirst=True)
        df = df[~df['Datum_Uhrzeit'].isna()].copy()

    # Zielspalten für Windgeschwindigkeit
    target_cols = ['KBR_WGK_TIS', 'KKB_WGK_TIS', 'KKK_WGK_TIS']
    available_cols = [c for c in target_cols if c in df.columns]

    if len(available_cols) == 0:
        raise ValueError(f'Benötigte Spalten für Windgeschwindigkeit nicht gefunden. Vorhanden: {df.columns.tolist()}')

    # Numerische Konvertierung (Komma zu Punkt)
    def to_num(x):
        if pd.isna(x):
            return np.nan
        if isinstance(x, (int, float, np.floating)):
            return float(x)
        if isinstance(x, str):
            s = x.strip()
            if s == '' or s.lower() in {'nan', 'na', 'none'}:
                return np.nan
            s = s.replace(',', '.')
            try:
                return float(s)
            except ValueError:
                return np.nan
        return np.nan

    for c in available_cols:
        df[c] = df[c].apply(to_num)

    # Daten für Visualisierung vorbereiten
    df_ws = df[available_cols].copy()
    df_ws = df_ws.dropna(how='all')

    # Statistiken speichern
    miss = df_ws.isna().mean().to_frame(name='Anteil_fehlender_Werte')
    miss.to_csv(os.path.join(output_dir, 'eda_anteil_fehlender_werte_wind.csv'), encoding='utf-8')
    
    desc = df_ws.describe(percentiles=[0.05, 0.25, 0.5, 0.75, 0.95]).T
    desc.to_csv(os.path.join(output_dir, 'eda_kennwerte_wind.csv'), encoding='utf-8')

    # Visualisierung
    sns.set_theme(style='whitegrid', context='talk', palette='colorblind')
    plt.figure(figsize=(12, 7))
    
    site_map = {'KBR_WGK_TIS': 'KBR', 'KKB_WGK_TIS': 'KKB', 'KKK_WGK_TIS': 'KKK'}
    colors = sns.color_palette('colorblind', n_colors=len(available_cols))

    # Histogramm erstellen
    for col, color in zip(available_cols, colors):
        data = df_ws[col].dropna().to_numpy()
        if data.size == 0:
            continue
        
        # Histogramm mit Transparenz für Überlagerung
        sns.histplot(data, bins=25, kde=True, label=site_map.get(col, col), 
                     color=color, alpha=0.4, element="step")
        
        # Median-Linie
        med = float(np.median(data))
        plt.axvline(med, color=color, linestyle='--', linewidth=2, alpha=0.8)

    plt.xlabel('Windgeschwindigkeit (m/s)')
    plt.ylabel('Häufigkeit')
    plt.title('Verteilung der Windgeschwindigkeiten an den Standorten\nMesszeitraum: 2023')
    plt.legend(title='Standort', frameon=True)
    
    plt.tight_layout()
    plot_path = os.path.join(output_dir, '2_windgeschwindigkeit_histogramm.png')
    plt.savefig(plot_path, dpi=300)
    plt.close()

except Exception as e:
    print(f"Fehler aufgetreten: {str(e)}")
