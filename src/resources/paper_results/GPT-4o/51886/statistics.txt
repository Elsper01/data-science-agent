library(dplyr)
library(readr)

# Korrektur der Pfade: Verwendung von Vorwärts-Slashes für bessere Kompatibilität
file_path <- './src/resources/data/study/51886/dataset.csv'

  select(time, value) %>%
  arrange(time)

  geom_line(color = "#0073C2FF", size = 1) +
  geom_point(color = "#EFC000FF", size = 2) +
  scale_x_continuous(breaks = seq(min(filtered_data$time), max(filtered_data$time), by = 1)) +
  labs(title = "Entwicklung der Baufertigstellungen in Oberschleißheim",
       subtitle = "Zeitraum 2014 - 2024 (Kategorie: Insgesamt)",
       x = "Jahr",
       y = "Anzahl der Baufertigstellungen",
       caption = "Quelle: Datensatz 51886") +
  theme_minimal() +
  theme(panel.grid.minor = element_blank())

# Pakete laden (ohne install.packages, da dies in eingeschränkten Umgebungen oft fehlschlägt)
# Es wird davon ausgegangen, dass die Standard-Bibliotheken vorhanden sind.
library(ggplot2)
library(dplyr)
library(readr)

# Daten laden
# Korrektur des Pfades (Backslashes in R müssen oft escaped werden oder durch Slashes ersetzt werden)
file_path <- "./src/resources/data/study/51886/dataset.csv"

# Sicherstellen, dass die Datei existiert, bevor sie gelesen wird
if (file.exists(file_path)) {
  df <- read_csv2(file_path, locale = locale(encoding = 'UTF-8'))

  # Daten filtern: 
  # 1. 'Insgesamt' ausschließen, da es die Summe der anderen Kategorien ist
  # 2. NAs in der relevanten Spalte entfernen
  df_filtered <- df %>%
    filter(!is.na(`2_variable_attribute_label`), 
           `2_variable_attribute_label` != "Insgesamt")

  # Daten für das gestapelte Balkendiagramm vorbereiten
  # Wir gruppieren nach Zeit und Raumgröße
  df_plot <- df_filtered %>%
    group_by(time, `2_variable_attribute_label`) %>%
    summarise(total_value = sum(value, na.rm = TRUE), .groups = 'drop')

  # Gestapeltes Balkendiagramm erstellen
  plot <- ggplot(df_plot, aes(x = factor(time), y = total_value, fill = `2_variable_attribute_label`)) +
    geom_bar(stat = "identity", position = "stack") +
    labs(title = "Verteilung der fertiggestellten Wohnungen nach Raumgrößen",
         subtitle = "Zeitraum: 2014 - 2024",
         x = "Jahr",
         y = "Anzahl der Wohnungen",
         fill = "Raumgrößen") +
    theme_minimal() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1),
          legend.position = "bottom")

  # Output-Verzeichnis sicherstellen (in R Skripten meist manuell oder durch ggsave)
  output_path <- "./src/resources/output/51886/1_verteilung_raumgroessen.png"
  
  # Plot speichern
  ggsave(filename = output_path, plot = plot, width = 10, height = 6, dpi = 300)
  
  print("Visualisierung erfolgreich erstellt und gespeichert.")
} else {
  stop(paste("Datei nicht gefunden:", file_path))
}

# Pakete laden (ohne install.packages, da dies in eingeschränkten Umgebungen oft fehlschlägt)
# Es wird davon ausgegangen, dass die Standard-Bibliotheken vorhanden sind.
library(ggplot2)
library(dplyr)
library(readr)

# Daten laden mit korrektem Pfad-Handling (Verwendung von Vorwärts-Slashes für R)
file_path <- "./src/resources/data/study/51886/dataset.csv"
df <- read_csv2(file_path, locale = locale(encoding = 'UTF-8'))

# Daten filtern: Nur negative Werte (Korrekturen/Rückbau)
# Wir stellen sicher, dass 'value' numerisch ist
df$value <- as.numeric(df$value)
negative_values <- df %>% filter(value < 0)

# Falls keine negativen Werte vorhanden sind, geben wir eine Meldung aus, 
# aber für die Visualisierung nehmen wir an, dass welche existieren.
if(nrow(negative_values) > 0) {
  
  # Balkendiagramm erstellen
  # Wir nutzen abs(value) für die Höhe oder lassen sie negativ für die Darstellung unter der Nulllinie
  plot <- ggplot(negative_values, aes(x = factor(time), y = value, fill = `2_variable_attribute_label`)) +
    geom_bar(stat = "identity", position = "dodge") +
    # Eine horizontale Linie bei 0 zur Verdeutlichung
    geom_hline(yintercept = 0, color = "black") +
    labs(title = "Jahre mit Korrekturen oder Rückbaumaßnahmen nach Wohnungsgröße",
         subtitle = "Negative Werte repräsentieren Abgänge/Korrekturen",
         x = "Jahr",
         y = "Anzahl der Korrekturen/Rückbauten (negativ)",
         fill = "Wohnungsgröße") +
    theme_minimal() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1))

  # Verzeichnis erstellen falls nicht vorhanden (optional, ggsave macht das oft nicht automatisch)
  output_path <- "./src/resources/output/51886/2_korrekturen_rueckbau.png"
  
  # Plot speichern
  ggsave(filename = output_path, plot = plot, width = 10, height = 6, dpi = 300)
  
  print("Visualisierung erfolgreich erstellt.")
} else {
  print("Keine negativen Werte im Datensatz gefunden, die visualisiert werden könnten.")
}
