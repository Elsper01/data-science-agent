df <- df %>%

# Load necessary libraries
library(readr)
library(dplyr)
library(ggplot2)
library(lubridate)
library(scales)

# Set paths
input_path <- file.path(".", "src", "resources", "data", "study", "16492", "dataset.csv")
output_dir <- file.path(".", "src", "resources", "output", "16492")
dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)
output_path <- file.path(output_dir, "2_streudiagramm_wert_vs_endzeit.png")

# Read data
df <- read_csv(input_path, locale = locale(encoding = "UTF-8"), show_col_types = FALSE)

# Data Cleaning and Transformation
df_clean <- df %>%
  mutate(
    # Convert time strings to POSIXct
    end_time = ymd_hms(end_measure, quiet = TRUE),
    # Convert value to numeric
    val_num = as.numeric(value),
    # Convert validated to logical (assuming 1/0 or TRUE/FALSE)
    is_valid = as.logical(as.integer(validated))
  ) %>%
  # Filter out rows with missing critical values
  filter(!is.na(val_num), !is.na(end_time)) %>%
  # Filter for validated data if the column exists and has values
  filter(is.na(is_valid) | is_valid == TRUE) %>%
  # Remove duplicates
  distinct(end_time, .keep_all = TRUE) %>%
  arrange(end_time)

# Check if we have data left to plot
if (nrow(df_clean) > 0) {
  
  # Calculate statistics for annotation and scaling
  mean_val <- mean(df_clean$val_num, na.rm = TRUE)
  min_time <- min(df_clean$end_time)
  max_time <- max(df_clean$end_time)
  time_range_hours <- as.numeric(difftime(max_time, min_time, units = "hours"))
  
  # Determine date breaks dynamically to avoid 'finite number' errors in seq.POSIXt
  # If range is small, use hours; if large, use days.
  date_break_str <- if(time_range_hours < 48) "6 hours" else "24 hours"
  
  # Create the plot
  p <- ggplot(df_clean, aes(x = end_time, y = val_num)) +
    # Scatter points colored by hour of day to highlight diurnal patterns
    geom_point(aes(color = hour(end_time)), size = 2, alpha = 0.7) +
    # LOESS smooth to show the trend
    geom_smooth(method = "loess", formula = y ~ x, se = TRUE, color = "#2C7BB6", fill = "#D1E5F0") +
    # Horizontal line for the mean
    geom_hline(yintercept = mean_val, linetype = "dashed", color = "red", alpha = 0.6) +
    # Scales
    scale_x_datetime(
      name = "Zeitpunkt der Messung (Ende)",
      date_labels = "%d.%m. %H:%M",
      date_breaks = date_break_str
    ) +
    scale_y_continuous(
      name = "Ortsdosisleistung (µSv/h)",
      labels = label_number(decimal.mark = ",", accuracy = 0.001)
    ) +
    scale_color_viridis_c(
      name = "Stunde (0-23)",
      option = "plasma",
      guide = guide_colorbar(barwidth = 15, barheight = 0.5, title.position = "top")
    ) +
    # Labels and Theme
    labs(
      title = "Zeitlicher Verlauf der Gamma-Ortsdosisleistung",
      subtitle = "Analyse von tageszeitlichen Schwankungen und Trends",
      caption = paste0("Datenquelle: BfS | Anzahl Messpunkte: ", nrow(df_clean))
    ) +
    theme_minimal(base_size = 12) +
    theme(
      legend.position = "bottom",
      axis.text.x = element_text(angle = 45, hjust = 1),
      plot.title = element_text(face = "bold"),
      panel.grid.minor = element_blank()
    )

  # Save the plot
  ggsave(filename = output_path, plot = p, width = 10, height = 6, dpi = 300)
} else {
  cat("No valid data points found for plotting.")
}
