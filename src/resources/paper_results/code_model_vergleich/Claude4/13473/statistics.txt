# Spaltennamen identifizieren
# Wir suchen nach den Spaltennamen, die "variable_attribute_label" und "value" enthalten.
# Da make.names() oder das Einlesen oft Präfixe wie "X" hinzufügen, suchen wir flexibel.
target_col_name <- grep("variable_attribute_label", colnames(df), value = TRUE)
value_col_name <- grep("value", colnames(df), value = TRUE)

# Falls grep mehrere oder keine Ergebnisse liefert, setzen wir Defaults basierend auf der Fehlermeldung
if(length(target_col_name) == 0) target_col_name <- "X2_variable_attribute_label"
if(length(value_col_name) == 0) value_col_name <- "value"

# WICHTIG: Wir nehmen nur das erste Element, falls grep einen Vektor zurückgibt
target_col <- target_col_name[1]
value_col <- value_col_name[1]

# Daten filtern und aggregieren
# Wir nutzen .data[[var]] für tidy evaluation innerhalb von dplyr
df_filtered <- df %>%
  filter(.data[[target_col]] %in% c('männlich', 'weiblich')) %>%
  group_by(.data[[target_col]]) %>%
  summarise(total = sum(as.numeric(gsub(",", ".", .data[[value_col]])), na.rm = TRUE)) %>%
  ungroup()

# Prozente berechnen
df_filtered <- df_filtered %>%
  mutate(percentage = total / sum(total) * 100,
         label = paste0(.data[[target_col]], "\n", 
                       format(total, big.mark = ".", decimal.mark = ","), 
                       " (", round(percentage, 1), "%)"))

# Farben definieren

# Daten einlesen
# Wir nutzen check.names=FALSE, um die Spaltennamen exakt so zu erhalten, wie sie in der CSV stehen.
data <- read.csv("./src/resources/data/study/13473/dataset.csv",

# Bereinigung der Spaltennamen von BOM oder Präfixen
colnames(data) <- gsub("^ï..", "", colnames(data))
colnames(data) <- gsub("^X\\.", "", colnames(data))

# Zielspalten definieren
target_cols <- c("1_variable_label", "1_variable_attribute_code", "2_variable_attribute_label", "value")

# Sicherstellen, dass die Spalten existieren
available_cols <- intersect(target_cols, colnames(data))
tabelle <- data %>% select(all_of(available_cols))

# Funktion zur Bereinigung von Texten, um Rendering-Fehler in gridExtra zu vermeiden.
# Wir nutzen iconv, um sicherzustellen, dass der Text valides ASCII/UTF-8 ist.
clean_text_safe <- function(x) {
  if(is.character(x)) {
    # Konvertiere zu ASCII und ersetze nicht darstellbare Zeichen durch Platzhalter
    # Dies verhindert den "invalid string" Fehler bei gsub
    x <- iconv(x, from = "UTF-8", to = "ASCII//TRANSLIT", sub = "?")
  }
  return(x)
}

tabelle <- tabelle %>% mutate(across(everything(), clean_text_safe))

# Spaltennamen für die Anzeige setzen (ASCII-kompatibel)
colnames(tabelle) <- c("Regionale Ebene", 
                       "Code",
