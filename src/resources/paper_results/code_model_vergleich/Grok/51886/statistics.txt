import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# The error 'UnicodeDecodeError: utf-8' with byte 0xdf (ß) suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# which is common for German datasets (e.g., Oberschleißheim).
try:
    df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='latin-1')

# Filter out 'Insgesamt' to focus on specific room counts
viz_df = df[df['2_variable_attribute_label'] != 'Insgesamt'].copy()

# Ensure 'value' is numeric
viz_df['value'] = pd.to_numeric(viz_df['value'], errors='coerce').fillna(0)

# Group by room count label and sum the values
sums = viz_df.groupby('2_variable_attribute_label', observed=True)['value'].sum().reset_index(name='total_value')

# Helper function to sort labels like '1 Raum', '2 Räume' correctly
def room_key(label):
    nums = ''.join([c for c in str(label) if c.isdigit()])
    return int(nums) if nums else 99

sums['sort_key'] = sums['2_variable_attribute_label'].apply(room_key)
sums_sorted = sums.sort_values('sort_key').reset_index(drop=True)

# Visualization
sns.set_style('whitegrid')
fig, ax = plt.subplots(figsize=(12, 8))

# Create the bar chart
sns.barplot(data=sums_sorted, x='2_variable_attribute_label', y='total_value', ax=ax, palette='viridis', hue='2_variable_attribute_label', legend=False)

# Formatting
ax.set_title('Verteilung der fertiggestellten Wohnungen nach Raumzahl\nOberschleißheim (Gesamtzeitraum)', fontsize=16, fontweight='bold', pad=20)
ax.set_xlabel('Anzahl der Räume', fontsize=14)
ax.set_ylabel('Summe der Wohnungen', fontsize=14)
ax.tick_params(axis='x', rotation=45)

# Add value labels on top of bars
for i, v in enumerate(sums_sorted['total_value']):
    ax.text(i, v + (max(sums_sorted['total_value']) * 0.01), str(int(v)), ha='center', fontweight='bold')

plt.tight_layout()

# Save the output
output_dir = './src/resources/output/51886/'
os.makedirs(output_dir, exist_ok=True)
plt.savefig(output_dir + '1_raumverteilung.png', dpi=300, bbox_inches='tight', facecolor='white')
plt.close()

print("Visualization created successfully.")

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Set visual style
sns.set_theme(style="whitegrid")

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xdf' 
# usually indicates 'latin-1' or 'iso-8859-1' encoding (0xdf is 'ß' in latin-1).
try:
    df = pd.read_csv(r'./src/resources/data/study/51886/dataset.csv', sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(r'./src/resources/data/study/51886/dataset.csv', sep=';', encoding='latin-1')

# Filter out 'Insgesamt' as per the goal
df_viz = df[df['2_variable_attribute_label'] != 'Insgesamt'].copy()

# Helper function to sort room labels (e.g., "1 Raum", "2 Räume", ..., "7 oder mehr Räume")
def room_key(label):
    try:
        if 'oder mehr' in label:
            return 99  # Put "or more" at the end
        # Extract the first number found in the string
        num_str = ''.join(filter(str.isdigit, label))
        return int(num_str) if num_str else 0
    except:
        return 0

# Get unique labels and sort them logically
room_labels = df_viz['2_variable_attribute_label'].unique()
room_order = sorted(room_labels, key=room_key)

# Pivot the data for stacked bar chart: index=time, columns=room_count, values=value
pivot_df = df_viz.pivot_table(index='time', 
                              columns='2_variable_attribute_label', 
                              values='value', 
                              aggfunc='sum')

# Reorder columns based on room count
pivot_df = pivot_df[room_order]

# Plotting
fig, ax = plt.subplots(figsize=(12, 7))
pivot_df.plot(kind='bar', stacked=True, ax=ax, colormap='viridis', width=0.8)

# Formatting
plt.title('Composition of Completed Apartments by Room Count (Year-over-Year)', fontsize=14, pad=15)
plt.xlabel('Year', fontsize=12)
plt.ylabel('Number of Apartments', fontsize=12)
plt.legend(title='Room Count', bbox_to_anchor=(1.05, 1), loc='upper left')
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)

plt.tight_layout()

# Save the output
plt.savefig(r'./src/resources/output/51886/2_zusammensetzung_raumzahl_stacked.png', dpi=300, bbox_inches='tight')
plt.close()
