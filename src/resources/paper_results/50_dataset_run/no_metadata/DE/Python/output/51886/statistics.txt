    # Fehlerbehebung: Encoding auf 'latin-1' oder 'iso-8859-1' ändern, da 'utf-8' fehlschlug
    # Deutsche Datensätze nutzen oft Windows-1252 oder ISO-8859-1 (wegen Umlauten wie ß in Oberschleißheim)

import matplotlib.pyplot as plt

    try:

    except Exception as e:
        print(f"Error reading CSV: {e}")
        raise

# Clean column names
df.columns = [c.strip() if isinstance(c, str) else c for c in df.columns]

# Ensure necessary columns exist
cols_needed = ['time', '2_variable_code', '2_variable_label', '2_variable_attribute_code', '2_variable_attribute_label', 'value']
for c in cols_needed:
    if c not in df.columns:
        df[c] = np.nan

# Data Cleaning
df['time'] = pd.to_numeric(df['time'], errors='coerce')
df['value'] = pd.to_numeric(df['value'], errors='coerce')

if '2_variable_code' in df.columns:

    raise SystemExit(0)

# Define a logical order for room sizes if possible
# Extracting numbers from labels like "1 Raum", "2 Räume" etc.
def sort_key(label):
    import re
    nums = re.findall(r'\d+', str(label))
    return int(nums[0]) if nums else 999

unique_labels = sorted(agg['2_variable_attribute_label'].unique(), key=sort_key)
