# Korrektur der Konvertierungsfunktion: Doppelte Backslashes für Regex in R
conv_de_number <- function(x) {
  x2 <- trimws(as.character(x))
  # Bekannte Nicht-Nachweis-Strings entfernen
  x2[x2 %in% c('n.n.', 'n.n', 'n. n.', 'n.v.', 'N.N.', 'N.N', '<LD', '< LD', '<LQ', '< LQ', '-', '')] <- NA_character_
  # Sonderzeichen und Leerzeichen entfernen
  x2 <- gsub('\u00A0', '', x2, perl = TRUE)
  x2 <- gsub('\u2007|\u202F', '', x2, perl = TRUE)
  x2 <- gsub('\u00B7', '', x2, perl = TRUE)
  # FEHLERBEHEBUNG: '\\s' statt '\s'
  x2 <- gsub('\\s+', '', x2)
  # Komma zu Punkt für numerische Konvertierung
  x2 <- gsub(',', '.', x2, fixed = TRUE)
  suppressWarnings(as.numeric(x2))
}

# Bereinigung der Texte (Umlaute ersetzen für grid-Kompatibilität)
clean_text <- function(x) {
  if(is.character(x)) {
    x <- iconv(x, from = "UTF-8", to = "UTF-8", sub = "")
    x <- str_replace_all(x, "ä", "ae")
    x <- str_replace_all(x, "ö", "oe")
    x <- str_replace_all(x, "ü", "ue")
    x <- str_replace_all(x, "Ä", "Ae")
    x <- str_replace_all(x, "Ö", "Oe")
    x <- str_replace_all(x, "Ü", "Ue")
    x <- str_replace_all(x, "ß", "ss")
  }
  return(x)
}

# Korrektur der combine_titles Funktion: 
# gtable_col erwartet 'width' (Singular) statt 'widths'
combine_titles <- function(g1, g2) {
  g <- gtable::gtable_col("title_block", 
                          grobs = list(g1, g2), 
                          width = unit(1, "null"), 
                          heights = unit.c(unit(1.5, "line"), unit(1, "line")))
  return(g)
}
