# Benötigte Bibliotheken laden
library(tidyverse)

# Daten einlesen
# Fehlerbehebung: Encoding auf 'latin1' oder 'ISO-8859-1' setzen, falls 'utf-8' fehlschlägt
# Da der Fehler "Invalid multibyte sequence" auftrat, versuchen wir eine robustere Methode
file_path <- './src/resources/data/study/51886/dataset.csv'

# Versuche das Einlesen mit automatischer Erkennung oder Fallback auf Latin1
data <- tryCatch({
  read_delim(file_path, delim = ';', locale = locale(encoding = 'UTF-8'), show_col_types = FALSE)
}, error = function(e) {
  read_delim(file_path, delim = ';', locale = locale(encoding = 'ISO-8859-1'), show_col_types = FALSE)
})

# Daten filtern und aggregieren
# Wir entfernen 'Insgesamt' und NA-Werte
data_filtered <- data %>%
  filter(!is.na(`2_variable_attribute_label`)) %>%
  filter(!str_detect(`2_variable_attribute_label`, "Insgesamt")) %>%
  filter(!is.na(value))

data_aggregated <- data_filtered %>%
  group_by(`2_variable_attribute_label`) %>%
  summarise(total_value = sum(value, na.rm = TRUE)) %>%
  arrange(desc(total_value))

# Verzeichnis erstellen
dir.create('./src/resources/output/51886', recursive = TRUE, showWarnings = FALSE)

# Visualisierung erstellen
# Wir nutzen coord_flip() für bessere Lesbarkeit der Kategorien
p <- ggplot(data_aggregated, aes(x = reorder(`2_variable_attribute_label`, total_value), y = total_value)) +
  geom_bar(stat = "identity", fill = "#2E86AB", color = "black", width = 0.7) +
  coord_flip() +
  labs(title = "Fertiggestellte Wohnungen nach Raumzahl",
       subtitle = "Gesamtsumme im betrachteten Zeitraum",
       x = "Wohnungsgroesse (Raumzahl)",
       y = "Anzahl fertiggestellter Wohnungen") +
  theme_minimal(base_size = 12) +
  theme(plot.title = element_text(face = "bold", size = 14, hjust = 0.5),
        plot.subtitle = element_text(size = 11, hjust = 0.5),
        axis.title = element_text(face = "bold"),
        axis.text = element_text(size = 10),
        panel.grid.major.y = element_blank()) +
  geom_text(aes(label = total_value), hjust = -0.2, size = 4)

# Speichern des Diagramms
# Wir verwenden ggsave, da es oft robuster mit Encodings in Pfaden umgeht
ggsave('./src/resources/output/51886/1_wohnungsgroessen_fertigstellungen.png', 
       plot = p, width = 10, height = 6, dpi = 150)

library(ggplot2)
library(dplyr)
library(tidyr)
library(readr)

# Daten einlesen mit Fehlerbehandlung für Encoding
# 'ISO-8859-1' oder 'latin1' wird oft benötigt, wenn 'utf-8' fehlschlägt (Invalid multibyte sequence)
file_path <- './src/resources/data/study/51886/dataset.csv'

df <- tryCatch({
  read_delim(file_path, delim = ';', locale = locale(encoding = 'utf-8'), show_col_types = FALSE)
}, error = function(e) {
  read_delim(file_path, delim = ';', locale = locale(encoding = 'ISO-8859-1'), show_col_types = FALSE)
})

# Datenbereinigung und Filterung
df_filtered <- df %>%
  filter(!is.na(`2_variable_attribute_label`),
         `2_variable_attribute_label` != "Insgesamt",
         !is.na(value),
         !is.na(time)) %>%
  mutate(
    value = as.numeric(value),
    time = as.numeric(time)
  ) %>%
  filter(!is.na(value)) # Falls Konvertierung fehlschlug

# Aggregation der Daten
df_aggregated <- df_filtered %>%
  group_by(time, `2_variable_attribute_label`) %>%
  summarise(total_value = sum(value, na.rm = TRUE), .groups = 'drop')

# Sortierung der Kategorien für die Legende (Vermeidung von Umlaut-Problemen im Code-Vergleich)
# Wir nutzen die vorhandenen Labels direkt aus den Daten
labels_present <- unique(df_aggregated$`2_variable_attribute_label`)
df_aggregated$`2_variable_attribute_label` <- factor(df_aggregated$`2_variable_attribute_label`)

# Verzeichnis erstellen
dir.create('./src/resources/output/51886', recursive = TRUE, showWarnings = FALSE)

# Visualisierung erstellen
png('./src/resources/output/51886/2_raumverteilung_zeitverlauf.png',
