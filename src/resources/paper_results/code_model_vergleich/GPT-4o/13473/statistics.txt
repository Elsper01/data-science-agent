import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Use raw strings for file paths
file_path = r'./src/resources/data/study/13473/dataset.csv'

# Check if file exists before reading
if not os.path.exists(file_path):
    file_path = 'dataset.csv'

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xf6' 
# suggests the file is likely encoded in 'latin-1' or 'ISO-8859-1' (common for German data with umlauts like 'Ã¶')
try:
    df = pd.read_csv(file_path, sep=';', encoding='latin-1')
except Exception as e:
    # Fallback to another common encoding if latin-1 fails
    try:
        df = pd.read_csv(file_path, sep=';', encoding='ISO-8859-1')
    except Exception as e:
        raise e

# Filter out the 'Insgesamt' category and ensure we only have gender labels
# We also handle potential whitespace or case sensitivity
filtered_df = df[df['2_variable_attribute_label'].notna()].copy()
filtered_df['2_variable_attribute_label'] = filtered_df['2_variable_attribute_label'].str.strip()
filtered_df = filtered_df[filtered_df['2_variable_attribute_label'] != 'Insgesamt']

# Create a bar chart for gender distribution
plt.figure(figsize=(10, 6))
sns.set_theme(style="whitegrid")
ax = sns.barplot(x='2_variable_attribute_label', y='value', data=filtered_df, palette='viridis')

# Add labels and title
plt.title('Gender Distribution of the Population in Hof', fontsize=14)
plt.xlabel('Gender', fontsize=12)
plt.ylabel('Population Count', fontsize=12)

# Add value labels on top of bars
for p in ax.patches:
    height = p.get_height()
    ax.annotate(format(height, '.0f'), 
                   (p.get_x() + p.get_width() / 2., height), 
                   ha = 'center', va = 'center', 
                   xytext = (0, 9), 
                   textcoords = 'offset points')

plt.tight_layout()

# Ensure output directory exists
output_dir = r'./src/resources/output/13473'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

# Save the plot
output_path = os.path.join(output_dir, '0_gender_distribution.png')
plt.savefig(output_path, format='png')
plt.close()
