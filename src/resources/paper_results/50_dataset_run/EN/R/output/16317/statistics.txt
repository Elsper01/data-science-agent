in_path <- "./src/resources/data/study/16317/dataset.csv"
out_path <- "./src/resources/output/16317/0_strahlungsrate_zeitverlauf.png"

dir.create(dirname(out_path), recursive = TRUE, showWarnings = FALSE)

# Load data
# Using read_csv with explicit column types to avoid guessing errors
df <- read_csv(in_path, show_col_types = FALSE, locale = locale(encoding = "UTF-8"))

# Data Preparation
# The error in parse_date_time often occurs with complex formats or locale mismatches.
# We'll use as.POSIXct or ymd_hms which are more robust for standard ISO-like strings.
df <- df %>%
  mutate(
    # Try standard lubridate parsing which is generally safer than parse_date_time with many orders
    end_dt = ymd_hms(end_measure, quiet = TRUE),
    value_num = as.numeric(value)
  ) %>%
  # Fallback for different formats if ymd_hms fails
  mutate(
    end_dt = if_else(is.na(end_dt), as.POSIXct(end_measure, tz = "UTC"), end_dt)
  ) %>%
  filter(!is.na(end_dt), !is.na(value_num))

if (nrow(df) == 0) {
  stop("No valid data points found after date/numeric conversion.")
}

# Calculate statistics
sum_stats <- df %>% 
  summarise(
    mean_val = mean(value_num, na.rm = TRUE),
    sd_val = sd(value_num, na.rm = TRUE),
    min_time = min(end_dt, na.rm = TRUE),
    max_time = max(end_dt, na.rm = TRUE)
  )

unit_label <- df %>% filter(!is.na(unit)) %>% distinct(unit) %>% slice(1) %>% pull(unit)
if (length(unit_label) == 0 || is.na(unit_label)) unit_label <- "µSv/h"

mean_val <- sum_stats$mean_val
sd_val <- sum_stats$sd_val
min_time <- sum_stats$min_time
max_time <- sum_stats$max_time

# Identify anomalies (points outside 2 SD)
anoms <- df %>% 
  filter(abs(value_num - mean_val) >= 2 * sd_val)

# Visualization
p <- ggplot(df, aes(x = end_dt, y = value_num)) +
  # Shaded area for +/- 1 Standard Deviation
  annotate("rect", xmin = min_time, xmax = max_time, 
           ymin = mean_val - sd_val, ymax = mean_val + sd_val, 
           fill = "#A6CEE3", alpha = 0.2) +
  # Main line
  geom_line(color = "#2C7FB8", linewidth = 0.6) +
  # Mean line
  geom_hline(yintercept = mean_val, linetype = "dashed", color = "#4D4D4D", linewidth = 0.5) +
  # Highlight anomalies
  geom_point(data = anoms, aes(x = end_dt, y = value_num), color = "#D7301F", size = 1, alpha = 0.6) +
  scale_x_datetime(labels = label_date_short(), breaks = breaks_pretty(n = 10)) +
  scale_y_continuous(labels = label_number(accuracy = 0.001)) +
  labs(
    title = "Zeitverlauf der Gamma-ODL-Dosisleistungsrate in Kornbach",
    subtitle = paste0(
      "Messzeitraum: ", format(min_time, "%d.%m.%Y"), " bis ", format(max_time, "%d.%m.%Y"),
      " | Mittelwert: ", round(mean_val, 3), " ", unit_label,
      " (±", round(sd_val, 3), " SD)"
    ),
    x = "Ende des Messintervalls (UTC)",
    y = paste0("Dosisleistungsrate (", unit_label, ")"),
    caption = "Quelle: BfS ODL-Messnetz | Rote Punkte markieren Abweichungen > 2 SD"
  ) +
  theme_minimal(base_size = 11) +
  theme(
    plot.title = element_text(face = "bold", size = 12),
    plot.subtitle = element_text(color = "grey30", size = 10),
    panel.grid.minor = element_blank(),
    plot.background = element_rect(fill = "white", color = NA)
  )
