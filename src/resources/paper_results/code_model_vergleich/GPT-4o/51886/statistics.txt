import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# The error occurred because 'data.csv' was not found. 
# In this environment, the dataset is provided as 'dataset.csv'.
file_path = 'dataset.csv'

try:
    # Load the dataset
    # Using sep=None and engine='python' to automatically detect the delimiter (comma or semicolon)
    df = pd.read_csv(file_path, sep=None, engine='python', encoding='utf-8')

    # Data cleaning and preparation
    # Ensure 'value' is numeric to avoid errors during summation
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
    
    # Filter out 'Insgesamt' and drop rows with NaN values in 'value'
    # We use case-insensitive comparison and strip whitespace
    df_filtered = df[df['2_variable_attribute_label'].str.strip() != 'Insgesamt'].dropna(subset=['value'])

    # Group by '2_variable_attribute_label' and sum the 'value'
    df_grouped = df_filtered.groupby('2_variable_attribute_label')['value'].sum().reset_index()

    # Sort the values to make the distribution logical on the x-axis
    # This helps if labels are like "1 Raum", "2 Räume", etc.
    df_grouped = df_grouped.sort_values('2_variable_attribute_label')

    # Plotting
    plt.figure(figsize=(12, 7))
    sns.set_style("whitegrid")
    
    # Create the bar chart
    ax = sns.barplot(
        data=df_grouped, 
        x='2_variable_attribute_label', 
        y='value', 
        palette='viridis', 
        hue='2_variable_attribute_label', 
        legend=False
    )
    
    plt.title('Distribution of Completed Apartment Sizes (Number of Rooms)\nOberschleißheim (Total)', fontsize=14)
    plt.xlabel('Number of Rooms / Category', fontsize=12)
    plt.ylabel('Total Number of Completed Apartments', fontsize=12)
    plt.xticks(rotation=45)
    
    # Add value labels on top of bars for better readability
    for p in ax.patches:
        height = p.get_height()
        if height > 0:
            ax.annotate(format(height, '.0f'), 
                        (p.get_x() + p.get_width() / 2., height), 
                        ha = 'center', va = 'center', 
                        xytext = (0, 9), 
                        textcoords = 'offset points')

    plt.tight_layout()
    plt.show()

except Exception as e:
    print(f"An error occurred: {e}")

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np
import os

# Use raw strings for file paths
file_path = r'./src/resources/data/study/51886/dataset.csv'

# Ensure the directory exists or handle potential path issues
if not os.path.exists(file_path):
    file_path = 'dataset.csv'

# The error 'UnicodeDecodeError: utf-8 codec can't decode byte 0xdf' 
# usually indicates 'latin-1' or 'iso-8859-1' encoding, common in German datasets (0xdf is 'ß').
try:
    df = pd.read_csv(file_path, sep=';', encoding='utf-8')
except (UnicodeDecodeError, UnicodeError):
    try:
        df = pd.read_csv(file_path, sep=';', encoding='latin-1')
    except Exception as e:
        import sys
        print(f"Error reading file: {e}")
        sys.exit(1)
except FileNotFoundError:
    import sys
    print(f"Error: File not found at {file_path}")
    sys.exit(1)

# Data cleaning and preparation
# Remove rows where '2_variable_attribute_label' is 'Insgesamt'
df = df[df['2_variable_attribute_label'] != 'Insgesamt'].copy()

# Handle missing values and negative values
# Replace negative values with NaN for visualization purposes as per rationale
df['value'] = pd.to_numeric(df['value'], errors='coerce')
df['value'] = df['value'].apply(lambda x: np.nan if x < 0 else x)

# Create a pivot table for the stacked bar chart
# Index: time (x-axis), Columns: 2_variable_attribute_label (legend), Values: value (y-axis)
df_pivot = df.pivot_table(index='time', columns='2_variable_attribute_label', values='value', aggfunc='sum')

# Plotting
fig, ax = plt.subplots(figsize=(12, 8))
df_pivot.plot(kind='bar', stacked=True, colormap='viridis', width=0.8, ax=ax)

# Customizing the plot
plt.title('Jährliche Zusammensetzung der Wohnungsgrößen', fontsize=14)
plt.xlabel('Jahr', fontsize=12)
plt.ylabel('Anzahl der Wohnungen', fontsize=12)
plt.xticks(rotation=45)
plt.legend(title='Wohnungsgröße', bbox_to_anchor=(1.05, 1), loc='upper left')
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.tight_layout()

# Save the plot
output_dir = r'./src/resources/output/51886'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)
    
output_path = os.path.join(output_dir, '2_stacked_bar_chart.png')
plt.savefig(output_path, dpi=300)
print(f"Plot saved to {output_path}")
plt.close()
