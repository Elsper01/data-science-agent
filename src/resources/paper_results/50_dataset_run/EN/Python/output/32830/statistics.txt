df = pd.read_csv(path, sep=',', encoding='utf-8')

if 'Unnamed__14' in df.columns:
    if df['Unnamed__14'].isna().all() or df['Unnamed__14'].eq('').all():
        df = df.drop(columns=['Unnamed__14'])

for c in df.columns:
    if isinstance(c, str):
        try:
            df[c] = df[c].apply(lambda x: x.strip() if isinstance(x, str) else x)
        except Exception:
            pass

def to_percent_fraction(x):
    if pd.isna(x):
        return np.nan
    if isinstance(x, (int, float)):
        return float(x) if float(x) <= 1 else float(x) / 100.0
    s = str(x).strip()
    if s in ['', '-', 'nan', 'None']:
        return np.nan
    s = s.replace('%', '').replace(',', '.')
    try:
        v = float(s)
    except Exception:
        return np.nan
    return v / 100.0 if v > 1 else v

def to_numeric_coerce(x):
    if pd.isna(x):
        return np.nan
    if isinstance(x, (int, float)):
        return float(x)
    s = str(x).strip()
    if s in ['', '-', 'nan', 'None']:
        return np.nan
    s = s.replace(',', '.')
    try:
        return float(s)
    except Exception:
        return np.nan

def to_seconds(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    if s in ['', '-', 'nan', 'None']:
        return np.nan
    try:
        return pd.to_timedelta(s).total_seconds()
    except Exception:
        return np.nan

describe_stats = df_numeric.describe().T
