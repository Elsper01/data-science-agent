    name_series = df['Name'] if 'Name' in df.columns else pd.Series([np.nan] * len(df))

import os
import sys
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

try:
    dataset_path = os.path.normpath('./src/resources/data/study\\32509\\dataset.csv')
    output_dir = os.path.normpath('./src/resources/output\\32509')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(dataset_path, sep=',', encoding='utf-8', dtype=str, low_memory=False)
    for c in df.columns:
        if df[c].dtype == object:
            df[c] = df[c].astype(str).str.strip()
    placeholders = {'-', '', 'na', 'n/a', 'none', 'null'}
    def is_missing(s):
        v = pd.Series(s, copy=False)
        return v.isna() | v.astype(str).str.strip().str.lower().isin(placeholders)
    missing_stats = []
    for col in df.columns:
        m = is_missing(df[col])
        missing_stats.append((col, int(m.sum()), float(m.mean())))
    missing_df = pd.DataFrame(missing_stats, columns=['Spalte', 'Anzahl_fehlend', 'Anteil_fehlend'])
    missing_df.to_csv(os.path.join(output_dir, '2_eda_missing_values.csv'), index=False, encoding='utf-8')
    def present_flag(x):
        if pd.isna(x):
            return 0
        s = str(x).strip()
        if s == '' or s.lower() in placeholders:
            return 0
        return 1
    df['Telefon_vorhanden'] = df['Telefon'].apply(present_flag).astype(int) if 'Telefon' in df.columns else 0
    df['Telefax_vorhanden'] = df['Telefax'].apply(present_flag).astype(int) if 'Telefax' in df.columns else 0
    if 'Kommune_Stadt' not in df.columns:
        raise ValueError('Spalte Kommune_Stadt fehlt im Datensatz.')
    status = df['Kommune_Stadt'].astype(str).str.lower().map(lambda x: 'Stadt' if 'stadt' in x else ('Gemeinde' if 'gemeinde' in x else 'Sonstige'))
    df['Status'] = status
    kanal_rates = pd.DataFrame({
        'Kanal': ['Telefon', 'Telefax'],
        'Anteil_vorhanden': [float(np.mean(df['Telefon_vorhanden'])), float(np.mean(df['Telefax_vorhanden']))]
    })
    kanal_rates.to_csv(os.path.join(output_dir, '2_eda_kanal_praesenz.csv'), index=False, encoding='utf-8')
    praesenz_tabelle = df[['Kommune_Stadt', 'Telefon_vorhanden', 'Telefax_vorhanden']].copy()
    praesenz_tabelle.to_csv(os.path.join(output_dir, '2_praesenz_telefon_telefax_tabelle.csv'), index=False, encoding='utf-8')
    plot_df = df[['Kommune_Stadt', 'Telefon_vorhanden', 'Telefax_vorhanden']].copy()
    order = plot_df.sort_values(['Telefax_vorhanden', 'Telefon_vorhanden', 'Kommune_Stadt'], ascending=[True, False, True])['Kommune_Stadt'].tolist()
    long_df = plot_df.melt(id_vars=['Kommune_Stadt'], value_vars=['Telefon_vorhanden', 'Telefax_vorhanden'], var_name='Kanal', value_name='Vorhanden')
    long_df['Kanal'] = long_df['Kanal'].map({'Telefon_vorhanden': 'Telefon', 'Telefax_vorhanden': 'Telefax'})
    sns.set_theme(style='whitegrid', context='talk')
    h = max(6, 0.6 * long_df['Kommune_Stadt'].nunique())
    fig, ax = plt.subplots(figsize=(12, h))
    palette = {'Telefon': '#1f77b4', 'Telefax': '#ff7f0e'}
    sns.barplot(data=long_df, y='Kommune_Stadt', x='Vorhanden', hue='Kanal', order=order, orient='h', dodge=True, palette=palette, ax=ax)
    ax.set_xlim(-0.05, 1.05)
    ax.set_xticks([0, 1])
    ax.set_xticklabels(['Nein', 'Ja'])
    ax.set_xlabel('Präsenz des Kanals')
    ax.set_ylabel('Kommune/Stadt')
    ax.set_title('Erreichbarkeit der Gleichstellungsbeauftragten nach Kommune\nPräsenz von Telefon und Telefax')
    ax.legend(title='Kommunikationskanal', loc='lower center', bbox_to_anchor=(0.5, 1.02), ncol=2, frameon=True)
    plt.tight_layout()
    output_path = os.path.join(output_dir, '2_telefon_vs_fax_praesenz.png')
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
    print(str(e))
    sys.exit(1)
