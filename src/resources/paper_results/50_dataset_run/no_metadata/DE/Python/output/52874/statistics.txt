    try:

        raise ValueError('Spalte "time" fehlt im Datensatz.')

    df_plot.loc[df_plot['value_variable_label'].str.contains('Klassen', case=False), 'category'] = 'Klassen'

    df_plot = df_plot.dropna(subset=['year_numeric', 'value', 'category'])
    
    # Aggregation (falls mehrere Einträge pro Jahr/Kategorie existieren)
    df_plot_agg = df_plot.groupby(['year_label', 'year_numeric', 'category'], as_index=False)['value'].sum()

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# Setzen des Designs
sns.set_theme(style="whitegrid", context="talk", palette="deep")

data_path = r"./src/resources/data/study/52874/dataset.csv"
out_path = r"./src/resources/output/52874/2_unterrichtsstunden_vs_lehrkraefte.png"

try:
    # Behebung des Encoding-Fehlers: 'latin-1' oder 'iso-8859-1' wird oft für deutsche Datensätze mit Umlauten verwendet, 
    # wenn utf-8 fehlschlägt.
    df = pd.read_csv(data_path, sep=';', encoding='latin-1', dtype=str)
    
    # Bereinigung der Spaltennamen
    df.columns = [c.strip() for c in df.columns]
    
    # Konvertierung der Werte
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
    
    # Extraktion des Jahres aus der 'time' Spalte
    df['year'] = df['time'].astype(str).str.extract(r'(\d{4})')[0]
    df['year'] = pd.to_numeric(df['year'], errors='coerce')
    
    # Filtern auf relevante Daten
    df = df.dropna(subset=['year', 'value', 'value_variable_label'])
    
    # Ziel-Labels definieren (Anpassung an exakte Schreibweise im Datensatz falls nötig)
    target_labels = ['Unterrichtswochenstunden', 'Voll- und teilzeitbeschäftigte Lehrkräfte']
    df_sel = df[df['value_variable_label'].isin(target_labels)].copy()
    
    # Pivotierung für die Visualisierung
    pivot = df_sel.pivot_table(index='year', columns='value_variable_label', values='value', aggfunc='last').sort_index()
    
    if pivot.empty or not all(label in pivot.columns for label in target_labels):
        # Fallback: Suche nach Teilstrings, falls exakte Übereinstimmung fehlschlägt
        cols = pivot.columns.tolist()
        stunden_col = next((c for c in cols if 'Unterrichtswochenstunden' in c), None)
        lehrer_col = next((c for c in cols if 'Lehrkräfte' in c), None)
        if stunden_col and lehrer_col:
            pivot = pivot.rename(columns={stunden_col: 'Unterrichtswochenstunden', lehrer_col: 'Voll- und teilzeitbeschäftigte Lehrkräfte'})
        else:
            raise ValueError(f'Benötigte Metriken nicht gefunden. Vorhanden: {cols}')

    # Berechnung von Metriken für den Untertitel
    pivot['Ratio'] = pivot['Unterrichtswochenstunden'] / pivot['Voll- und teilzeitbeschäftigte Lehrkräfte']
    avg_ratio = pivot['Ratio'].mean()
    correlation = pivot['Unterrichtswochenstunden'].corr(pivot['Voll- und teilzeitbeschäftigte Lehrkräfte'])
    
    years = pivot.index.astype(int)
    
    # Erstellung des Diagramms
    fig, ax1 = plt.subplots(figsize=(12, 8))
    
    color_hours = sns.color_palette()[0]
    color_teachers = sns.color_palette()[3]
    
    # Balkendiagramm für Unterrichtsstunden
    ax1.bar(years, pivot['Unterrichtswochenstunden'], color=color_hours, alpha=0.7, label='Unterrichtswochenstunden')
    ax1.set_xlabel('Jahr')
    ax1.set_ylabel('Unterrichtswochenstunden', color=color_hours)
    ax1.tick_params(axis='y', labelcolor=color_hours)
    
    # Zweite Achse für Lehrkräfte
    ax2 = ax1.twinx()
    ax2.plot(years, pivot['Voll- und teilzeitbeschäftigte Lehrkräfte'], color=color_teachers, marker='o', linewidth=3, markersize=8, label='Lehrkräfte')
    ax2.set_ylabel('Anzahl Lehrkräfte', color=color_teachers)
    ax2.tick_params(axis='y', labelcolor=color_teachers)
    
    # Titel und Beschriftung
    plt.title('Verhältnis Unterrichtsstunden zu Lehrkräften\n' + 
              f'Ø {avg_ratio:.1f} Std./Lehrkraft | Korrelation: {correlation:.2f}', pad=20)
    
    # Legende zusammenführen
    lines, labels = ax1.get_legend_handles_labels()
    lines2, labels2 = ax2.get_legend_handles_labels()
    ax1.legend(lines + lines2, labels + labels2, loc='upper left')
    
    plt.xticks(years)
    plt.tight_layout()
    
    # Speichern
    plt.savefig(out_path, dpi=150, bbox_inches='tight')
    plt.close(fig)

except Exception as e:
    print(f"Fehler aufgetreten: {str(e)}")
