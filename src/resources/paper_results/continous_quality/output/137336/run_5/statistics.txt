def load_dataset():
    path_raw = './src/resources/data/study\\137336\\dataset.csv'
    path_norm = path_raw.replace('\\', '/')
    if os.path.exists(path_raw):
        return pd.read_csv(path_raw, sep=',', encoding='utf-8')
    if os.path.exists(path_norm):
        return pd.read_csv(path_norm, sep=',', encoding='utf-8')
    raise FileNotFoundError('Datendatei wurde nicht gefunden.')

df = load_dataset()

eda = {}
eda['row_count'] = len(df)
eda['col_count'] = df.shape[1]
missing_prop = df.isna().mean()
eda['missing_prop'] = missing_prop.to_dict()

if 'Ergebnis' not in df.columns:
    raise KeyError('Spalte "Ergebnis" fehlt in den Daten.')
if 'Pruefmerkmal' not in df.columns:
    raise KeyError('Spalte "Pruefmerkmal" fehlt in den Daten.')
if 'Einheit' not in df.columns:
    df['Einheit'] = np.nan

def is_censored_value(x):
    if pd.isna(x):
        return False
    s = str(x).strip().lower()
    if s == '':
        return False
    if s.startswith('<') or s.startswith('≤'):
        return True
    if 'n.n' in s or 'nicht nachweis' in s or 'unter loq' in s or 'loq' in s:
        return True
    return False

def parse_numeric(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip().lower()
    if s == '' or 'n.n' in s or 'nicht nachweis' in s or s.startswith('<') or s.startswith('≤'):
        return np.nan
    s = re.sub(r'[^0-9,\.\-]', '', s)
    s = s.replace(',', '.')
    try:
        return float(s)
    except Exception:
        return np.nan

erg_raw = df['Ergebnis'].astype(str).str.strip()

df['Ergebnis_num'] = erg_raw.apply(parse_numeric)

df['censored'] = erg_raw.apply(is_censored_value)

df['detected'] = df['Ergebnis_num'] > 0

total_n = len(df)
n_detected = int(df['detected'].sum())
n_zero = int((df['Ergebnis_num'] == 0).sum())
n_censored = int(df['censored'].sum())
n_missing = int(df['Ergebnis_num'].isna().sum() - df['censored'].sum())

plot_df = df[(df['detected']) & df['Pruefmerkmal'].notna()].copy()
