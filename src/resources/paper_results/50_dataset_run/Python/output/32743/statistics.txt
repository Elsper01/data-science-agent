def parse_percent(series):
    s = series.replace({None: np.nan, 'nan': np.nan})
    s = s.astype(str).str.replace('%', '', regex=False).str.replace(',', '.', regex=False)
    return pd.to_numeric(s, errors='coerce')

def parse_time_to_seconds(series):
    def _to_sec(x):
        if x is None:
            return np.nan
        xs = str(x).strip()
        if xs == '' or xs.lower() == 'nan':
            return np.nan
        parts = xs.split(':')
        try:
            parts = [int(float(p)) for p in parts]
            if len(parts) == 1:
                h, m, s = 0, 0, parts[0]
            elif len(parts) == 2:
                h, m, s = 0, parts[0], parts[1]
            else:
                h, m, s = parts[-3], parts[-2], parts[-1]
            return h * 3600 + m * 60 + s
        except Exception:
            return np.nan
    return series.apply(_to_sec)

# Pfade definieren

# Daten laden
df = pd.read_csv(input_path, sep=",", encoding="utf-8")

# Datenbereinigung
if "Unnamed__11" in df.columns:
    df = df.drop(columns=["Unnamed__11"])

# Konvertierung der Zeitspalte
# Der Fehler lag vermutlich an der Inkompatibilität von Timedelta-Objekten in regplot.
# Wir stellen sicher, dass wir rein numerische Floats für die Berechnung nutzen.
if "Zeit_im_Besuch" in df.columns:
    # Umwandlung in Timedelta und dann explizit in Sekunden (Float)
    td = pd.to_timedelta(df["Zeit_im_Besuch"], errors="coerce")
    df["Zeit_im_Besuch_min"] = td.dt.total_seconds() / 60.0
else:
    df["Zeit_im_Besuch_min"] = np.nan

# Konvertierung der Seiten-Spalte
if "Seiten_pro_Besuch" in df.columns:
    # Sicherstellen, dass Kommas durch Punkte ersetzt werden, falls es als String vorliegt
    if df["Seiten_pro_Besuch"].dtype == object:
        df["Seiten_pro_Besuch"] = df["Seiten_pro_Besuch"].str.replace(',', '.', regex=False)
    df["Seiten_pro_Besuch"] = pd.to_numeric(df["Seiten_pro_Besuch"], errors="coerce")

# Filtern der relevanten Daten für den Plot (keine NaNs)
plot_df = df[["Seiten_pro_Besuch", "Zeit_im_Besuch_min"]].dropna().copy()

# Sicherstellen, dass die Daten wirklich numerisch sind (float64)
plot_df["Seiten_pro_Besuch"] = plot_df["Seiten_pro_Besuch"].astype(float)
plot_df["Zeit_im_Besuch_min"] = plot_df["Zeit_im_Besuch_min"].astype(float)

# Statistiken berechnen
n_points = len(plot_df)
if n_points > 1:
    pearson_r = plot_df["Seiten_pro_Besuch"].corr(plot_df["Zeit_im_Besuch_min"], method="pearson")
    spearman_r = plot_df["Seiten_pro_Besuch"].corr(plot_df["Zeit_im_Besuch_min"], method="spearman")
else:
    pearson_r, spearman_r = np.nan, np.nan

# Visualisierung
plt.figure(figsize=(10, 7))

if n_points > 0:
    ax = sns.regplot(
        data=plot_df,

def _to_pct(x):
    if pd.isna(x):
        return np.nan
    if isinstance(x, str):
        s = x.strip().replace('%', '').replace(',', '.')
        if s == '':
            return np.nan
        try:
            v = float(s)
        except Exception:
            return np.nan
        if v > 1.0:
            return v / 100.0
        return v
    if isinstance(x, (int, float, np.floating)):
        v = float(x)
        if v > 1.0:
            return v / 100.0
        return v
    return np.nan
