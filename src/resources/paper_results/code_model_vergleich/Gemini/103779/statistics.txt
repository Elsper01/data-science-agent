import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Create output directory if it doesn't exist
output_dir = './src/resources/output/103779'
os.makedirs(output_dir, exist_ok=True)

# Load the dataset
file_path = './src/resources/data/study/103779/dataset.csv'
df = pd.read_csv(file_path, sep=',', encoding='utf-8')

# Basic EDA: Check for missing values in relevant columns
missing_gemeinde = df['Gemeinde'].isnull().sum()
missing_objekt = df['Objektnummer'].isnull().sum()

# Group by 'Gemeinde' and count 'Objektnummer'
# We drop rows where Gemeinde is missing to ensure accurate geographical representation
gemeinde_counts = df.dropna(subset=['Gemeinde']).groupby('Gemeinde')['Objektnummer'].count().reset_index()
gemeinde_counts = gemeinde_counts.sort_values(by='Objektnummer', ascending=False)

# Select top 20 for better readability if the list is too long, 
# but the goal asks for the concentration, so we'll show the most significant ones.
# Given there are 65 municipalities, we'll plot the top 25 to keep the chart clean.
top_gemeinden = gemeinde_counts.head(25)

# Visualization
plt.figure(figsize=(12, 10))
sns.set_style("whitegrid")

# Create horizontal bar chart
barplot = sns.barplot(
    data=top_gemeinden, 
    y='Gemeinde', 
    x='Objektnummer', 
    palette='viridis'
)

# Add labels and title in German
plt.title('Konzentration von Kulturdenkmalen nach Gemeinde im Kreis Segeberg (Top 25)', fontsize=16, pad=20)
plt.xlabel('Anzahl der Kulturdenkmale (Objektnummern)', fontsize=12)
plt.ylabel('Gemeinde', fontsize=12)

# Add data labels to the bars
for i, v in enumerate(top_gemeinden['Objektnummer']):
    barplot.text(v + 0.5, i, str(int(v)), color='black', va='center', fontweight='bold')

# Adjust layout to prevent clipping
plt.tight_layout()

# Save the plot
plt.savefig(f'{output_dir}/0_denkmal_konzentration_gemeinde.png', dpi=300)
plt.close()

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Load the dataset
file_path = './src/resources/data/study/103779/dataset.csv'
df = pd.read_csv(file_path)

# EDA: Check for missing values in 'Begr_ndung'
missing_justifications = df['Begr_ndung'].isna().sum()
total_entries = len(df)

# Process the 'Begr_ndung' column
# The column contains comma-separated values (e.g., "Geschichtlich, Künstlerisch")
# We need to split these and count individual occurrences.
justification_series = df['Begr_ndung'].dropna().str.split(', ')
all_justifications = [item for sublist in justification_series for item in sublist]
justification_counts = pd.Series(all_justifications).value_counts()

# Create the visualization
plt.figure(figsize=(12, 7))
sns.set_theme(style="whitegrid")

# Plotting
barplot = sns.barplot(x=justification_counts.values, y=justification_counts.index, palette='viridis')

# Add labels and title
plt.title('Primäre rechtliche Begründungen für den Denkmalschutz im Kreis Segeberg', fontsize=16, pad=20)
plt.xlabel('Anzahl der Denkmäler', fontsize=12)
plt.ylabel('Begründung', fontsize=12)

# Add data labels to the bars
for i, v in enumerate(justification_counts.values):
    barplot.text(v + 1, i, str(v), color='black', va='center', fontweight='bold')

# Add a note about missing data if significant
plt.figtext(0.9, 0.02, f'Hinweis: {missing_justifications} von {total_entries} Einträgen ohne explizite Begründung.', 
            horizontalalignment='right', fontsize=10, style='italic')

# Adjust layout to prevent clipping
plt.tight_layout()

# Save the plot
output_dir = './src/resources/output/103779'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

plt.savefig(f'{output_dir}/2_denkmalschutz_begruendungen.png', dpi=300)
plt.close()
