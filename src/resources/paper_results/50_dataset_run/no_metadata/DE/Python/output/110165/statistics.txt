import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

def load_csv_with_encoding_fallback(file_path):
    """Versucht eine CSV-Datei mit verschiedenen Encodings zu laden."""
    encodings = ['utf-8', 'latin1', 'iso-8859-1', 'cp1252']
    for enc in encodings:
        try:
            return pd.read_csv(file_path, sep=",", encoding=enc, low_memory=False)
        except (UnicodeDecodeError, pd.errors.ParserError):
            continue
    # Letzter Versuch mit automatischer Erkennung (falls chardet installiert wäre, hier manuell)
    raise ValueError(f"Datei konnte mit keinem der Encodings {encodings} geladen werden.")

try:
    sns.set_theme(style="whitegrid", context="talk")
    
    # Pfade definieren

    # Daten laden mit Encoding-Fallback (behebt den 'utf-8' Fehler)
    df = load_csv_with_encoding_fallback(input_path)

import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

if __name__ == "__main__":
    try:
        # Set visual style
        sns.set_theme(style="whitegrid", context="talk")
        
        data_path = r"./src/resources/data/study/110165/dataset.csv"
        output_dir = r"./src/resources/output/110165"
        os.makedirs(output_dir, exist_ok=True)
        
        # The error 'utf-8' codec can't decode byte 0xfc (ü) indicates the file is likely encoded in ISO-8859-1 (Latin-1)
        # We try latin-1 which is common for German datasets containing umlauts.
        try:
            df = pd.read_csv(data_path, sep=",", encoding="utf-8")
        except UnicodeDecodeError:
            df = pd.read_csv(data_path, sep=",", encoding="latin-1")

        col = "Begr_ndung"
        if col not in df.columns:
            # Fallback check if the column name itself has encoding issues
            possible_cols = [c for c in df.columns if "Begr" in c]
            if possible_cols:
                col = possible_cols[0]
            else:
                raise ValueError(f"Erforderliche Spalte '{col}' fehlt im Datensatz. Vorhanden: {df.columns.tolist()}")

        # Data Cleaning and Processing
        s = df[col].dropna().astype(str)
        s = s[s.str.strip() != ""]
        
        # Split comma-separated values and explode into individual rows
        tokens = s.str.split(",")
        exploded = tokens.explode().astype(str).str.strip()
        
        # Filter out noise and normalize
        exploded = exploded[exploded != ""]
        noise = ["nan", "na", "none", "-", "k.a.", "k.a", "null"]
        exploded = exploded[~exploded.str.lower().isin(noise)]
        
        # Fix common encoding/typo issues in the values and capitalize
        exploded = exploded.str.replace(r"\s+", " ", regex=True)
        exploded = exploded.str.capitalize()
        
        # Map specific known values if they appear broken
        mapping = {"Stadtebaulich": "Städtebaulich", "Kunstlerisch": "Künstlerisch"}
        exploded = exploded.replace(mapping)

        counts = exploded.value_counts()
        if counts.empty:
            raise ValueError("Keine auswertbaren Werte in der Spalte für Schutzbegründungen gefunden.")

        # Prepare Plot Data
        top10 = counts.head(10).reset_index()
        top10.columns = ["Begründung", "Anzahl"]
        top10 = top10.sort_values("Anzahl", ascending=False)

        # Visualization
        plt.figure(figsize=(12, 8))
        ax = sns.barplot(
            data=top10, 
            x="Anzahl", 
            y="Begründung", 
            palette="viridis"
        )
        
        ax.set_title("Top 10 häufigste Schutzbegründungen für Denkmale", pad=20)
        ax.set_xlabel("Anzahl der Nennungen")
        ax.set_ylabel("Schutzbegründung")

        # Add value labels to the bars
        x_max = top10["Anzahl"].max()
        for i, p in enumerate(ax.patches):
            width = p.get_width()
            ax.text(
                width + (x_max * 0.01), 
                p.get_y() + p.get_height()/2, 
                f'{int(width)}', 
                va='center'
            )

        plt.tight_layout()
        out_path = os.path.join(output_dir, "2_top10_schutzbegruendungen.png")
        plt.savefig(out_path, dpi=150, bbox_inches="tight")
        plt.close()

    except Exception as e:
        print(f"Fehler aufgetreten: {str(e)}")
        sys.exit(1)
