import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

def run_visualization():
    try:
        input_path = os.path.normpath(r'./src/resources/data/study/51886/dataset.csv')
        output_dir = os.path.normpath(r'./src/resources/output/51886')
        os.makedirs(output_dir, exist_ok=True)

        # The error 'utf-8' codec can't decode byte 0xdf (ß) suggests the file is likely encoded in ISO-8859-1 (Latin-1)
        # or cp1252, which are common for German datasets.
        try:
            df = pd.read_csv(input_path, sep=';', encoding='utf-8', dtype=str)
        except UnicodeDecodeError:
            df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1', dtype=str)

        # Basic cleaning
        for col in df.columns:
            if df[col].dtype == object:
                df[col] = df[col].str.strip()

        # Convert numeric columns
        df['time'] = pd.to_numeric(df.get('time', np.nan), errors='coerce')
        df['value'] = pd.to_numeric(df.get('value', np.nan), errors='coerce')

        # Filter for the specific visualization goal
        # 2_variable_attribute_label contains the room counts (e.g., "1 Raum", "2 Räume", etc.)
        rooms_df = df.copy()
        
        label_col = '2_variable_attribute_label'
        if label_col not in rooms_df.columns:
            print(f"Fehler: Spalte '{label_col}' nicht gefunden.")
            sys.exit(1)

        # Exclude 'Insgesamt' (Total)
        rooms_df = rooms_df[~rooms_df[label_col].fillna('').str.lower().str.contains('insgesamt')]
        rooms_df = rooms_df[rooms_df[label_col].notna()]
        
        # Filter for valid values
        rooms_df = rooms_df[rooms_df['value'].notna()]

        if rooms_df.empty:
            print('Fehler: Keine Daten nach Filterung vorhanden.')
            sys.exit(1)

        # Helper to extract numeric order from labels like "1 Raum", "2 Räume", "7 oder mehr Räume"
        def get_room_order(label):
            digits = ''.join(filter(str.isdigit, str(label)))
            return int(digits) if digits else 99

        rooms_df['order'] = rooms_df[label_col].apply(get_room_order)
        
        # Aggregate values by room category
        agg = rooms_df.groupby([label_col, 'order'], as_index=False)['value'].sum()
        agg = agg.sort_values('order').reset_index(drop=True)

        # Plotting
        sns.set_theme(style='whitegrid', context='talk')
        fig, ax = plt.subplots(figsize=(12, 7), dpi=150)
        
        palette = sns.color_palette('viridis', n_colors=len(agg))
        bars = sns.barplot(data=agg, x=label_col, y='value', ax=ax, palette=palette)

        # Formatting
        ax.set_xlabel('Anzahl der Räume', fontsize=12, labelpad=10)
        ax.set_ylabel('Summe der fertiggestellten Einheiten', fontsize=12, labelpad=10)
        
        year_min = int(df['time'].min()) if 'time' in df.columns and not df['time'].isna().all() else ""
        year_max = int(df['time'].max()) if 'time' in df.columns and not df['time'].isna().all() else ""
        title_period = f"({year_min}–{year_max})" if year_min and year_max else ""
        
        ax.set_title(f'Verteilung der Wohnungsgrößen nach Raumzahl\nOberschleißheim {title_period}', 
                     fontsize=14, fontweight='bold', pad=20)

        # Add value labels on top of bars
        for p in ax.patches:
            height = p.get_height()
            ax.annotate(f'{int(height)}',
                        (p.get_x() + p.get_width() / 2., height),
                        ha='center', va='bottom',
                        xytext=(0, 5),
                        textcoords='offset points',
                        fontsize=10, fontweight='bold')

        # Thousands separator for y-axis
        ax.yaxis.set_major_formatter(FuncFormatter(lambda x, pos: f'{int(x):,}'.replace(',', '.')))

        plt.tight_layout()
        
        output_path = os.path.join(output_dir, 'apartment_size_distribution.png')
        plt.savefig(output_path, dpi=150)
        plt.close(fig)
        
    except Exception as e:
        print(f'Fehler bei der Ausführung: {e}')
        sys.exit(1)

if __name__ == "__main__":
    run_visualization()

import os
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Use raw strings or forward slashes for paths to avoid escape sequence errors
input_path = r'./src/resources/data/study/51886/dataset.csv'
output_path = r'./src/resources/output/51886/2_heatmap_negative_raumzahl.png'

# Ensure output directory exists
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# Load dataset with encoding handling
# The error '0xdf' suggests ISO-8859-1 (Latin-1) or cp1252 encoding (common in German datasets for 'ß')
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8', na_values=['', 'nan', 'NaN'])
except (UnicodeDecodeError, pd.errors.ParserError):
    try:
        df = pd.read_csv(input_path, sep=';', encoding='latin-1', na_values=['', 'nan', 'NaN'])
    except FileNotFoundError:
        # Fallback for different environments
        input_path = 'dataset.csv'
        df = pd.read_csv(input_path, sep=';', encoding='latin-1', na_values=['', 'nan', 'NaN'])

# Data Cleaning and Preprocessing
if 'time' in df.columns:
    df['time'] = pd.to_numeric(df['time'], errors='coerce')
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for "Raumzahl" related data
if '2_variable_label' in df.columns:
    df = df[df['2_variable_label'].astype(str).str.contains('Raumzahl', na=False, case=False)]

# Drop rows with missing essential values for the heatmap
df = df.dropna(subset=['time', '2_variable_attribute_label', 'value'])

# Helper to order room sizes logically (1 Raum, 2 Räume, etc.)
def _extract_room_num(label):
    import re
    nums = re.findall(r'\d+', str(label))
    return int(nums[0]) if nums else 99

df['room_order'] = df['2_variable_attribute_label'].apply(_extract_room_num)

# Aggregate data to ensure unique index/column pairs for pivot
df_agg = df.groupby(['time', '2_variable_attribute_label', 'room_order'], as_index=False)['value'].sum()
df_agg = df_agg.sort_values(['room_order', 'time'])

# Create Pivot Table
pivot = df_agg.pivot_table(index='2_variable_attribute_label', columns='time', values='value')

# Re-sort index based on room count
sorted_labels = df_agg[['2_variable_attribute_label', 'room_order']].drop_duplicates().sort_values('room_order')['2_variable_attribute_label'].tolist()
pivot = pivot.reindex(sorted_labels)

# Visualization
sns.set_theme(style='white')
fig, ax = plt.subplots(figsize=(12, 7), dpi=200)

# Use a diverging colormap to highlight negative (red) vs positive (blue)
if not pivot.empty:
    max_val = np.nanmax(np.abs(pivot.values))
    # Ensure max_val is not 0 to avoid issues
    max_val = max_val if max_val != 0 else 1
    cmap = sns.color_palette('RdBu', as_cmap=True)

    sns.heatmap(pivot, 
                ax=ax, 
                cmap=cmap, 
                center=0, 
                vmin=-max_val, 
                vmax=max_val, 
                annot=True, 
                fmt='.0f', 
                linewidths=.5, 
                cbar_kws={'label': 'Nettoveränderung (Anzahl)'})

    # Highlight negative cells with a border
    for i in range(pivot.shape[0]):
        for j in range(pivot.shape[1]):
            if pivot.iloc[i, j] < 0:
                ax.add_patch(plt.Rectangle((j, i), 1, 1, fill=False, edgecolor='black', lw=2))

    neg_count = (pivot < 0).sum().sum()
    caption = f"Hinweis: Schwarze Rahmen markieren negative Nettoveränderungen (Abgänge oder statistische Korrekturen).\nAnzahl negativer Instanzen: {neg_count}."
else:
    ax.text(0.5, 0.5, "Keine Daten für die gewählten Kriterien gefunden.", ha='center')
    caption = ""

ax.set_title('Nettoveränderung der Wohnungsbestände nach Raumzahl\n(Hervorhebung negativer Korrekturen/Abgänge)', pad=20, fontsize=14)
ax.set_xlabel('Jahr')
ax.set_ylabel('Wohnungsgröße')

if caption:
    plt.figtext(0.5, 0.01, caption, ha='center', fontsize=10, bbox={"facecolor":"orange", "alpha":0.1, "pad":5})

plt.tight_layout(rect=[0, 0.05, 1, 0.95])
plt.savefig(output_path, bbox_inches='tight')
plt.show()
