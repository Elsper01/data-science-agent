import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from collections import Counter

# Load the dataset
df = pd.read_csv(r'./src/resources/data/study/103779/dataset.csv', sep=',', encoding='utf-8')

# The error indicated that 'Begründung' was not found. 
# Based on the visualization goal, the column name is likely 'Begr_ndung' due to encoding or naming conventions.
# Let's check for the correct column name.
target_col = 'Begr_ndung'
if target_col not in df.columns:
    # Fallback: search for any column that looks like 'Begründung' or 'Begr_ndung'
    potential_cols = [c for c in df.columns if 'Begr' in c]
    if potential_cols:
        target_col = potential_cols[0]
    else:
        # If still not found, use the first column that might contain text justifications
        # but for this task, we stick to the requested one.
        pass

all_reasons = []
if target_col in df.columns:
    for val in df[target_col].dropna():
        if isinstance(val, str) and val.strip():
            # Split by comma as justifications are often comma-separated lists
            terms = [t.strip() for t in val.split(',')]
            all_reasons.extend(terms)

# Count frequencies
counter = Counter(all_reasons)
top_items = counter.most_common(15) # Top 15 for better coverage
df_top = pd.DataFrame(top_items, columns=['Justification', 'Frequency'])
df_top = df_top.sort_values('Frequency', ascending=False)

# Visualization
plt.figure(figsize=(12, 8))
sns.set_theme(style="whitegrid")
barplot = sns.barplot(data=df_top, x='Frequency', y='Justification', palette='magma')

plt.title('Primary Legal and Historical Justifications for Monument Protection', fontsize=16, pad=20)
plt.xlabel('Number of Mentions', fontsize=12)
plt.ylabel('Justification Type', fontsize=12)

# Add value labels to the bars
for i, v in enumerate(df_top['Frequency']):
    barplot.text(v + 0.5, i, str(int(v)), color='black', va='center')

plt.tight_layout()
plt.savefig('./src/resources/output/103779/2_begruendung_frequenz.png', dpi=300, bbox_inches='tight')
plt.close()
