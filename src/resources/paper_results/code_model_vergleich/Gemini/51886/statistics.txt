import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Set plot style
sns.set_theme(style="whitegrid")

# Load the dataset
file_path = './src/resources/data/study/51886/dataset.csv'

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xdf' 
# suggests the file is likely encoded in 'latin-1' (ISO-8859-1) or 'cp1252', 
# which is common for German datasets (0xdf is the 'ß' character).
try:
    df = pd.read_csv(file_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(file_path, sep=';', encoding='latin-1')

# Data Cleaning and Preparation
# Ensure 'value' is numeric
df['value'] = pd.to_numeric(df['value'], errors='coerce').fillna(0)

# Filter out 'Insgesamt' (Total) to focus on specific apartment sizes
# We use a case-insensitive check and strip whitespace to be robust
df_filtered = df[df['2_variable_attribute_label'].str.contains('Insgesamt', case=False) == False].copy()

# Group by apartment size and sum the values
distribution = df_filtered.groupby('2_variable_attribute_label')['value'].sum().reset_index()

# Define a logical order for the rooms (1 to 7+)
def extract_room_number(label):
    import re
    nums = re.findall(r'\d+', str(label))
    if nums:
        return int(nums[0])
    return 99

distribution['sort_key'] = distribution['2_variable_attribute_label'].apply(extract_room_number)
distribution = distribution.sort_values('sort_key')

# Visualization
plt.figure(figsize=(12, 7))
colors = sns.color_palette("viridis", len(distribution))
bars = plt.bar(distribution['2_variable_attribute_label'], distribution['value'], color=colors, edgecolor='black', alpha=0.8)

# Add labels and title
plt.title('Verteilung der Wohnungsfertigstellungen nach Raumzahl (2014-2024)', fontsize=14, fontweight='bold', pad=20)
plt.xlabel('Anzahl der Räume', fontsize=12)
plt.ylabel('Summe der fertiggestellten Wohnungen', fontsize=12)
plt.xticks(rotation=45, ha='right')

# Add value labels on top of bars
for bar in bars:
    yval = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, yval + 0.1, int(yval), ha='center', va='bottom', fontweight='bold')

# Adjust layout and save
plt.tight_layout()
output_dir = './src/resources/output/51886'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

plt.savefig(f'{output_dir}/1_apartment_size_distribution.png', dpi=300)
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Create output directory if it doesn't exist
output_dir = './src/resources/output/51886'
os.makedirs(output_dir, exist_ok=True)

# Load the dataset
file_path = './src/resources/data/study/51886/dataset.csv'

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xdf' 
# suggests the file is likely encoded in 'latin-1' or 'ISO-8859-1' (common for German datasets with 'ß').
try:
    df = pd.read_csv(file_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(file_path, sep=';', encoding='latin-1')

# Data Cleaning and Preparation
# Filter out 'Insgesamt' as per visualization goal
df_filtered = df[df['2_variable_attribute_label'] != 'Insgesamt'].copy()

# Ensure 'value' is numeric and handle potential missing values
df_filtered['value'] = pd.to_numeric(df_filtered['value'], errors='coerce').fillna(0)

# Pivot the data for the heatmap: Rows = Room categories, Columns = Years
heatmap_data = df_filtered.pivot_table(
    index='2_variable_attribute_label', 
    columns='time', 
    values='value', 
    aggfunc='sum'
)

# Sort index to have a logical order (1 room to 7+ rooms)
def sort_key(label):
    if not isinstance(label, str): return 0
    if '7' in label: return 7
    try:
        # Extract first number found in the string
        import re
        match = re.search(r'\d+', label)
        return int(match.group()) if match else 0
    except:
        return 0

sorted_index = sorted(heatmap_data.index, key=sort_key)
heatmap_data = heatmap_data.reindex(sorted_index)

# Visualization
plt.figure(figsize=(14, 8))
sns.set_theme(style="white")

# Create heatmap with a diverging color map to highlight negative values (losses)
# Using 'RdBu' where Red indicates losses and Blue indicates completions
# We set center=0 so that 0 is white, positive is blue, negative is red
cmap = sns.diverging_palette(10, 240, as_cmap=True)

ax = sns.heatmap(
    heatmap_data, 
    annot=True, 
    fmt=".0f", 
    cmap=cmap, 
    center=0, 
    linewidths=.5, 
    cbar_kws={'label': 'Anzahl der Fertigstellungen / Korrekturen'}
)

plt.title('Wohnungsfertigstellungen und administrative Korrekturen (2014-2024)', fontsize=16, pad=20)
plt.xlabel('Berichtsjahr', fontsize=12)
plt.ylabel('Wohnungsgröße (Anzahl der Räume)', fontsize=12)

# Adjust layout to prevent clipping
plt.tight_layout()

# Save the plot
plt.savefig(f'{output_dir}/2_heatmap_construction_losses.png', dpi=300)
plt.close()
