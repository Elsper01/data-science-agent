# Read data
# Using read_delim with semicolon as seen in previous attempt
df <- readr::read_delim(input_path, delim = ";", locale = readr::locale(encoding = "UTF-8"), show_col_types = FALSE)

# The error "Objekt '2_variable_attribute_code' nicht gefunden" happened because 
# janitor::clean_names() converts "2_variable_attribute_code" to "x2_variable_attribute_code" 
# (R variables cannot start with numbers).
# We will clean names but be mindful of the transformation.
df_clean <- janitor::clean_names(df)

# Identify the correct column names after cleaning
# "2_variable_attribute_label" becomes "x2_variable_attribute_label"
# "2_variable_attribute_code" becomes "x2_variable_attribute_code"
label_col <- "x2_variable_attribute_label"
code_col <- "x2_variable_attribute_code"

# Filter and prepare data
# 1. Exclude 'Insgesamt'
# 2. Ensure time is numeric
# 3. Ensure value is numeric
comp_data <- df_clean %>%
  filter(!is.na(value)) %>%
  filter(!is.na(!!sym(label_col))) %>%
  filter(!!sym(label_col) != "Insgesamt") %>%
  mutate(time = suppressWarnings(as.integer(time))) %>%
  filter(!is.na(time))

# Define a logical order for the room categories if possible
# Usually RAUM-1 to RAUM-7UM
raum_order_codes <- c("RAUM-1", "RAUM-2", "RAUM-3", "RAUM-4", "RAUM-5", "RAUM-6", "RAUM-7UM")

# Create a factor for the labels based on the code order
label_map <- comp_data %>% 
  select(all_of(c(code_col, label_col))) %>% 
  distinct()

ordered_labels <- label_map %>% 
  mutate(ord = match(!!sym(code_col), raum_order_codes)) %>% 
  arrange(ord) %>% 
  filter(!is.na(ord)) %>% 
  pull(!!sym(label_col)) %>% 
  unique()

# If the codes don't match the expected RAUM-X pattern, fallback to alphabetical or existing order
if(length(ordered_labels) == 0) {
  ordered_labels <- sort(unique(comp_data[[label_col]]))
}

comp_data <- comp_data %>% 
  mutate(raumzahl = factor(!!sym(label_col), levels = ordered_labels))

# Create the visualization
