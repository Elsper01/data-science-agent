if (!require("pacman")) install.packages("pacman")
pacman::p_load(tidyverse)

# Daten einlesen
df <- read.csv("./src/resources/data/study/13473/dataset.csv", sep = ";", encoding = "UTF-8", stringsAsFactors = FALSE)

# Fehlerbehebung: Spaltennamen prüfen und sicherstellen, dass sie korrekt angesprochen werden
# In R können Spaltennamen, die mit Zahlen beginnen, beim Einlesen verändert werden (z.B. X1_variable_label)
# Wir nutzen janitor-ähnliche Bereinigung oder sprechen sie über die Position/Backticks an.
# Zudem konvertieren wir 'value' sicher zu numerisch (ersetzen Komma falls vorhanden).

df$value <- as.numeric(gsub(",", ".", as.character(df$value)))

# Dynamische Identifikation der Spalten, falls die Namen durch R leicht verändert wurden
col_1 <- grep("1_variable_label", names(df), value = TRUE)
col_2 <- grep("2_variable_attribute_label", names(df), value = TRUE)

# Falls grep nichts findet (z.B. wegen X-Präfix), nehmen wir die ersten beiden Spalten basierend auf dem Wissen der Struktur
if(length(col_1) == 0) col_1 <- names(df)[1]
if(length(col_2) == 0) col_2 <- names(df)[2]

# Vorbereitung der Plot-Daten
plot_data <- df %>%
  filter(!is.na(value)) %>%
  mutate(
    level = as.factor(get(col_1)),
    attribute = as.factor(get(col_2))
  )

# Verzeichnis erstellen
dir.create("./src/resources/output/13473", recursive = TRUE, showWarnings = FALSE)

# Visualisierung
p <- ggplot(plot_data, aes(x = attribute, y = value, fill = level)) +
  geom_bar(stat = "identity", position = position_dodge(width = 0.8), width = 0.7) +
  geom_text(aes(label = format(value, big.mark = ".", decimal.mark = ",")), 
            position = position_dodge(width = 0.8), 
            vjust = -0.5, 
            size = 3.5) +
  # Farbskala flexibel halten, falls Labels leicht abweichen
  scale_fill_brewer(palette = "Paired") +
  labs(title = "Vergleich der Bevölkerungszahlen: Kreis- vs. Gemeindeebene",
       subtitle = "Stadt Hof - Validierung der Datenredundanz (Identität der Werte)",
       x = "Kategorie (Geschlecht)",
       y = "Anzahl Personen",
       fill = "Regionale Ebene") +
  theme_minimal() +
  theme(legend.position = "bottom",
        plot.title = element_text(face = "bold", size = 12),
        axis.text.x = element_text(angle = 0),
        panel.grid.major.x = element_blank()) +
  scale_y_continuous(expand = expansion(mult = c(0, 0.15)))

# Speichern
ggsave("./src/resources/output/13473/2_vergleich_kreis_gemeinde_ebene.png", plot = p, width = 10, height = 7, dpi = 300)
