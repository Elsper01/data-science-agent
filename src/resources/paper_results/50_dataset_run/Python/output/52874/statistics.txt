import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Der Fehler 'utf-8' codec can't decode byte 0xe4 deutet auf eine ISO-8859-1 (Latin-1) Kodierung hin (0xe4 ist 'ä').
# Wir versuchen 'latin-1' oder 'cp1252', um die deutschen Umlaute korrekt zu lesen.
try:
    df = pd.read_csv(r'./src/resources/data/study/52874/dataset.csv', sep=';', encoding='latin-1')
except Exception as e:
    # Fallback auf cp1252 falls latin-1 fehlschlägt
    df = pd.read_csv(r'./src/resources/data/study/52874/dataset.csv', sep=';', encoding='cp1252')

df.columns = [str(c).strip() for c in df.columns]

# Datenbereinigung und Vorbereitung
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
else:
    df['value'] = np.nan

if 'time' in df.columns:
    # Extrahiere das Jahr aus dem Zeit-String (z.B. "2014/15" -> 2014)
    df['time_year'] = pd.to_numeric(df['time'].astype(str).str.extract(r'(\d{4})')[0], errors='coerce')
else:
    df['time_year'] = np.nan

# Filterung auf die gewünschten Kategorien

vars_of_interest = ['Schüler', 'Voll- und teilzeitbeschäftigte Lehrkräfte']

# Entferne Zeilen ohne Werte oder Zeitangaben

# Aggregation (falls mehrere Schulen/Einträge pro Jahr vorhanden sind)

df_plot = df_plot.sort_values(['time_year', 'value_variable_label'])

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Set style
sns.set_theme(style="whitegrid", context="talk")

file_path = r"./src/resources/data/study\52874\dataset.csv"
output_path = r"./src/resources/output\52874/1_balkendiagramm_ressourcen_2024.png"
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# The error was a UnicodeDecodeError with utf-8. 
# German datasets often use 'iso-8859-1' or 'latin-1' due to umlauts (ä, ö, ü).
try:
    df = pd.read_csv(file_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(file_path, sep=';', encoding='iso-8859-1')
    except Exception:
        df = pd.read_csv(file_path, sep=';', encoding='cp1252')

# Clean string columns
for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].astype(str).str.strip()

# Convert value to numeric
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Extract year from time column
if 'time' in df.columns:
    df['year'] = pd.to_numeric(df['time'].astype(str).str.extract(r'(\d{4})')[0], errors='coerce')
else:
    df['year'] = np.nan

# Filter for the year 2024 as requested
target_year = 2024
df_2024 = df[df['year'] == target_year].copy()

# If 2024 is not available, fall back to the latest available year
if df_2024.empty and not df['year'].dropna().empty:
    latest_year = int(df['year'].max())
    df_2024 = df[df['year'] == latest_year].copy()
    target_year = latest_year

# Define the relevant variables for the visualization
subset_vars = ['SCHULE', 'KLASSE', 'HBLEHR']
plot_df = df_2024[df_2024['value_variable_code'].isin(subset_vars)].copy()

# If codes are not found, try to use labels or just take what's available in the filtered year
if plot_df.empty:
    plot_df = df_2024.copy()

# Sort labels for better presentation
label_order = ['Schulen', 'Klassen', 'Voll- und teilzeitbeschäftigte Lehrkräfte']
if 'value_variable_label' in plot_df.columns:
    # Ensure labels exist in the data before ordering
    present_labels = plot_df['value_variable_label'].unique()
    ordered_categories = [l for l in label_order if l in present_labels]
    for l in present_labels:
        if l not in ordered_categories:
            ordered_categories.append(l)
    
    plot_df['value_variable_label'] = pd.Categorical(
        plot_df['value_variable_label'], 
        categories=ordered_categories, 
        ordered=True
    )
    plot_df = plot_df.sort_values('value_variable_label')

# Drop rows with missing values for the plot
plot_df = plot_df.dropna(subset=['value'])

# Create the plot
plt.figure(figsize=(12, 7))
if not plot_df.empty:
    ax = sns.barplot(
        data=plot_df, 
        x='value_variable_label', 
        y='value', 
        palette='viridis',
        hue='value_variable_label',
        legend=False
    )
    
    # Add labels on top of bars
    for p in ax.patches:
        height = p.get_height()
        ax.annotate(f'{int(height)}',
                    (p.get_x() + p.get_width() / 2., height),
                    ha='center', va='bottom',
                    xytext=(0, 5),
                    textcoords='offset points',
                    fontsize=12,
                    fontweight='bold')

    plt.title(f'Ressourcenverteilung in Stadtlauringen ({target_year})', pad=20)
    plt.xlabel('Kennzahl')
    plt.ylabel('Anzahl')
    
    # Adjust y-axis to give space for labels
    max_val = plot_df['value'].max()
    plt.ylim(0, max_val * 1.15)
else:
    plt.text(0.5, 0.5, 'Keine Daten für die Visualisierung verfügbar', ha='center')

plt.tight_layout()
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close()

Indem wir 'value' für die Kategorie 'Schüler' gegen 'value' für die Kategorie 'Unterrichtswochenstunden' (aus 'value_variable_label') über die Jahre 'time' hinweg betrachten, lässt sich feststellen, ob mehr Schüler automatisch zu mehr Unterrichtsstunden führen oder ob andere Faktoren (wie Lehrermangel oder Lehrplanänderungen) den Ausschlag geben."

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Behebung der FutureWarning: use_inf_as_na ist veraltet
# pd.set_option('mode.use_inf_as_na', True) # Entfernt

input_path = os.path.normpath('./src/resources/data/study/52874/dataset.csv')
output_dir = os.path.normpath('./src/resources/output/52874')
plot_path = os.path.join(output_dir, '2_scatter_schueler_uwostd.png')
os.makedirs(output_dir, exist_ok=True)

# Behebung des UnicodeDecodeError: Versuche 'latin-1' (ISO-8859-1), da 'utf-8' fehlschlug
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8', dtype=str)
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1', dtype=str)

# Datenbereinigung
for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].str.strip()

missing_tokens = {'', 'NA', 'NaN', 'nan', 'None', 'null', 'NULL'}
# Ersetze inf durch NaN manuell, um die Warnung zu vermeiden
df = df.replace(['inf', '-inf'], np.nan)
df = df.apply(lambda col: col.map(lambda x: np.nan if isinstance(x, str) and x in missing_tokens else x))

# Konvertierung der Werte
df['value'] = pd.to_numeric(df['value'], errors='coerce')
# Extrahiere Jahr aus 'time' für die Beschriftung
df['year_label'] = df['time'].astype(str)

# Filtern der relevanten Kategorien
sub = df[df['value_variable_label'].isin(['Schüler', 'Unterrichtswochenstunden'])].copy()
sub = sub[['year_label', 'value_variable_label', 'value']].dropna(subset=['value'])

# Pivotierung, um Schüler und Stunden in separate Spalten zu bekommen
# Wir nutzen 'year_label' als Index
wide = sub.pivot_table(index='year_label', columns='value_variable_label', values='value', aggfunc='first')
wide = wide.dropna(subset=['Schüler', 'Unterrichtswochenstunden']).sort_index()

# Korrelation berechnen
corr = float(wide[['Schüler', 'Unterrichtswochenstunden']].corr(method='pearson').iloc[0, 1]) if len(wide) > 1 else np.nan

# Visualisierung
sns.set_theme(style='whitegrid', context='talk')
plt.rcParams.update({'axes.titlesize': 14, 'axes.labelsize': 11, 'legend.fontsize': 10, 'xtick.labelsize': 9, 'ytick.labelsize': 9})
fig, ax = plt.subplots(figsize=(10, 7), dpi=150)

if not wide.empty:
    # Scatter Plot mit Regressionslinie
    sns.regplot(data=wide.reset_index(), x='Schüler', y='Unterrichtswochenstunden', 
                ci=None, scatter_kws={'s':100, 'color':'#1f77b4', 'alpha':0.8, 'edgecolors':'white'}, 
                line_kws={'color':'#d62728', 'lw':2}, ax=ax)
    
    # Datenpunktbeschriftung mit 'time' (year_label)
    for idx, r in wide.reset_index().iterrows():
        ax.annotate(str(r['year_label']), (r['Schüler'], r['Unterrichtswochenstunden']), 
                    xytext=(5, 5), textcoords='offset points', fontsize=8, alpha=0.8)
    
    # Korrelationskoeffizient anzeigen
    ax.text(0.05, 0.95, f'Pearson r = {corr:.2f}' if np.isfinite(corr) else 'r nicht berechenbar', 
            transform=ax.transAxes, verticalalignment='top', bbox=dict(boxstyle='round', facecolor='white', alpha=0.5))
    
    ax.set_title('Zusammenhang: Schülerzahl vs. Unterrichtswochenstunden\n(Stadtlauringen)', pad=20)
    ax.set_xlabel('Anzahl Schüler')
    ax.set_ylabel('Anzahl Unterrichtswochenstunden')
else:
    ax.text(0.5, 0.5, 'Keine Daten für Schüler und Unterrichtswochenstunden gefunden.', 
            ha='center', va='center')
    ax.set_title('Daten nicht verfügbar')

plt.tight_layout()
fig.savefig(plot_path, facecolor='white')
plt.close(fig)
