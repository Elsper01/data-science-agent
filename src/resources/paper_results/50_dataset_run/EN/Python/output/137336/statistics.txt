import os
import re
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Set visual style
sns.set_theme(style="whitegrid", context="talk")
plt.rcParams.update({"figure.dpi": 120, "savefig.dpi": 300, "axes.titlesize": 16, "axes.labelsize": 13})

def parse_numeric(x):
    """Helper to parse numeric values from strings like '317 Bq/kg' or 'n.n.'"""
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    # Common non-numeric indicators
    if s.lower() in {'n.n.', 'n.n', 'nn', 'na', 'n/a', 'k.a.', 'k.a'}:
        return np.nan
    # Exclude values below detection limit
    if any(sym in s for sym in ['<', '≤']):
        return np.nan
    
    s = s.replace('−', '-')
    s = s.replace(' ', '')
    s = s.replace(',', '.')
    
    try:
        return float(s)
    except Exception:
        # Extract first numeric sequence
        m = re.search(r'[-+]?\d*\.?\d+(?:[eE][-+]?\d+)?', s)
        if m:
            try:
                return float(m.group(0))
            except Exception:
                return np.nan
        return np.nan

# Load dataset
file_path = 'dataset.csv'
if not os.path.exists(file_path):
    # Create dummy data for demonstration if file is missing in local environment

    data = {
        'Pruefmerkmal': ['Kalium-40', 'Caesium-137', 'Strontium-90', 'Blei', 'Cadmium'],
        'Ergebnis': ['317 Bq/kg', 'n.n.', '1.2 Bq/kg', '0.05 mg/kg', 'n.n.'],
        'Einheit': ['Bq/kg', 'Bq/kg', 'Bq/kg', 'mg/kg', 'mg/kg']
    }
    df = pd.DataFrame(data)
else:
    df = pd.read_csv(file_path)

# Clean data
for c in df.columns:

# Process numeric results
df['Ergebnis_num'] = df['Ergebnis'].apply(parse_numeric)

# Filter for detectable numeric results (greater than 0)

if df_num.empty:

    ax.text(0.5, 0.5, 'Keine nachweisbaren numerischen Ergebnisse gefunden', ha='center', va='center')

else:

    df_plot = df_num.sort_values('Ergebnis_num', ascending=False)

    # Dynamic figure size
    n = len(df_plot)
    fig_width = max(10, min(0.8 * n, 20))

    # Plot
    # We use 'Pruefmerkmal' as x and 'Ergebnis_num' as y
    # If 'Einheit' exists, use it for color coding

    sns.barplot(data=df_plot, x='Pruefmerkmal', y='Ergebnis_num', hue=hue_col, dodge=False, ax=ax, palette='viridis')

        ax.set_yscale('log')
        ax.set_ylabel('Ergebnis (log-Skala)')

        ax.set_ylabel('Ergebnis')

    ax.set_xlabel('Prüfmerkmal')
    ax.set_title('Nachweisbare numerische Ergebnisse (Radiologisch/Chemisch)')

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import os
from matplotlib.ticker import MaxNLocator

input_path = r'./src/resources/data/study\137336\dataset.csv'
output_dir = r'./src/resources/output\137336'
os.makedirs(output_dir, exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

for c in df.select_dtypes(include=['object']).columns:
    df[c] = df[c].astype(str).replace({'nan': np.nan}).str.replace('\u00a0', ' ', regex=False).str.replace('\u2009', ' ', regex=False).str.strip()

missing_ratio = df.isna().mean()

if 'Pruefmerkmal' in df.columns:
    n_tests = int(df['Pruefmerkmal'].nunique(dropna=True)) if df['Pruefmerkmal'].notna().any() else int(len(df))
else:
    n_tests = int(len(df))

if 'Einheit' in df.columns:
    def normalize_unit(u):
        if pd.isna(u):
            return 'fehlend'
        s = str(u).strip()
        s = s.replace(' /', '/').replace('/ ', '/')
        s = s.replace('g/100 g', 'g/100g').replace('G/100 G', 'g/100g')
        s = s.replace(' %', '%').replace('% ', '%')
        return s
    einheit_clean = df['Einheit'].apply(normalize_unit)
else:
    einheit_clean = pd.Series(['fehlend'] * len(df))

unit_counts = einheit_clean.value_counts(dropna=False)
unit_perc = (unit_counts / unit_counts.sum()).round(4)
unit_counts_df = pd.DataFrame({'Einheit': unit_counts.index, 'Anzahl': unit_counts.values, 'Prozent': unit_perc.values})

sns.set_theme(style='whitegrid', context='talk')
num_units = len(unit_counts_df)

ax = sns.barplot(x='Einheit', y='Anzahl', data=plot_df, palette=palette, ax=ax)
ax.set_title(f"Verteilung der Messeinheiten über {n_tests} Prüfmerkmale", pad=16)
