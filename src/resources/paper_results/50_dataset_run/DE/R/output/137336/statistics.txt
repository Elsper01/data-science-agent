parse_ergebnis_to_numeric <- function(x) {
  if (is.null(x)) return(NULL)
  x_chr <- as.character(x)
  x_chr_trim <- stringr::str_trim(x_chr)
  y <- rep(NA_real_, length(x_chr_trim))
  is_missing <- is.na(x_chr_trim) | tolower(x_chr_trim) %in% c("n.n.", "n.n", "nn", "na", "n/a", "-") | x_chr_trim == ""
  is_lt <- stringr::str_detect(x_chr_trim, "^\\s*<\\s*[0-9]+([\\.,][0-9]+)?\\s*$")
  if (any(is_lt, na.rm = TRUE)) {
    num <- x_chr_trim[is_lt]
    num <- gsub("^\\s*<\\s*", "", num)
    num <- gsub("\\s*$", "", num)
    num <- gsub(",", ".", num, fixed = TRUE)
    suppressWarnings(num <- as.numeric(num))
    y[is_lt] <- num/2
  }
  is_num <- !(is_missing | is_lt)
  if (any(is_num, na.rm = TRUE)) {
    num2 <- gsub(",", ".", x_chr_trim[is_num], fixed = TRUE)
    suppressWarnings(y[is_num] <- as.numeric(num2))
  }
  y
}
