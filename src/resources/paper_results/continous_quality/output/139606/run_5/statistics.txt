import os
import sys
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

try:
    # Define paths
    input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '139606', 'dataset.csv')
    output_dir = os.path.join('.', 'src', 'resources', 'output', '139606', 'run_5')
    os.makedirs(output_dir, exist_ok=True)

    # Load data
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')

    # Data Cleaning: Convert 'Ergebnis' to numeric
    # pd.to_numeric does not have a 'decimal' argument. 
    # We must replace the decimal comma with a dot before conversion.
    df['Ergebnis_clean'] = df['Ergebnis'].astype(str).str.replace('\u202f', '', regex=True).str.strip()
    df['Ergebnis_clean'] = df['Ergebnis_clean'].str.replace(',', '.', regex=False)
    df['Ergebnis_num'] = pd.to_numeric(df['Ergebnis_clean'], errors='coerce')

    # Identify non-detects (n.n.)
    non_detect_mask = df['Ergebnis'].astype(str).str.lower().str.contains('n.n', regex=False)
    total_params = len(df)
    non_detects = int(non_detect_mask.sum())
    
    # Filter for detected values (numeric and not NaN)
    det_df = df[df['Ergebnis_num'].notna()].copy()
    
    # Filter for relevant units if applicable (Bq/kg)
    if 'Einheit' in det_df.columns:
        det_df = det_df[det_df['Einheit'].astype(str).str.contains('Bq/kg', na=False)]
    
    detects = len(det_df)
    det_df = det_df.sort_values('Ergebnis_num', ascending=False)

    # Visualization
    sns.set_theme(style='whitegrid', font_scale=1.1)
    fig, ax = plt.subplots(figsize=(12, 7))

    if det_df.empty:
        ax.axis('off')
        ax.text(0.5, 0.6, 'Keine nachweisbaren Radionuklide gefunden', ha='center', va='center', fontsize=16)
        subtitle = f'Parameter gesamt: {total_params}, Nicht nachweisbar: {non_detects}, Nachweise: 0'
        ax.text(0.5, 0.45, subtitle, ha='center', va='center', fontsize=11)
    else:
        # Create bar chart
        palette = sns.color_palette('viridis', n_colors=len(det_df))
        sns.barplot(data=det_df, x='Pruefmerkmal', y='Ergebnis_num', palette=palette, ax=ax)

        # Use log scale if values vary significantly
        if det_df['Ergebnis_num'].max() / det_df['Ergebnis_num'].min() > 20:
            ax.set_yscale('log')
            y_suffix = ' (log scale)'
        else:
            y_suffix = ''

        ax.set_xlabel('Radionuklid (Prüfmerkmal)')
        y_label_unit = det_df['Einheit'].iloc[0] if 'Einheit' in det_df.columns else 'Bq/kg'
        ax.set_ylabel(f'Aktivität [{y_label_unit}]{y_suffix}')

        # Add value labels on top of bars
        for i, p in enumerate(ax.patches):
            val = det_df['Ergebnis'].iloc[i]
            ax.annotate(f'{val}', 
                        (p.get_x() + p.get_width() / 2., p.get_height()), 
                        ha='center', va='center', 
                        xytext=(0, 9), 
                        textcoords='offset points',
                        fontsize=10, fontweight='bold')

        # Title and Subtitle
        title_parts = []
        if 'Bezeichnung' in df.columns and not df['Bezeichnung'].isna().all():
            title_parts.append(str(df['Bezeichnung'].dropna().iloc[0]))
        title_context = f" ({' | '.join(title_parts)})" if title_parts else ""
        
        ax.set_title(f'Nachgewiesene Radionuklide in der Probe{title_context}', pad=20)
        
        subtitle = f'Gesamtanzahl Parameter: {total_params} | Nicht nachweisbar: {non_detects} | Nachweise: {detects}'
        plt.figtext(0.5, 0.01, subtitle, ha='center', fontsize=10, bbox={"facecolor":"orange", "alpha":0.1, "pad":5})

    plt.xticks(rotation=30, ha='right')
    plt.tight_layout(rect=[0, 0.03, 1, 0.95])

    # Save output
    output_path = os.path.join(output_dir, 'radionuklide_levels_barchart.png')
    fig.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close(fig)

except Exception as e:
    print(f"Error: {str(e)}")
    sys.exit(1)

import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

input_path = './src/resources/data/study\\139606\\dataset.csv'
output_dir = './src/resources/output\\139606\\run_5'
plot_filename = '1_verteilung_nachweisbarkeit.png'
plot_path = os.path.join(output_dir, plot_filename)
os.makedirs(output_dir, exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

df['Ergebnis'] = df['Ergebnis'].astype(str).str.strip().replace({'': np.nan})

def parse_german_number(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    if s.lower() in ['n.n.', 'n.n', 'nn']:
        return np.nan
    if ',' in s:
        s2 = s.replace('.', '').replace(',', '.')
    else:
        s2 = s
    return pd.to_numeric(s2, errors='coerce')

df['Ergebnis_num'] = df['Ergebnis'].apply(parse_german_number)

non_detect_mask = df['Ergebnis'].str.lower().eq('n.n.') | df['Ergebnis'].str.lower().eq('n.n') | df['Ergebnis'].str.lower().eq('nn')
cat = np.full(len(df), 'Fehlend/sonstiges', dtype=object)
cat[non_detect_mask.fillna(False)] = 'Nicht nachweisbar (n.n.)'
cat[df['Ergebnis_num'].notna()] = 'Nachweisbar'
df['Ergebniskategorie'] = pd.Categorical(cat, categories=['Nicht nachweisbar (n.n.)', 'Nachweisbar', 'Fehlend/sonstiges'], ordered=True)

missing_ratio = df.isna().mean()
num_summary = df[['Ergebnis_num']].describe()

counts = df['Ergebniskategorie'].value_counts(dropna=False).rename_axis('Kategorie').reset_index(name='Anzahl')
counts['Prozent'] = counts['Anzahl'] / counts['Anzahl'].sum() * 100
present_categories = [c for c in ['Nicht nachweisbar (n.n.)', 'Nachweisbar', 'Fehlend/sonstiges'] if c in counts['Kategorie'].tolist()]

sns.set_theme(style='whitegrid')
plt.figure(figsize=(8, 6), dpi=150)
palette = {'Nicht nachweisbar (n.n.)': '#7f8c8d', 'Nachweisbar': '#2ecc71', 'Fehlend/sonstiges': '#f39c12'}
ax = sns.barplot(data=counts, x='Kategorie', y='Anzahl', order=present_categories, palette=[palette.get(k, '#3498db') for k in present_categories])
ax.set_xlabel('Kategorie des Ergebnisses')
ax.set_ylabel(f'Anzahl der Prüfergebnisse (n={int(counts["Anzahl"].sum())})')
ax.set_title('Verteilung nachweisbarer vs. nicht nachweisbarer Ergebnisse über alle Radionuklide')
for p in ax.patches:
    height = p.get_height()
    if height is not None and not np.isnan(height):
        total = counts['Anzahl'].sum()
        pct = 100.0 * float(height) / float(total) if total else 0.0
        ax.annotate(f'{int(height)}\n({pct:.1f}%)', (p.get_x() + p.get_width() / 2., height), ha='center', va='bottom', fontsize=10)
plt.xticks(rotation=0)
plt.tight_layout()
plt.savefig(plot_path, bbox_inches='tight')
plt.close()

    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    str_cols = df.select_dtypes(include=['object', 'string']).columns
    for c in str_cols:
        df[c] = df[c].astype(str).str.strip()
    def parse_num(x):
        if pd.isna(x):
            return np.nan
        s = str(x).strip()
        if s == '' or s.lower().startswith('n'):
            return np.nan
        s = s.replace('.', '')
        s = s.replace(',', '.')
        try:
            return float(s)
        except Exception:
            return np.nan
    df['Ergebnis_num'] = df['Ergebnis'].apply(parse_num)
    _missing_ratio = df.replace({'': np.nan}).isna().mean()
