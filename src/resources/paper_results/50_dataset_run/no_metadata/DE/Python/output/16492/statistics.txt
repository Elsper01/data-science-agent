import os
import sys
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

try:
    data_path = os.path.normpath('./src/resources/data/study\\16492\\dataset.csv')
    output_dir = os.path.normpath('./src/resources/output\\16492')
    os.makedirs(output_dir, exist_ok=True)

    df = pd.read_csv(data_path, sep=',', encoding='utf-8', low_memory=False)
    df = df.replace(r'^\s*$', np.nan, regex=True)

    df['value_num'] = pd.to_numeric(df['value'], errors='coerce') if 'value' in df.columns else np.nan
    s = df['value_num'].dropna() if 'value_num' in df.columns else pd.Series(dtype=float)
    if s.empty:
        print('Fehler: Keine numerischen Werte in der Spalte "value" gefunden oder Datensatz leer.', file=sys.stderr)
        sys.exit(1)

    missing_pct = df.isna().mean().mul(100).round(2).rename('Anteil_Fehlende_%').to_frame()
    missing_pct.to_csv(os.path.join(output_dir, '1_eda_missing_values.csv'), encoding='utf-8')

    stats_series = pd.Series({
        'Anzahl': int(s.count()),
        'Mittelwert_µSv/h': float(s.mean()),
        'Median_µSv/h': float(s.median()),
        'Stdabw_µSv/h': float(s.std(ddof=1)),
        'Varianz_µSv/h²': float(s.var(ddof=1)),
        'Min_µSv/h': float(s.min()),
        'Q1_µSv/h': float(s.quantile(0.25)),
        'Q3_µSv/h': float(s.quantile(0.75)),
        'Max_µSv/h': float(s.max()),
        'IQR_µSv/h': float(s.quantile(0.75) - s.quantile(0.25)),
        'Schiefe': float(s.skew()),
        'Wölbung': float(s.kurt())
    })
    stats_series.to_frame(name='Wert').to_csv(os.path.join(output_dir, '1_eda_stats_value.csv'), encoding='utf-8')

    df_num = df.apply(pd.to_numeric, errors='coerce')
    if df_num.shape[1] > 0 and df_num.notna().sum().sum() > 0:
        corr = df_num.corr()
        corr.to_csv(os.path.join(output_dir, '1_eda_correlation_numeric.csv'), encoding='utf-8')

    sns.set_theme(style='whitegrid', context='talk', font_scale=0.9)

    n = s.shape[0]
    q25, q75 = s.quantile([0.25, 0.75])
    iqr = q75 - q25
    data_range = float(s.max() - s.min())
    if data_range == 0:
        center = float(s.iloc[0])
        edges = np.linspace(center - 0.005, center + 0.005, 11)
    else:
        if iqr > 0:
            bw = 2 * iqr * np.power(n, -1/3)
        else:
            bw = None
        if bw is None or bw <= 0:
            bins = int(np.ceil(np.log2(n) + 1))
            bins = min(50, max(5, bins))
            counts, edges = np.histogram(s, bins=bins)
        else:
            bins = int(np.ceil(data_range / bw))
            bins = min(60, max(5, bins))
            counts, edges = np.histogram(s, bins=bins)

    counts, edges = np.histogram(s, bins=edges)
    idx_max = int(np.argmax(counts)) if counts.size > 0 else 0
    left = float(edges[idx_max]) if edges.size > 1 else float(s.min())
    right = float(edges[idx_max + 1]) if edges.size > 1 else float(s.max())
    max_count = int(counts[idx_max]) if counts.size > 0 else 0
    max_pct = (max_count / n * 100.0) if n > 0 else 0.0

    fig, ax = plt.subplots(figsize=(10, 6), constrained_layout=True)
    sns.histplot(s, bins=edges, color='#2b8cbe', edgecolor='white', alpha=0.9, ax=ax, stat='count')

    mean_v = float(s.mean())
    median_v = float(s.median())
    max_v = float(s.max())

    ax.axvline(mean_v, color='#e6550d', linestyle='-', linewidth=1.8, label=f'Mittelwert {mean_v:.3f} µSv/h')
    ax.axvline(median_v, color='#31a354', linestyle='--', linewidth=1.8, label=f'Median {median_v:.3f} µSv/h')
    ax.axvline(max_v, color='#756bb1', linestyle=':', linewidth=1.5, label=f'Max {max_v:.3f} µSv/h')

    ymax = counts.max() if counts.size > 0 else 0
    if ymax > 0:
        ax.annotate(
            f'Häufigster Bereich: [{left:.3f}, {right:.3f}] µSv/h\nn={max_count} ({max_pct:.1f}%)',
            xy=((left + right) / 2.0, max_count),
            xytext=((left + right) / 2.0, ymax * 1.15),
            ha='center', va='bottom',
            arrowprops=dict(arrowstyle='->', color='black', lw=1.0),
            bbox=dict(boxstyle='round,pad=0.25', fc='white', ec='gray', alpha=0.95),
            fontsize=10,
        )

    ax.set_title('Verteilung der Gamma-Ortsdosisleistung (ODL) – Lauf an der Pegnitz, Januar 2026')
    ax.set_xlabel('Gamma-Ortsdosisleistung (µSv/h)')
    ax.set_ylabel('Häufigkeit (Anzahl)')
    ax.legend(frameon=True, title='Referenzwerte')

    plot_path = os.path.join(output_dir, '1_histogram_odl_value.png')
    fig.savefig(plot_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
    print(f'Fehler bei der Verarbeitung: {e}', file=sys.stderr)
    sys.exit(1)

import os
import re
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import geopandas as gpd

try:
    data_path = './src/resources/data/study\\16492\\dataset.csv'
    output_dir = os.path.normpath('./src/resources/output\\16492')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(data_path, sep=',', encoding='utf-8')
    for c in df.columns:
        if df[c].dtype == object:
            df[c] = df[c].astype(str).str.strip()
    for c in ['value', 'validated', 'kenn']:
        if c in df.columns:
            df[c] = pd.to_numeric(df[c], errors='coerce')
    for c in ['start_measure', 'end_measure']:
        if c in df.columns:
            df[c] = pd.to_datetime(df[c], errors='coerce')
    missing = df.isna().mean().to_frame('anteil_fehlend').reset_index().rename(columns={'index': 'spalte'})
    missing_path = os.path.join(output_dir, '2_eda_anteil_fehlender_werte.csv')
    missing.to_csv(missing_path, index=False, encoding='utf-8')
    num_cols = df.select_dtypes(include=[np.number]).columns.tolist()
    if len(num_cols) > 0:
        desc = df[num_cols].describe().T.reset_index().rename(columns={'index': 'variable'})
        desc_path = os.path.join(output_dir, '2_eda_numerische_kennwerte.csv')
        desc.to_csv(desc_path, index=False, encoding='utf-8')
    if len(num_cols) >= 2:
        corr = df[num_cols].corr(method='pearson')
        corr_path = os.path.join(output_dir, '2_eda_korrelationen.csv')
        corr.to_csv(corr_path, encoding='utf-8')
    const_cols = [c for c in ['id', 'name', 'nuclide', 'unit', 'duration', 'geom'] if c in df.columns]
    if len(const_cols) > 0:
        const_info = pd.DataFrame({
            'spalte': const_cols,
            'anzahl_auspraegungen': [df[c].nunique(dropna=True) for c in const_cols],
            'beispiel': [df[c].dropna().astype(str).iloc[0] if df[c].notna().any() else '' for c in const_cols]
        })
        const_path = os.path.join(output_dir, '2_eda_konstante_parameter.csv')
        const_info.to_csv(const_path, index=False, encoding='utf-8')
    if 'geom' not in df.columns or df['geom'].dropna().empty:
        raise ValueError('Spalte geom fehlt oder enthält keine gültigen Werte.')
    geom_str = df['geom'].dropna().astype(str).iloc[0]
    m = re.search(r'POINT\s*\(\s*([+-]?\d+\.?\d*)\s+([+-]?\d+\.?\d*)\s*\)', geom_str)
    if not m:
        raise ValueError('Geom-Format nicht als WKT POINT erkennbar.')
    a = float(m.group(1))
    b = float(m.group(2))
    def plausible_germany(lat, lon):
        return (47.0 <= lat <= 56.0) and (5.0 <= lon <= 16.0)
    if plausible_germany(a, b):
        lat, lon = a, b
    elif plausible_germany(b, a):
        lat, lon = b, a
    else:
        lon, lat = a, b
    name = df['name'].dropna().astype(str).iloc[0] if 'name' in df.columns and df['name'].notna().any() else 'Unbekannt'
    sid = df['id'].dropna().astype(str).iloc[0] if 'id' in df.columns and df['id'].notna().any() else 'n/a'
    nuclide = df['nuclide'].dropna().astype(str).iloc[0] if 'nuclide' in df.columns and df['nuclide'].notna().any() else 'n/a'
    unit = df['unit'].dropna().astype(str).iloc[0] if 'unit' in df.columns and df['unit'].notna().any() else 'n/a'
    world = gpd.read_file(gpd.datasets.get_path('naturalearth_lowres'))
    sns.set_theme(style='white', context='talk')
    fig, ax = plt.subplots(figsize=(8, 8), dpi=150)
    world.plot(ax=ax, color='#f2f2f2', edgecolor='#9e9e9e', linewidth=0.5, zorder=1)
    ax.scatter(lon, lat, s=140, c='#d62728', edgecolors='white', linewidths=1.2, zorder=5)
    dx = 3.0
    dy = 3.0
    ax.set_xlim(lon - dx, lon + dx)
    ax.set_ylim(lat - dy, lat + dy)
    info_text = f"{name}\nID: {sid}\nNuklid: {nuclide}\nEinheit: {unit}"
    ax.annotate(info_text, xy=(lon, lat), xycoords='data', xytext=(lon + 0.2, lat + 0.9), textcoords='data', fontsize=10, ha='left', va='bottom', bbox=dict(boxstyle='round,pad=0.35', fc='white', ec='#333333', alpha=0.95), arrowprops=dict(arrowstyle='->', color='#333333', lw=1.0), zorder=6)
    ax.set_title('Geografische Lage der Messstation und zugehörige Metadaten', fontsize=14, pad=12)
    ax.set_xlabel('Längengrad [°]')
    ax.set_ylabel('Breitengrad [°]')
    ax.grid(True, which='major', linestyle='--', linewidth=0.5, color='#d0d0d0')
    ax.set_aspect('equal', adjustable='box')
    plot_path = os.path.join(output_dir, '2_karte_messstation.png')
    plt.tight_layout()
    plt.savefig(plot_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
    print(str(e))
