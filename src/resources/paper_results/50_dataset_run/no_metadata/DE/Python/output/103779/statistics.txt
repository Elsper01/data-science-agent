import os
import sys
import traceback
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

try:
    input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '103779', 'dataset.csv')
    output_dir = os.path.join('.', 'src', 'resources', 'output', '103779')
    os.makedirs(output_dir, exist_ok=True)

    df = pd.read_csv(
        input_path,
        sep=',',
        encoding='utf-8',
        na_values=['nan', 'NaN', 'None', 'NULL', ''],
        keep_default_na=True,
        engine='python'
    )

    df = df.replace(r'^\s+$', np.nan, regex=True)

    missing_share = df.isna().mean().sort_values(ascending=False).to_frame('anteil_fehlend_prozent') * 100
    missing_share.index.name = 'spalte'
    missing_share = missing_share.reset_index()
    missing_share['anteil_fehlend_prozent'] = missing_share['anteil_fehlend_prozent'].round(2)
    missing_path = os.path.join(output_dir, '1_missing_values.csv')
    missing_share.to_csv(missing_path, index=False, encoding='utf-8')

    num_df = df.select_dtypes(include=[np.number])
    if num_df.shape[1] > 0:
        desc = num_df.describe().T
        desc_path = os.path.join(output_dir, '1_numeric_summary.csv')
        desc.to_csv(desc_path, encoding='utf-8')
        if num_df.shape[1] > 1:
            corr = num_df.corr(numeric_only=True)
            corr_path = os.path.join(output_dir, '1_numeric_correlations.csv')
            corr.to_csv(corr_path, encoding='utf-8')
    else:
        pd.DataFrame().to_csv(os.path.join(output_dir, '1_numeric_summary.csv'), encoding='utf-8')

    col_candidates = {c.lower().strip(): c for c in df.columns}
    target_col_key = None
    for key in col_candidates.keys():
        if key == 'kulturdenkmaltyp':
            target_col_key = key
            break
    if target_col_key is None:
        raise KeyError('Spalte "Kulturdenkmaltyp" nicht gefunden.')
    target_col = col_candidates[target_col_key]

    typ_series = df[target_col].astype(str).where(~df[target_col].isna(), other=np.nan)
    typ_series = typ_series.str.strip()
    typ_series = typ_series.replace({'': np.nan})
    typ_series = typ_series.fillna('Unbekannt')

    counts = typ_series.value_counts(dropna=False).rename_axis('Kulturdenkmaltyp').reset_index(name='Anzahl')
    counts_path = os.path.join(output_dir, '1_kulturdenkmaltyp_counts.csv')
    counts.to_csv(counts_path, index=False, encoding='utf-8')

    sns.set_theme(style='whitegrid', context='talk')
    plt.rcParams['figure.dpi'] = 120
    plt.rcParams['savefig.dpi'] = 300

    fig, ax = plt.subplots(figsize=(12, max(4, 0.6 * len(counts))))
    palette = sns.color_palette('Blues_r', n_colors=max(3, len(counts)))
    sns.barplot(data=counts, y='Kulturdenkmaltyp', x='Anzahl', ax=ax, palette=palette)

    ax.set_title('Häufigkeit der Kulturdenkmaltypen im Kreis Segeberg')
    ax.set_xlabel('Anzahl der Objekte')
    ax.set_ylabel('Kulturdenkmaltyp')

    for p in ax.patches:
        width = p.get_width()
        y = p.get_y() + p.get_height() / 2
        ax.text(width + max(counts['Anzahl']) * 0.01, y, f'{int(width)}', va='center', ha='left', fontsize=11)

    ax.margins(x=0.05)
    plt.tight_layout()
    plot_path = os.path.join(output_dir, '1_kulturdenkmaltyp_balken.png')
    plt.savefig(plot_path, bbox_inches='tight', facecolor='white')
    plt.close(fig)
except Exception as e:
    print(str(e))
    traceback.print_exc()
    sys.exit(1)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

try:
    file_path = os.path.normpath(r'./src/resources/data/study\103779\dataset.csv')
    output_dir = os.path.normpath(r'./src/resources/output\103779')
    os.makedirs(output_dir, exist_ok=True)
    plot_path = os.path.join(output_dir, '2_begruendung_balkendiagramm.png')

    df = pd.read_csv(file_path, sep=',', encoding='utf-8')

    _missing_ratio = df.isna().mean().sort_values(ascending=False)
    _num_cols = df.select_dtypes(include=[np.number]).columns.tolist()
    if len(_num_cols) > 0:
        _desc = df[_num_cols].describe().T
        _corr = df[_num_cols].corr()
    else:
        _desc = pd.DataFrame()
        _corr = pd.DataFrame()

    if 'Begr_ndung' in df.columns:
        col = df['Begr_ndung'].astype(str)
        col = col[~col.str.lower().isin(['nan', 'none', ''])]
        exploded = col.str.split(',').explode().astype(str).str.strip()
        exploded = exploded[~exploded.str.lower().isin(['nan', 'none', ''])]
        exploded = exploded.replace('', np.nan).dropna()
        counts = exploded.value_counts().sort_values(ascending=False)
    else:
        counts = pd.Series(dtype=int)

    if counts.empty:
        fig, ax = plt.subplots(figsize=(8, 4))
        ax.axis('off')
        ax.text(0.5, 0.5, 'Keine gültigen Daten in der Spalte "Begr_ndung" verfügbar', ha='center', va='center', fontsize=12)
        plt.savefig(plot_path, dpi=200, bbox_inches='tight')
        plt.close(fig)
    else:
        sns.set_theme(style='whitegrid', font_scale=1.05)
        fig, ax = plt.subplots(figsize=(12, 7))
        palette = sns.color_palette('Blues_d', n_colors=len(counts))
        sns.barplot(x=counts.index, y=counts.values, palette=palette, ax=ax)
        ax.set_title('Häufigkeit der Begründungskategorien für Denkmalschutz im Kreis Segeberg')
        ax.set_xlabel('Begründungskategorie')
        ax.set_ylabel('Anzahl der Nennungen')
        ax.set_xticklabels(ax.get_xticklabels(), rotation=25, ha='right')
        y_max = counts.values.max()
        total = counts.values.sum()
        for i, v in enumerate(counts.values):
            ax.text(i, v + max(0.01 * y_max, 0.1), f'{int(v)} ({(v/total)*100:.1f}%)', ha='center', va='bottom', fontsize=9)
        ax.set_ylim(0, y_max + max(0.1 * y_max, 1))
        plt.tight_layout()
        plt.savefig(plot_path, dpi=200, bbox_inches='tight')
        plt.close(fig)
except Exception as e:
    print(str(e))
