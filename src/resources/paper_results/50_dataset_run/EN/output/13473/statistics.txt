import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

plt.switch_backend('Agg')

def run_visualization():
    input_path = r'./src/resources/data/study/13473/dataset.csv'
    output_dir = r'./src/resources/output/13473'
    os.makedirs(output_dir, exist_ok=True)

    # The error 'utf-8' codec can't decode byte 0xf6 (ö) suggests the file is likely encoded in ISO-8859-1 (latin1)
    # which is common for German datasets containing umlauts.
    try:
        df = pd.read_csv(input_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=';', encoding='latin1')

    # Data Cleaning
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')

    # EDA Outputs
    na_share = df.isna().mean().to_frame('fehlendenanteil')
    na_share.to_csv(os.path.join(output_dir, '0_eda_na_anteile.csv'), encoding='utf-8')

    num_cols = df.select_dtypes(include=[np.number]).columns.tolist()
    if len(num_cols) > 0:
        desc = df[num_cols].describe().T
        desc.to_csv(os.path.join(output_dir, '0_eda_numeric_describe.csv'), encoding='utf-8')

    # Visualization Logic
    level_col = '1_variable_code'
    gender_label_col = '2_variable_attribute_label'
    value_col = 'value'

    if level_col not in df.columns or gender_label_col not in df.columns or value_col not in df.columns:
        print("Required columns missing.")
        return

    # Filter for administrative level 'KREISE' as requested
    if (df[level_col] == 'KREISE').any():
        df_level = df[df[level_col] == 'KREISE'].copy()
        level_name = 'Kreise'
    else:
        # Fallback to first available level if KREISE is not found
        first_level = df[level_col].dropna().unique()[0]
        df_level = df[df[level_col] == first_level].copy()
        level_name = str(first_level)

    # Aggregate data for the bar chart
    df_plot = (
        df_level[[gender_label_col, value_col]]
        .dropna(subset=[gender_label_col, value_col])
        .groupby(gender_label_col, as_index=False)[value_col]
        .sum()
    )

    # Define order and colors
    preferred_order = ['männlich', 'weiblich', 'Insgesamt']
    available = df_plot[gender_label_col].astype(str).tolist()
    ordered = [x for x in preferred_order if x in available] + [x for x in available if x not in preferred_order]
    
    df_plot[gender_label_col] = pd.Categorical(df_plot[gender_label_col], categories=ordered, ordered=True)
    df_plot = df_plot.sort_values(gender_label_col)

    # Formatting helper
    def format_de(x):
        return f"{int(x):,}".replace(',', '.')

    # Plotting
    sns.set_theme(style='whitegrid')
    fig, ax = plt.subplots(figsize=(10, 6))
    
    # Map colors
    color_palette = {'männlich': '#4C78A8', 'weiblich': '#F58518', 'Insgesamt': '#76b7b2'}
    palette = [color_palette.get(str(cat), '#BAB0AC') for cat in ordered]

    sns.barplot(data=df_plot, x=gender_label_col, y=value_col, palette=palette, ax=ax)

    # Labels and Title
    ax.set_xlabel('Geschlecht', fontsize=12)
    ax.set_ylabel('Bevölkerung (Anzahl)', fontsize=12)
    
    time_info = ""
    if 'time' in df.columns:
        time_info = f" ({df['time'].iloc[0]})"
    
    ax.set_title(f'Geschlechterverteilung der Bevölkerung in Hof{time_info}\nEbene: {level_name}', fontsize=14, pad=20)
    ax.yaxis.set_major_formatter(FuncFormatter(lambda x, p: format_de(x)))

    # Add value labels on top of bars
    for i, p in enumerate(ax.patches):
        val = df_plot.iloc[i][value_col]
        ax.annotate(format_de(val), 
                    (p.get_x() + p.get_width() / 2., p.get_height()), 
                    ha='center', va='center', 
                    xytext=(0, 9), 
                    textcoords='offset points',
                    fontsize=11)

    plt.tight_layout()
    output_path = os.path.join(output_dir, '0_geschlechterverteilung_kreise.png')
    plt.savefig(output_path, dpi=200)
    plt.close()

if __name__ == "__main__":
    try:
        run_visualization()
    except Exception as e:
        print(f"An error occurred: {e}")

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter
import os

# Set visual style
sns.set_theme(style='whitegrid', context='talk')

# File paths
input_path = r'./src/resources/data/study/13473/dataset.csv'
output_dir = r'./src/resources/output/13473'
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '1_vergleich_berichtsebenen_geschlecht.png')

# Fix: Handle UnicodeDecodeError by trying 'latin-1' or 'iso-8859-1' which are common for German datasets
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8', parse_dates=['time'], dayfirst=False)
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1', parse_dates=['time'], dayfirst=False)

# Filter data for the specific visualization goal
# We want to compare KREISE and GEMEIN reporting levels for the 'Geschlecht' (Gender) variable
mask = (df['1_variable_code'].isin(['KREISE', 'GEMEIN'])) & \
       (df['2_variable_label'] == 'Geschlecht') & \
       (~df['value'].isna())

dfp = df.loc[mask, ['1_variable_code', '2_variable_attribute_label', 'value']].copy()

# Define order for consistency
order_x = ['KREISE', 'GEMEIN']
order_hue = ['männlich', 'weiblich', 'Insgesamt']

# Ensure categories are ordered correctly
dfp['1_variable_code'] = pd.Categorical(dfp['1_variable_code'], categories=order_x, ordered=True)
# Filter hue categories to only those present in the data to avoid empty bars
available_hues = [h for h in order_hue if h in dfp['2_variable_attribute_label'].unique()]
dfp['2_variable_attribute_label'] = pd.Categorical(dfp['2_variable_attribute_label'], categories=available_hues, ordered=True)

# Check if values are identical between levels for the caption
pvt = dfp.pivot_table(index='2_variable_attribute_label', columns='1_variable_code', values='value', aggfunc='sum')
if pvt.shape[1] == 2:
    eq_all = np.allclose(pvt.iloc[:, 0], pvt.iloc[:, 1], equal_nan=True)
    caption = 'Hinweis: Werte für KREISE und GEMEIN sind identisch.' if eq_all else 'Hinweis: Vergleich der Berichtsebenen.'
else:
    caption = ''

# Plotting
plt.figure(figsize=(12, 7))
ax = sns.barplot(
    data=dfp, 
    x='1_variable_code', 
    y='value', 
    hue='2_variable_attribute_label', 
    palette='viridis'
)

ax.set_title('Bevölkerung in Hof: Vergleich der Berichtsebenen (KREISE vs. GEMEIN)', pad=20)
ax.set_xlabel('Berichtsebene (1_variable_code)')
ax.set_ylabel('Bevölkerungszahl')
ax.legend(title='Kategorie', loc='upper right', bbox_to_anchor=(1.15, 1))

# Format y-axis with German thousands separator
formatter = FuncFormatter(lambda x, pos: f"{int(x):,}".replace(',', '.'))
ax.yaxis.set_major_formatter(formatter)

# Add value labels on top of bars
for p in ax.patches:
    h = p.get_height()
    if np.isfinite(h) and h > 0:
        ax.annotate(f"{int(round(h)):,}".replace(',', '.'), 
                    (p.get_x() + p.get_width() / 2, h), 
                    ha='center', va='bottom', 
                    fontsize=10, xytext=(0, 3), 
                    textcoords='offset points')

plt.figtext(0.5, 0.01, caption, ha='center', fontsize=10, style='italic')
plt.tight_layout()

# Save the plot
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close()

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from matplotlib.patches import FancyBboxPatch

# The error 'utf-8' codec can't decode byte 0xf6 suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
# common for German datasets (ö = 0xf6 in latin-1).

try:
    sns.set_theme(context='talk', style='white')
    dataset_path = r'./src/resources/data/study/13473/dataset.csv'
    
    # Attempt reading with latin-1 encoding to handle German umlauts
    try:
        df = pd.read_csv(dataset_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(dataset_path, sep=';', encoding='latin-1')

    # Data Cleaning
    for c in df.select_dtypes(include=['object', 'string']).columns:
        df[c] = df[c].astype(str).str.strip().replace({'': np.nan, 'nan': np.nan})
    
    if 'value' in df.columns:
        df['value'] = pd.to_numeric(df['value'], errors='coerce')
    if 'time' in df.columns:
        df['time'] = pd.to_datetime(df['time'], errors='coerce')

    # Filter for the specific indicator: 'KREISE' and 'Insgesamt'
    # Based on the goal: 'value' filtered where '2_variable_attribute_label' is 'Insgesamt' and '1_variable_code' is 'KREISE'
    mask = (df['1_variable_code'] == 'KREISE') & (df['2_variable_attribute_label'] == 'Insgesamt')
    total_indicator = df[mask]

    if total_indicator.empty:
        # Fallback if KREISE is not found, try to find any 'Insgesamt'
        total_indicator = df[df['2_variable_attribute_label'] == 'Insgesamt']
        
    if total_indicator.empty:
        raise ValueError('Keine Daten für "Insgesamt" gefunden.')

    # Get the latest record if multiple exist
    if 'time' in total_indicator.columns:
        total_indicator = total_indicator.sort_values('time', ascending=False)
    
    latest_row = total_indicator.iloc[0]
    total_value = latest_row['value']
    
    if pd.isna(total_value):
        raise ValueError('Der Wert für die Gesamtbevölkerung ist nicht numerisch verfügbar.')

    # Metadata for display
    ref_date = latest_row['time'] if 'time' in latest_row and pd.notna(latest_row['time']) else None
    ref_date_str = ref_date.strftime('%d.%m.%Y') if ref_date is not None else '2024'
    unit = latest_row['value_unit'] if 'value_unit' in latest_row and pd.notna(latest_row['value_unit']) else 'Einwohner'
    region_label = latest_row['1_variable_label'] if '1_variable_label' in latest_row and pd.notna(latest_row['1_variable_label']) else 'Hof (Krfr.St)'
    stat_label = df['statistics_label'].dropna().unique()[0] if 'statistics_label' in df.columns and not df['statistics_label'].dropna().empty else 'Amtliche Statistik'

    def fmt_de(n):
        try:
            return f"{int(round(n)):,.0f}".replace(',', '.')
        except Exception:
            return str(n)

    # Visualization
    output_dir = r'./src/resources/output/13473'
    os.makedirs(output_dir, exist_ok=True)
    output_path = os.path.join(output_dir, '2_indikator_gesamtbevoelkerung_hof.png')

    fig = plt.figure(figsize=(10, 6), dpi=200)
    ax = fig.add_axes([0, 0, 1, 1])
    ax.set_xlim(0, 1)
    ax.set_ylim(0, 1)
    ax.axis('off')

    # Background Card
    card = FancyBboxPatch((0.1, 0.2), 0.8, 0.6, boxstyle='round,pad=0.02,rounding_size=0.05', 
                          linewidth=0, facecolor='#F8FAFC', edgecolor='none')
    ax.add_patch(card)

    # Title
    ax.text(0.5, 0.72, f'Gesamtbevölkerung in {region_label}', ha='center', va='center', 
            fontsize=20, color='#1E293B', fontweight='bold')
    
    # Subtitle (Date)
    ax.text(0.5, 0.65, f'Stand: {ref_date_str}', ha='center', va='center', 
            fontsize=14, color='#64748B')

    # Main Value
    ax.text(0.5, 0.45, fmt_de(total_value), ha='center', va='center', 
            fontsize=80, color='#0891B2', fontweight='bold')

    # Unit/Label
    ax.text(0.5, 0.32, unit, ha='center', va='center', 
            fontsize=18, color='#475569', fontweight='medium')

    # Footer
    footer_text = f'Quelle: {stat_label}'
    ax.text(0.5, 0.12, footer_text, ha='center', va='center', fontsize=10, color='#94A3B8')

    plt.savefig(output_path, dpi=300, bbox_inches='tight', facecolor='white')
    plt.close(fig)

except Exception as e:
    print(f"Error occurred: {e}")
