import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="white")

file_path = r'./src/resources/data/study\32509\dataset.csv'
out_path = r'./src/resources/output\32509/0_tabelle_kommune_name.png'

obj_cols = df.select_dtypes(include=['object']).columns.tolist()
for c in obj_cols:
    df[c] = df[c].apply(lambda x: np.nan if pd.isna(x) or str(x).strip() in {'', '-'} else str(x).strip())

missing_prop = df.isna().mean()
row_count = int(df.shape[0])

ax.axis('off')

table = ax.table(cellText=cell_text, colLabels=col_labels, cellLoc='left', colLoc='left', cellColours=cell_colours, loc='center')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

sns.set_theme(style="whitegrid", context="talk")

fp = r"./src/resources/data/study\32509\dataset.csv"
df = pd.read_csv(fp, sep=",", encoding="utf-8")

for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].apply(lambda x: x.strip() if isinstance(x, str) else x)
        df[c] = df[c].replace({"-": np.nan, "–": np.nan, "—": np.nan, "": np.nan})

missing_prop = df.isna().mean()
num_cols = df.select_dtypes(include=[np.number]).columns.tolist()
if len(num_cols) > 0:
    corr_num = df[num_cols].corr()

post_series = df["Postleitzahl_Ort"].apply(lambda x: x.strip() if isinstance(x, str) else x)
post_series = post_series.replace({"-": np.nan, "–": np.nan, "—": np.nan, "": np.nan})
counts = post_series.fillna("Unbekannt").value_counts(dropna=False).reset_index()
counts.columns = ["Postleitzahl_Ort", "Anzahl"]
counts = counts.sort_values(["Anzahl", "Postleitzahl_Ort"], ascending=[False, True]).reset_index(drop=True)

plt.figure(figsize=(12, 7))
palette = sns.color_palette("viridis", n_colors=len(counts))
ax = sns.barplot(data=counts, x="Postleitzahl_Ort", y="Anzahl", palette=palette)
ax.set_title("Geografische Verteilung der Verwaltungsstellen nach Postleitzahl/Ort (Kreis Kleve, 2021)")
ax.set_xlabel("Postleitzahl und Ort")
ax.set_ylabel("Anzahl der Einträge [Stück]")
plt.xticks(rotation=35, ha="right")
for p in ax.patches:
    h = p.get_height()
    ax.annotate(f"{int(h)}", (p.get_x() + p.get_width() / 2, h), ha="center", va="bottom", fontsize=11)
plt.tight_layout()

out_dir = r"./src/resources/output\32509"

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(context='notebook', style='white')

input_path = r'./src/resources/data/study\32509\dataset.csv'
output_dir = r'./src/resources/output\32509'
os.makedirs(output_dir, exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

def _strip(x):
    if isinstance(x, str):
        return x.strip()
    return x

df = df.applymap(_strip)
df.replace({'-': np.nan, '': np.nan}, inplace=True)

contacts_df = df[['Kommune_Stadt', 'Telefon', 'E_Mail']].copy()

n_rows = len(df)
missing_counts_all = df.isna().sum()
missing_props_all = (df.isna().mean()).round(3)
missing_counts = contacts_df.isna().sum()
missing_props = (contacts_df.isna().mean()).round(3)
unique_kommune = df['Kommune_Stadt'].nunique(dropna=True)
complete_contacts = int(((~contacts_df['Telefon'].isna()) & (~contacts_df['E_Mail'].isna())).sum())

contacts_df_sorted = contacts_df.sort_values(by=['Kommune_Stadt'], key=lambda s: s.str.lower() if s.dtype == 'O' else s)
contacts_df_display = contacts_df_sorted.fillna('Keine Angabe')

fig, ax = plt.subplots(figsize=(18, 9))
ax.axis('off')

header_labels = ['Kommune/Stadt', 'Telefon', 'E-Mail']
cell_text = contacts_df_display.values.tolist()

rows = contacts_df_display.shape[0]
cols = contacts_df_display.shape[1]
cell_colours = [[(1, 1, 1, 1) for _ in range(cols)] for _ in range(rows)]
for i, (_, row) in enumerate(contacts_df_sorted.iterrows()):
    if pd.isna(row['Telefon']):
        cell_colours[i][1] = (1.0, 0.88, 0.88, 1)
    if pd.isna(row['E_Mail']):
        cell_colours[i][2] = (1.0, 0.88, 0.88, 1)

header_colours = [(0.90, 0.94, 0.98, 1)] * cols

table = ax.table(cellText=cell_text,
                 colLabels=header_labels,
                 cellColours=cell_colours,
                 colColours=header_colours,
                 loc='center')

table.auto_set_font_size(False)
table.set_fontsize(11)
table.scale(1, 1.4)

for (row, col), cell in table.get_celld().items():
    if row == 0:
        cell.set_text_props(weight='bold', color='black')
        cell.set_height(cell.get_height() * 1.2)
    cell.set_edgecolor((0.85, 0.85, 0.85, 1))

try:
    table.auto_set_column_width(col=list(range(cols)))
except Exception:
    pass

fig.suptitle('Primäre Kontaktwege je Kommune: Telefon und E-Mail\nGleichstellungsbeauftragte, Kreis Kleve (2021)', x=0.5, y=0.97, fontsize=16, fontweight='bold')

eda_lines = []
eda_lines.append(f'Einträge: {n_rows}, Kommunen: {unique_kommune}, vollständige Kontakte (Telefon & E-Mail): {complete_contacts}')
eda_lines.append(f'Fehlende Angaben – Telefon: {int(missing_counts["Telefon"])} ({int(missing_props["Telefon"]*100)}%), E-Mail: {int(missing_counts["E_Mail"])} ({int(missing_props["E_Mail"]*100)}%)')
eda_lines.append('Hinweis: "-" wurde als fehlend interpretiert; fehlende Felder in Hellrot hervorgehoben.')
footer_text = '\n'.join(eda_lines)

fig.text(0.5, 0.06, footer_text, ha='center', va='center', fontsize=10, color='dimgray')

output_path = os.path.join(output_dir, '2_kontakt_wege_tabelle.png')
plt.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close(fig)
