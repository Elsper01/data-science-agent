    def parse_erg(x):
        if pd.isna(x):
            return np.nan
        s = str(x).strip().lower()
        if s in {'n.n.', 'n.n', 'nn', 'nicht nachweisbar', 'nicht nachw.', 'na', ''}:
            return np.nan
        s = s.replace('≤', '').replace('≥', '').replace('<', '').replace('>', '')
        s = s.replace(' ', '')
        if ',' in s and not any(ch in s for ch in ['e+', 'e-']):
            s = s.replace('.', '').replace(',', '.') if s.count(',') == 1 and s.count('.') > 1 else s.replace(',', '.')
        try:
            return float(s)
        except Exception:
            try:
                return pd.to_numeric(s, errors='coerce')
            except Exception:
                return np.nan
    if 'Ergebnis' in df.columns:
        df['Ergebnis_num'] = df['Ergebnis'].apply(parse_erg)
        erg_stats = df['Ergebnis_num'].describe(percentiles=[0.25, 0.5, 0.75]).to_frame(name='Ergebnis_num')

import os
import re
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

pd.set_option('display.max_columns', None)
sns.set_theme(style='whitegrid', context='notebook')
plt.rcParams['axes.titlesize'] = 14
plt.rcParams['axes.labelsize'] = 12
plt.rcParams['xtick.labelsize'] = 10
plt.rcParams['ytick.labelsize'] = 10
plt.rcParams['figure.dpi'] = 150

def normpath(p):
    return os.path.normpath(p)

data_path = normpath('./src/resources/data/study\\137336\\dataset.csv')
out_dir = normpath('./src/resources/output\\137336\\run_4')
os.makedirs(out_dir, exist_ok=True)

df = pd.read_csv(data_path, sep=',', encoding='utf-8')

def clean_unit(u):
    if pd.isna(u):
        return 'Fehlend/ohne Einheit'
    s = str(u).strip()
    s = re.sub(r'\s+', ' ', s)
    return s if s != '' else 'Fehlend/ohne Einheit'

def parse_ergebnis_value(x):
    if pd.isna(x):
        return np.nan
    s = str(x).strip()
    if s.lower() in {'n.n.', 'n.n', 'nn', 'na', 'n/a', 'k.a.', 'k.a', 'nan', ''}:
        return np.nan
    s2 = s.replace(',', '.')
    m = re.search(r'[-+]?\d*\.?\d+(?:[eE][-+]?\d+)?', s2)
    return float(m.group(0)) if m else np.nan

def is_censored(x):
    if pd.isna(x):
        return False
    return '<' in str(x)

df['Einheit_clean'] = df['Einheit'].apply(clean_unit)
df['Ergebnis_num'] = df['Ergebnis'].apply(parse_ergebnis_value)
df['Ergebnis_censored'] = df['Ergebnis'].apply(is_censored)

missing_df = pd.DataFrame({'missing_count': df.isna().sum(), 'missing_share': df.isna().mean()})
missing_df.to_csv(os.path.join(out_dir, 'eda_missing_values.csv'), encoding='utf-8')

summary = df['Ergebnis_num'].describe(percentiles=[0.25, 0.5, 0.75]).to_frame(name='Ergebnis_num')
summary.to_csv(os.path.join(out_dir, 'eda_ergebnis_summary.csv'), encoding='utf-8')

unit_counts = df['Einheit_clean'].value_counts(dropna=False)
unit_perc = unit_counts / unit_counts.sum()

order_idx = unit_counts.index.tolist()
values = unit_counts.values
percs = unit_perc.values

fig_h = max(3, min(20, 0.5 * len(order_idx) + 1))
plt.figure(figsize=(10, fig_h))
palette = sns.color_palette('viridis', n_colors=len(order_idx))
sns.barplot(x=values, y=order_idx, palette=palette, orient='h')
plt.xlabel('Anzahl der Prüfmerkmale')
plt.ylabel('Einheit')
plt.title(f'Verteilung der verwendeten Messeinheiten über {len(df)} Prüfmerkmale')
max_val = values.max() if len(values) > 0 else 0
for i, (v, p) in enumerate(zip(values, percs)):
    plt.text(v + max(1, 0.01 * max_val), i, f'{int(v)} ({p*100:.1f}%)', va='center', ha='left', fontsize=10)
plt.tight_layout()
plt.savefig(os.path.join(out_dir, '2_verteilung_einheiten.png'), dpi=300, bbox_inches='tight')
plt.close()
