options(warn=-1)
packages <- c("tidyverse", "lubridate", "janitor", "scales")

  if (!requireNamespace(p, quietly = TRUE)) install.packages(p, dependencies = TRUE, quiet = TRUE) 
}
invisible(lapply(packages, function(p) suppressPackageStartupMessages(library(p, character.only = TRUE))))

# Create output directory
output_dir <- "./src/resources/output/32743"
dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)

# Load data
path_data <- "./src/resources/data/study/32743/dataset.csv"
df <- readr::read_csv(path_data, locale = readr::locale(encoding = "UTF-8"), show_col_types = FALSE)

# Clean and transform
df <- janitor::remove_empty(df, which = c("cols"))

# Helper functions
conv_pct <- function(x) suppressWarnings(readr::parse_number(as.character(x)) / 100)

df_clean <- df %>%
  mutate(
    Datum = suppressWarnings(lubridate::dmy(as.character(Datum))),
    Bounce_Rate_num = conv_pct(Bounce_Rate),
    Seiten_pro_Besuch = as.numeric(Seiten_pro_Besuch)
  ) %>%
  filter(!is.na(Datum), !is.na(Bounce_Rate_num), !is.na(Seiten_pro_Besuch))

# Statistics for subtitle
n_days <- nrow(df_clean)
r_pearson <- suppressWarnings(cor(df_clean$Bounce_Rate_num, df_clean$Seiten_pro_Besuch, use = "complete.obs", method = "pearson"))
avg_pages <- mean(df_clean$Seiten_pro_Besuch, na.rm = TRUE)

subtitle_txt <- paste0(
  "n = ", n_days, " Tage; Korrelation (Pearson) = ", sprintf("%.2f", r_pearson),
  "; Ø Seiten/Besuch = ", sprintf("%.2f", avg_pages)
)

# Visualization
p <- ggplot(df_clean, aes(x = Bounce_Rate_num, y = Seiten_pro_Besuch)) +
  geom_point(color = "#2C7FB8", fill = "#A6CEE3", size = 3, alpha = 0.8, shape = 21) +
  geom_smooth(method = "lm", se = TRUE, color = "#D95F0E", fill = "#FDBB84", linewidth = 1) +
  scale_x_continuous(labels = scales::label_percent(accuracy = 1), name = "Absprungrate (Bounce Rate)") +
  scale_y_continuous(name = "Seiten pro Besuch") +
  labs(
    title = "Zusammenhang: Absprungrate vs. Seiten pro Besuch",
    subtitle = subtitle_txt,
    caption = "Quelle: Webanalyse kreis-kleve.de (Dezember 2021)"
  ) +
  theme_minimal(base_size = 12) +
  theme(
    plot.title = element_text(face = "bold", size = 14),
    plot.subtitle = element_text(size = 10, color = "grey30"),
    panel.grid.minor = element_blank()
  )

# Save plot
ggsave(
  filename = "1_scatter_absprungrate_vs_seiten_pro_besuch.png", 
  plot = p, 
  path = output_dir, 
  width = 8, 
  height = 6, 
  dpi = 300
)
