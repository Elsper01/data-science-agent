set.seed(1)
input_path <- file.path("src","resources","data","study","51886","dataset.csv")

dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)

# Fix for UTF-8 / Encoding issues: Read with Latin1 if UTF-8 fails or check common German encodings
# The error "invalid UTF-8" often happens with German Umlauts (ä, ö, ü, ß) in CSVs from German sources
df <- suppressMessages(readr::read_delim(
  file = input_path, 
  delim = ";", 
  locale = readr::locale(encoding = "ISO-8859-1"), # Common for German CSVs if UTF-8 fails
  trim_ws = TRUE, 
  show_col_types = FALSE, 
  progress = FALSE
))

# Clean column names
colnames(df) <- make.names(colnames(df), unique = TRUE)

# Identify columns
col_value <- if ("value" %in% names(df)) "value" else names(df)[grepl("value", names(df))][1]
col_label2 <- if ("X2_variable_attribute_label" %in% names(df)) "X2_variable_attribute_label" else names(df)[grepl("X2_variable_attribute_label", names(df))][1]

# Data Processing
# We use iconv to ensure strings are valid UTF-8 before string operations like trimws/tolower
plot_df <- df %>%
  mutate(across(where(is.character), ~iconv(., from = "ISO-8859-1", to = "UTF-8", sub = ""))) %>%
  filter(!is.na(.data[[col_label2]])) %>%
  # Filter out "Insgesamt" (Total)
  filter(!str_detect(tolower(trimws(.data[[col_label2]])), "^insgesamt$")) %>%
  mutate(across(all_of(col_value), as.numeric)) %>%
  group_by(Kategorie_Raw = .data[[col_label2]]) %>%
  summarise(Summe = sum(.data[[col_value]], na.rm = TRUE), .groups = "drop") %>%
  filter(Summe > 0) %>%
  mutate(Kategorie = forcats::fct_reorder(Kategorie_Raw, Summe))

# Save summary data
readr::write_csv(plot_df, file.path(output_dir, "eda_raumzahl_summen.csv"))

# Visualization
p <- ggplot(plot_df, aes(x = Summe, y = Kategorie)) +
  geom_col(fill = "#2E86AB", width = 0.7) +
  geom_text(aes(label = scales::number(Summe, big.mark = ".", decimal.mark = ",")), 
            hjust = -0.2, size = 3.5) +
  scale_x_continuous(
    labels = function(x) scales::number(x, big.mark = ".", decimal.mark = ","),
    expand = expansion(mult = c(0, 0.15))
  ) +
  labs(
    title = "Fertiggestellte Wohnungen nach Raumzahl",
    subtitle = "Gemeinde Oberschleißheim; Summe über den gesamten Zeitraum",
    x = "Summe der Fertigstellungen (Anzahl)",
    y = "Wohnungsgröße (Raumzahl)"
  ) +
  theme_minimal(base_family = "sans") +
  theme(
    plot.title = element_text(face = "bold", size = 12),
    plot.subtitle = element_text(size = 10),
    axis.title = element_text(size = 10),
    axis.text = element_text(size = 9),
    panel.grid.major.y = element_blank(),
    panel.grid.minor = element_blank(),
    plot.margin = margin(10, 40, 10, 10)
  )

cat("Visualization created successfully at:", plot_path)
