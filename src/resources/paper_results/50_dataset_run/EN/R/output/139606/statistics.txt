df <- df %>%

  filter(
    !is.na(ergebnis_num),
    !str_detect(tolower(ergebnis), "^n\\.n\\."),
    !str_detect(tolower(pruefmerkmal), "messzeit")
  ) %>%
  mutate(kategorie = case_when(
    str_detect(tolower(pruefmerkmal), "kalium-40|k-40") ~ "Natürliche Radionuklide",
    str_detect(tolower(pruefmerkmal), "cäsium-137|caesium-137|cs-137") ~ "Künstliche Radionuklide",
