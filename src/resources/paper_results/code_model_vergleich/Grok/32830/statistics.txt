import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import seaborn as sns
sns.set_style('whitegrid')
df = pd.read_csv('./src/resources/data/study\\32830\\dataset.csv', sep=',', encoding='utf-8')
df.drop(columns=['Unnamed__14'], inplace=True, errors='ignore')
df_daily = df[df['Datum'].notna()].copy()
df_daily['Datum'] = pd.to_datetime(df_daily['Datum'], format='%d.%m.%Y')
df_daily = df_daily.sort_values('Datum').reset_index(drop=True)
df_daily['Seiten_pro_Besuch'] = pd.to_numeric(df_daily['Seiten_pro_Besuch'].astype(str).str.strip(), errors='coerce')
df_daily['Bounce_Rate'] = pd.to_numeric(df_daily['Bounce_Rate'].astype(str).str.strip().str.replace('%', ''), errors='coerce') / 100
df_daily = df_daily.dropna(subset=['Seiten_pro_Besuch', 'Bounce_Rate'])
corr = df_daily['Seiten_pro_Besuch'].corr(df_daily['Bounce_Rate'])
fig, ax1 = plt.subplots(figsize=(14, 7))
line1 = ax1.plot(df_daily['Datum'], df_daily['Seiten_pro_Besuch'], color='tab:blue', linewidth=2, marker='o')
ax1.set_xlabel('Datum', fontsize=12)
ax1.set_ylabel('Seiten pro Besuch', color='tab:blue', fontsize=12)
ax1.tick_params(axis='y', labelcolor='tab:blue')
ax1.grid(True, alpha=0.3)
ax2 = ax1.twinx()
line2 = ax2.plot(df_daily['Datum'], df_daily['Bounce_Rate'], color='tab:red', linewidth=2, marker='s')
ax2.set_ylabel('Bounce Rate', color='tab:red', fontsize=12)
ax2.tick_params(axis='y', labelcolor='tab:red')
lines = line1 + line2
labels = [l.get_label() for l in lines]
ax1.legend(lines, labels, loc='upper right', fontsize=10)
plt.title('Täglicher Zusammenhang zwischen Seiten pro Besuch und Bounce Rate\n(Korrelation: {:.3f})'.format(corr), fontsize=14, pad=20)
fig.autofmt_xdate(rotation=45)
plt.tight_layout()
plt.savefig('./src/resources/output\\32830/1_seiten_bounce_zusammenhang.png', dpi=300, bbox_inches='tight')
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_style('whitegrid')


df = pd.read_csv(r'./src/resources/data/study\32830\dataset.csv', sep=',', encoding='utf-8')

df.drop('Unnamed__14', axis=1, inplace=True)

df = df[df['Datum'].notna()].copy()

df['Datum'] = pd.to_datetime(df['Datum'], format='%d.%m.%Y')

df.sort_values('Datum', inplace=True)


df['Anteil_Besuche_mit_Cookies_num'] = pd.to_numeric(df['Anteil_Besuche_mit_Cookies'].str.rstrip('%'), errors='coerce') / 100


print('Statistiken für Anteil_Besuche_mit_Cookies:')
print(df['Anteil_Besuche_mit_Cookies_num'].describe())


plt.figure(figsize=(14, 7))
plt.plot(df['Datum'], df['Anteil_Besuche_mit_Cookies_num'], marker='o', linewidth=2, markersize=6, color='steelblue')
plt.fill_between(df['Datum'], df['Anteil_Besuche_mit_Cookies_num'], alpha=0.3, color='steelblue')

plt.title('Entwicklung des Anteils der Besuche mit Cookies (Anteil_Besuche_mit_Cookies) im September 2024', fontsize=16, pad=20)
plt.xlabel('Datum', fontsize=12)
plt.ylabel('Anteil in %', fontsize=12)

plt.xticks(rotation=45)
plt.grid(True, alpha=0.3)

plt.tight_layout()

plt.savefig(r'./src/resources/output\32830\2_anteil_besuche_mit_cookies_over_datum.png', dpi=300, bbox_inches='tight')
