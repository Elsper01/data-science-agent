import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from pathlib import Path

# Define paths
input_path = Path(r'./src/resources/data/study/13979/dataset.csv')
output_dir = Path(r'./src/resources/output/13979')
output_dir.mkdir(parents=True, exist_ok=True)

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xfc' 
# suggests the file is likely encoded in 'latin-1' or 'iso-8859-1' (0xfc is 'ü' in latin-1).
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Data Cleaning
# Ensure 'value' and 'time' are numeric
for col in ['value', 'time']:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='coerce')

# EDA: Missing values
eda_missing = df.isna().mean().reset_index()
eda_missing.columns = ['spalte', 'anteil_missing']
eda_missing.to_csv(output_dir / 'eda_missing_values.csv', index=False, encoding='utf-8')

# Filter for the specific visualization goal
# Goal: '2_variable_attribute_label' is 'Insgesamt'
label_col = '2_variable_attribute_label'
if label_col in df.columns:
    # Use case-insensitive matching and strip whitespace
    mask_insgesamt = df[label_col].astype(str).str.strip().str.lower() == 'insgesamt'
    ts = df[mask_insgesamt].copy()
else:
    ts = df.copy()

# Filter for the timeframe 2014-2024
if 'time' in ts.columns:
    ts = ts[(ts['time'] >= 2014) & (ts['time'] <= 2024)]

# Aggregate if there are multiple entries per year (though usually one for 'Insgesamt')
if not ts.empty and 'time' in ts.columns and 'value' in ts.columns:
    ts = ts.groupby('time', as_index=False)['value'].sum()
    ts = ts.sort_values('time')

# Visualization
sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(12, 6), constrained_layout=True)

if not ts.empty:
    sns.lineplot(data=ts, x='time', y='value', marker='o', linewidth=2.5, color='#1f77b4', ax=ax)
    
    # Set integer ticks for years
    years = sorted(ts['time'].unique().astype(int))
    ax.set_xticks(years)
    
    # Add data labels
    for x, y in zip(ts['time'], ts['value']):
        if pd.notnull(x) and pd.notnull(y):
            ax.text(x, y, f"{int(y)}", ha='center', va='bottom', fontsize=10, fontweight='bold', color='#1f77b4')

ax.set_title('Fertiggestellte Wohnungen (Insgesamt) in Sünching, 2014–2024', loc='left', pad=20)
ax.set_xlabel('Jahr')
ax.set_ylabel('Anzahl Wohnungen')
ax.grid(True, which='major', axis='both', linestyle='--', linewidth=0.6, alpha=0.6)

plt.savefig(output_dir / '0_baufertigstellungen_insgesamt_zeitreihe.png', dpi=200)
plt.close()

# Save descriptive stats for the filtered series
if not ts.empty:
    ts.describe().to_csv(output_dir / 'eda_timeseries_insgesamt_stats.csv', encoding='utf-8')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os
import re
from matplotlib.ticker import MaxNLocator

# Set visual style
sns.set_theme(style="whitegrid")

# Data path
data_path = r'./src/resources/data/study/13979/dataset.csv'

# Load data with appropriate encoding
try:
    df = pd.read_csv(data_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(data_path, sep=';', encoding='latin-1')

# Clean string columns and ensure they are strings before using .str accessor
for c in df.columns:
    df[c] = df[c].astype(str).str.strip()

# Convert 'value' to numeric safely
# The error was caused by calling .str on a column that might have been inferred as numeric or mixed.
# We ensure it's string first, then replace comma with dot for German decimal format.
if 'value' in df.columns:
    df['value_num'] = pd.to_numeric(df['value'].astype(str).str.replace(',', '.'), errors='coerce')
else:
    df['value_num'] = np.nan

# Filter for 'Raumzahl' (Room count)
# Based on the dataset structure, we look for the variable label indicating room count
mask = (df['2_variable_label'] == 'Raumzahl') if '2_variable_label' in df.columns else pd.Series([True] * len(df))
f = df[mask].copy()

# Exclude 'Insgesamt' (Total) as per the goal
if '2_variable_attribute_label' in f.columns:
    f = f[~f['2_variable_attribute_label'].str.contains('insgesamt', case=False, na=False)]

# Drop rows with missing values in the target column
f = f.dropna(subset=['value_num'])

# Group by room count label and sum the values
# We use both code and label to ensure unique grouping
group_cols = []
if '2_variable_attribute_code' in f.columns: group_cols.append('2_variable_attribute_code')
if '2_variable_attribute_label' in f.columns: group_cols.append('2_variable_attribute_label')

if group_cols:
    g = f.groupby(group_cols, dropna=False, as_index=False)['value_num'].sum(min_count=1)
else:
    # Fallback if columns are missing
    g = pd.DataFrame(columns=['2_variable_attribute_label', 'value_num'])

# Helper to extract number for sorting (e.g., "1 Raum" -> 1, "5 oder mehr Räume" -> 5)
def extract_room_number(x):
    if isinstance(x, str):
        m = re.search(r'(\d+)', x)
        if m:
            try:
                return int(m.group(1))
            except Exception:
                return 999
    return 999

if '2_variable_attribute_label' in g.columns and not g.empty:
    g['order'] = g['2_variable_attribute_label'].apply(extract_room_number)
    g = g.sort_values('order').reset_index(drop=True)
    # Create categorical type for correct plotting order
    g['2_variable_attribute_label'] = pd.Categorical(g['2_variable_attribute_label'], 
                                                     categories=g['2_variable_attribute_label'].unique(), 
                                                     ordered=True)

# Plotting
if not g.empty:
    plt.figure(figsize=(10, 6))
    ax = sns.barplot(data=g, x='2_variable_attribute_label', y='value_num', palette='viridis', hue='2_variable_attribute_label', legend=False)

    ax.set_title('Verteilung der fertiggestellten Wohnungen nach Raumzahl in Sünching', fontsize=13, pad=15)
    ax.set_xlabel('Raumzahl der Wohnungen', fontsize=11)
    ax.set_ylabel('Summe der Wohnungen (Anzahl)', fontsize=11)

    # Ensure y-axis uses integers
    ax.yaxis.set_major_locator(MaxNLocator(integer=True))

    # Add value labels on top of bars
    for p in ax.patches:
        h = p.get_height()
        if not np.isnan(h):
            ax.annotate(f'{int(h):d}', (p.get_x() + p.get_width() / 2, h), 
                        ha='center', va='bottom', fontsize=10, xytext=(0, 3), textcoords='offset points')

    plt.xticks(rotation=15, ha='right')
    plt.tight_layout()

    # Save output
    out_path = r'./src/resources/output/13979/1_verteilung_wohnungen_nach_raumzahl.png'
    os.makedirs(os.path.dirname(os.path.normpath(out_path)), exist_ok=True)
    plt.savefig(out_path, dpi=300, bbox_inches='tight')
    plt.close()
else:
    print("No data available for plotting after filtering.")

import os
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# Set visual style
sns.set_theme(style="whitegrid", context="talk")

input_path = r'./src/resources/data/study/13979/dataset.csv'
output_dir = r'./src/resources/output/13979'
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '2_mix_wohnungsgrößen_stacked_bars.png')

# The error 'UnicodeDecodeError: 'utf-8' codec can't decode byte 0xfc' 
# usually indicates 'latin-1' or 'iso-8859-1' encoding (0xfc is 'ü' in latin-1).
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Clean string columns
for col in df.columns:
    if df[col].dtype == object:
        df[col] = df[col].astype(str).str.strip()

# Convert numeric columns
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
if 'time' in df.columns:
    df['time'] = pd.to_numeric(df['time'], errors='coerce')

# Filter for the specific location and variable as per the goal
# We assume 'Sünching' and 'Raumzahl' based on the previous context, 
# but we make it robust to case sensitivity.
if '1_variable_attribute_label' in df.columns:
    df = df[df['1_variable_attribute_label'].str.lower() == 'sünching']
if '2_variable_label' in df.columns:
    df = df[df['2_variable_label'].str.lower().str.contains('raumzahl', na=False)]

# Exclude 'Insgesamt' (Total) to show the mix of sizes
if '2_variable_attribute_label' in df.columns:
    df = df[~df['2_variable_attribute_label'].str.lower().str.contains('insgesamt', na=False)]

# Drop rows with missing essential data
df = df.dropna(subset=['time', 'value', '2_variable_attribute_label'])

if df.empty:
    fig, ax = plt.subplots(figsize=(10, 6))
    ax.text(0.5, 0.5, 'Keine Daten für die Visualisierung verfügbar', ha='center', va='center')
    ax.set_axis_off()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
else:
    # Helper to sort room sizes logically (1 room, 2 rooms ... 7 or more)
    def parse_room_order(label):
        l = str(label).lower()
        if '7' in l or 'mehr' in l:
            return 7
        # Extract first digit found
        import re
        match = re.search(r'\d+', l)
        if match:
            return int(match.group())
        return 99

    # Create a sorted list of labels
    unique_labels = df['2_variable_attribute_label'].unique()
    sorted_labels = sorted(unique_labels, key=parse_room_order)

    # Pivot data for stacked bar chart
    pivot = df.pivot_table(
        index='time',
        columns='2_variable_attribute_label',
        values='value',
        aggfunc='sum',
        fill_value=0
    )
    
    # Reorder columns based on logical room count
    pivot = pivot.reindex(columns=[l for l in sorted_labels if l in pivot.columns])
    pivot.index = pivot.index.astype(int)
    pivot = pivot.sort_index()

    # Plotting
    fig, ax = plt.subplots(figsize=(14, 8))
    n_cats = len(pivot.columns)
    colors = sns.color_palette('viridis', n_colors=n_cats)
    
    pivot.plot(kind='bar', stacked=True, ax=ax, color=colors, width=0.8, edgecolor='white', linewidth=0.5)

    ax.set_xlabel('Jahr')
    ax.set_ylabel('Anzahl fertiggestellter Wohnungen')
    
    start_year = pivot.index.min()
    end_year = pivot.index.max()
    ax.set_title(f'Mix der Wohnungsgrößen in Sünching ({start_year}–{end_year})', pad=20)

    ax.legend(title='Raumzahl', loc='center left', bbox_to_anchor=(1.02, 0.5), frameon=False)
    ax.grid(axis='y', linestyle='--', alpha=0.3)
    ax.set_axisbelow(True)
    
    # Clean up x-axis labels
    plt.xticks(rotation=45)

    # Add total labels on top of bars
    totals = pivot.sum(axis=1)
    for i, total in enumerate(totals):
        if total > 0:
            ax.text(i, total + (totals.max() * 0.01), f'{int(total)}', ha='center', va='bottom', fontsize=10)

    plt.tight_layout()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
