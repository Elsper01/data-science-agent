library(dplyr)
library(ggplot2)
library(scales)

# Daten laden
data_path <- "./src/resources/data/study/13473/dataset.csv"
df <- read.csv(data_path, sep = ";", encoding = "UTF-8", stringsAsFactors = FALSE, na.strings = c("", "NA", "nan"))

# Output Verzeichnis erstellen
output_dir <- "./src/resources/output/13473"
if (!dir.exists(output_dir)) dir.create(output_dir, recursive = TRUE)

# Spaltennamen bereinigen (Punkte statt Leerzeichen oder Sonderzeichen durch read.csv)
# Der Fehler "Objekt '1_variable_attribute_label' nicht gefunden" deutet darauf hin, 
# dass R die Spaltennamen beim Einlesen verändert hat (z.B. X1_variable_attribute_label).
colnames(df) <- make.names(colnames(df))

# Identifikation der korrekten Spaltennamen nach der Bereinigung
# 1_variable_attribute_label wird meist zu X1_variable_attribute_label
col_region <- grep("variable_attribute_label", colnames(df), value = TRUE)[1]
col_gender <- grep("variable_attribute_label", colnames(df), value = TRUE)[2]

# Datenaufbereitung
# Wir filtern auf die Stadt Hof und die spezifischen Geschlechterkategorien
geschlecht_data <- df %>%
  filter(grepl("Hof", .data[[col_region]]),
         .data[[col_gender]] %in% c("männlich", "weiblich")) %>%
  mutate(value = as.numeric(gsub(",", ".", value))) %>% # Sicherstellen, dass value numerisch ist
  filter(!is.na(value))

# Falls keine Daten gefunden wurden (z.B. wegen Encoding-Problemen bei "männlich")
if(nrow(geschlecht_data) == 0) {
  # Alternativer Versuch ohne Umlaute falls nötig
  geschlecht_data <- df %>%
    filter(grepl("Hof", .data[[col_region]])) %>%
    filter(grepl("m.nnlich|weiblich", .data[[col_gender]])) %>%
    mutate(value = as.numeric(gsub(",", ".", value))) %>%
    filter(!is.na(value))
}

# Gesamtsumme für die Prozentberechnung
total_val <- sum(geschlecht_data$value)

# Labels für das Diagramm vorbereiten
geschlecht_data <- geschlecht_data %>%
  mutate(
    anteil = value / total_val,
    label_text = paste0(.data[[col_gender]], "\n", 
                        format(value, big.mark = ".", decimal.mark = ","), "\n",
                        "(", percent(anteil, accuracy = 0.1, decimal.mark = ","), ")")
  )

# Kreisdiagramm erstellen
p <- ggplot(geschlecht_data, aes(x = "", y = value, fill = .data[[col_gender]])) +
  geom_bar(stat = "identity", width = 1, color = "white") +
  coord_polar("y", start = 0) +
  geom_text(aes(label = label_text), 
            position = position_stack(vjust = 0.5), 
            size = 4.5, 
            fontface = "bold",
            color = "white") +
  scale_fill_manual(values = c("männlich" = "#2C7BB6", "weiblich" = "#D7191C"),
                    name = "Geschlecht") +
  labs(
    title = "Geschlechterverteilung der Bevölkerung in Hof",
    subtitle = paste0("Gesamtbevölkerung (m/w): ", format(total_val, big.mark = ".", decimal.mark = ","), " Personen"),
    caption = "Quelle: Amtliche Fortschreibung des Bevölkerungsstandes"
  ) +
  theme_void() +
  theme(
    plot.title = element_text(hjust = 0.5, size = 16, face = "bold", margin = margin(b = 10)),
    plot.subtitle = element_text(hjust = 0.5, size = 12, margin = margin(b = 20)),
    plot.caption = element_text(hjust = 0.5, size = 9, color = "grey40"),
    legend.position = "right"
  )

# Speichern der Visualisierung
output_file <- file.path(output_dir, "1_geschlechterverteilung_hof.png")
ggsave(filename = output_file, plot = p, width = 8, height = 7, dpi = 300, bg = "white")

cat("Visualisierung erfolgreich erstellt:", output_file, "\n")

# Pakete laden
library(readr)
library(dplyr)
library(gt)

# Verzeichnis erstellen
output_dir <- "./src/resources/output/13473"
dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)

# Daten einlesen
file_path <- "./src/resources/data/study/13473/dataset.csv"

if (file.exists(file_path)) {
  # Einlesen mit expliziter Kodierung und Fehlerbehandlung für Sonderzeichen
  df <- read_delim(file_path, 
                   delim = ";", 
                   locale = locale(encoding = "UTF-8"), 
                   show_col_types = FALSE)

  # Datenaufbereitung
  # Wir bereinigen die Spaltennamen und Inhalte von potenziellen UTF-8 Problemen
  df_table_data <- df %>%
    select(`1_variable_code`, `2_variable_attribute_label`, value) %>%
    mutate(
      # Sicherstellen, dass numerische Werte korrekt konvertiert werden
      value_num = as.numeric(gsub(",", ".", as.character(value))),
      # Bereinigung von Strings, um "invalid UTF-8" Fehler in gt/gsub zu vermeiden
      `1_variable_code` = iconv(`1_variable_code`, "UTF-8", "UTF-8", sub = ""),
      `2_variable_attribute_label` = iconv(`2_variable_attribute_label`, "UTF-8", "UTF-8", sub = "")
    ) %>%
    filter(!is.na(`1_variable_code`))

  # Erstellung der gt Tabelle
  # Um den "invalid UTF-8" Fehler in gtsave/webshot zu umgehen, 
  # nutzen wir eine robuste Formatierung.
  gt_plot <- df_table_data %>%
    select(`1_variable_code`, `2_variable_attribute_label`, value_num) %>%
    gt() %>%
    tab_header(
      title = "Regionale Aggregationsstufen und Datenwerte",
      subtitle = "Vergleich der Werte (Kreisfreie Stadt Hof)"
    ) %>%
    cols_label(
      `1_variable_code` = "Aggregations-Code",
      `2_variable_attribute_label` = "Merkmal / Geschlecht",
      value_num = "Wert"
    ) %>%
    fmt_number(columns = value_num, decimals = 0, sep_mark = ".", dec_mark = ",") %>%
    tab_options(
      table.width = pct(100),
      table.font.size = px(12)
    )

  # Speichern der Tabelle
  # Der Fehler "input string 1 is invalid UTF-8" tritt oft beim Rendering von HTML zu PNG auf.
  # Wir versuchen zuerst HTML (stabiler) und dann PNG.
  
  html_file <- file.path(output_dir, "2_tabelle_regional_aggregationsstufen.html")
  png_file <- file.path(output_dir, "2_tabelle_regional_aggregationsstufen.png")
  
  tryCatch({
    # HTML ist die sicherste Variante in R-Umgebungen mit Encoding-Problemen
    gtsave(gt_plot, filename = html_file)
    
    # Versuche PNG-Export (benötigt funktionierendes webshot2/chromote)
    # Wir nutzen expand = 5 um Abschneiden zu verhindern
    gtsave(gt_plot, filename = png_file, expand = 5)
    cat("Tabelle erfolgreich als PNG gespeichert.\n")
  }, error = function(e) {
    cat("Hinweis: PNG-Speicherung fehlgeschlagen, HTML wurde erstellt. Fehler: ", e$message, "\n")
  })

  # Validierungsausgabe
  cat("Dimensions:", nrow(df_table_data), "rows\n")
  print(head(df_table_data))

} else {
  cat("Datei nicht gefunden:", file_path, "\n")
}
