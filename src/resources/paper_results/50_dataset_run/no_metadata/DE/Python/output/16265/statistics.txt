import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style='whitegrid')

input_path = os.path.normpath('./src/resources/data/study\\16265\\dataset.csv')
output_dir = os.path.normpath('./src/resources/output\\16265')
os.makedirs(output_dir, exist_ok=True)

df = pd.read_csv(input_path, sep=',', encoding='utf-8')
if 'start_measure' in df.columns:
    df['start_measure'] = pd.to_datetime(df['start_measure'], errors='coerce')
if 'end_measure' in df.columns:
    df['end_measure'] = pd.to_datetime(df['end_measure'], errors='coerce')

value = pd.to_numeric(df['value'], errors='coerce') if 'value' in df.columns else pd.Series(dtype=float)
value = value.dropna()

missing_counts = df.isna().sum()
missing_share = df.isna().mean()
numeric_df = df.select_dtypes(include=[np.number])
corr_matrix = numeric_df.corr() if not numeric_df.empty else pd.DataFrame()

n = int(value.shape[0])
mean_v = float(value.mean()) if n > 0 else np.nan
std_v = float(value.std(ddof=1)) if n > 1 else np.nan
min_v = float(value.min()) if n > 0 else np.nan
q25 = float(value.quantile(0.25)) if n > 0 else np.nan
median_v = float(value.median()) if n > 0 else np.nan
q75 = float(value.quantile(0.75)) if n > 0 else np.nan
max_v = float(value.max()) if n > 0 else np.nan
iqr = q75 - q25 if n > 0 else np.nan

if n > 1 and iqr is not None and np.isfinite(iqr):
    if iqr == 0:
        bins = min(15, max(5, int(np.ceil(np.sqrt(n)))))
    else:
        bw = 2 * iqr / (n ** (1 / 3))
        if bw <= 0:
            bins = 'auto'
        else:
            span = max_v - min_v
            bins_est = int(np.ceil(span / bw)) if span > 0 else 5
            bins = int(np.clip(bins_est, 8, 40))
else:
    bins = 'auto'

fig, ax = plt.subplots(figsize=(10, 6), dpi=150)
color = sns.color_palette('Blues', 6)[4]
edge_c = 'white'
sns.histplot(value, bins=bins, kde=False, color=color, edgecolor=edge_c, linewidth=0.5, ax=ax)
ax.set_title('Histogramm der Gamma-Ortsdosisleistung (24h-Mittelwerte)\nMessstation Dorfprozelten (DEZ1778), Jan 2025 – Jan 2026', fontsize=14)
ax.set_xlabel('Gamma-Ortsdosisleistung [µSv/h]', fontsize=12)
ax.set_ylabel('Häufigkeit (Anzahl Tage)', fontsize=12)
if np.isfinite(mean_v):
    ax.axvline(mean_v, color='#d62728', linewidth=2, linestyle='-', label=f'Mittelwert: {mean_v:.4f} µSv/h')
if np.isfinite(median_v):
    ax.axvline(median_v, color='#2ca02c', linewidth=2, linestyle='--', label=f'Median: {median_v:.4f} µSv/h')
text_lines = []
if np.isfinite(std_v):
    text_lines.append(f'σ = {std_v:.4f} µSv/h')
if np.isfinite(min_v) and np.isfinite(max_v):
    text_lines.append(f'Min/Max = {min_v:.3f} / {max_v:.3f} µSv/h')
if np.isfinite(iqr):
    text_lines.append(f'IQR = {iqr:.4f} µSv/h')
text_lines.append(f'n = {n}')
textbox = '\n'.join(text_lines)
ax.text(0.98, 0.95, textbox, transform=ax.transAxes, ha='right', va='top', fontsize=10, bbox=dict(boxstyle='round', facecolor='white', alpha=0.8, edgecolor='#bbbbbb'))
if np.isfinite(min_v) and np.isfinite(max_v) and max_v > min_v:
    span = max_v - min_v
    ax.set_xlim(min_v - 0.05 * span, max_v + 0.05 * span)
ax.legend(frameon=True, loc='upper left')
plt.tight_layout()
output_path = os.path.normpath(os.path.join(output_dir, '1_histogramm_wertverteilung.png'))
fig.savefig(output_path, dpi=300, bbox_inches='tight')
plt.close(fig)

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import geopandas as gpd

# Setze Theme
sns.set_theme(style='whitegrid', context='talk')

# Pfade definieren
data_path = os.path.normpath(r'./src/resources/data/study/16265/dataset.csv')
output_dir = os.path.normpath(r'./src/resources/output/16265')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '2_karte_messstation.png')

# Daten laden
df = pd.read_csv(data_path, sep=',', encoding='utf-8')

# Datentypen konvertieren
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
for c in ['start_measure', 'end_measure']:
    if c in df.columns:
        df[c] = pd.to_datetime(df[c], errors='coerce')

# Aggregation der Daten pro Station
agg = df.groupby(['id', 'name', 'geom'], dropna=False, as_index=False).agg(
    mean_value=('value', 'mean'), 
    n=('value', 'size')
)

# Koordinaten aus 'geom' extrahieren (Format: POINT (LON LAT))
coords = agg['geom'].astype(str).str.extract(r'POINT\s*\(\s*([\-0-9\.]+)\s+([\-0-9\.]+)\s*\)')
coords.columns = ['c1', 'c2']
coords = coords.apply(pd.to_numeric, errors='coerce')
agg = pd.concat([agg, coords], axis=1)

# Heuristik zur Bestimmung von Lat/Lon (Deutschland liegt ca. 47-55N, 5-15E)
lat1, lon1 = agg['c2'], agg['c1']
lat2, lon2 = agg['c1'], agg['c2']
mask1 = (lat1.between(47, 56)) & (lon1.between(5, 16))
mask2 = (lat2.between(47, 56)) & (lon2.between(5, 16))

if mask1.sum() >= mask2.sum():
    agg['lat'], agg['lon'] = lat1, lon1
else:
    agg['lat'], agg['lon'] = lat2, lon2

agg = agg.dropna(subset=['lat', 'lon'])

if agg.empty:
    raise ValueError('Keine gültigen Geokoordinaten aus der Spalte "geom" extrahiert.')

# Fehlerbehebung: Geopandas Datasets Deprecation
# Wir nutzen eine alternative Quelle oder laden die Daten direkt via URL
world_url = "https://naciscdn.org/naturalearth/110m/cultural/ne_110m_admin_0_countries.zip"
try:
    world = gpd.read_file(world_url)
except Exception:
    # Fallback: Falls URL nicht erreichbar, erstelle leeres GeoDataFrame für den Plot-Rahmen
    from shapely.geometry import box
    world = gpd.GeoDataFrame({'geometry': [box(-180, -90, 180, 90)]}, crs="EPSG:4326")

# Informationen für die erste Station (Dorfprozelten) extrahieren
pt_lat = float(agg.iloc[0]['lat'])
pt_lon = float(agg.iloc[0]['lon'])
name = str(agg.iloc[0]['name'])
mean_val = agg.iloc[0]['mean_value']

# Zeitraum bestimmen
start_date = df['start_measure'].min()
end_date = df['end_measure'].max()
period_str = f" ({start_date.strftime('%d.%m.%Y')} – {end_date.strftime('%d.%m.%Y')})" if pd.notna(start_date) else ""

# Visualisierung
fig, ax = plt.subplots(figsize=(10, 8), dpi=150)

# Kartenausschnitt festlegen (Zoom auf Deutschland/Region)
margin = 1.5
minx, maxx = pt_lon - margin, pt_lon + margin
miny, maxy = pt_lat - margin, pt_lat + margin

# Hintergrundkarte plotten
world.plot(ax=ax, color='#f2f2f2', edgecolor='#adadad', linewidth=0.5)

# Messstation einzeichnen
ax.scatter([pt_lon], [pt_lat], s=200, c='#e74c3c', edgecolors='white', linewidths=1.5, zorder=5, label='Messstation')

# Label/Tooltip-Annotation
label_text = f"{name}\nØ-Wert: {mean_val:.3f} µSv/h"
ax.annotate(
    label_text, 
    xy=(pt_lon, pt_lat), 
    xytext=(15, 15), 
    textcoords='offset points',
    fontsize=12, 
    fontweight='bold',
    bbox=dict(boxstyle='round,pad=0.5', fc='white', ec='#2c3e50', alpha=0.85),
    arrowprops=dict(arrowstyle='->', connectionstyle='arc3,rad=0.1', color='#2c3e50')
)

# Achsen und Titel
ax.set_xlim(minx, maxx)
ax.set_ylim(miny, maxy)
ax.set_xlabel('Längengrad (°E)')
ax.set_ylabel('Breitengrad (°N)')
ax.set_title(f'Standort der Messstation und Gamma-ODL Mittelwert{period_str}', pad=20)

# Layout-Feinschliff
ax.set_aspect('equal')
plt.tight_layout()

# Speichern
plt.savefig(output_path, bbox_inches='tight')
plt.close()
