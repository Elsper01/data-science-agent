import os
import pandas as pd
import numpy as np

import matplotlib.pyplot as plt

input_path = r'./src/resources/data/study\32743\dataset.csv'
output_dir = r'./src/resources/output\32743'
os.makedirs(output_dir, exist_ok=True)

try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
except Exception as e:
    print(str(e))
    raise SystemExit(1)

for c in df.columns:
    if isinstance(c, str):
        df.rename(columns={c: c.strip()}, inplace=True)

if 'Unnamed__11' in df.columns and df['Unnamed__11'].isna().all():
    df = df.drop(columns=['Unnamed__11'])

if 'Datum' in df.columns:
    df['Datum'] = pd.to_datetime(df['Datum'], dayfirst=True, errors='coerce')

obj_cols = [c for c in df.columns if df[c].dtype == object]
percent_cols = []
for c in obj_cols:
    s = df[c].astype(str)
    if s.str.contains('%').any():
        percent_cols.append(c)
for c in percent_cols:
    df[c] = (
        df[c]
        .astype(str)
        .str.replace('%', '', regex=False)
        .str.replace(',', '.', regex=False)
        .str.strip()
    )
    df[c] = pd.to_numeric(df[c], errors='coerce')

for time_col in ['Zeit_im_Besuch', 'Bounce_Zeit']:
    if time_col in df.columns:
        td = pd.to_timedelta(df[time_col], errors='coerce')
        df[time_col + '_Sek'] = td.dt.total_seconds()

num_like_cols = [c for c in df.columns if c not in (['Datum', 'Zeit_im_Besuch', 'Bounce_Zeit'] + percent_cols)]
for c in num_like_cols:
    if df[c].dtype == object:
        df[c] = pd.to_numeric(df[c].astype(str).str.replace(',', '.', regex=False), errors='coerce')

df_daily = df[df['Datum'].notna()].copy() if 'Datum' in df.columns else df.copy()

missing_overall = df.isna().mean().to_frame('Anteil_NA')
missing_daily = df_daily.isna().mean().to_frame('Anteil_NA_taeglich')
missing_overall.to_csv(os.path.join(output_dir, '1_eda_missing_overall.csv'), encoding='utf-8')
missing_daily.to_csv(os.path.join(output_dir, '1_eda_missing_daily.csv'), encoding='utf-8')

num_cols_daily = df_daily.select_dtypes(include=[np.number]).columns.tolist()
describe_daily = df_daily[num_cols_daily].describe().T
describe_daily.to_csv(os.path.join(output_dir, '1_eda_describe_daily.csv'), encoding='utf-8')

corr_daily = df_daily[num_cols_daily].corr()
corr_daily.to_csv(os.path.join(output_dir, '1_eda_corr_daily.csv'), encoding='utf-8')

cols_needed = []
if 'Seiten_pro_Besuch' in df_daily.columns:
    cols_needed.append('Seiten_pro_Besuch')
if 'Bounce_Rate' in df_daily.columns:
    cols_needed.append('Bounce_Rate')

scatter_df = df_daily[cols_needed].dropna().copy()

r_value = np.nan
if set(['Seiten_pro_Besuch', 'Bounce_Rate']).issubset(scatter_df.columns) and len(scatter_df) > 1:
    x = scatter_df['Seiten_pro_Besuch'].to_numpy()
    y = scatter_df['Bounce_Rate'].to_numpy()
    if np.isfinite(x).all() and np.isfinite(y).all():
        r_matrix = np.corrcoef(x, y)

sns.set_theme(style='whitegrid', context='talk')

    ax.set_ylim(y_min - pad, y_max + pad)

    ax.set_xlim(x_min - pad_x, x_max + pad_x)
