import os
import sys
import re
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns


def ensure_dir(path):
    if not os.path.isdir(path):
        os.makedirs(path, exist_ok=True)


def load_dataset(path, sep=',', encoding='utf-8'):
    try:
        df = pd.read_csv(path, sep=sep, encoding=encoding)
        return df
    except Exception as e:
        print(str(e))
        sys.exit(1)


def coerce_numeric_from_string(series):
    s = series.astype(str)
    s = s.replace({'n.n.': np.nan, 'n.n': np.nan, 'NN': np.nan, 'NaN': np.nan, 'nan': np.nan}, regex=False)
    s = s.str.replace('\xa0', ' ', regex=False)
    s = s.str.strip()
    s = s.str.replace(',', '.', regex=False)
    s = s.str.replace(r'(?i)^<\s*([^\d\-\.]*)', '<', regex=True)
    s = s.str.replace(r'(?i)^>\s*([^\d\-\.]*)', '>', regex=True)
    num = pd.to_numeric(s.str.replace(r'[^\d\.-]', '', regex=True), errors='coerce')
    return num


def eda_and_save(df, out_dir):
    dtypes_df = pd.DataFrame({'Spalte': df.columns, 'DatenTyp': [str(t) for t in df.dtypes]})
    dtypes_df.to_csv(os.path.join(out_dir, 'eda_dtypes.csv'), index=False, encoding='utf-8')
    missing = df.isna().mean().sort_values(ascending=False).rename('Anteil_fehlend')
    missing.to_csv(os.path.join(out_dir, 'eda_missingness.csv'), header=True, encoding='utf-8')
    if 'Ergebnis' in df.columns:
        df['Ergebnis_num'] = coerce_numeric_from_string(df['Ergebnis'])
    numeric_cols = df.select_dtypes(include=[np.number]).columns.tolist()
    if len(numeric_cols) > 0:
        numeric_desc = df[numeric_cols].describe(percentiles=[0.05, 0.25, 0.5, 0.75, 0.95]).T
        numeric_desc.to_csv(os.path.join(out_dir, 'eda_numeric_descriptives.csv'), encoding='utf-8')
        if len(numeric_cols) >= 2:
            corr = df[numeric_cols].corr(method='pearson', min_periods=2)
            corr.to_csv(os.path.join(out_dir, 'eda_numeric_correlation.csv'), encoding='utf-8')
        else:
            pd.DataFrame().to_csv(os.path.join(out_dir, 'eda_numeric_correlation.csv'), encoding='utf-8')
    else:
        pd.DataFrame().to_csv(os.path.join(out_dir, 'eda_numeric_descriptives.csv'), encoding='utf-8')
        pd.DataFrame().to_csv(os.path.join(out_dir, 'eda_numeric_correlation.csv'), encoding='utf-8')


def plot_methods_frequency(df, out_png):
    sns.set_theme(context='talk', style='whitegrid')
    col = 'Methode'
    if col not in df.columns:
        print('Spalte "Methode" nicht gefunden.')
        sys.exit(1)
    methods = df[col].astype(str)
    methods = methods.str.replace('\xa0', ' ', regex=False).str.strip()
    methods = methods.replace({'': np.nan, 'None': np.nan, 'none': np.nan, 'NaN': np.nan, 'nan': np.nan})
    counts = methods.dropna().value_counts()
    if counts.empty:
        print('Keine gültigen Werte in der Spalte "Methode" vorhanden.')
        sys.exit(1)
    max_bars = 25
    truncated = False
    if len(counts) > max_bars:
        top_counts = counts.iloc[:max_bars].copy()
        other_sum = counts.iloc[max_bars:].sum()
        if other_sum > 0:
            top_counts.loc['Weitere Methoden'] = other_sum
        counts_plot = top_counts
        truncated = True
    else:
        counts_plot = counts
    counts_plot = counts_plot.sort_values(ascending=True)
    n_bars = len(counts_plot)
    height = max(5, int(0.45 * n_bars + 2))
    fig, ax = plt.subplots(figsize=(14, height))
    palette = sns.color_palette('viridis', n_colors=n_bars)
    ax.barh(y=np.arange(n_bars), width=counts_plot.values, color=palette)
    ax.set_yticks(np.arange(n_bars))
    ax.set_yticklabels(counts_plot.index)
    total_n = int(df.shape[0])
    title_main = 'Häufigkeit analytischer Methoden in der Probe'
    subtitle = f'Gezählt nach Feld "Methode"; n={total_n} Prüfmerkmale'
    if truncated:
        subtitle += f' | Top {max_bars} + Weitere'
    ax.set_title(title_main + '\n' + subtitle)
    ax.set_xlabel('Anzahl der Prüfmerkmale')
    ax.set_ylabel('Analytische Methode')
    max_val = counts_plot.values.max()
    for i, v in enumerate(counts_plot.values):
        ax.text(v + max_val * 0.01, i, str(int(v)), va='center', ha='left', fontsize=11)
    ax.set_xlim(0, max_val * 1.15)
    sns.despine(left=False, bottom=True)
    plt.tight_layout()
    plt.savefig(out_png, dpi=200, bbox_inches='tight')
    plt.close(fig)


def main():
    data_path = os.path.normpath('./src/resources/data/study\\137336/dataset.csv')
    out_dir = os.path.normpath('./src/resources/output\\137336')
    ensure_dir(out_dir)
    df = load_dataset(data_path, sep=',', encoding='utf-8')
    eda_and_save(df, out_dir)
    out_png = os.path.join(out_dir, '0_methoden_haeufigkeit.png')
    plot_methods_frequency(df, out_png)


if __name__ == '__main__':
    main()

if 'Ergebnis' not in df.columns or 'Pruefmerkmal' not in df.columns:

missing_per_col = df.isna().sum()

def parse_result(val):
    if pd.isna(val):
        return np.nan, 'Unter Nachweisgrenze'
    s = str(val).strip()
    s_low = s.lower().strip()
    s_compact = s_low.replace(' ', '').replace('.', '')
    if s_compact in {'nn', 'nichtnachweisbar'}:
        return np.nan, 'Unter Nachweisgrenze'
    if s_low.startswith('≤') or s_low.startswith('<'):
        num = s_low.lstrip('≤<').strip().replace(',', '.')
        try:
            _ = float(num)
        except Exception:
            _ = np.nan
        return np.nan, 'Unter Nachweisgrenze'
    if s_low.startswith('≥') or s_low.startswith('>'):
        num = s_low.lstrip('≥>').strip().replace(',', '.')
        try:
            val_float = float(num)
            return val_float, 'Nachweisbar'
        except Exception:
            return np.nan, 'Nachweisbar'
    num = s_low.replace(',', '.')
    try:
        val_float = float(num)
        return val_float, 'Nachweisbar'
    except Exception:
        return np.nan, 'Unter Nachweisgrenze'

parsed = df['Ergebnis'].apply(parse_result)
df['Ergebnis_Wert'] = parsed.apply(lambda x: x[0])
