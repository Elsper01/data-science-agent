# Helper functions for cleaning
na_like <- function(x) {
  x2 <- tolower(trimws(x))

}

conv_num <- function(x) {

# Define sites and identify columns
site_codes <- c("KBR", "KKB", "KKK")

get_imk_cols <- function(site, nms) {
  grep(paste0("^", site, "_\\d{4}_IMK$"), nms, value = TRUE)
}

get_precip_col <- function(site, nms) {
  opts <- c(paste0(site, "_NIED_MET"), paste0(site, "_NID1_MET"))
  cand <- opts[opts %in% nms]
  if (length(cand) == 0) {
    pat <- paste0("^", site, ".*NIED.*")
    cand <- grep(pat, nms, value = TRUE, ignore.case = TRUE)
  }
  if (length(cand) == 0) NA_character_ else cand[1]
}

# Calculate median Dose Rate per site
comp_median <- function(d, cols) {
  if (length(cols) == 0) return(rep(NA_real_, nrow(d)))
  # Use rowMeans or apply(..., 1, median)
  apply(as.matrix(d[, cols, drop = FALSE]), 1, function(x) {
    if(all(is.na(x))) return(NA_real_)
    median(x, na.rm = TRUE)
  })
}
