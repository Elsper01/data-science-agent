import os
import sys
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

try:
    input_path = os.path.normpath('./src/resources/data/study/51886/dataset.csv')
    output_dir = os.path.normpath('./src/resources/output/51886')
    os.makedirs(output_dir, exist_ok=True)

    # Der Fehler 'utf-8' codec can't decode byte 0xdf deutet auf ein Sonderzeichen (ß) hin, 
    # das oft in 'latin-1' oder 'iso-8859-1' kodiert ist. Wir versuchen 'iso-8859-1'.
    try:
        df = pd.read_csv(input_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1')

    # Datenbereinigung und Konvertierung
    for col in ['time', 'value']:
        if col in df.columns:
            df[col] = pd.to_numeric(df[col], errors='coerce')

    # Filterung auf 'Insgesamt'
    # Wir nutzen str.contains oder strip, um robust gegen Leerzeichen oder Encoding-Reste zu sein
    if '2_variable_attribute_label' in df.columns:
        df['2_variable_attribute_label'] = df['2_variable_attribute_label'].astype(str).str.strip()
        subset_mask = df['2_variable_attribute_label'].str.lower() == 'insgesamt'
        df_insgesamt = df.loc[subset_mask, ['time', 'value']].copy()
    else:
        print("Fehler: Spalte '2_variable_attribute_label' nicht gefunden.")
        sys.exit(1)

    df_insgesamt = df_insgesamt.dropna(subset=['time', 'value'])
    
    if df_insgesamt.empty:
        print('Fehler: Keine Daten für "Insgesamt" gefunden oder alle Werte fehlen.')
        sys.exit(1)

    # Aggregation nach Zeit (falls mehrere Einträge pro Jahr existieren)
    trend = df_insgesamt.groupby('time', as_index=False)['value'].sum()
    trend = trend.sort_values('time')

    # Deskriptive Statistiken speichern
    trend.describe().to_csv(os.path.join(output_dir, 'eda_insgesamt_deskriptiv.csv'), encoding='utf-8')

    # Visualisierung
    sns.set_theme(style='whitegrid', context='talk')
    plt.figure(figsize=(12, 7), dpi=150)
    
    ax = sns.lineplot(data=trend, x='time', y='value', color='#1f77b4', marker='o', linewidth=2.5, markersize=8)
    
    ax.set_title('Entwicklung der Baufertigstellungen in Oberschleißheim', pad=20, fontsize=16)
    ax.set_xlabel('Jahr', fontsize=12)
    ax.set_ylabel('Anzahl fertiggestellter Wohnungen', fontsize=12)

    # X-Achsen Ticks optimieren
    years = trend['time'].astype(int).unique()
    if len(years) > 15:
        ax.set_xticks(years[::2]) # Nur jedes zweite Jahr anzeigen bei vielen Daten
    else:
        ax.set_xticks(years)
    
    plt.xticks(rotation=45)

    # Highlights: Min und Max
    if not trend.empty:
        idx_min = trend['value'].idxmin()
        idx_max = trend['value'].idxmax()
        
        x_min, y_min = trend.loc[idx_min, 'time'], trend.loc[idx_min, 'value']
        x_max, y_max = trend.loc[idx_max, 'time'], trend.loc[idx_max, 'value']
        
        ax.scatter([x_min, x_max], [y_min, y_max], color='red', zorder=5)
        ax.annotate(f'Min: {int(y_min)}', xy=(x_min, y_min), xytext=(0, -15), 
                    textcoords='offset points', ha='center', color='red', fontsize=10)
        ax.annotate(f'Max: {int(y_max)}', xy=(x_max, y_max), xytext=(0, 10), 
                    textcoords='offset points', ha='center', color='red', fontsize=10)

    plt.tight_layout()
    output_path = os.path.join(output_dir, '0_baufertigstellungen_gesamttrend.png')
    plt.savefig(output_path, dpi=200)
    plt.close()

except Exception as e:
    print(f'Fehler bei der Ausführung: {e}')
    sys.exit(1)

import os
import sys
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Set visual style
sns.set_theme(style="whitegrid", context="talk")

# Define paths
output_dir = os.path.join('.', 'src', 'resources', 'output', '51886')
os.makedirs(output_dir, exist_ok=True)
data_path = os.path.join('.', 'src', 'resources', 'data', 'study', '51886', 'dataset.csv')

# Load data with robust encoding handling
# The error 'utf-8' codec can't decode byte 0xdf (ß) suggests 'latin-1' or 'iso-8859-1'
try:
    df = pd.read_csv(data_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(data_path, sep=';', encoding='latin-1')
    except Exception as e:
        sys.stderr.write(f"Error loading CSV: {e}")
        sys.exit(1)
except Exception as e:
    sys.stderr.write(f"Error loading CSV: {e}")
    sys.exit(1)

# Data Cleaning
for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].astype(str).str.strip()
        df.loc[df[c].str.lower() == 'nan', c] = np.nan

# Convert numeric columns
if 'time' in df.columns:
    df['time'] = pd.to_numeric(df['time'], errors='coerce')
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for relevant years (2014-2024)
start_year = 2014
end_year = 2024
if 'time' in df.columns:
    df = df[(df['time'] >= start_year) & (df['time'] <= end_year)]

# Filter for visualization goal
# We need '2_variable_attribute_label' and 'value'
if '2_variable_attribute_label' in df.columns and 'value' in df.columns:
    plot_df = df.copy()
    # Remove 'Insgesamt' (Total) to focus on specific room counts
    plot_df = plot_df[plot_df['2_variable_attribute_label'].notna()]
    plot_df = plot_df[~plot_df['2_variable_attribute_label'].str.contains('insgesamt', case=False, na=False)]
    
    # Aggregate: Sum of values per room category
    aggr = plot_df.groupby('2_variable_attribute_label', as_index=False)['value'].sum()
    aggr = aggr.sort_values('value', ascending=False)
    
    if not aggr.empty:
        # Visualization
        fig, ax = plt.subplots(figsize=(12, 8), dpi=150)
        
        # Create horizontal bar plot
        sns.barplot(
            data=aggr, 
            y='2_variable_attribute_label', 
            x='value', 
            palette='viridis', 
            ax=ax, 
            orient='h'
        )

        # Labels and Title
        ax.set_title(f'Baufertigstellungen nach Raumzahl in Oberschleißheim\n(Summe {start_year}–{end_year})', pad=20)
        ax.set_xlabel('Summe der fertiggestellten Wohnungen')
        ax.set_ylabel('Wohnungsgröße (Raumzahl)')

        # Add value labels to bars
        for p in ax.patches:
            width = p.get_width()
            if not np.isnan(width):
                ax.text(
                    width + (aggr['value'].max() * 0.01), 
                    p.get_y() + p.get_height() / 2, 
                    f'{int(width)}', 
                    va='center', 
                    fontsize=11
                )

        # Clean up spines
        sns.despine(left=True, bottom=True)
        plt.tight_layout()

        # Save plot
        plot_path = os.path.join(output_dir, '1_baufertigstellungen_raumzahl_summe.png')
        plt.savefig(plot_path, dpi=150, bbox_inches='tight')
        plt.close(fig)
        print(f"Visualization saved to {plot_path}")
    else:
        print("No data available for the specified filters.")
else:
    sys.stderr.write("Required columns '2_variable_attribute_label' or 'value' not found.")
    sys.exit(1)

import os
import sys
import re
import numpy as np
import pandas as pd
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
import seaborn as sns

def solve():
    try:
        output_dir = './src/resources/output/51886'
        os.makedirs(output_dir, exist_ok=True)
        plot_path = os.path.join(output_dir, '2_gestapelte_saeulen_raumkategorien.png')

        # The error 'utf-8' codec can't decode byte 0xdf indicates the file is likely encoded in ISO-8859-1 (Latin-1)
        # 0xdf is the German 'ß'.
        try:
            df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='utf-8')
        except UnicodeDecodeError:
            df = pd.read_csv('./src/resources/data/study/51886/dataset.csv', sep=';', encoding='iso-8859-1')

        def fix_text(s):
            if pd.isna(s):
                return s
            t = str(s)
            # Common encoding artifacts if file was read with wrong encoding or contains mixed symbols
            rep = {
                'Oberschleiﬂheim': 'Oberschleißheim',
                'R‰ume': 'Räume',
                'r‰ume': 'räume',
                '‰': 'ä',
                'Ã¶': 'ö',
                'Ã¼': 'ü',
                'Ã¤': 'ä',
                'ÃŸ': 'ß'
            }
            for k, v in rep.items():
                t = t.replace(k, v)
            return t.strip()

        for col in df.columns:
            if df[col].dtype == object:
                df[col] = df[col].apply(fix_text)

        # Convert numeric columns
        if 'time' in df.columns:
            df['time'] = pd.to_numeric(df['time'], errors='coerce')
        if 'value' in df.columns:
            df['value'] = pd.to_numeric(df['value'], errors='coerce')

        # Filter for Room categories based on the variable label
        if '2_variable_label' in df.columns:
            df = df[df['2_variable_label'].astype(str).str.contains('Raum', case=False, na=False)]

        if '2_variable_attribute_label' not in df.columns:
            raise ValueError('Spalte 2_variable_attribute_label fehlt im Datensatz.')

        # Filter out 'Insgesamt' (Total) to show the distribution
        mask_valid = df['2_variable_attribute_label'].str.strip().str.lower() != 'insgesamt'
        df_cat = df[mask_valid].copy()

        df_cat = df_cat.dropna(subset=['time', 'value'])
        
        # Helper to sort room categories (e.g., "1 Raum", "2 Räume" ...)
        def room_order_key(s):
            if pd.isna(s): return 999
            t = str(s).strip()
            m = re.match(r'^(\d+)', t)
            if m:
                return int(m.group(1))
            if 'oder mehr' in t.lower():
                m2 = re.search(r'(\d+)', t)
                if m2: return int(m2.group(1))
            return 999

        # Aggregate data
        grp = df_cat.groupby(['time', '2_variable_attribute_label'], as_index=False)['value'].sum()
        pivot = grp.pivot(index='time', columns='2_variable_attribute_label', values='value').fillna(0)
        pivot = pivot.sort_index()

        if pivot.empty:
            raise ValueError('Keine Daten für die Visualisierung vorhanden.')

        # Sort columns by room count
        ordered_cols = sorted(list(pivot.columns), key=room_order_key)
        pivot = pivot[ordered_cols]

        # Plotting
        sns.set_theme(style='whitegrid')
        fig, ax = plt.subplots(figsize=(12, 7), dpi=150)

        years = pivot.index.values
        bottom = np.zeros(len(years))
        
        n_cat = len(pivot.columns)
        colors = sns.color_palette('viridis', n_colors=n_cat)

        for i, col in enumerate(pivot.columns):
            values = pivot[col].values
            ax.bar(years, values, bottom=bottom, label=col, color=colors[i], edgecolor='white', width=0.7)
            bottom += values

        ax.set_title('Verteilung der fertiggestellten Wohnungen nach Raumkategorien (Oberschleißheim)', fontsize=14, pad=20)
        ax.set_xlabel('Jahr', fontsize=12)
        ax.set_ylabel('Anzahl fertiggestellter Wohnungen', fontsize=12)
        
        # Ensure X-axis shows all years as integers
        ax.set_xticks(years)
        ax.set_xticklabels([str(int(y)) for y in years])

        ax.legend(title='Raumkategorie', bbox_to_anchor=(1.02, 1), loc='upper left')
        
        plt.tight_layout()
        plt.savefig(plot_path, dpi=300, bbox_inches='tight')
        plt.close(fig)

    except Exception as e:
        print(f"Error: {str(e)}")
        sys.exit(1)

if __name__ == "__main__":
    solve()
