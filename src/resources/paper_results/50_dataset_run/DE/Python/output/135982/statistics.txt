def _to_float_de(val):
    if pd.isna(val):
        return np.nan
    s = str(val).strip()
    if s == "":
        return np.nan
    sl = s.lower().replace(" ", "")
    if sl in {"n.n.", "n.n", "nn", "na", "n/a", "k.a.", "k.a", "ka", "<ld>", "n.v.", "n.v"}:
        return np.nan
    s = s.replace(" ", "").replace("\xa0", "")
    s = s.replace(",", ".")
    try:
        return float(s)
    except Exception:
        return np.nan

def _format_de_scientific(x):
    if pd.isna(x):
        return ""
    s = f"{x:.2e}"
    s = s.replace("e", "E")
    s = s.replace(".", ",")
    return s + " Bq/l"

try:
    df = pd.read_csv(input_path, sep=",", encoding="utf-8")
