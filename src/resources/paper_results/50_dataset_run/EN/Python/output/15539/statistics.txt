import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os
import matplotlib.ticker as mticker

# Set display options
pd.set_option('display.max_columns', None)

# Paths
input_path = './src/resources/data/study/15539/dataset.csv'
output_dir = './src/resources/output/15539'
os.makedirs(output_dir, exist_ok=True)

# Load data with robust encoding handling
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(input_path, sep=';', encoding='iso-8859-1')
    except Exception:
        df = pd.read_csv(input_path, sep=';', encoding='cp1252')

# Data Cleaning
for c in ['time', 'value']:
    if c in df.columns:
        if df[c].dtype == object:
            df[c] = df[c].astype(str).str.replace(',', '.')
        df[c] = pd.to_numeric(df[c], errors='coerce')

# EDA: Missing values
eda_missing = (df.isna().mean().to_frame('fehlenden_anteil')).reset_index().rename(columns={'index': 'spalte'})
eda_missing.to_csv(os.path.join(output_dir, '0_eda_fehlende_werte.csv'), index=False, encoding='utf-8')

# Filter for 'Insgesamt' (Total)
mask_total = df['2_variable_attribute_label'].astype(str).str.strip().str.lower() == 'insgesamt'
df_total = df.loc[mask_total].copy()

# Fallback if 'Insgesamt' is not found
if df_total.empty:
    df_total = df.copy()

# Map labels for better readability
mapping = {
    'Gästeankünfte in Beherbergungsbetrieben': 'Ankünfte',
    'Gästeübernachtungen in Beherbergungsbetrieben': 'Übernachtungen'
}
df_total['serie'] = df_total['value_variable_label'].map(mapping).fillna(df_total['value_variable_label'])

# EDA: Statistics
# Fix: The previous error was caused by reset_index() creating an extra column or 
# the aggregation returning a different number of columns than expected.
# We'll define the columns explicitly after aggregation.
eda_stats = df_total.groupby('serie')['value'].agg(['count', 'mean', 'std', 'min', 'max']).reset_index()
# Ensure the column names match the number of columns in eda_stats (which is 6: serie + 5 stats)
eda_stats.columns = ['serie', 'anzahl', 'mittelwert', 'standardabweichung', 'minimum', 'maximum']
eda_stats.to_csv(os.path.join(output_dir, '0_eda_statistiken.csv'), index=False, encoding='utf-8')

# Visualization
sns.set_theme(style='whitegrid', context='talk')
plt.figure(figsize=(12, 7), dpi=150)

# Prepare plotting data
plot_df = df_total.dropna(subset=['time', 'value']).copy()
plot_df = plot_df.groupby(['time', 'serie'], as_index=False)['value'].sum().sort_values('time')

# Create the line chart
unique_series = plot_df['serie'].unique()

ax = plt.gca()

# Add legend
ax.legend(title='Kennzahl', loc='best', frameon=True)

if plot_df['time'].min() <= 2020 <= plot_df['time'].max():

import os
import sys
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import PercentFormatter

# Set visual style
sns.set_theme(style="whitegrid", context="talk")

input_path = r'./src/resources/data/study/15539/dataset.csv'
output_dir = r'./src/resources/output/15539'
os.makedirs(output_dir, exist_ok=True)

# The error 'utf-8' codec can't decode byte 0xe4 (which is 'ä') suggests the file 
# is likely encoded in ISO-8859-1 (Latin-1) or CP1252, common for German datasets.
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    try:
        df = pd.read_csv(input_path, sep=';', encoding='latin-1')
    except Exception as e:
        print(f"Failed to load data: {e}")
        sys.exit(1)
except Exception as e:
    print(f"An unexpected error occurred: {e}")
    sys.exit(1)

# Clean string columns
for col in df.select_dtypes(include=['object', 'string']).columns:
    df[col] = df[col].astype(str).str.strip()

# Convert numeric columns
if 'time' in df.columns:
    df['time'] = pd.to_numeric(df['time'], errors='coerce').astype('Int64')
if 'value' in df.columns:
    df['value'] = pd.to_numeric(df['value'], errors='coerce')

# Filter for the specific metric
label_filter = 'Gästeübernachtungen in Beherbergungsbetrieben'
df_ue = df[df['value_variable_label'] == label_filter].copy()

if df_ue.empty:
    print('No data found for the specified label.')
    sys.exit(1)

# Filter for Domestic (Inland) and International (Ausland)
# We exclude 'Insgesamt' to show the proportion of the components
mask_attr = df_ue['2_variable_attribute_label'].isin(['Inland', 'Ausland'])
df_ue = df_ue[mask_attr].copy()

if df_ue.empty:
    print('No data found for Inland/Ausland categories.')
    sys.exit(1)

# Pivot data for plotting
pivot_abs = (
    df_ue.pivot_table(
        index='time',
        columns='2_variable_attribute_label',
        values='value',
        aggfunc='sum'
    )
    .sort_index()
    .fillna(0)
)

# Ensure columns are in a logical order if they exist
ordered_cols = [c for c in ['Inland', 'Ausland'] if c in pivot_abs.columns]
pivot_abs = pivot_abs[ordered_cols]

# Calculate percentages for the stacked bar chart
totals = pivot_abs.sum(axis=1).replace(0, np.nan)
pivot_share = (pivot_abs.div(totals, axis=0).fillna(0) * 100)

# Save processed data for reference
share_tbl = pivot_abs.copy()
share_tbl['Total'] = totals.fillna(0)
for c in ordered_cols:
    share_tbl[f'{c}_Share_%'] = pivot_share[c].round(2) if c in pivot_share.columns else 0
share_tbl.to_csv(os.path.join(output_dir, 'overnight_stays_shares.csv'), encoding='utf-8')

# Visualization
fig, ax = plt.subplots(figsize=(12, 7))
colors = ['#4C72B0', '#C44E52'] # Distinct colors for Inland/Ausland

pivot_share.plot(
    kind='bar', 
    stacked=True, 
    ax=ax, 
    color=colors, 
    width=0.8, 
    edgecolor='white'
)

# Formatting
ax.set_ylim(0, 100)
ax.yaxis.set_major_formatter(PercentFormatter(xmax=100))
ax.set_title('Proportion of Domestic vs. International Overnight Stays\nBad Berneck i.Fichtelgebirge', pad=20)
ax.set_xlabel('Year')
ax.set_ylabel('Share of Total Overnight Stays (%)')
ax.legend(title='Guest Origin', bbox_to_anchor=(1.05, 1), loc='upper left')

plt.xticks(rotation=45)
plt.tight_layout()

# Save plot
plot_path = os.path.join(output_dir, 'domestic_vs_international_shares.png')
plt.savefig(plot_path, dpi=300, bbox_inches='tight')
plt.close()

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import os

# The error 'utf-8' codec can't decode byte 0xe4 suggests the file is likely encoded in ISO-8859-1 (Latin-1)
# which is common for German datasets containing umlauts (ä, ö, ü).

try:
    input_path = r'./src/resources/data/study/15539/dataset.csv'
    output_dir = r'./src/resources/output/15539'
    os.makedirs(output_dir, exist_ok=True)

    # Attempt reading with latin-1 encoding to handle German characters
    try:
        df = pd.read_csv(input_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(input_path, sep=';', encoding='latin-1')

    # Data Cleaning
    for c in df.columns:
        if df[c].dtype == object:
            df[c] = df[c].astype(str).str.strip()

    df['time'] = pd.to_numeric(df['time'], errors='coerce')
    df['value'] = pd.to_numeric(df['value'], errors='coerce')
    
    # Filter for 'Insgesamt' (Total)
    # We use a case-insensitive check and handle potential encoding artifacts
    df['attr_label_clean'] = df['2_variable_attribute_label'].str.lower()
    dff = df[df['attr_label_clean'].str.contains('insgesamt', na=False)].copy()
    
    # Map the variable labels to shorter versions for the legend
    # Handling both possible encodings of umlauts in the data
    valid_labels_map = {
        'Gästeankünfte in Beherbergungsbetrieben': 'Ankünfte',
        'Gaesteankuenfte in Beherbergungsbetrieben': 'Ankünfte',
        'Gästeübernachtungen in Beherbergungsbetrieben': 'Übernachtungen',
        'Gaesteuebernachtungen in Beherbergungsbetrieben': 'Übernachtungen'
    }
    
    dff['var_kurz'] = dff['value_variable_label'].map(lambda x: valid_labels_map.get(x, x))
    dff = dff[dff['var_kurz'].isin(['Ankünfte', 'Übernachtungen'])]
    dff = dff.dropna(subset=['time', 'value'])

    # Pivot for plotting
    pivot = dff.pivot_table(index='time', columns='var_kurz', values='value', aggfunc='sum').sort_index()

    if pivot.empty or 'Ankünfte' not in pivot.columns or 'Übernachtungen' not in pivot.columns:
        raise ValueError('Required data columns (Ankünfte/Übernachtungen) missing after filtering.')

    # Visualization
    sns.set_theme(style='whitegrid')
    plt.rcParams['font.family'] = 'DejaVu Sans'
    
    years = pivot.index.astype(int)
    x = np.arange(len(years))
    width = 0.35

    fig, ax = plt.subplots(figsize=(12, 7))
    
    rects1 = ax.bar(x - width/2, pivot['Ankünfte'], width, label='Gästeankünfte', color='#66c2a5')
    rects2 = ax.bar(x + width/2, pivot['Übernachtungen'], width, label='Gästeübernachtungen', color='#fc8d62')

    ax.set_xlabel('Jahr')
    ax.set_ylabel('Anzahl')
    ax.set_title('Vergleich: Gästeankünfte vs. Gästeübernachtungen (Insgesamt)\nBad Berneck i. Fichtelgebirge', fontsize=14, pad=20)
    ax.set_xticks(x)
    ax.set_xticklabels(years)
    ax.legend()

    # Add a caption with insights
    avg_stay = (pivot['Übernachtungen'] / pivot['Ankünfte']).mean()
    caption = f"Durchschnittliche Aufenthaltsdauer über den Zeitraum: {avg_stay:.2f} Nächte."
    plt.figtext(0.5, 0.01, caption, wrap=True, horizontalalignment='center', fontsize=10, style='italic')

    plt.tight_layout(rect=[0, 0.05, 1, 1])
    
    output_path = os.path.join(output_dir, '2_gaesteankuenfte_uebernachtungen_gruppiert.png')
    plt.savefig(output_path, dpi=200)
    plt.close(fig)

except Exception as e:
    print(f"An error occurred: {e}")
