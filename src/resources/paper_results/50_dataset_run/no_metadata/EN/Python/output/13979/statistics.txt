def load_csv_with_encodings(path, sep=';'):
    """Try different encodings to handle potential UnicodeDecodeErrors."""
    encodings = ['utf-8', 'latin1', 'iso-8859-1', 'cp1252']
    for enc in encodings:
        try:
            return pd.read_csv(path, sep=sep, encoding=enc)
        except (UnicodeDecodeError, pd.errors.ParserError):
            continue
    # If all fail, try with engine='python' and errors='replace'
    return pd.read_csv(path, sep=sep, encoding='utf-8', errors='replace')

try:
    sns.set_theme(style="whitegrid", context="talk")
    
    input_path = './src/resources/data/study/13979/dataset.csv'
    df = load_csv_with_encodings(input_path, sep=';')
    
    # Data Cleaning
    if 'time' in df.columns:
        df['time'] = pd.to_numeric(df['time'], errors='coerce')
    if 'value' in df.columns:
        # Handle potential string numbers with commas or dots
        if df['value'].dtype == object:
            df['value'] = df['value'].str.replace(',', '.')
        df['value'] = pd.to_numeric(df['value'], errors='coerce')
    
    # Filter for 'Insgesamt' (Total)
    label_col = '2_variable_attribute_label'
    if label_col in df.columns:
        # Use case-insensitive and stripped comparison to be robust
        df_filtered = df[df[label_col].astype(str).str.strip().str.lower() == 'insgesamt'].copy()
    else:
        # Fallback if column is missing
        df_filtered = df.copy()
        
    # Filter for the requested time range 2014-2024
    df_filtered = df_filtered[(df_filtered['time'] >= 2014) & (df_filtered['time'] <= 2024)]
    
    # Aggregate and sort
    plot_data = df_filtered.groupby('time', as_index=False)['value'].sum().sort_values('time')
    
    # Visualization
    plt.figure(figsize=(12, 7))
    
    if not plot_data.empty:
        line_color = '#2c7fb8'
        sns.lineplot(data=plot_data, x='time', y='value', marker='o',

        plt.xlabel('Jahr', fontsize=14, fontweight='bold')
        plt.ylabel('Anzahl Baufertigstellungen', fontsize=14, fontweight='bold')
        
        # Ensure x-axis shows all years as integers

import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Set visual style
sns.set_theme(style='whitegrid', context='talk')

# File paths
input_path = os.path.join('.', 'src', 'resources', 'data', 'study', '13979', 'dataset.csv')
output_dir = os.path.join('.', 'src', 'resources', 'output', '13979')
os.makedirs(output_dir, exist_ok=True)
output_path = os.path.join(output_dir, '1_verteilung_wohnungsraumzahl.png')

# Handle encoding error: 'utf-8' failed with 0xfc (ü), which suggests 'latin-1' or 'iso-8859-1'
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin-1')

# Clean string columns
for c in df.select_dtypes(include=['object', 'string']).columns:
    df[c] = df[c].astype(str).str.strip()

# Filter for Sünching and Room Count variable
# Note: 'Sünching' might be encoded differently depending on the file, 
# but 'latin-1' usually handles the 'ü' correctly.
subset = df[(df['1_variable_attribute_label'] == 'Sünching') & (df['2_variable_label'] == 'Raumzahl')]

# Convert value to numeric
pre_len = len(subset)
subset['value_num'] = pd.to_numeric(subset['value'], errors='coerce')
value_missing_pct = 0.0
if pre_len > 0:
    value_missing_pct = (subset['value_num'].isna().sum() / pre_len) * 100

# Drop missing values and exclude 'Insgesamt' (Total)
subset = subset[subset['value_num'].notna()]
subset = subset[~subset['2_variable_attribute_label'].str.contains('insgesamt', case=False, na=False)]

# Get time range for title
subset['time_num'] = pd.to_numeric(subset['time'], errors='coerce')
year_min = int(subset['time_num'].min()) if subset['time_num'].notna().any() else None
year_max = int(subset['time_num'].max()) if subset['time_num'].notna().any() else None

# Aggregate values by room count label
agg = subset.groupby(['2_variable_attribute_code', '2_variable_attribute_label'], dropna=False, as_index=False)['value_num'].sum()

# Custom sorting for room counts (RAUM-1, RAUM-2, ..., RAUM-7UM)
def _sort_key(code):
    s = str(code)
    if 'RAUM-' in s:
        r = s.split('RAUM-')[-1]
        if 'UM' in r:
            return 99  # Put "7 or more" at the end
        try:
            return int(r)
        except ValueError:
            return 100
    return 101

agg['sort_key'] = agg['2_variable_attribute_code'].apply(_sort_key)
agg = agg.sort_values('sort_key').reset_index(drop=True)

labels = agg['2_variable_attribute_label'].tolist()
values = agg['value_num'].astype(float).tolist()
total_units = sum(values) if values else 0

# Plotting
plt.figure(figsize=(12, 7))
if values:
    ax = sns.barplot(x=labels, y=values, palette='viridis')
    
    # Add labels on top of bars
    ymax = max(values) if values else 1
    y_offset = ymax * 0.02
    for i, p in enumerate(ax.patches):
        h = p.get_height()
        pct = (h / total_units * 100) if total_units > 0 else 0
        ax.text(p.get_x() + p.get_width()/2., h + y_offset, f'{int(h)}\n({pct:.1f}%)', 
                ha="center", va="bottom", fontsize=10)
    
    ax.set_ylim(0, ymax * 1.15)
else:
    plt.text(0.5, 0.5, "Keine Daten verfügbar", ha='center')

# Labels and Title
ax.set_xlabel('Wohnungsgröße (Anzahl Räume)')
ax.set_ylabel('Summe fertiggestellter Wohnungen')
title = 'Verteilung der fertiggestellten Wohnungen nach Raumzahl in Sünching'
if year_min and year_max:
    title += f' ({year_min}–{year_max})'
ax.set_title(title, pad=20)

# Footnote
note = f"Gesamt: {int(total_units)} Wohnungen | Datenquelle: Statistik der Baufertigstellungen"
plt.gcf().text(0.02, 0.01, note, fontsize=10, alpha=0.7)

plt.tight_layout()
plt.savefig(output_path, dpi=200, bbox_inches='tight')
plt.close()

import os
import re
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

def solve():
    # Define paths
    data_path = r"./src/resources/data/study/13979/dataset.csv"
    output_dir = os.path.normpath(r"./src/resources/output/13979")
    os.makedirs(output_dir, exist_ok=True)

    # The error 'utf-8' codec can't decode byte 0xfc suggests the file is likely encoded in 'latin-1' or 'iso-8859-1'
    # (0xfc is 'ü' in latin-1, common in German datasets like Sünching).
    try:
        df = pd.read_csv(data_path, sep=';', encoding='utf-8')
    except UnicodeDecodeError:
        df = pd.read_csv(data_path, sep=';', encoding='latin-1')

    # Clean string columns
    for col in df.select_dtypes(include=['object', 'string']).columns:
        df[col] = df[col].astype(str).str.strip()
        df[col] = df[col].replace({'': np.nan, 'nan': np.nan})

    # Convert numeric columns
    df['time'] = pd.to_numeric(df.get('time'), errors='coerce')
    df['value'] = pd.to_numeric(df.get('value'), errors='coerce')

    # Filter for apartment size variables (Raumzahl) and exclude 'Insgesamt'
    # Based on the goal: '2_variable_attribute_label' as category, excluding 'Insgesamt'
    df_filtered = df.copy()
    
    # Filter for rows related to room counts if the column exists
    if '2_variable_label' in df_filtered.columns:
        df_filtered = df_filtered[df_filtered['2_variable_label'].fillna('').str.contains('Raumzahl|Wohnungen', case=False, na=False)]

    # Exclude 'Insgesamt' (Total)
    if '2_variable_attribute_label' in df_filtered.columns:
        df_filtered = df_filtered[~df_filtered['2_variable_attribute_label'].str.lower().str.contains('insgesamt', na=False)]
    
    # Drop rows with missing essential data
    df_filtered = df_filtered.dropna(subset=['time', 'value', '2_variable_attribute_label'])

    # Helper to sort room labels numerically (e.g., "1 Raum", "2 Räume")
    def extract_room_order(label):
        m = re.findall(r"\d+", str(label))
        return int(m[0]) if m else 999

    df_filtered['room_order'] = df_filtered['2_variable_attribute_label'].apply(extract_room_order)

    # Aggregate data
    grouped = (
        df_filtered
        .groupby(['time', '2_variable_attribute_label', 'room_order'], as_index=False)['value']
        .sum()
    )

    # Pivot for stacked bar chart
    pivot = grouped.pivot_table(
        index='time', 
        columns='2_variable_attribute_label', 
        values='value', 
        aggfunc='sum', 
        fill_value=0
    )

    # Sort columns based on the room order
    sorted_cols = grouped[['2_variable_attribute_label', 'room_order']].drop_duplicates().sort_values('room_order')['2_variable_attribute_label'].tolist()
    pivot = pivot.reindex(columns=[c for c in sorted_cols if c in pivot.columns])

    # Plotting
    sns.set_theme(style="whitegrid", context="talk")
    fig, ax = plt.subplots(figsize=(12, 7))
    
    colors = sns.color_palette("viridis", n_colors=len(pivot.columns))
    pivot.plot(kind='bar', stacked=True, ax=ax, color=colors, width=0.8, edgecolor='white', linewidth=0.5)

    # Formatting
    start_year = int(pivot.index.min())
    end_year = int(pivot.index.max())
    ax.set_title(f"Mix der Wohnungsgrößen bei Baufertigstellungen ({start_year}–{end_year})", pad=20)
    ax.set_xlabel("Berichtsjahr")
    ax.set_ylabel("Anzahl fertiggestellter Wohnungen")
    
    ax.legend(title="Wohnungsgröße", bbox_to_anchor=(1.02, 1), loc='upper left')
    
    # Clean up x-axis labels
    ax.set_xticklabels([str(int(x)) for x in pivot.index], rotation=45)
    ax.yaxis.set_major_locator(plt.MaxNLocator(integer=True))

    plt.tight_layout()
    
    # Save
    output_path = os.path.join(output_dir, "apartment_size_mix_year_over_year.png")
    plt.savefig(output_path, dpi=200, bbox_inches='tight')
    plt.close(fig)

if __name__ == "__main__":
    solve()
