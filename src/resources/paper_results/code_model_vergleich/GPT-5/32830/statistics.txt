import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

input_path = r'./src/resources/data/study\32830\dataset.csv'
output_path = r'./src/resources/output\32830/0_traffic_besuche_besucher.png'

sns.set_theme(style='whitegrid', context='talk')

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

df = df.replace('-', np.nan)
if 'Unnamed__14' in df.columns:
    df = df.drop(columns=['Unnamed__14'])
if 'Datum' in df.columns:
    df['Datum'] = df['Datum'].replace({'nan': np.nan, 'NaN': np.nan})
    df['Datum'] = pd.to_datetime(df['Datum'], dayfirst=True, errors='coerce')

for col in ['Besuche', 'Besucher', 'Seitenaufrufe', 'Seiten_pro_Besuch']:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='coerce')

for tcol in ['Zeit_im_Besuch', 'Bounce_Zeit']:
    if tcol in df.columns:
        df[tcol + '_Sekunden'] = pd.to_timedelta(df[tcol], errors='coerce').dt.total_seconds()

missing_ratio = df.isna().mean()
num_df = df.select_dtypes(include=[np.number])
central_stats = num_df.agg(['count', 'mean', 'std', 'min', 'median', 'max']).T
cor_matrix = num_df.corr(numeric_only=True)

df_plot = df.copy()
df_plot = df_plot[df_plot['Datum'].notna()]
df_plot = df_plot.sort_values('Datum')
df_plot = df_plot[(df_plot['Besuche'].notna()) & (df_plot['Besucher'].notna())]

plt.figure(figsize=(14, 7), dpi=150)
ax = plt.gca()

sns.lineplot(data=df_plot, x='Datum', y='Besuche', marker='o', linewidth=2.2, color='#1f77b4', label='Besuche', ax=ax)
sns.lineplot(data=df_plot, x='Datum', y='Besucher', marker='o', linewidth=2.2, color='#ff7f0e', label='Besucher', ax=ax)

weekend_dates = df_plot.loc[df_plot['Datum'].dt.weekday >= 5, 'Datum']
for d in weekend_dates:
    ax.axvspan(d, d + pd.Timedelta(days=1), color='#999999', alpha=0.08, lw=0)

ax.set_title('Besuche und Besucher im September 2024 – kreis-kleve.de (Monatssumme ausgeschlossen)', fontsize=16)
ax.set_xlabel('Datum')
ax.set_ylabel('Anzahl')
ax.legend(title='Metrik')
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.savefig(output_path, dpi=150, bbox_inches='tight')
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid", context="talk")

fp = r"./src/resources/data/study\32830\dataset.csv"
df = pd.read_csv(fp, sep=",", encoding="utf-8")

if "Unnamed__14" in df.columns:
    if df["Unnamed__14"].isna().all():
        df = df.drop(columns=["Unnamed__14"])    

def _to_percent_numeric(s):
    return pd.to_numeric(pd.Series(s, dtype=str).str.replace("%", "", regex=False).str.replace(",", ".", regex=False).str.strip(), errors="coerce")

if "Bounce_Rate" in df.columns:
    df["Bounce_Rate_pct"] = _to_percent_numeric(df["Bounce_Rate"])    

if "Seiten_pro_Besuch" in df.columns:
    df["Seiten_pro_Besuch"] = pd.to_numeric(df["Seiten_pro_Besuch"], errors="coerce")

if "Datum" in df.columns:
    df["Datum_dt"] = pd.to_datetime(df["Datum"], format="%d.%m.%Y", errors="coerce")

eda_missing = df.isna().mean()

num_df = df.select_dtypes(include=[np.number])
eda_corr = num_df.corr(numeric_only=True)

clean = df.dropna(subset=["Datum_dt"]).copy()
clean = clean.sort_values("Datum_dt")
clean = clean.dropna(subset=["Seiten_pro_Besuch", "Bounce_Rate_pct"]) 

seiten_mean = float(clean["Seiten_pro_Besuch"].mean()) if not clean.empty else np.nan
seiten_std = float(clean["Seiten_pro_Besuch"].std()) if not clean.empty else np.nan
bounce_mean = float(clean["Bounce_Rate_pct"].mean()) if not clean.empty else np.nan
bounce_std = float(clean["Bounce_Rate_pct"].std()) if not clean.empty else np.nan
pearson_r = float(clean[["Seiten_pro_Besuch", "Bounce_Rate_pct"]].corr().iloc[0, 1]) if len(clean) > 1 else np.nan

fig, ax1 = plt.subplots(figsize=(14, 7))
ax2 = ax1.twinx()

x = clean["Datum_dt"]
y1 = clean["Seiten_pro_Besuch"]
y2 = clean["Bounce_Rate_pct"]

c1 = "#1f77b4"
c2 = "#d62728"

ax1.plot(x, y1, color=c1, marker="o", linewidth=2, markersize=5, label="Seiten pro Besuch")
ax2.plot(x, y2, color=c2, marker="s", linewidth=2, markersize=5, label="Absprungrate (%)")

ax1.set_xlabel("Datum")
ax1.set_ylabel("Seiten pro Besuch", color=c1)
ax2.set_ylabel("Absprungrate (%)", color=c2)

for tl in ax1.get_yticklabels():
    tl.set_color(c1)
for tl in ax2.get_yticklabels():
    tl.set_color(c2)

fig.suptitle("Zusammenhang zwischen Seiten pro Besuch und Absprungrate – September 2024 (kreis-kleve.de)")
subtitle = f"Mittelwert Seiten/Besuch: {seiten_mean:.2f} (±{seiten_std:.2f}) | Mittelwert Absprungrate: {bounce_mean:.1f}% (±{bounce_std:.1f}%) | Pearson r: {pearson_r:.2f}"
ax1.set_title(subtitle, fontsize=12, pad=14)

ax1.grid(True, which="major", axis="both", linestyle="--", alpha=0.3)
ax1.tick_params(axis="x", rotation=45)

lines1, labels1 = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax1.legend(lines1 + lines2, labels1 + labels2, loc="upper right", frameon=True)

plt.tight_layout(rect=[0, 0.03, 1, 0.95])

out_fp = r"./src/resources/output\32830/1_seiten_pro_besuch_vs_bounce_rate.png"
try:
    plt.savefig(out_fp, dpi=300, bbox_inches="tight")
except Exception as e:
    print(str(e))
plt.close(fig)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

input_path = r'./src/resources/data/study\32830\dataset.csv'
output_path = r'./src/resources/output\32830/2_cookie_optin_bars.png'

sns.set_theme(style='white', context='talk')

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

df_eda = df.copy()
df_eda = df_eda.replace({'-': np.nan, '': np.nan, ' ': np.nan})
missing_ratio = df_eda.isna().mean()

for c in ['Besuche', 'Besucher', 'Seitenaufrufe', 'Seiten_pro_Besuch', 'Conversions__Last_Ad_', 'Conversion_Wert__Last_Ad_']:
    if c in df_eda.columns:
        df_eda[c] = pd.to_numeric(df_eda[c], errors='coerce')

for c in ['Bounce_Rate', 'Conversion_Rate__Last_Ad_']:
    if c in df_eda.columns:
        s = df_eda[c].astype(str).str.replace('%', '', regex=False).str.replace(',', '.', regex=False)
        df_eda[c] = pd.to_numeric(s, errors='coerce')

df['Datum_dt'] = pd.to_datetime(df['Datum'], dayfirst=True, errors='coerce')

def parse_percent(series):
    s = series.astype(str).str.replace('%', '', regex=False).str.replace(',', '.', regex=False).str.strip()
    s = s.replace({'-': np.nan, 'nan': np.nan, 'None': np.nan})
    return pd.to_numeric(s, errors='coerce')

if 'Anteil_Besuche_mit_Cookies' in df.columns and 'Anteil_Besucher_mit_Cookies' in df.columns:
    df['Anteil_Besuche_mit_Cookies_num'] = parse_percent(df['Anteil_Besuche_mit_Cookies'])
    df['Anteil_Besucher_mit_Cookies_num'] = parse_percent(df['Anteil_Besucher_mit_Cookies'])
else:
    df['Anteil_Besuche_mit_Cookies_num'] = np.nan
    df['Anteil_Besucher_mit_Cookies_num'] = np.nan

plot_df = df.loc[df['Datum_dt'].notna(), ['Datum_dt', 'Anteil_Besuche_mit_Cookies_num', 'Anteil_Besucher_mit_Cookies_num']].copy()
plot_df = plot_df.sort_values('Datum_dt')

plot_df['Tag'] = plot_df['Datum_dt'].dt.strftime('%d.%m.')

visits_pct = plot_df['Anteil_Besuche_mit_Cookies_num'].astype(float).to_numpy()
visitors_pct = plot_df['Anteil_Besucher_mit_Cookies_num'].astype(float).to_numpy()

mean_visits = np.nanmean(visits_pct) if np.isfinite(np.nanmean(visits_pct)) else 0.0
mean_visitors = np.nanmean(visitors_pct) if np.isfinite(np.nanmean(visitors_pct)) else 0.0

x = np.arange(len(plot_df))
width = 0.42

fig, ax = plt.subplots(figsize=(16, 7))
colors = sns.color_palette('tab10', 2)
bar1 = ax.bar(x - width/2, visits_pct, width=width, color=colors[0], label='Anteil Besuche mit Cookies')
bar2 = ax.bar(x + width/2, visitors_pct, width=width, color=colors[1], label='Anteil Besucher mit Cookies')

ax.axhline(mean_visits, color=colors[0], linestyle='--', linewidth=1.5, alpha=0.9, label=f'Durchschnitt Besuche: {mean_visits:.2f}%')
ax.axhline(mean_visitors, color=colors[1], linestyle='--', linewidth=1.5, alpha=0.9, label=f'Durchschnitt Besucher: {mean_visitors:.2f}%')

ax.set_xticks(x)
ax.set_xticklabels(plot_df['Tag'], rotation=60, ha='right')
ax.set_ylabel('Anteil mit Cookies (%)')
ax.set_xlabel('Datum (September 2024)')
ax.set_title('Cookie-Einwilligungen nach Tag: Anteil an Besuchen und Besuchern mit Cookies (September 2024)')

max_val = np.nanmax([visits_pct.max() if len(visits_pct) else 0, visitors_pct.max() if len(visitors_pct) else 0])
if not np.isfinite(max_val):
    max_val = 0.1
ylim_top = max(0.1, float(max_val) * 1.6 if max_val > 0 else 0.1)
ax.set_ylim(0, ylim_top)

ax.grid(axis='y', linestyle=':', alpha=0.4)
ax.legend(ncol=2, frameon=False, loc='upper right')
fig.tight_layout()

os.makedirs(os.path.dirname(output_path), exist_ok=True)
fig.savefig(output_path, dpi=300)
plt.close(fig)
