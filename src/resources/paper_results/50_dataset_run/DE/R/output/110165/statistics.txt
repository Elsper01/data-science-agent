to_na <- function(x){
  y <- as.character(x)
  y <- stringr::str_squish(y)
  yl <- tolower(y)
  y[is.na(y) | yl %in% c("", "na", "nan", "null", "n/a", "keine", "none")] <- NA_character_
  y
}
