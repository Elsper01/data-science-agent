import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter
import os

# File paths
input_path = './src/resources/data/study/15653/dataset.csv'
output_dir = './src/resources/output/15653'
os.makedirs(output_dir, exist_ok=True)

# The error 'utf-8' codec can't decode byte 0xe4 (ä) suggests the file is likely encoded in ISO-8859-1 (latin1) or cp1252.
# We will try 'latin1' which is common for German datasets containing umlauts.
try:
    df = pd.read_csv(input_path, sep=';', encoding='utf-8')
except UnicodeDecodeError:
    df = pd.read_csv(input_path, sep=';', encoding='latin1')

# Data Cleaning
pd.options.mode.chained_assignment = None

# Convert numeric columns
df['time'] = pd.to_numeric(df.get('time'), errors='coerce')
df['value'] = pd.to_numeric(df.get('value'), errors='coerce')

# Define columns based on the goal
origin_col = '2_variable_attribute_label'
metric_col = 'value_variable_label'

# Filter for Inland/Ausland and ensure valid data
mask_valid = (
    df[origin_col].isin(['Inland', 'Ausland']) & 
    df['time'].notna() & 
    df['value'].notna() & 
    df[metric_col].notna()
)
df_valid = df.loc[mask_valid].copy()

if df_valid.empty:
    print("No valid data found after filtering.")
else:
    # Get the latest 3 years
    years = sorted(df_valid['time'].astype(int).unique())
    latest_years = years[-3:] if len(years) >= 3 else years
    plot_df = df_valid[df_valid['time'].isin(latest_years)].copy()
    plot_df['time'] = plot_df['time'].astype(int)

    # Calculate shares for labels
    totals = plot_df.groupby(['time', metric_col], as_index=False)['value'].sum().rename(columns={'value': 'total_value'})
    plot_df = plot_df.merge(totals, on=['time', metric_col], how='left')
    plot_df['share'] = np.where(plot_df['total_value'] > 0, plot_df['value'] / plot_df['total_value'], 0)

    # Visualization setup
    sns.set_theme(style='whitegrid', context='talk')
    order_hue = ['Inland', 'Ausland']
    palette = {'Inland': '#1f77b4', 'Ausland': '#ff7f0e'}

    g = sns.catplot(
        data=plot_df,
        kind='bar',
        x='time',
        y='value',
        hue=origin_col,
        col=metric_col,
        hue_order=order_hue,
        palette=palette,
        height=5,
        aspect=1.2,
        legend=True
    )

    g.set_axis_labels('Jahr', 'Anzahl')
    g.set_titles(col_template='{col_name}')

    # Formatting axes and adding percentage labels
    share_lookup = plot_df.set_index(['time', origin_col, metric_col])['share'].to_dict()

    for ax in g.axes.flatten():
        # Format Y axis with German thousands separator
        ax.yaxis.set_major_formatter(FuncFormatter(lambda x, p: f"{int(x):,}".replace(',', '.')))
        
        facet_label = ax.get_title()
        # Add percentage labels on top of bars
        for container in ax.containers:
            # Identify which hue this container belongs to
            hue_label = container.get_label()
            for i, bar in enumerate(container):
                # Get the year from the x-axis tick label
                try:
                    year_val = int(ax.get_xticklabels()[i].get_text())
                    share = share_lookup.get((year_val, hue_label, facet_label), 0)
                    if share > 0:
                        height = bar.get_height()
                        ax.text(
                            bar.get_x() + bar.get_width() / 2,
                            height,
                            f"{share*100:.1f}%",
                            ha='center', va='bottom', fontsize=10, fontweight='bold'
                        )
                except (ValueError, IndexError):
                    continue

    # Title and Layout
    years_str = f"{min(latest_years)}–{max(latest_years)}" if len(latest_years) > 1 else str(latest_years[0])
    plt.subplots_adjust(top=0.85)
    g.fig.suptitle(f'Tourismus in Rettenberg: Inland vs. Ausland ({years_str})', fontsize=18)

    output_path = os.path.join(output_dir, '1_tourismus_inland_ausland_vergleich.png')
    g.savefig(output_path, dpi=200, bbox_inches='tight')
    plt.close('all')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from matplotlib.ticker import FuncFormatter, MaxNLocator

# Set visual style
sns.set_theme(style="whitegrid")

file_path = r'./src/resources/data/study/15653/dataset.csv'
output_path = r'./src/resources/output/15653/2_uebernachtungen_vs_ankuenfte.png'

# The error 'utf-8' codec can't decode byte 0xe4 suggests the file is likely encoded in ISO-8859-1 (Latin-1) 
# which is common for German datasets containing umlauts (ä, ö, ü).
try:
    df = pd.read_csv(file_path, sep=';', encoding='iso-8859-1')
except Exception as e:
    # Fallback to latin-1 if iso-8859-1 fails, though they are very similar
    try:
        df = pd.read_csv(file_path, sep=';', encoding='latin-1')
    except Exception as e2:
        print(f"Failed to load data: {e2}")
        raise SystemExit(1)

# Data Cleaning and Filtering
# Ensure numeric types
for col in ['time', 'value']:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='coerce')

# Filter for Rettenberg (based on the context of the study)
df = df[df['1_variable_attribute_label'].astype(str).str.contains('Rettenberg', na=False)]

# Filter for the specific KPIs requested
kpis = ['Gästeankünfte in Beherbergungsbetrieben', 'Gästeübernachtungen in Beherbergungsbetrieben']
df_kpi = df[df['value_variable_label'].isin(kpis)].copy()

# Filter for 'Insgesamt' as requested in the visualization goal
mask_total = df_kpi['2_variable_attribute_label'].astype(str) == 'Insgesamt'
df_total = df_kpi[mask_total].copy()

# Pivot data for plotting
pivot = df_total.pivot_table(index='time', columns='value_variable_label', values='value', aggfunc='sum')

# Rename for cleaner legend
rename_map = {
    'Gästeübernachtungen in Beherbergungsbetrieben': 'Gästeübernachtungen',
    'Gästeankünfte in Beherbergungsbetrieben': 'Gästeankünfte'
}
pivot = pivot.rename(columns=rename_map)
pivot = pivot.sort_index()

# Check if we have data to plot
if pivot.empty or not any(col in pivot.columns for col in ['Gästeübernachtungen', 'Gästeankünfte']):
    print('Required KPIs missing after filtering.')
    raise SystemExit(1)

# Calculate metrics for annotation
ue = pivot['Gästeübernachtungen'] if 'Gästeübernachtungen' in pivot.columns else pd.Series(dtype=float)
ank = pivot['Gästeankünfte'] if 'Gästeankünfte' in pivot.columns else pd.Series(dtype=float)
common = pd.concat([ue, ank], axis=1).dropna()

avg_los = (ue.sum() / ank.sum()) if not ank.empty and ank.sum() != 0 else np.nan

# Plotting
fig, ax1 = plt.subplots(figsize=(12, 7))

color_ue = '#2E86AB' # Blue
color_ank = '#E67E22' # Orange

# Primary Axis: Overnight Stays
l1 = None
if 'Gästeübernachtungen' in pivot.columns:
    l1, = ax1.plot(pivot.index, pivot['Gästeübernachtungen'], color=color_ue, marker='o', 
                   linewidth=2.5, markersize=8, label='Gästeübernachtungen')

# Secondary Axis: Arrivals
ax2 = ax1.twinx()
l2 = None
if 'Gästeankünfte' in pivot.columns:
    l2, = ax2.plot(pivot.index, pivot['Gästeankünfte'], color=color_ank, marker='s', 
                   linewidth=2.5, markersize=8, label='Gästeankünfte')

# Formatting
ax1.set_xlabel('Jahr', fontsize=12)
ax1.set_ylabel('Gästeübernachtungen (Anzahl)', color=color_ue, fontsize=12, fontweight='bold')
ax2.set_ylabel('Gästeankünfte (Anzahl)', color=color_ank, fontsize=12, fontweight='bold')

# Thousands separator for German locale (using dot)
formatter = FuncFormatter(lambda x, p: f"{int(x):,}".replace(',', '.'))
ax1.yaxis.set_major_formatter(formatter)
ax2.yaxis.set_major_formatter(formatter)

ax1.set_ylim(bottom=0)
ax2.set_ylim(bottom=0)

plt.title('Rettenberg: Gästeübernachtungen vs. Gästeankünfte (Insgesamt)', fontsize=14, pad=20)

# Ensure all years are shown on X-axis
years = pivot.index.astype(int)
ax1.xaxis.set_major_locator(MaxNLocator(integer=True))
ax1.set_xticks(years)

# Legend
lines = [l for l in [l1, l2] if l is not None]
labels = [l.get_label() for l in lines]
ax1.legend(lines, labels, loc='upper left', frameon=True, shadow=True)

# Annotation for average length of stay
if not np.isnan(avg_los):
    info_text = f'Ø Aufenthaltsdauer: {avg_los:.2f} Nächte'
    ax1.text(0.02, 0.85, info_text, transform=ax1.transAxes, fontsize=11, 
             bbox=dict(boxstyle='round', facecolor='white', alpha=0.7))

ax1.grid(True, axis='y', linestyle='--', alpha=0.6)
plt.tight_layout()

# Save output
try:
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
except Exception as e:
    print(f"Error saving plot: {e}")
finally:
    plt.close()
