    df = pd.read_csv(input_path, sep=',', encoding='utf-8', dtype=str)

    print(f'Fehler beim Laden der Daten: {e}')
    sys.exit(1)

try:
    df.columns = [c.strip() for c in df.columns]
    total_rows = len(df)
    missing_ratio = df.isna().mean()
    dtypes_overview = df.dtypes.astype(str)

        print('Fehlende Spalte: Methode')

    method_counts = methode.value_counts(dropna=False)
    method_share = (method_counts / method_counts.sum()).sort_values(ascending=False)
    plot_df = (
        pd.DataFrame({'Methode': method_counts.index, 'Anzahl': method_counts.values})

    plt.figure(figsize=(12, max(4, 0.35 * len(plot_df))))

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import os
import sys
import re

plt.rcParams.update({"figure.dpi": 120, "savefig.dpi": 300, "font.size": 10, "axes.titlesize": 12, "axes.labelsize": 11})
sns.set_theme(style="whitegrid")

try:
    input_path = os.path.normpath('./src/resources/data/study\\137336\\dataset.csv')
    output_dir = os.path.normpath('./src/resources/output\\137336\\run_1')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(input_path, sep=',', encoding='utf-8', dtype=str)
    for c in df.columns:
        df[c] = df[c].astype(str).str.strip()
    def parse_ergebnis(val):
        if val is None:
            return np.nan, False, np.nan
        s = str(val).strip().lower()
        if s in ['', 'nan']:
            return np.nan, False, np.nan
        s = s.replace(' ', '')
        if s in ['n.n.', 'n.n', 'nn', 'nichtnachweisbar']:
            return np.nan, False, np.nan
        s = s.replace(',', '.')
        if s.startswith('<'):
            try:
                num = float(re.sub(r'[^0-9.+-eE]', '', s[1:]))
            except Exception:
                num = np.nan
            return np.nan, True, num
        if s.startswith('>'):
            try:
                num = float(re.sub(r'[^0-9.+-eE]', '', s[1:]))
            except Exception:
                num = np.nan
            return num, True, num
        try:
            num = float(re.sub(r'[^0-9.+-eE]', '', s))
            return num, False, np.nan
        except Exception:
            return np.nan, False, np.nan
    parsed = df['Ergebnis'].apply(parse_ergebnis)
    df['Ergebnis_num'] = parsed.apply(lambda x: x[0])
    df['Ergebnis_zensiert'] = parsed.apply(lambda x: x[1])
    df['Ergebnis_Grenze'] = parsed.apply(lambda x: x[2])
    num_cols = ['Ergebnis_num', 'Ergebnis_Grenze']
    for c in num_cols:
        df[c] = pd.to_numeric(df[c], errors='coerce')
    missing_props = df.replace({'': np.nan}).isna().mean().rename('Anteil_fehlend')
    types = df.dtypes.astype(str).rename('dtype')
    overview = pd.concat([missing_props, types], axis=1)
    overview_path = os.path.join(output_dir, 'eda_uebersicht.csv')
    overview.to_csv(overview_path, index=True, encoding='utf-8')
    unit_stats = df.groupby('Einheit')['Ergebnis_num'].agg(['count', 'nunique', 'min', 'max', 'mean', 'std']).rename(columns={'count': 'anzahl', 'nunique': 'anzahl_unikate'})
    unit_stats_path = os.path.join(output_dir, 'eda_einheit_stats.csv')
    unit_stats.to_csv(unit_stats_path, encoding='utf-8')
    cens_summary = pd.DataFrame({
        'gesamt': [len(df)],
        'anzahl_num': [df['Ergebnis_num'].notna().sum()],
        'anzahl_zensiert': [df['Ergebnis_zensiert'].sum()],
        'anteil_zensiert': [float(df['Ergebnis_zensiert'].mean())]
    })
    cens_path = os.path.join(output_dir, 'eda_zensierte_werte.csv')
    cens_summary.to_csv(cens_path, index=False, encoding='utf-8')
    detected = df.copy()
    detected = detected[df['Ergebnis_num'].notna()]
    detected = detected[~detected['Ergebnis_zensiert']] 
    detected = detected[detected['Ergebnis_num'] > 0]
    if detected.empty:
        fig = plt.figure(figsize=(8, 4))
        plt.text(0.5, 0.6, 'Keine nachgewiesenen Analyten mit quantifizierbaren Werten (>0) gefunden', ha='center', va='center')
        plt.axis('off')
        plt.suptitle('Nachgewiesene Analyten im Heu – Übersicht', y=0.98)
        plt.figtext(0.5, 0.04, 'Filter: Ausschluss von n.n. und <Nachweisgrenze; es wurden keine Werte >0 gefunden', ha='center')
        out_path = os.path.join(output_dir, '1_nachgewiesene_analyten.png')
        plt.savefig(out_path, bbox_inches='tight', facecolor='white')
        plt.close()
        sys.exit(0)
    detected['Einheit'] = detected['Einheit'].fillna('ohne Einheit')
    units = list(detected['Einheit'].value_counts().index)
    n_units = len(units)
    ncols = 2 if n_units > 1 else 1
    nrows = int(np.ceil(n_units / ncols))
    counts_per_unit = detected.groupby('Einheit')['Pruefmerkmal'].count()
    max_bars = int(counts_per_unit.max()) if len(counts_per_unit) > 0 else 5
    height_per_row = max(2.0, min(0.35 * max_bars, 10.0))
    fig_w = ncols * 7
    fig_h = nrows * height_per_row + 1.0
    fig, axes = plt.subplots(nrows=nrows, ncols=ncols, figsize=(fig_w, fig_h), squeeze=False, sharex=False)
    palette = sns.color_palette('viridis')
    idx = 0
    for r in range(nrows):
        for c in range(ncols):
            ax = axes[r, c]
            if idx >= n_units:
                ax.axis('off')
                continue
            unit = units[idx]
            sub = detected[detected['Einheit'] == unit].copy()
            sub = sub.sort_values('Ergebnis_num', ascending=True)
            y_labels = sub['Pruefmerkmal'].astype(str).tolist()
            x_vals = sub['Ergebnis_num'].values
            colors = sns.color_palette('viridis', n_colors=len(sub))
            ax.barh(range(len(sub)), x_vals, color=colors, edgecolor='black', linewidth=0.3)
            ax.set_yticks(range(len(sub)))
            ax.set_yticklabels(y_labels, fontsize=9)
            ax.set_xlabel(f'Messwert ({unit})')
            ax.set_title(f'Einheit: {unit} (n={len(sub)})')
            ax.grid(True, axis='x', linestyle='--', alpha=0.5)
            ax.grid(False, axis='y')
            idx += 1
    for r in range(nrows):
        for c in range(ncols):
            axes[r, c].tick_params(axis='x', labelrotation=0)
    plt.suptitle('Nachgewiesene Analyten (Prüfmerkmal) im Heu mit quantifizierbaren Werten, nach Einheit', y=0.995)
    plt.figtext(0.5, 0.01, 'Darstellung umfasst nur Werte > 0 ohne n.n. und ohne <Nachweisgrenze. Einheiten werden in Facetten getrennt.', ha='center', fontsize=9)
    plt.tight_layout(rect=[0, 0.03, 1, 0.97])
    out_path = os.path.join(output_dir, '1_nachgewiesene_analyten.png')
    plt.savefig(out_path, bbox_inches='tight', facecolor='white')
    plt.close()
except Exception as e:
    print(str(e))

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

plt.rcParams["figure.dpi"] = 150
plt.rcParams["savefig.dpi"] = 300
plt.rcParams["font.family"] = "DejaVu Sans"

try:
    df = pd.read_csv('./src/resources/data/study\\137336\\dataset.csv', sep=',', encoding='utf-8')
except Exception as e:
    print(str(e))
    raise

missing_ratio = df.isna().mean()

if 'Einheit' in df.columns:
    einheit = df['Einheit'].astype(str).replace({'nan': np.nan})
else:
    einheit = pd.Series(np.nan, index=df.index)

def _clean_unit(u):
    if pd.isna(u):
        return np.nan
    s = str(u).strip()
    s = s.replace('\u00a0', ' ')
    s = ' '.join(s.split())
    if s == '' or s.lower() in ['nan', 'none']: 
        return np.nan
    return s

einheit_clean = einheit.map(_clean_unit)
einheit_clean = einheit_clean.fillna('keine Angabe')
counts = einheit_clean.value_counts(dropna=False).sort_values(ascending=False)

unique_units = counts.index.tolist()
counts_values = counts.values.tolist()

sns.set_theme(style='whitegrid', context='talk')
fig, ax = plt.subplots(figsize=(10, max(4, 0.4*len(unique_units)+1)))
palette = sns.color_palette('viridis', n_colors=len(unique_units))
ax.barh(range(len(unique_units)), counts_values, color=palette, edgecolor='black', linewidth=0.5)
ax.set_yticks(range(len(unique_units)))
ax.set_yticklabels(unique_units)
ax.invert_yaxis()
ax.set_xlabel('Anzahl der Prüfmerkmale')
ax.set_ylabel('Messeinheit')
ax.set_title('Häufigkeit der verwendeten Messeinheiten in der Analyse eines Heu-Samples\nAnzahl der unterschiedlichen Einheiten: '+str(len(unique_units))+
             ' | Gesamtzahl der Prüfmerkmale: '+str(len(df)))
for i, v in enumerate(counts_values):
    ax.text(v + max(counts_values)*0.01, i, str(v), va='center', ha='left', fontsize=10)
plt.tight_layout()
plt.subplots_adjust(left=0.32)

out_path = './src/resources/output\\137336\\run_1/2_einheiten_haeufigkeit.png'
try:
    fig.savefig(out_path, bbox_inches='tight')
except Exception as e:
    print(str(e))
    raise
plt.close(fig)
