import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

df = pd.read_csv('./src/resources/data/study/103779/dataset.csv', sep=',', encoding='utf-8')

df['has_FotoURL'] = df['FotoURL'].notna()
df['is_Alteintragung'] = df['Beschreibung'].astype(str).str.contains('Alteintragung', case=False, na=False)

print("=== Datenqualitätsanalyse ===")
print(f"\nGesamtanzahl der Einträge: {len(df)}")
print(f"\nAnteil mit FotoURL: {df['has_FotoURL'].sum() / len(df) * 100:.1f}%")
print(f"Anteil ohne FotoURL: {(~df['has_FotoURL']).sum() / len(df) * 100:.1f}%")
print(f"\nAnteil Alteintragung: {df['is_Alteintragung'].sum() / len(df) * 100:.1f}%")

print("\n=== Verteilung nach Kulturdenkmaltyp ===")
print(df['Kulturdenkmaltyp'].value_counts())

foto_by_type = df.groupby(['Kulturdenkmaltyp', 'has_FotoURL']).size().unstack(fill_value=0)
print("\n=== FotoURL-Status nach Kulturdenkmaltyp ===")
print(foto_by_type)

foto_by_type_pct = foto_by_type.div(foto_by_type.sum(axis=1), axis=0) * 100
print("\n=== FotoURL-Status nach Kulturdenkmaltyp (Prozent) ===")
print(foto_by_type_pct)

fig, ax = plt.subplots(figsize=(12, 7))

x = np.arange(len(foto_by_type.index))
width = 0.35

bars1 = ax.bar(x - width/2, foto_by_type[False], width, label='Ohne FotoURL', color='#d62728', alpha=0.8)
bars2 = ax.bar(x + width/2, foto_by_type[True], width, label='Mit FotoURL', color='#2ca02c', alpha=0.8)

ax.set_xlabel('Kulturdenkmaltyp', fontsize=12, fontweight='bold')
ax.set_ylabel('Anzahl der Einträge', fontsize=12, fontweight='bold')
ax.set_title('Dokumentationsstatus der Kulturdenkmäler:\nVisuelle Dokumentation nach Denkmaltyp', 
             fontsize=14, fontweight='bold', pad=20)
ax.set_xticks(x)
ax.set_xticklabels(foto_by_type.index, rotation=45, ha='right')
ax.legend(fontsize=11, loc='upper right')
ax.grid(axis='y', alpha=0.3, linestyle='--')

for bars in [bars1, bars2]:
    for bar in bars:
        height = bar.get_height()
        if height > 0:
            ax.text(bar.get_x() + bar.get_width()/2., height,
                   f'{int(height)}',
                   ha='center', va='bottom', fontsize=9)

plt.tight_layout()
plt.savefig('./src/resources/output/103779/2_dokumentationsstatus_nach_denkmaltyp.png', dpi=300, bbox_inches='tight')
plt.close()

print("\n=== Visualisierung erfolgreich erstellt ===")
