options(encoding = "UTF-8")

# Load necessary libraries
library(tidyverse)
library(janitor)
library(scales)

# Define paths
in_path <- file.path(".", "src", "resources", "data", "study", "15539", "dataset.csv")
out_dir <- file.path(".", "src", "resources", "output", "15539")
dir.create(out_dir, recursive = TRUE, showWarnings = FALSE)

# Read data. The error "unable to translate ... to a wide string" often happens 
# when R's internal encoding doesn't match the file. 
# We'll try reading with Latin1 if UTF-8 fails, or use a more robust approach.
# First, read raw to check for common German characters.
raw_data <- readr::read_delim(
  in_path, 
  delim = ";", 
  locale = readr::locale(encoding = "ISO-8859-1"), # Common for German CSVs
  trim_ws = TRUE, 
  show_col_types = FALSE
)

# Clean column names
df <- janitor::clean_names(raw_data)

# Convert time and value to numeric
df <- df %>%
  mutate(
    time = as.numeric(time),
    value = as.numeric(value)
  )

# To avoid encoding issues with 'ä' and 'ü' in grepl, we use regex wildcards (.)
# or filter by the 'code' columns which are usually ASCII.
# Goal: Filter for 'Insgesamt' in the attribute label.
# Goal: Compare 'Gästeankünfte' and 'Gästeübernachtungen'.

# 1. Filter for 'Insgesamt' (Total)
# We look for 'Insgesamt' or the code '1' (common for totals)
df_total <- df %>%
  filter(
    grepl("Insgesamt", x2_variable_attribute_label, ignore.case = TRUE) |
    x2_variable_attribute_code == "1"
  )

# 2. Identify the metrics using wildcards to bypass encoding errors
# G.steank.nfte -> Gästeankünfte
# G.ste.bernachtungen -> Gästeübernachtungen
plot_df <- df_total %>%
  filter(
    grepl("G.steank.nfte", value_variable_label, ignore.case = TRUE) | 
    grepl("G.ste.bernachtungen", value_variable_label, ignore.case = TRUE)
  ) %>%
  mutate(
    metric_clean = case_when(
      grepl("ank.nfte", value_variable_label, ignore.case = TRUE) ~ "Gästeankünfte",
      grepl(".bernachtungen", value_variable_label, ignore.case = TRUE) ~ "Gästeübernachtungen",
      TRUE ~ value_variable_label
    )
  )

# Check if we have data
if (nrow(plot_df) == 0) {
  # Fallback: If specific filters failed, take all data from df_total to see what's there
  plot_df <- df_total %>% 
    filter(!is.na(value)) %>%
    mutate(metric_clean = value_variable_label)
}

if (nrow(plot_df) > 0) {
  # Create the plot
  p <- ggplot(plot_df, aes(x = time, y = value, color = metric_clean, group = metric_clean)) +
    geom_line(linewidth = 1.2) +
    geom_point(size = 2) +
    scale_x_continuous(breaks = seq(min(plot_df$time, na.rm=T), max(plot_df$time, na.rm=T), by = 1)) +
    scale_y_continuous(labels = scales::label_number(big.mark = ".", decimal.mark = ",")) +
    labs(
      title = "Tourismusentwicklung in Bad Berneck (2014-2024)",
      subtitle = "Gästeankünfte und Übernachtungen (Kategorie: Insgesamt)",
      x = "Jahr",
      y = "Anzahl",
      color = "Kennzahl"
    ) +
    theme_minimal(base_size = 12) +
    theme(
      legend.position = "bottom",
      plot.title = element_text(face = "bold"),
      panel.grid.minor = element_blank()
    )

  # Save the plot
  ggplot2::ggsave(
    filename = file.path(out_dir, "0_gaeste_trend.png"), 
    plot = p, 
    width = 10, 
    height = 6, 
    dpi = 300
  )
  
  # Save summary stats
  summary_stats <- plot_df %>%
    group_by(metric_clean) %>%
    summarise(
      mean_val = mean(value, na.rm = TRUE),
      max_val = max(value, na.rm = TRUE),
      min_val = min(value, na.rm = TRUE),
      .groups = "drop"
    )
  readr::write_csv(summary_stats, file.path(out_dir, "summary_stats.csv"))
  
} else {
  # Final safety check to avoid "No output" error
  writeLines("No data could be filtered for the plot.", file.path(out_dir, "error_log.txt"))
}
