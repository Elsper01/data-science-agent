import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Set plot style

df = pd.read_csv(file_path, sep=',', encoding='utf-8')

df_clean['Besucher'] = pd.to_numeric(df_clean['Besucher'], errors='coerce')

# EDA: Basic statistics for the report (optional but good practice)
stats = df_clean[['Besuche', 'Besucher']].describe()

plt.figure(figsize=(12, 6))

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# Load the dataset
file_path = './src/resources/data/study/32830/dataset.csv'
df = pd.read_csv(file_path, sep=',', encoding='utf-8')

# Data Cleaning
# 1. Remove the aggregate row (Index 0 where Datum is NaN)
df_clean = df.dropna(subset=['Datum']).copy()

# 2. Remove the empty column 'Unnamed: 14'
if 'Unnamed__14' in df_clean.columns:
    df_clean = df_clean.drop(columns=['Unnamed__14'])

# 3. Convert 'Datum' to datetime objects
df_clean['Datum'] = pd.to_datetime(df_clean['Datum'], format='%d.%m.%Y')

# 4. Ensure numerical columns are correctly typed
df_clean['Besuche'] = pd.to_numeric(df_clean['Besuche'], errors='coerce')
df_clean['Seiten_pro_Besuch'] = pd.to_numeric(df_clean['Seiten_pro_Besuch'], errors='coerce')

# Sort by date for the line chart
df_clean = df_clean.sort_values('Datum')

# EDA: Correlation between traffic volume and engagement depth
correlation = df_clean[['Besuche', 'Seiten_pro_Besuch']].corr().iloc[0, 1]

# Visualization: Dual-axis line chart
fig, ax1 = plt.subplots(figsize=(12, 6))

# Plot Besuche (Visits) on the first y-axis
color_visits = '#1f77b4'
ax1.set_xlabel('Datum (September 2024)', fontsize=12)
ax1.set_ylabel('Anzahl der Besuche', color=color_visits, fontsize=12, fontweight='bold')
line1 = ax1.plot(df_clean['Datum'], df_clean['Besuche'], color=color_visits, marker='o', label='Besuche', linewidth=2)
ax1.tick_params(axis='y', labelcolor=color_visits)
ax1.grid(True, linestyle='--', alpha=0.6)

# Create a second y-axis for Seiten_pro_Besuch (Pages per Visit)
ax2 = ax1.twinx()
color_pages = '#d62728'
ax2.set_ylabel('Seiten pro Besuch (Tiefe)', color=color_pages, fontsize=12, fontweight='bold')
line2 = ax2.plot(df_clean['Datum'], df_clean['Seiten_pro_Besuch'], color=color_pages, marker='s', linestyle='--', label='Seiten pro Besuch', linewidth=2)
ax2.tick_params(axis='y', labelcolor=color_pages)

# Title and Legend
plt.title(f'Verhältnis zwischen Traffic-Volumen und Nutzungstiefe (Korrelation: {correlation:.2f})\nDomain: kreis-kleve.de', fontsize=14, pad=20)

# Combine legends from both axes
lines = line1 + line2
labels = [l.get_label() for l in lines]
ax1.legend(lines, labels, loc='upper left', frameon=True)

# Formatting x-axis dates
plt.xticks(rotation=45)
fig.tight_layout()

# Save the plot
plt.savefig('./src/resources/output/32830/1_traffic_vs_engagement_depth.png', dpi=300)
plt.close()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime

def time_to_seconds(time_str):
    if pd.isna(time_str) or not isinstance(time_str, str):
        return np.nan
    try:
        h, m, s = map(int, time_str.split(':'))
        return h * 3600 + m * 60 + s
    except:
        return np.nan

file_path = './src/resources/data/study/32830/dataset.csv'
df = pd.read_csv(file_path)

# Data Cleaning
# Remove the aggregate row (Index 0 where Datum is NaN)
df_clean = df.dropna(subset=['Datum']).copy()

# Convert Datum to datetime
df_clean['Datum'] = pd.to_datetime(df_clean['Datum'], format='%d.%m.%Y')

# Convert Bounce_Rate from string percentage to float
df_clean['Bounce_Rate_Num'] = df_clean['Bounce_Rate'].str.replace('%', '').astype(float)

# Convert Zeit_im_Besuch (HH:MM:SS) to total seconds
df_clean['Zeit_im_Besuch_Sec'] = df_clean['Zeit_im_Besuch'].apply(time_to_seconds)

# Sort by date for the line chart
df_clean = df_clean.sort_values('Datum')

# Statistical Analysis: Correlation
correlation = df_clean[['Bounce_Rate_Num', 'Zeit_im_Besuch_Sec']].corr().iloc[0, 1]

# Visualization
fig, ax1 = plt.subplots(figsize=(12, 7))

# Plot Bounce Rate
color_bounce = '#e74c3c'
ax1.set_xlabel('Datum', fontsize=12)
ax1.set_ylabel('Absprungrate (%)', color=color_bounce, fontsize=12)
line1 = ax1.plot(df_clean['Datum'], df_clean['Bounce_Rate_Num'], color=color_bounce, marker='o', linewidth=2, label='Absprungrate (%)')
ax1.tick_params(axis='y', labelcolor=color_bounce)
ax1.grid(True, linestyle='--', alpha=0.6)

# Create a second y-axis for session duration
ax2 = ax1.twinx()
color_time = '#2980b9'
ax2.set_ylabel('Besuchsdauer (Sekunden)', color=color_time, fontsize=12)
line2 = ax2.plot(df_clean['Datum'], df_clean['Zeit_im_Besuch_Sec'], color=color_time, marker='s', linewidth=2, label='Besuchsdauer (Sek)')
ax2.tick_params(axis='y', labelcolor=color_time)

# Title and Legend
plt.title(f'Tägliche Absprungrate vs. Besuchsdauer (Sept 2024)\nKorrelation: {correlation:.2f}', fontsize=14, fontweight='bold', pad=20)
lines = line1 + line2
labels = [l.get_label() for l in lines]
ax1.legend(lines, labels, loc='upper left', frameon=True)

# Formatting x-axis dates
plt.xticks(rotation=45)
fig.tight_layout()

# Save the plot
import os
output_dir = './src/resources/output/32830/'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

plt.savefig(f'{output_dir}2_bounce_rate_vs_duration.png', dpi=300)
plt.close()
