import os
import sys
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

plt.switch_backend('Agg')
sns.set_theme(style='whitegrid', context='talk')

input_path = os.path.normpath('./src/resources/data/study\\69089\\dataset.csv')
output_dir = os.path.normpath('./src/resources/output/69089')
plot_filename = '2_hist_alter_nach_geschlecht.png'

try:
    os.makedirs(output_dir, exist_ok=True)
except Exception as e:
    print(f'Fehler beim Erstellen des Output-Verzeichnisses: {e}')
    sys.exit(1)

try:
    df = pd.read_csv(input_path, sep=',', encoding='utf-8', keep_default_na=True)
except Exception as e:
    print(f'Fehler beim Einlesen der Datei: {e}')
    sys.exit(1)

if 'Einzugsdatum' in df.columns:
    df['Einzugsdatum'] = pd.to_datetime(df['Einzugsdatum'], errors='coerce', dayfirst=True)
if 'Registrierungssdatum' in df.columns:
    df['Registrierungssdatum'] = pd.to_datetime(df['Registrierungssdatum'], errors='coerce', dayfirst=True)

missing_share = df.isna().mean().rename('anteil_fehlend')
missing_share.to_csv(os.path.join(output_dir, '2_eda_anteil_fehlender_werte.csv'), encoding='utf-8')

num_df = df.apply(pd.to_numeric, errors='coerce')
describe_num = num_df.describe().T
describe_num.to_csv(os.path.join(output_dir, '2_eda_kennwerte_numerisch.csv'), encoding='utf-8')

corr_num = num_df.corr(numeric_only=True)
corr_num.to_csv(os.path.join(output_dir, '2_eda_korrelationen_numerisch.csv'), encoding='utf-8')

alter_series = pd.to_numeric(df['Alter'], errors='coerce') if 'Alter' in df.columns else pd.Series(dtype=float)
geschl_series = pd.to_numeric(df['Geschlecht'], errors='coerce') if 'Geschlecht' in df.columns else pd.Series(dtype=float)

map_geschl = {1: 'männlich', 2: 'weiblich'}

df_plot = pd.DataFrame({'Alter': alter_series, 'Geschlecht': geschl_series}).copy()
df_plot['Geschlecht'] = df_plot['Geschlecht'].map(map_geschl)
df_plot = df_plot.dropna(subset=['Alter', 'Geschlecht'])
df_plot = df_plot[(df_plot['Alter'] >= 0) & (df_plot['Alter'] <= 99)]

if not df_plot.empty:
    age_gender_stats = df_plot.groupby('Geschlecht')['Alter'].agg(['count', 'mean', 'median', 'std', 'min', 'max']).rename(columns={'count': 'n'})
    age_gender_stats.to_csv(os.path.join(output_dir, '2_eda_alter_nach_geschlecht.csv'), encoding='utf-8')
else:
    pd.DataFrame(columns=['Geschlecht', 'n', 'mean', 'median', 'std', 'min', 'max']).to_csv(os.path.join(output_dir, '2_eda_alter_nach_geschlecht.csv'), index=False, encoding='utf-8')

palette = {'männlich': '#4C78A8', 'weiblich': '#F58518'}

plt.figure(figsize=(12, 7))
if df_plot.empty:
    fig = plt.gcf()
    fig.patch.set_facecolor('white')
    plt.axis('off')
    plt.text(0.5, 0.6, 'Keine gültigen Daten für Alter und Geschlecht verfügbar', ha='center', va='center', fontsize=16)
    plt.text(0.5, 0.45, 'Erwartet: Alter in [0, 99] und Geschlecht ∈ {1, 2}', ha='center', va='center', fontsize=12)
else:
    bins = np.arange(0, 105, 5)
    ax = sns.histplot(data=df_plot, x='Alter', hue='Geschlecht', bins=bins, multiple='layer', stat='count', palette=palette, edgecolor='white', alpha=0.65)
    ax.set_xlim(0, 100)
    ax.set_xlabel('Alter (Jahre)')
    ax.set_ylabel('Anzahl Personen')
    counts = df_plot['Geschlecht'].value_counts().to_dict()
    legend = ax.get_legend()
    if legend is not None:
        labels = []
        for text in legend.texts:
            key = text.get_text()
            n = counts.get(key, 0)
            labels.append(f'{key} (n={n})')
        for txt, new_label in zip(legend.texts, labels):
            txt.set_text(new_label)
    plt.xticks(np.arange(0, 101, 10))
    plt.yticks()
    plt.grid(axis='y', alpha=0.25)
    total_n = len(df_plot)
    n_m = counts.get('männlich', 0)
    n_w = counts.get('weiblich', 0)
    plt.title(f'Altersverteilung nach Geschlecht der registrierten Personen (2022)\nInsgesamt n={total_n}, männlich n={n_m}, weiblich n={n_w}')
    plt.tight_layout()

out_path = os.path.join(output_dir, plot_filename)
try:
    plt.savefig(out_path, dpi=200, bbox_inches='tight', facecolor='white')
    plt.close()
except Exception as e:
    print(f'Fehler beim Speichern der Grafik: {e}')
    sys.exit(1)
