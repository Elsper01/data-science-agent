import pandas as pd
import numpy as np

sns.set_theme(style="whitegrid", context="talk")

file_path = r"./src/resources/data/study\16492\dataset.csv"
output_path = r"./src/resources/output\16492/1_histogram_strahlungswerte.png"

df = pd.read_csv(file_path, sep=",", encoding="utf-8")

for col in df.columns:
    if df[col].dtype == object:
        try:
            df[col] = df[col].str.strip()
        except Exception:
            pass

missing_prop = df.isna().mean()

value = pd.to_numeric(df.get("value"), errors="coerce")
unit_col = df.get("unit")
validated_col = pd.to_numeric(df.get("validated"), errors="coerce") if "validated" in df.columns else None

if unit_col is not None:
    mask_unit = unit_col == "µSv/h"
else:
    mask_unit = pd.Series([True] * len(df))

if validated_col is not None:
    mask_valid = validated_col == 1
else:
    mask_valid = pd.Series([True] * len(df))

mask_numeric = value.notna()
mask = mask_unit & mask_valid & mask_numeric
v = value[mask]

n = int(v.shape[0])
mean_v = float(v.mean()) if n > 0 else np.nan
std_v = float(v.std(ddof=1)) if n > 1 else np.nan
min_v = float(v.min()) if n > 0 else np.nan
max_v = float(v.max()) if n > 0 else np.nan
median_v = float(v.median()) if n > 0 else np.nan
skew_v = float(v.skew()) if n > 0 else np.nan
miss_value_pct = float((~mask_numeric).mean() * 100)

bins = "auto"
if n > 1:
    q75, q25 = np.percentile(v, [75, 25])
    iqr = q75 - q25
    data_range = max_v - min_v if max_v is not None and min_v is not None else 0
    if iqr > 0 and data_range > 0:
        bw = 2 * iqr * (n ** (-1/3))
        if bw > 0:
            est_bins = int(np.ceil(data_range / bw))
            est_bins = max(5, min(est_bins, 40))
            bins = est_bins
    elif data_range > 0:
        est_bins = int(np.ceil(np.sqrt(n)))
        est_bins = max(5, min(est_bins, 40))
        bins = est_bins

plt.figure(figsize=(10, 6))
ax = plt.gca()

hist_color = "#4C78A8"
kde_color = "#B279A2"
norm_color = "#59A14F"

sns.histplot(v, bins=bins, stat="density", kde=False, color=hist_color, edgecolor="white", linewidth=0.7, alpha=0.9, ax=ax)

try:
    sns.kdeplot(v, color=kde_color, linewidth=2.0, ax=ax, label="KDE-Dichte")
except Exception:
    pass

if np.isfinite(mean_v) and np.isfinite(std_v) and std_v > 0:
    x_min = min_v - 3 * std_v
    x_max = max_v + 3 * std_v
    if np.isfinite(x_min) and np.isfinite(x_max) and x_max > x_min:
        xs = np.linspace(x_min, x_max, 300)
        pdf = (1 / (std_v * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((xs - mean_v) / std_v) ** 2)
        ax.plot(xs, pdf, color=norm_color, linewidth=2.0, label="Normalverteilung (μ, σ)")

if np.isfinite(mean_v):
    ax.axvline(mean_v, color="#E15759", linestyle="-", linewidth=2, label=f"Mittelwert: {mean_v:.3f} µSv/h")
if np.isfinite(median_v):

ax.set_title("Verteilung der Gamma-ODL-Messwerte (µSv/h) – Station DEZ1715, Jan 2026", pad=14)

import os
import sys
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

try:
    input_path = os.path.normpath(r'./src/resources/data/study\16492\dataset.csv')
    output_dir = os.path.normpath(r'./src/resources/output\16492')
    os.makedirs(output_dir, exist_ok=True)
    df = pd.read_csv(input_path, sep=',', encoding='utf-8')
    for col in ['start_measure', 'end_measure']:
        if col in df.columns:
            df[col] = pd.to_datetime(df[col], errors='coerce')
    numeric_value = pd.to_numeric(df['value'], errors='coerce') if 'value' in df.columns else pd.Series(dtype=float)
    unit = df['unit'].dropna().astype(str).mode().iloc[0] if 'unit' in df.columns and df['unit'].notna().any() else 'µSv/h'
    value_missing_ratio = float(numeric_value.isna().mean()) if len(numeric_value) > 0 else 1.0
    values = numeric_value.dropna()
    if values.empty:
        print('Fehler: Keine numerischen Werte in der Spalte value verfügbar.')
        sys.exit(1)
    n = int(values.shape[0])
    vmin = float(values.min())
    q1 = float(values.quantile(0.25))
    median = float(values.median())
    q3 = float(values.quantile(0.75))
    vmax = float(values.max())
    mean = float(values.mean())
    std = float(values.std(ddof=1)) if n > 1 else 0.0
    iqr = q3 - q1
    p5 = float(values.quantile(0.05))
    p95 = float(values.quantile(0.95))
    start_dt = df['start_measure'].min() if 'start_measure' in df.columns else None
    end_dt = df['end_measure'].max() if 'end_measure' in df.columns else None
    period_str = ''
    if pd.notna(start_dt) and pd.notna(end_dt):
        period_str = f" ({start_dt.strftime('%d.%m.%Y')}–{end_dt.strftime('%d.%m.%Y')})"
    sns.set_theme(style='whitegrid', context='talk')
    fig, ax = plt.subplots(figsize=(8, 6))
    bp = ax.boxplot(values.values, vert=True, notch=True, patch_artist=True, labels=['Gamma-ODL-Brutto'], whis=1.5, showmeans=True, meanline=False, meanprops=dict(marker='D', markerfacecolor='#D62728', markeredgecolor='white', markersize=8))
    for box in bp['boxes']:
        box.set(facecolor='#4C78A8', alpha=0.35, edgecolor='#4C78A8', linewidth=2)
    for whisker in bp['whiskers']:
        whisker.set(color='#4C78A8', linewidth=2)
    for cap in bp['caps']:
        cap.set(color='#4C78A8', linewidth=2)
    for median_line in bp['medians']:
        median_line.set(color='#2CA02C', linewidth=2.5)
    if 'fliers' in bp:
        for flier in bp['fliers']:
            flier.set(marker='o', markerfacecolor='#FF7F0E', markeredgecolor='white', alpha=0.7)
    ax.set_title(f'Verteilung der Ortsdosisleistungsrate in Lauf an der Pegnitz{period_str}', fontsize=14)
    ax.set_ylabel(f'ODL Dosisleistungsrate ({unit})', fontsize=12)
    ax.set_xlabel('Messgröße', fontsize=12)
    rng = vmax - vmin
    pad = rng * 0.1 if rng > 0 else 0.01
    ax.set_ylim(vmin - pad, vmax + pad)
    ytxt = (
        f'N = {n}\n'
        f'Mittelwert = {mean:.3f} {unit}\n'
        f'Median = {median:.3f} {unit}\n'
        f'Stdabw. = {std:.3f} {unit}\n'
        f'Min/Max = {vmin:.3f}/{vmax:.3f} {unit}\n'
        f'Q1/Q3 = {q1:.3f}/{q3:.3f} {unit}\n'
        f'IQR = {iqr:.3f} {unit}\n'
        f'P5/P95 = {p5:.3f}/{p95:.3f} {unit}\n'
        f'Fehlende Werte (value) = {value_missing_ratio*100:.1f}%'

    for spine in ['top', 'right']:
        ax.spines[spine].set_visible(False)
    output_path = os.path.join(output_dir, '2_boxplot_odl_value.png')
    plt.tight_layout()
    plt.savefig(output_path, dpi=200, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
    print(str(e))
    sys.exit(1)
