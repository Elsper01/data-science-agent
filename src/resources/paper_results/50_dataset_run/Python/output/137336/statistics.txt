import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style='whitegrid', context='talk')

path = './src/resources/data/study/137336/dataset.csv'
df = pd.read_csv(path, sep=',', encoding='utf-8', dtype=str, keep_default_na=True)

for c in df.columns:
    if df[c].dtype == object:
        df[c] = df[c].astype(str).str.replace('\u00a0', ' ', regex=False).str.strip()

missing_share = df.replace({'': np.nan}).isna().mean()
unit_counts = df['Einheit'].fillna('Unbekannt').value_counts(dropna=False)

def to_numeric_value(x):
    if x is None or (isinstance(x, float) and np.isnan(x)):
        return np.nan
    s = str(x).strip().lower()
    if s in {'', 'nan', 'na', 'n.a.', 'n/a', 'none'}:
        return np.nan
    s = s.replace(' ', '')
    s = s.replace(',', '.')
    if s in {'n.n.', 'n.n', 'nn'}:
        return np.nan
    if s.startswith('<') or s.startswith('>') or s.startswith('≤') or s.startswith('≥'):
        return np.nan
    try:
        return float(s)
    except Exception:
        return np.nan

if 'Ergebnis' in df.columns:
    df['Ergebnis_num'] = df['Ergebnis'].apply(to_numeric_value)
else:
    df['Ergebnis_num'] = np.nan

else:

mask_bqkg_tm = df['Einheit_clean'] == 'bq/kg(tm)'

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os
import re

try:
    data_path = os.path.normpath('./src/resources/data/study\\137336/dataset.csv')
    out_dir = os.path.normpath('./src/resources/output\\137336')
    os.makedirs(out_dir, exist_ok=True)
    plot_path = os.path.join(out_dir, '2_einheiten_countplot.png')
    eda_cols_path = os.path.join(out_dir, 'eda_spaltenuebersicht.csv')
    eda_num_summary_path = os.path.join(out_dir, 'eda_numeric_summary.csv')
    eda_corr_path = os.path.join(out_dir, 'eda_korrelationen.csv')

    df = pd.read_csv(data_path, sep=',', encoding='utf-8')
    df.columns = [str(c).strip() for c in df.columns]

    def clean_unit(x):
        if pd.isna(x):
            return 'fehlend'
        s = str(x).strip()
        s = re.sub(r'\s*/\s*', '/', s)
        s = s.replace('ug/', 'µg/').replace('μg/', 'µg/').replace('U g/', 'µg/')
        s = s.replace('g/100g', 'g/100 g')
        s = s.replace(' Bq', ' Bq').replace('Bq /', 'Bq/')
        s = s.replace(' %', '%')
        return s

    if 'Einheit' in df.columns:
        df['Einheit_clean'] = df['Einheit'].apply(clean_unit)
    else:
        df['Einheit_clean'] = 'fehlend'

    def parse_numeric(v):
        if pd.isna(v):
            return np.nan
        s = str(v).strip().lower()
        if s in {'n.n.', 'n.n', 'nn', 'na', 'nan', ''}:
            return np.nan
        s = s.replace(',', '.')
        s = s.replace('<', '').replace('>', '').replace('≤', '').replace('≥', '')
        m = re.findall(r'[-+]?\d*\.?\d+(?:[eE][-+]?\d+)?', s)
        if len(m) == 0:
            return np.nan
        try:
            return float(m[0])
        except Exception:
            return np.nan

    if 'Ergebnis' in df.columns:
        df['Ergebnis_num'] = df['Ergebnis'].apply(parse_numeric)

    col_summary = pd.DataFrame({
        'dtype': df.dtypes.astype(str),
        'anzahl_auspraegungen': df.nunique(dropna=True),
        'fehlend_anzahl': df.isna().sum(),
        'fehlend_anteil': (df.isna().mean().round(4))
    })
    col_summary.index.name = 'spalte'
    col_summary.reset_index().to_csv(eda_cols_path, index=False, encoding='utf-8')

    df_num = df.select_dtypes(include=[np.number]).copy()
    nunique_num = df_num.nunique(dropna=True)
    keep_numeric = nunique_num[nunique_num > 1].index.tolist()
    df_num_var = df_num[keep_numeric] if len(keep_numeric) > 0 else pd.DataFrame()

    if df_num_var.shape[1] > 0:
        desc = df_num_var.describe(percentiles=[0.05, 0.25, 0.5, 0.75, 0.95]).T
        desc.to_csv(eda_num_summary_path, encoding='utf-8')
    else:
        pd.DataFrame().to_csv(eda_num_summary_path, encoding='utf-8')

    if df_num_var.shape[1] >= 2:
        corr = df_num_var.corr(numeric_only=True, method='pearson')
        corr.to_csv(eda_corr_path, encoding='utf-8')
    else:
        pd.DataFrame().to_csv(eda_corr_path, encoding='utf-8')

    base = df.copy()
    if 'Pruefmerkmal' in base.columns:
        base = base[base['Pruefmerkmal'].notna()]

    unit_counts = base['Einheit_clean'].value_counts(dropna=False)
    units_order = unit_counts.index.tolist()
    n_units = len(units_order)
    max_count = int(unit_counts.max()) if n_units > 0 else 0

    sns.set_theme(style='whitegrid', context='talk')
    plt.rcParams['font.family'] = 'DejaVu Sans'
    width = 12
    height = max(6, 0.5 * n_units + 1)
    fig, ax = plt.subplots(figsize=(width, height))
    sns.countplot(data=base, y='Einheit_clean', order=units_order, color='#4C72B0', ax=ax)
    ax.set_xlabel('Anzahl der Prüfmerkmale')
    ax.set_ylabel('Maßeinheit')
    n_features = int(base['Pruefmerkmal'].nunique(dropna=True)) if 'Pruefmerkmal' in base.columns else int(base.shape[0])
    ax.set_title(f'Verteilung der Prüfmerkmale nach Maßeinheit (n={n_features})')
    ax.spines['right'].set_visible(False)
    ax.spines['top'].set_visible(False)
    for p in ax.patches:
        w = p.get_width()
        y = p.get_y() + p.get_height() / 2
        if np.isfinite(w):
            ax.text(w + max(0.02 * max_count, 0.1), y, f'{int(w)}', va='center', ha='left', fontsize=11)
    fig.tight_layout()
    fig.savefig(plot_path, dpi=200, bbox_inches='tight')
    plt.close(fig)
except Exception as e:
    import sys
    print(str(e))
    sys.exit(1)
