import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

if 'Einheit' in df.columns:
    df['Einheit'] = df['Einheit'].astype(str)

    df['Einheit'] = ''

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

input_path = './src/resources/data/study\\139606\\dataset.csv'
output_path = './src/resources/output\\139606\\run_1/1_aktivitaet_nachgewiesene_radionuklide.png'

sns.set_theme(style='whitegrid', context='talk')

df = pd.read_csv(input_path, sep=',', encoding='utf-8')

_ = df.isna().mean()

s = df['Ergebnis'].astype(str).str.strip().str.lower()

x = s.str.replace(',', '.', regex=False)

def _to_float(v):
    try:
        return float(v)
    except Exception:
        return np.nan

df['Ergebnis_num'] = x.apply(_to_float)

mask_unit = df['Einheit'].astype(str).str.strip() == 'Bq/kg(FM)'
mask_detected = df['Ergebnis_num'].notna()
df_plot = df.loc[mask_unit & mask_detected, ['Pruefmerkmal', 'Ergebnis_num']].copy()

df_plot = df_plot.sort_values('Ergebnis_num', ascending=False)

fig, ax = plt.subplots(figsize=(10, 6))

if df_plot.empty:
    ax.axis('off')
    fig.suptitle('Keine nachgewiesenen spezifischen Aktivitäten (Bq/kg FM) vorhanden', fontsize=16)
else:
    order = df_plot['Pruefmerkmal'].tolist()
    base_palette = sns.color_palette('colorblind', n_colors=max(3, len(order)))
    colors = []
    for name in order:
        if ('Cs-137' in str(name)) or ('Cäsium-137' in str(name)) or ('Caesium-137' in str(name)):
            colors.append((0.80, 0.20, 0.20))
        else:
            colors.append(base_palette[0])
    sns.barplot(data=df_plot, x='Pruefmerkmal', y='Ergebnis_num', order=order, palette=colors, ax=ax, edgecolor='black')
    ax.set_yscale('log')
    ymin = max(1e-6, df_plot['Ergebnis_num'].min() * 0.5)
    ymax = df_plot['Ergebnis_num'].max() * 10
    if ymax <= ymin:
        ymax = ymin * 10
    ax.set_ylim(ymin, ymax)
    ax.set_xlabel('Radionuklid (Prüfmerkmal)')
    ax.set_ylabel('Spezifische Aktivität [Bq/kg FM] (log)')
    title_date = str(df.loc[df.index.min(), 'Probenahme_Beginn_Datum']) if 'Probenahme_Beginn_Datum' in df.columns else ''
    sample_label = str(df.loc[df.index.min(), 'Bezeichnung']) if 'Bezeichnung' in df.columns else ''
    fig.suptitle('Nachgewiesene spezifische Aktivitätskonzentrationen in {} (Bq/kg FM)'.format(sample_label if sample_label != 'nan' else ''), fontsize=16)
    ax.set_title('IMIS-Gammaspektrometrie, Probenahme: {}'.format(title_date if title_date != 'nan' else ''), fontsize=12)
    ax.tick_params(axis='x', rotation=25)
    for p in ax.patches:
        v = p.get_height()
        if np.isfinite(v) and v > 0:
            if v >= 100:
                label = '{:.0f}'.format(v)
            elif v >= 1:
                label = '{:.2f}'.format(v)
            else:
                label = '{:.3f}'.format(v)
            x_pos = p.get_x() + p.get_width() / 2
            y_pos = v * 1.15
            ax.text(x_pos, y_pos, label, ha='center', va='bottom', fontsize=10)
    detected_n = int(df_plot.shape[0])
    min_v = float(df_plot['Ergebnis_num'].min())
    max_v = float(df_plot['Ergebnis_num'].max())
    median_v = float(df_plot['Ergebnis_num'].median())
    caption = 'Gefiltert: Einheit = Bq/kg(FM); Nachgewiesen (n) = {} | Median = {:.3g} Bq/kg | Min = {:.3g} | Max = {:.3g}'.format(detected_n, median_v, min_v, max_v)
    fig.text(0.01, 0.01, caption, ha='left', va='bottom', fontsize=10)

plt.tight_layout(rect=[0, 0.03, 1, 0.95])
fig.savefig(output_path, dpi=300)

def parse_german_number(x):
    if pd.isna(x):
        return np.nan
    if isinstance(x, (int, float, np.integer, np.floating)):
        return float(x)
    s = str(x).strip()
    if s == '':
        return np.nan
    sl = s.lower()
    if sl in {'n.n.', 'n.n', 'nn', 'na', 'n/a', 'nicht nachweisbar'}:
        return np.nan
    s = s.replace(' ', '')
    if ',' in s:
        s = s.replace('.', '')
        s = s.replace(',', '.')
    allowed = set('0123456789eE+-.')
    s = ''.join(ch for ch in s if ch in allowed)
    try:
        return float(s)
    except Exception:
        return np.nan

df = pd.read_csv(os.path.normpath(data_path_raw), sep=',', encoding='utf-8')

for c in df.columns:
