import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Create output directory
os.makedirs('./src/resources/output/103779', exist_ok=True)

# Load dataset
df = pd.read_csv('./src/resources/data/study/103779/dataset.csv', sep=',', encoding='utf-8')

# The error log shows the column is named 'Begr_ndung' instead of 'Begründung'
# Let's normalize column names to handle potential encoding/naming issues
col_map = {
    'Begr_ndung': 'Begruendung',
    'Begründung': 'Begruendung',
    'Kulturdenkmaltyp': 'Kulturdenkmaltyp'
}
df = df.rename(columns=col_map)

# Verify column existence
target_col = 'Begruendung'
if target_col not in df.columns:
    # Fallback: find the column that looks like Begruendung if renaming failed
    potential_cols = [c for c in df.columns if 'Begr' in c]
    if potential_cols:
        df = df.rename(columns={potential_cols[0]: 'Begruendung'})
        target_col = 'Begruendung'

print('Columns in dataset:', df.columns.tolist())

# Data Cleaning: Filter rows where both key columns are present
df_viz = df[df[target_col].notna() & df['Kulturdenkmaltyp'].notna()].copy()

# Process the 'Begruendung' column which contains comma-separated values
# e.g., "geschichtlich, städtebaulich"
df_viz['Begruendung_list'] = df_viz[target_col].str.split(r',\s*')
exploded = df_viz.explode('Begruendung_list')
exploded['Begruendung_clean'] = exploded['Begruendung_list'].str.strip()

# Remove empty strings or nulls after explosion
exploded = exploded[exploded['Begruendung_clean'].str.len() > 0]

# Create a crosstab for the stacked bar chart
# We limit to top categories if there are too many to keep the chart readable
crosstab = pd.crosstab(exploded['Kulturdenkmaltyp'], exploded['Begruendung_clean'])

# Plotting
plt.figure(figsize=(14, 8))
# Use a nice color palette
colors = sns.color_palette("viridis", n_colors=len(crosstab.columns))
crosstab.plot(kind='bar', stacked=True, color=colors, ax=plt.gca(), edgecolor='white', linewidth=0.5)

plt.title('Häufigkeit der Schutzbegründungen nach Kulturdenkmaltyp', fontsize=16, pad=20)
plt.xlabel('Kulturdenkmaltyp', fontsize=12)
plt.ylabel('Anzahl der Nennungen', fontsize=12)
plt.xticks(rotation=30, ha='right')
plt.legend(title='Begründung', bbox_to_anchor=(1.05, 1), loc='upper left', fontsize=10)

plt.tight_layout()
plt.savefig('./src/resources/output/103779/1_stacked_begruendung_kulturdenkmaltyp.png', dpi=300, bbox_inches='tight')
plt.close()

print("Visualization saved successfully.")

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

sns.set_style('whitegrid')

df = pd.read_csv('./src/resources/data/study/103779/dataset.csv', sep=',', encoding='utf-8')

alteintragung = (df['Beschreibung'] == 'Alteintragung (Aktualisierung vorgesehen)').sum()
has_photo = df['FotoURL'].notna().sum()

totals = df.groupby('Kulturdenkmaltyp').size()
alte_counts = df.groupby('Kulturdenkmaltyp').apply(lambda x: (x['Beschreibung'] == 'Alteintragung (Aktualisierung vorgesehen)').sum())
photo_counts = df.groupby('Kulturdenkmaltyp').apply(lambda x: x['FotoURL'].notna().sum())

summary = pd.DataFrame({
    'Alteintragungen': alte_counts,
    'Mit Foto': photo_counts,
    'Gesamt': totals
}).fillna(0)

summary = summary.sort_values('Gesamt', ascending=False)

fig, ax = plt.subplots(figsize=(14, 8))

x = np.arange(len(summary.index))
width = 0.35

bars1 = ax.bar(x - width/2, summary['Alteintragungen'], width, label='Alteintragungen\n(Aktualisierung vorgesehen)', color='orange', alpha=0.8)
bars2 = ax.bar(x + width/2, summary['Mit Foto'], width, label='Einträge mit FotoURL', color='green', alpha=0.8)

ax.set_xlabel('Kulturdenkmaltyp', fontsize=12)
ax.set_ylabel('Anzahl', fontsize=12)
ax.set_title('Status der digitalen Dokumentation und Modernisierung\nnach Kulturdenkmaltyp', fontsize=14, fontweight='bold', pad=20)
ax.set_xticks(x)
ax.set_xticklabels(summary.index, rotation=45, ha='right', fontsize=11)
ax.legend(loc='upper right', fontsize=11)

plt.tight_layout()
plt.savefig('./src/resources/output/103779/2_status_digital_dokumentation.png', dpi=300, bbox_inches='tight')
plt.close()
