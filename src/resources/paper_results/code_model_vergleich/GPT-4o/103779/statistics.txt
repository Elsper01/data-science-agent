import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Use raw strings for file paths to avoid escape sequence issues
file_path = r'./src/resources/data/103779/dataset.csv'

# Ensure the directory exists or handle potential path issues
try:
    df = pd.read_csv(file_path, sep=',', encoding='utf-8')
except FileNotFoundError:
    # Fallback for different environments if the relative path structure varies
    import glob
    possible_files = glob.glob('**/103779/dataset.csv', recursive=True)
    if possible_files:
        df = pd.read_csv(possible_files[0], sep=',', encoding='utf-8')
    else:
        raise FileNotFoundError(f"Could not find dataset.csv in expected paths.")

# Count the number of unique monuments (Objektnummer) per municipality (Gemeinde)
# Grouping by Gemeinde and counting Objektnummer
municipality_counts = df.groupby('Gemeinde')['Objektnummer'].count().sort_values(ascending=False).head(10)

# Plotting
plt.figure(figsize=(12, 8))
sns.barplot(x=municipality_counts.values, y=municipality_counts.index, hue=municipality_counts.index, palette='viridis', legend=False)

plt.title('Top 10 Gemeinden mit der höchsten Konzentration an Denkmälern', fontsize=16)
plt.xlabel('Anzahl der registrierten Denkmäler (Objektnummer)', fontsize=14)
plt.ylabel('Gemeinde', fontsize=14)
plt.xticks(fontsize=12)
plt.yticks(fontsize=12)
plt.tight_layout()

# Save the plot
output_dir = r'./src/resources/output/103779'
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

import pandas as pd
import matplotlib.pyplot as plt
import io

# Since the file 'data.csv' was not found in the previous environment, 
# we will use the provided data loading mechanism which typically involves 
# reading from the environment's data source. 
# In this context, we assume the data is available as 'data.csv'.

def create_visualization():
    try:
        # Load the dataset
        df = pd.read_csv('data.csv')

        # The goal is to analyze the 'Begr_ndung' column.
        # This column often contains multiple justifications separated by commas (e.g., "Geschichtlich, Städtebaulich")
        if 'Begr_ndung' in df.columns:
            # Drop missing values and convert to string
            justifications = df['Begr_ndung'].dropna().astype(str)

            # Split the justifications into individual reasons and flatten the list
            # We use a comma as the delimiter
            all_reasons = []
            for entry in justifications:
                reasons = [r.strip() for r in entry.split(',') if r.strip()]
                all_reasons.extend(reasons)

            # Count the occurrences of each justification
            justification_counts = pd.Series(all_reasons).value_counts()

            # Plotting the justifications
            plt.figure(figsize=(12, 8))
            justification_counts.plot(kind='bar', color='skyblue', edgecolor='black')
            
            plt.title('Primary Legal Justifications for Monument Protection', fontsize=16)
            plt.xlabel('Justification (Begründung)', fontsize=12)
            plt.ylabel('Frequency', fontsize=12)
            plt.xticks(rotation=45, ha='right')
            plt.grid(axis='y', linestyle='--', alpha=0.7)
            plt.tight_layout()

            # Display the plot
            plt.show()
            
        else:
            print(f"Error: Column 'Begr_ndung' not found. Available columns: {df.columns.tolist()}")

    except FileNotFoundError:
        # If the file is missing, we'll print a helpful message. 
        # In a real environment, ensure the file path is correct.
        print("Error: 'data.csv' not found. Please ensure the dataset is uploaded correctly.")
    except Exception as e:
        print(f"An unexpected error occurred: {e}")

create_visualization()
