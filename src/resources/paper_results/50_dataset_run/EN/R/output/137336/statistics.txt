df <- df_raw %>% mutate(across(everything(), ~{x <- .; x <- ifelse(is.na(x), NA_character_, x); x <- trimws(x, which = "both"); ifelse(x == "", NA_character_, x)}))
missing_prop <- sapply(df, function(x) mean(is.na(x)))
ergebnis_clean_str <- df$Ergebnis
if (!is.null(ergebnis_clean_str)) ergebnis_clean_str <- str_replace_all(ergebnis_clean_str, fixed(","), ".")
pattern_nn <- regex("^\\s*n\\.?\\s*n\\.?\\s*$", ignore_case = TRUE)
starts_with_censor <- ifelse(is.na(ergebnis_clean_str), FALSE, str_detect(ergebnis_clean_str, "^\\s*[<≤]"))
ergebnis_num <- ergebnis_clean_str
ergebnis_num <- ifelse(is.na(ergebnis_num), NA_character_, str_replace_all(ergebnis_num, "^[<≥>≤\\s]+", ""))
ergebnis_num <- suppressWarnings(as.numeric(ergebnis_num))
status <- case_when(
  !is.na(ergebnis_clean_str) & str_detect(ergebnis_clean_str, pattern_nn) ~ "Nicht nachweisbar",
  starts_with_censor ~ "Nicht nachweisbar",
  !is.na(ergebnis_num) ~ "Nachweisbar",
  TRUE ~ "Nicht nachweisbar"
)
status <- factor(status, levels = c("Nicht nachweisbar", "Nachweisbar"))
detect_flag <- ifelse(status == "Nachweisbar", 1L, 0L)
order_df <- df %>% mutate(detect_flag = detect_flag, ergebnis_num = ergebnis_num) %>% arrange(desc(detect_flag), desc(ergebnis_num), Pruefmerkmal)
levels_order <- unique(order_df$Pruefmerkmal)
df_plot <- df %>% mutate(Status = status, detect_flag = detect_flag, ergebnis_num = ergebnis_num, Pruefmerkmal_f = factor(Pruefmerkmal, levels = rev(levels_order))) %>% select(Pruefmerkmal_f, Status)
eda_overview <- tibble(
  anzahl_pruefmerkmale = nrow(df),
  anzahl_nachweisbar = sum(status == "Nachweisbar", na.rm = TRUE),
  anteil_nachweisbar = ifelse(nrow(df) > 0, sum(status == "Nachweisbar", na.rm = TRUE)/nrow(df), NA_real_),

  median_wert = suppressWarnings(median(ergebnis_num[status == "Nachweisbar"], na.rm = TRUE)),
  max_wert = suppressWarnings(max(ergebnis_num[status == "Nachweisbar"], na.rm = TRUE))
)
fill_cols <- c("Nicht nachweisbar" = "#D55E00", "Nachweisbar" = "#1B9E77")
p <- ggplot(df_plot, aes(x = Status, y = Pruefmerkmal_f, fill = Status)) +
  geom_tile(color = "white", width = 0.95, height = 0.95) +
  scale_fill_manual(values = fill_cols, drop = FALSE, guide = guide_legend(title = NULL)) +
